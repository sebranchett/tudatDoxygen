480 template< int ObservationSize >
481 std::shared_ptr< ObservationPartial< ObservationSize > > createObservationPartialWrtLinkProperty(
482  const observation_models::LinkEnds& linkEnds,
483  const observation_models::ObservableType observableType,
484  const std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > parameterToEstimate,
485  const bool useBiasPartials = true )
486 {
487  std::shared_ptr< ObservationPartial< ObservationSize > > observationPartial;
488 
489  // Check parameter type
490  switch( parameterToEstimate->getParameterName( ).first )
491  {
492  case estimatable_parameters::constant_additive_observation_bias:
493  {
494  if( useBiasPartials )
495  {
496  // Check input consistency
497  std::shared_ptr< estimatable_parameters::ConstantObservationBiasParameter > constantBias =
498  std::dynamic_pointer_cast< estimatable_parameters::ConstantObservationBiasParameter >(
499  parameterToEstimate );
500  if( constantBias == nullptr )
501  {
502  throw std::runtime_error( "Error when making partial w.r.t. observation bias, type is inconsistent" );
503  }
504  else
505  {
506  // Check dependency between parameter and link properties.
507  if( linkEnds == constantBias->getLinkEnds( ) && observableType == constantBias->getObservableType( ) )
508  {
509  observationPartial = std::make_shared< ObservationPartialWrtConstantAbsoluteBias< ObservationSize > >(
510  observableType, linkEnds );
511  }
512  }
513  }
514  break;
515  }
516  case estimatable_parameters::arcwise_constant_additive_observation_bias:
517  {
518  if( useBiasPartials )
519  {
520  // Check input consistency
521  std::shared_ptr< estimatable_parameters::ArcWiseObservationBiasParameter > arcwiseBias =
522  std::dynamic_pointer_cast< estimatable_parameters::ArcWiseObservationBiasParameter >(
523  parameterToEstimate );
524  if( arcwiseBias == nullptr )
525  {
526  throw std::runtime_error( "Error when making partial w.r.t. arcwise observation bias, type is inconsistent" );
527  }
528  else
529  {
530  // Check dependency between parameter and link properties.
531  if( linkEnds == arcwiseBias->getLinkEnds( ) && observableType == arcwiseBias->getObservableType( ) )
532  {
533  observationPartial = std::make_shared< ObservationPartialWrtArcWiseAbsoluteBias< ObservationSize > >(
534  observableType, linkEnds,
535  arcwiseBias->getLookupScheme( ),
536  arcwiseBias->getLinkEndIndex( ),
537  arcwiseBias->getArcStartTimes( ).size( ) );
538  }
539  }
540  }
541  break;
542  }
543  case estimatable_parameters::constant_relative_observation_bias:
544  {
545  if( useBiasPartials )
546  {
547  // Check input consistency
548  std::shared_ptr< estimatable_parameters::ConstantObservationBiasParameter > constantBias =
549  std::dynamic_pointer_cast< estimatable_parameters::ConstantObservationBiasParameter >(
550  parameterToEstimate );
551  if( constantBias == nullptr )
552  {
553  throw std::runtime_error( "Error when making partial w.r.t. observation bias, type is inconsistent" );
554  }
555  else
556  {
557  // Check dependency between parameter and link properties.
558  if( linkEnds == constantBias->getLinkEnds( ) && observableType == constantBias->getObservableType( ) )
559  {
560  observationPartial = std::make_shared< ObservationPartialWrtConstantRelativeBias< ObservationSize > >(
561  observableType, linkEnds );
562 
563  }
564  }
565  break;
566  }
567  }
568  case estimatable_parameters::arcwise_constant_relative_observation_bias:
569  {
570  if( useBiasPartials )
571  {
572  // Check input consistency
573  std::shared_ptr< estimatable_parameters::ArcWiseObservationBiasParameter > arcwiseBias =
574  std::dynamic_pointer_cast< estimatable_parameters::ArcWiseObservationBiasParameter >(
575  parameterToEstimate );
576  if( arcwiseBias == nullptr )
577  {
578  throw std::runtime_error( "Error when making partial w.r.t. arcwise relative observation bias, type is inconsistent" );
579  }
580  else
581  {
582  // Check dependency between parameter and link properties.
583  if( linkEnds == arcwiseBias->getLinkEnds( ) && observableType == arcwiseBias->getObservableType( ) )
584  {
585  observationPartial = std::make_shared< ObservationPartialWrtArcWiseRelativeBias< ObservationSize > >(
586  observableType, linkEnds,
587  arcwiseBias->getLookupScheme( ),
588  arcwiseBias->getLinkEndIndex( ),
589  arcwiseBias->getArcStartTimes( ).size( ) );
590  }
591  }
592  }
593  break;
594  }
595  default:
596  break;
597  }
598 
599  return observationPartial;
600 }
