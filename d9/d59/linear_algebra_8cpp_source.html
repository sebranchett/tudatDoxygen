48 Eigen::Vector4d quaternionProduct( const Eigen::Vector4d& firstQuaternion, const Eigen::Vector4d& secondQuaternion )
49 {
50  Eigen::Vector4d resultantQuaternion;
51  resultantQuaternion[ 0 ] = firstQuaternion[ 0 ] * secondQuaternion[ 0 ] -
52  firstQuaternion.segment( 1, 3 ).dot( secondQuaternion.segment( 1, 3 ) );
53  resultantQuaternion.segment( 1, 3 ) = firstQuaternion[ 0 ] * secondQuaternion.segment( 1, 3 ) +
54  secondQuaternion[ 0 ] * firstQuaternion.segment( 1, 3 ) +
55  getCrossProductMatrix( firstQuaternion.segment( 1, 3 ) ) * secondQuaternion.segment( 1, 3 );
56  return resultantQuaternion;

79 double computeCosineOfAngleBetweenVectors( const Eigen::VectorXd& vector0,
80  const Eigen::VectorXd& vector1 )
81 {
82  if( !( vector0.size( ) == vector1.size( ) ) )
83  {
84  throw std::runtime_error( "Error when computing angle between vectors; size is incompatible" );
85  }
86 
87  // Get the cosine of the angle by dotting the normalized vectors.
88  double dotProductOfNormalizedVectors = vector0.normalized( ).dot( vector1.normalized( ) );
89 
90  // Explicitly define the extreme cases, which can give problems with the acos function.
91  if ( dotProductOfNormalizedVectors >= 1.0 )
92  {
93  return 1.0;
94  }
95 
96  else if ( dotProductOfNormalizedVectors <= -1.0 )
97  {
98  return -1.0;
99  }
100  // Determine the actual angle.
101  else
102  {
103  return dotProductOfNormalizedVectors;
104  }
105 }

108 double computeAngleBetweenVectors( const Eigen::VectorXd& vector0, const Eigen::VectorXd& vector1 )
109 {
110  // Determine the cosine of the angle by using another routine.
111  double dotProductOfNormalizedVectors = computeCosineOfAngleBetweenVectors( vector0, vector1 );
112 
113  // Return arccosine of the above, which is effectively the angle.
114  return std::acos( dotProductOfNormalizedVectors );
115 }

154 double getVectorEntryRootMeanSquare( const Eigen::VectorXd& inputVector )
155 {
