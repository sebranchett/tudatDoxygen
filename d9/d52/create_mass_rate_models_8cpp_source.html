21 std::shared_ptr< basic_astrodynamics::MassRateModel >
22 createMassRateModel(
23  const std::string& bodyWithMassRate,
24  const std::shared_ptr< MassRateModelSettings > massRateModelSettings,
25  const NamedBodyMap& bodyMap,
26  const basic_astrodynamics::AccelerationMap& accelerationModels )
27 {
28  std::shared_ptr< basic_astrodynamics::MassRateModel > massRateModel;
29 
30  // Check type of mass rate model.
31  switch( massRateModelSettings->massRateType_ )
32  {
33  case basic_astrodynamics::custom_mass_rate_model:
34  {
35  // Check input consistency
36  std::shared_ptr< CustomMassRateModelSettings > customMassRateModelSettings =
37  std::dynamic_pointer_cast< CustomMassRateModelSettings >( massRateModelSettings );
38  if( customMassRateModelSettings == nullptr )
39  {
40  throw std::runtime_error( "Error when making cusom mass rate model, input is inconsistent" );
41  }
42  else
43  {
44  massRateModel = std::make_shared< basic_astrodynamics::CustomMassRateModel >(
45  customMassRateModelSettings->massRateFunction_ );
46  }
47  break;
48  }
49  case basic_astrodynamics::from_thrust_mass_rate_model:
50  {
51  // Check input consistency
52  std::shared_ptr< FromThrustMassModelSettings > fromThrustMassModelSettings =
53  std::dynamic_pointer_cast< FromThrustMassModelSettings >( massRateModelSettings );
54  if( fromThrustMassModelSettings == nullptr )
55  {
56  throw std::runtime_error( "Error when making from-engine mass rate model, input is inconsistent" );
57  }
58  else
59  {
60  std::vector< std::shared_ptr< basic_astrodynamics::AccelerationModel3d > >
61  thrustAccelerations;
62  if( accelerationModels.count( bodyWithMassRate ) != 0 )
63  {
64  if( accelerationModels.at( bodyWithMassRate ).count( bodyWithMassRate ) != 0 )
65  {
66  thrustAccelerations = basic_astrodynamics::getAccelerationModelsOfType(
67  accelerationModels.at( bodyWithMassRate ).at( bodyWithMassRate ),
68  basic_astrodynamics::thrust_acceleration );
69  }
70  }
71 
72  if( thrustAccelerations.size( ) == 0 )
73  {
74  std::cerr << "Warning when making from-thrust mass-rate model, no thrust model is found; no thust is used" << std::endl;
75  }
76 
77  std::vector< std::shared_ptr< propulsion::ThrustAcceleration > >
78  explicitThrustAccelerations;
79 
80  if( fromThrustMassModelSettings->useAllThrustModels_ == 0 )
81  {
82  // Retrieve thrust models with the correct id (should be 1)
83  for( unsigned int i = 0; i < thrustAccelerations.size( ); i++ )
84  {
85  if( std::dynamic_pointer_cast< propulsion::ThrustAcceleration >(
86  thrustAccelerations.at( i ) )->getAssociatedThrustSource( ) ==
87  fromThrustMassModelSettings->associatedThrustSource_ )
88  {
89  explicitThrustAccelerations.push_back( std::dynamic_pointer_cast< propulsion::ThrustAcceleration >(
90  thrustAccelerations.at( i ) ) );
91  }
92  }
93 
94  if( explicitThrustAccelerations.size( ) != 1 )
95  {
96  std::cerr << "Warning when making from-thrust mass-rate model, did not find exactly 1 thrust model with correct identifier" << std::endl;
97  }
98  }
99  else
100  {
101  // Combine all thrust models into list
102  for( unsigned int i = 0; i < thrustAccelerations.size( ); i++ )
103  {
104  explicitThrustAccelerations.push_back( std::dynamic_pointer_cast< propulsion::ThrustAcceleration >(
105  thrustAccelerations.at( i ) ) );
106  }
107  }
108 
109  // Create mass rate model
110  massRateModel = std::make_shared< propulsion::FromThrustMassRateModel >(
111  explicitThrustAccelerations );
112  }
113  break;
114  }
115  default:
116  throw std::runtime_error( "Error when making mass rate model, type not recognized" );
117 
118  }
119  return massRateModel;
120 }

124 basic_astrodynamics::MassRateModelMap createMassRateModelsMap(
125  const NamedBodyMap& bodyMap,
126  const SelectedMassRateModelMap& massRateModelSettings,
127  const basic_astrodynamics::AccelerationMap& accelerationModels )
128 {
129  // Iterate over all bodies
130  std::map< std::string, std::vector< std::shared_ptr< basic_astrodynamics::MassRateModel > > > massRateModels;
131  for( std::map< std::string, std::vector< std::shared_ptr< MassRateModelSettings > > >::const_iterator settingsIterator =
132  massRateModelSettings.begin( ); settingsIterator != massRateModelSettings.end( ); settingsIterator++)
133  {
134  // Iterate over all mass model settings for current body.
135  for( unsigned int i = 0; i < settingsIterator->second.size( ); i++ )
136  {
137  massRateModels[ settingsIterator->first ].push_back(
138  createMassRateModel( settingsIterator->first, settingsIterator->second.at( i ), bodyMap,
139  accelerationModels ) );
140  }
141  }
142  return massRateModels;
143
