136  void updateFilter( const DependentVector& currentMeasurementVector )
137  {
138  // Compute sigma points
139  computeSigmaPoints( this->aPosterioriStateEstimate_, this->aPosterioriCovarianceEstimate_ );
140  historyOfSigmaPoints_[ this->currentTime_ ] = mapOfSigmaPoints_; // store points
141 
142  // Prediction step
143  // Compute series of state estimates based on sigma points
144  std::map< unsigned int, DependentVector > sigmaPointsStateEstimates;
145  for ( sigmaPointConstantIterator_ = mapOfSigmaPoints_.begin( );
146  sigmaPointConstantIterator_ != mapOfSigmaPoints_.end( ); sigmaPointConstantIterator_++ )
147  {
148  currentSigmaPoint_ = sigmaPointConstantIterator_->first;
149  sigmaPointsStateEstimates[ currentSigmaPoint_ ] = this->predictState(
150  sigmaPointConstantIterator_->second.segment( 0, stateDimension_ ) );
151  }
152 
153  // Compute the weighted average to find the a-priori state vector
154  DependentVector aPrioriStateEstimate = DependentVector::Zero( stateDimension_ );
155  computeWeightedAverageFromSigmaPointEstimates( aPrioriStateEstimate, sigmaPointsStateEstimates );
156 
157  // Compute the weighted average to find the a-priori covariance matrix
158  DependentMatrix aPrioriCovarianceEstimate = DependentMatrix::Zero( stateDimension_, stateDimension_ );
159  computeWeightedAverageFromSigmaPointEstimates( aPrioriCovarianceEstimate, aPrioriStateEstimate, sigmaPointsStateEstimates );
160 
161  // Re-compute sigma points
162  computeSigmaPoints( aPrioriStateEstimate, aPrioriCovarianceEstimate );
163 
164  // Compute series of measurement estimates based on sigma points
165  std::map< unsigned int, DependentVector > sigmaPointsMeasurementEstimates;
166  for ( sigmaPointConstantIterator_ = mapOfSigmaPoints_.begin( );
167  sigmaPointConstantIterator_ != mapOfSigmaPoints_.end( ); sigmaPointConstantIterator_++ )
168  {
169  currentSigmaPoint_ = sigmaPointConstantIterator_->first;
170  sigmaPointsMeasurementEstimates[ currentSigmaPoint_ ] = this->measurementFunction_(
171  this->currentTime_, sigmaPointConstantIterator_->second.segment( 0, stateDimension_ ) );
172  }
173 
174  // Compute the weighted average to find the expected measurement vector
175  DependentVector measurementEstimate = DependentVector::Zero( measurementDimension_ );
176  computeWeightedAverageFromSigmaPointEstimates( measurementEstimate, sigmaPointsMeasurementEstimates );
177 
178  // Compute innovation and cross-correlation matrices
179  DependentMatrix innovationMatrix = DependentMatrix::Zero( measurementDimension_, measurementDimension_ );
180  computeWeightedAverageFromSigmaPointEstimates( innovationMatrix, measurementEstimate, sigmaPointsMeasurementEstimates );
181  DependentMatrix crossCorrelationMatrix = DependentMatrix::Zero( stateDimension_, measurementDimension_ );
182  for ( sigmaPointConstantIterator_ = sigmaPointsStateEstimates.begin( );
183  sigmaPointConstantIterator_ != sigmaPointsStateEstimates.end( ); sigmaPointConstantIterator_++ )
184  {
185  crossCorrelationMatrix += covarianceEstimationWeights_.at( sigmaPointConstantIterator_->first ) *
186  ( sigmaPointConstantIterator_->second - aPrioriStateEstimate ) *
187  ( sigmaPointsMeasurementEstimates[ sigmaPointConstantIterator_->first ] - measurementEstimate ).transpose( );
188  }
189 
190  // Compute Kalman gain
191  DependentMatrix kalmanGain = crossCorrelationMatrix * innovationMatrix.inverse( );
192 
193  // Correction step
194  this->currentTime_ += this->filteringStepSize_;
195  this->correctState( aPrioriStateEstimate, currentMeasurementVector, measurementEstimate, kalmanGain );
196  correctCovariance( aPrioriCovarianceEstimate, innovationMatrix, kalmanGain );
197  }

313  void computeSigmaPoints( const DependentVector& currentStateEstimate, const DependentMatrix& currentCovarianceEstimate )
314  {
315  // Update augmented state and covariance matrix to new values
316  augmentedStateVector_.segment( 0, stateDimension_ ) = currentStateEstimate;
317  augmentedCovarianceMatrix_.topLeftCorner( stateDimension_, stateDimension_ ) = currentCovarianceEstimate;
318 
319  // Pre-compute square root of augmented covariance matrix
320  DependentMatrix augmentedCovarianceMatrixSquareRoot;
321  if ( augmentedCovarianceMatrix_.determinant( ) != static_cast< DependentVariableType >( 0.0 ) )
322  {
323  // Matrix is invertible, so take matrix square-root directly
324  augmentedCovarianceMatrixSquareRoot = augmentedCovarianceMatrix_.sqrt( );
325  }
326  else
327  {
328  // Check if matrix is diagonal
329  DependentVector augmentedCovarianceMatrixDiagonal = augmentedCovarianceMatrix_.diagonal( );
330  if ( augmentedCovarianceMatrix_.isApprox( DependentMatrix( augmentedCovarianceMatrixDiagonal.asDiagonal( ) ) ) )
331  {
332  augmentedCovarianceMatrixDiagonal = augmentedCovarianceMatrixDiagonal.array( ).sqrt( );
333  augmentedCovarianceMatrixSquareRoot = augmentedCovarianceMatrixDiagonal.asDiagonal( );
334  }
335  else
336  {
337  // Otherwise, matrix is singular and the square-root cannot be computed
338  throw std::runtime_error( "Error in unscented Kalman filter. Augemented covariance matrix is singular. "
339  "Its square-root cannot be computed." );
340  }
341  }
342 
343  // Loop over sigma points and assign value
344  for ( unsigned int i = 0; i < numberOfSigmaPoints_; i++ )
345  {
346  if ( i == 0 )
347  {
348  mapOfSigmaPoints_[ i ] = augmentedStateVector_;
349  }
350  else if ( i < ( augmentedCovarianceMatrixSquareRoot.cols( ) + 1 ) )
351  {
352  mapOfSigmaPoints_[ i ] = augmentedStateVector_ + constantParameters_.at( gamma_index ) *
353  augmentedCovarianceMatrixSquareRoot.col( i - 1 );
354  }
355  else
356  {
357  mapOfSigmaPoints_[ i ] = augmentedStateVector_ - constantParameters_.at( gamma_index ) *
358  augmentedCovarianceMatrixSquareRoot.col( ( i - 1 ) - augmentedCovarianceMatrixSquareRoot.cols( ) );
359  }
360  }
361  }

495 template< typename IndependentVariableType, typename DependentVariableType >
496 void UnscentedKalmanFilter< IndependentVariableType, DependentVariableType >::setConstantParameterValues(
497  const ConstantParameterReferences constantValueReference,
498  const std::pair< DependentVariableType, DependentVariableType >& customConstantParameters )
499 {
500  // Assign size to vector
501  constantParameters_.resize( 5 );
502 
503  // Set parameters based on input
504  switch ( constantValueReference )
505  {
506  case reference_Wan_and_Van_der_Merwe:
507  {
508  constantParameters_.at( alpha_index ) = static_cast< DependentVariableType >( 0.003 );
509  constantParameters_.at( kappa_index ) = static_cast< DependentVariableType >( 0.0 );
510  break;
511  }
512  case reference_Lisano_and_Born_and_Axelrad:
513  {
514  constantParameters_.at( alpha_index ) = static_cast< DependentVariableType >( 1.0 );
515  constantParameters_.at( kappa_index ) = static_cast< DependentVariableType >( 3.0 - stateDimension_ );
516  break;
517  }
518  case reference_Challa_and_Moore_and_Rogers:
519  {
520  constantParameters_.at( alpha_index ) = static_cast< DependentVariableType >( 0.001 );
521  constantParameters_.at( kappa_index ) = static_cast< DependentVariableType >( 1.0 );
522  break;
523  }
524  case custom_parameters:
525  {
526  // Check that the values have been set
527  if ( customConstantParameters.first == static_cast< DependentVariableType >( TUDAT_NAN ) ||
528  customConstantParameters.second == static_cast< DependentVariableType >( TUDAT_NAN ) )
529  {
530  throw std::runtime_error( "Error in unscented Kalman filter. The value of the alpha and kappa parameters "
531  "have not been specified, but the selected method is custom_parameters." );
532  }
533 
534  // Assign values to parameters
535  constantParameters_.at( alpha_index ) = customConstantParameters.first;
536  constantParameters_.at( kappa_index ) = customConstantParameters.second;
537  break;
538  }
539  default:
540  throw std::runtime_error( "Error in unscented Kalman filter. The name of the reference for the alpha and kappa "
541  "parameters is not recognized. To enter a custom pair of coefficients, "
542  "use the value custom_parameters." );
543  }
544 
545  // Set augmented state and sigma parameters
546  augmentedStateDimension_ = 2 * stateDimension_ + measurementDimension_;
547  numberOfSigmaPoints_ = 2 * augmentedStateDimension_ + 1;
548 
549  // Set remaining parameters
550  constantParameters_.at( beta_index ) = 2.0;
551  constantParameters_.at( lambda_index ) = std::pow( constantParameters_.at( alpha_index ), 2 ) *
552  ( augmentedStateDimension_ + constantParameters_.at( kappa_index ) ) - augmentedStateDimension_;
553  constantParameters_.at( gamma_index ) = std::sqrt( augmentedStateDimension_ + constantParameters_.at( lambda_index ) );
554 }

557 template< typename IndependentVariableType, typename DependentVariableType >
558 void UnscentedKalmanFilter< IndependentVariableType, DependentVariableType >::generateEstimationWeights( )
559 {
560  // Generate state and covariance estimation weights
561  stateEstimationWeights_.push_back( constantParameters_.at( lambda_index ) /
562  ( augmentedStateDimension_ + constantParameters_.at( lambda_index ) ) );
563  for ( unsigned int i = 1; i < numberOfSigmaPoints_; i++ )
564  {
565  stateEstimationWeights_.push_back( 1.0 / ( 2.0 * ( augmentedStateDimension_ + constantParameters_.at( lambda_index ) ) ) );
566  }
567  covarianceEstimationWeights_ = stateEstimationWeights_;
568  covarianceEstimationWeights_.at( 0 ) += 1.0 - std::pow( constantParameters_.at( alpha_index ), 2 ) +
569  constantParameters_.at( beta_index );
570 }
