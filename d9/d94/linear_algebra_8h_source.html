142 static inline void flipMatrixRows( Eigen::MatrixXd& matrixToFlip )
143 {
144  // Loop through each row.
145  for ( int i = 0; i < static_cast< int >( matrixToFlip.rows( ) / 2 ); i++ )
146  {
147  // Save top row in a temporary matrix.
148  const Eigen::MatrixXd temporaryRow = matrixToFlip.row( i );
149 
150  // Set top row to bottom row.
151  matrixToFlip.row( i ) = matrixToFlip.row( matrixToFlip.rows( ) - 1 - i );
152 
153  // Set bottom row to temporarily stored matrix.
154  matrixToFlip.row( matrixToFlip.rows( ) - 1 - i ) = temporaryRow;
155  }
156 }

184 template< typename StateScalarType, int NumberOfRows, int NumberOfColumns >
185 bool doesMatrixHaveNanEntries( const Eigen::Matrix< StateScalarType, NumberOfRows, NumberOfColumns > matrixToCheck )
186 {
187  bool areNanEntriesPresent = false;
188  for( int i = 0; i < matrixToCheck.rows( ); i++ )
189  {
190  for( int j = 0; j < matrixToCheck.cols( ); j++ )
191  {
192  if( !( matrixToCheck( i, j ) == matrixToCheck( i, j ) ) )
193  {
194  areNanEntriesPresent = true;
195  }
196  }
197  }
198  return areNanEntriesPresent;
199 }
