62 template< typename ScalarType = double >
63 Eigen::Matrix< ScalarType, 6, 1 > convertKeplerianToModifiedEquinoctialElements(
64  const Eigen::Matrix< ScalarType, 6, 1 >& keplerianElements,
65  const bool flipSingularityToZeroInclination )
66 {
67  using mathematical_constants::getPi;
68  using mathematical_constants::getFloatingInteger;
69 
70  // Declaring eventual output vector.
71  Eigen::Matrix< ScalarType, 6, 1 > modifiedEquinoctialState( 6 );
72 
73  // Compute semi-latus rectum.
74  ScalarType singularityTolerance = 5.0 * std::numeric_limits< ScalarType >::epsilon( );
75 
76  // Extracting eccentricity for ease of referencing.
77  ScalarType eccentricity = keplerianElements( eccentricityIndex );
78 
79  // If e is (very near) one, then semi-major axis is undefined and thus the first kepler
80  // element is the semi-latus rectum.
81  if ( std::fabs( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) < singularityTolerance )
82  {
83  modifiedEquinoctialState( semiLatusRectumIndex )
84  = keplerianElements( semiLatusRectumIndex );
85  }
86  else // eccentricity is significantly away from singularity and can be computed with p=a(1-e).
87  {
88  modifiedEquinoctialState( semiLatusRectumIndex ) =
89  keplerianElements( semiMajorAxisIndex )
90  * ( getFloatingInteger< ScalarType >( 1 ) - eccentricity * eccentricity );
91  }
92 
93  // Determine prograde-ness, as other five parameters depend on that fact.
94  ScalarType inclination = keplerianElements( inclinationIndex );
95 
96  // If inclination is outside range [0,PI].
97  if ( ( inclination < getFloatingInteger< ScalarType >( 0 ) ) || ( inclination > getPi< ScalarType >( ) ) )
98  {
99  // Define the error message.
100  throw std::runtime_error( "Inclination is expected in range [0," + std::to_string( getPi< ScalarType >( ) ) + "]\n"
101  + "Specified inclination: " + std::to_string( inclination ) + " rad." );
102  }
103  //Else, nothing wrong and continue.
104 
105  // Compute set dependant helper parameters.
106  ScalarType argumentOfPeriapsisAndAscendingNode = getFloatingInteger< ScalarType >( 0 );
107  ScalarType tangentOfHalfInclination = getFloatingInteger< ScalarType >( 0 );
108 
109  // If normal set of equations is to be used (i.e. not inverse set for singularity).
110  if ( !flipSingularityToZeroInclination )
111  {
112  // Add (+) argument of periapsis and longitude of ascending node.
113  argumentOfPeriapsisAndAscendingNode = keplerianElements( argumentOfPeriapsisIndex )
114  + keplerianElements( longitudeOfAscendingNodeIndex );
115 
116  // Take the tangent of half the inclination.
117  tangentOfHalfInclination = std::tan( inclination / getFloatingInteger< ScalarType >( 2 ) );
118  }
119  else // the orbit is retrograde.
120  {
121  // Subtract (-) longitude of ascending node from argument of periapsis.
122  argumentOfPeriapsisAndAscendingNode = keplerianElements( argumentOfPeriapsisIndex )
123  - keplerianElements( longitudeOfAscendingNodeIndex );
124 
125  // Take the inverse of the tangent of half the inclination to avoid singularity at tan
126  // PI/2.
127  tangentOfHalfInclination = getFloatingInteger< ScalarType >( 1 ) /
128  std::tan( inclination / getFloatingInteger< ScalarType >( 2 ) );
129  }
130 
131  // Compute f-element.
132  modifiedEquinoctialState( fElementIndex ) = eccentricity
133  * std::cos( argumentOfPeriapsisAndAscendingNode );
134 
135  // Compute g-element.
136  modifiedEquinoctialState( gElementIndex ) = eccentricity
137  * std::sin( argumentOfPeriapsisAndAscendingNode );
138 
139  // Compute h-element.
140  modifiedEquinoctialState( hElementIndex ) = tangentOfHalfInclination
141  * std::cos( keplerianElements( longitudeOfAscendingNodeIndex ) );
142 
143  // Compute k-element.
144  modifiedEquinoctialState( kElementIndex ) = tangentOfHalfInclination
145  * std::sin( keplerianElements( longitudeOfAscendingNodeIndex ) );
146 
147  // Compute true longitude (modulo 2 PI to keep within interval -2PI to 2PI).
148  modifiedEquinoctialState( trueLongitudeIndex )
149  = basic_mathematics::computeModulo(
150  argumentOfPeriapsisAndAscendingNode
151  + keplerianElements( trueAnomalyIndex ), getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
152 
153  // Give back result.
154  return modifiedEquinoctialState;
155 }

178 Eigen::Matrix< ScalarType, 6, 1 > convertKeplerianToModifiedEquinoctialElements(
179  const Eigen::Matrix< ScalarType, 6, 1 >& keplerianElements )
180 // Based on Hintz, 2008.
181 {
182  // Check if orbit is retrograde
183  bool flipSingularityToZeroInclination =
184  mission_geometry::isOrbitRetrograde( keplerianElements );

188  flipSingularityToZeroInclination );

217 template< typename ScalarType = double >
218 Eigen::Matrix< ScalarType, 6, 1 > convertModifiedEquinoctialToKeplerianElements(
219  const Eigen::Matrix< ScalarType, 6, 1 >& modifiedEquinoctialElements,
220  const bool flipSingularityToZeroInclination )
221 {
222  using mathematical_constants::getPi;
223  using mathematical_constants::getFloatingInteger;
224 
225  // Declaration of output vector.
226  Eigen::Matrix< ScalarType, 6, 1 > convertedKeplerianElements = Eigen::Matrix< ScalarType, 6, 1 >::Zero( 6 );
227 
228  // for ease of referencing, almost all modified equinoctial elements.
229  ScalarType fElement = modifiedEquinoctialElements( fElementIndex );
230  ScalarType gElement = modifiedEquinoctialElements( gElementIndex );
231  ScalarType hElement = modifiedEquinoctialElements( hElementIndex );
232  ScalarType kElement = modifiedEquinoctialElements( kElementIndex );
233 
234  // Tolerance for singularities.
235  ScalarType singularityTolerance = 5.0 * std::numeric_limits< ScalarType >::epsilon( );
236 
237  // Compute eccentricity.0
238  ScalarType eccentricity = std::sqrt( fElement * fElement + gElement * gElement );
239  convertedKeplerianElements( eccentricityIndex ) = eccentricity;
240 
241  // Compute semi-major axis.
242  // If eccentricity is not near-parabolic.
243  if ( std::fabs( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) > singularityTolerance )
244  {
245  // Use semi-latus rectum and eccentricity to calculate semi-major axis with a=p/(1-e^2).
246  convertedKeplerianElements( semiMajorAxisIndex ) =
247  modifiedEquinoctialElements( semiLatusRectumIndex ) /
248  ( getFloatingInteger< ScalarType >( 1 ) - eccentricity * eccentricity );
249  }
250  // This is (almost) a parabola and the semimajor axis will tend to infinity and be useless.
251  else
252  {
253  // Give semi-latus rectum instead.
254  convertedKeplerianElements( semiLatusRectumIndex ) =
255  modifiedEquinoctialElements( semiLatusRectumIndex );
256  }
257 
258  // Compute longitude of ascending node.
259 
260  // Compute solution in [-PI,PI] interval with atan2.
261  ScalarType longitudeOfAscendingNode = std::atan2( kElement, hElement );
262 
263  // Store longitude of ascending node.
264  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) = basic_mathematics::
265  computeModulo( longitudeOfAscendingNode, getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
266 
267  // Compute inclination.
268 
269  // If prograde factor must be positive, otherwise negative.
270  ScalarType retrogradeFactor =
271  flipSingularityToZeroInclination ? -getFloatingInteger< ScalarType >( 1 ) : getFloatingInteger< ScalarType >( 1 );
272 
273  // Calculate magnitude of inclination with retrogradeFactor (derived based on Hintz, 2008).
274  ScalarType hSquaredPlusKSquared = std::pow( hElement * hElement + kElement * kElement ,
275  retrogradeFactor );
276 
277  convertedKeplerianElements( inclinationIndex ) =
278  getFloatingInteger< ScalarType >( 2 ) * std::atan( std::sqrt( hSquaredPlusKSquared ) );
279  // Was: std::atan2(2 * std::sqrt(hSquaredPlusKSquared) , (1.0 - hSquaredPlusKSquared));
280 
281  // Compute argument of periapsis.
282 
283  // Helper quantity (omega + I * OMEGA ), for argument of periapsis and true anomaly.
284  ScalarType argumentOfPeriapsisAndLongitude;
285 
286  // If eccentricity is (near) circular
287  if ( eccentricity < singularityTolerance )
288  // Then the composite argument of periapsis and longitude together cannot be determined
289  // because both arguments of the atan2 functions will be zero.
290  {
291  // Set argument of periapsis to zero.
292  convertedKeplerianElements( argumentOfPeriapsisIndex ) = getFloatingInteger< ScalarType >( 0 );
293 
294  // Set composite argument to longitude only, since argument of periapsis is now zero.
295  argumentOfPeriapsisAndLongitude = retrogradeFactor * longitudeOfAscendingNode;
296  }
297  else
298  // Argument of periapsis can be found from composite argument.
299  {
300  // Compute composite argument.
301  argumentOfPeriapsisAndLongitude = std::atan2( gElement, fElement );
302 
303  // Compute argument of periapsis.
304  convertedKeplerianElements( argumentOfPeriapsisIndex ) = basic_mathematics::
305  computeModulo( argumentOfPeriapsisAndLongitude
306  - retrogradeFactor * longitudeOfAscendingNode,
307  getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
308  }
309 
310  // Compute true anomaly.
311  convertedKeplerianElements( trueAnomalyIndex ) = basic_mathematics::
312  computeModulo( modifiedEquinoctialElements( trueLongitudeIndex )
313  - argumentOfPeriapsisAndLongitude, getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
314 
315  // Return converted elements.
316  return convertedKeplerianElements;
317 }

432 template< typename ScalarType = double >
433 Eigen::Matrix< ScalarType, 6, 1 > convertModifiedEquinoctialToCartesianElementsViaKeplerElements(
434  const Eigen::Matrix< ScalarType, 6, 1 >& modifiedEquinoctialElements,
435  const ScalarType centralBodyGravitationalParameter,
436  const bool flipSingularityToZeroInclination )
437 // Using unnamed pdf and code archive Bart Rmgens.
438 {
439  using mathematical_constants::getFloatingInteger;
440 
441  // Creating output vector.
442  Eigen::Matrix< ScalarType, 6, 1 > convertedCartesianElements = Eigen::Matrix< ScalarType, 6, 1 >::Zero( 6 );
443 
444  // If the prograde equations are to be used.
445  if ( !flipSingularityToZeroInclination )
446  {
447  // Local storage of elements for ease of access.
448  ScalarType semiLatusRectum = modifiedEquinoctialElements( semiLatusRectumIndex );
449  ScalarType fElement = modifiedEquinoctialElements( fElementIndex );
450  ScalarType gElement = modifiedEquinoctialElements( gElementIndex );
451  ScalarType hElement = modifiedEquinoctialElements( hElementIndex );
452  ScalarType kElement = modifiedEquinoctialElements( kElementIndex );
453  ScalarType trueLongitude = modifiedEquinoctialElements( trueLongitudeIndex );
454 
455  // Computing helper parameters.
456 
457  // Square-root
458  ScalarType squareRootOfGravitationalParameterOverSemiLatusRectum =
459  std::sqrt( centralBodyGravitationalParameter / semiLatusRectum );
460 
461  // Cosines and sine.
462  ScalarType cosineTrueLongitude = std::cos( trueLongitude );
463  ScalarType sineTrueLongitude = std::sin( trueLongitude );
464 
465  // s, a, w, r from references.
466  ScalarType sSquaredParameter = getFloatingInteger< ScalarType >( 1 ) + hElement * hElement + kElement * kElement;
467  ScalarType aSquaredParameter = hElement * hElement - kElement*kElement;
468  ScalarType wParameter = getFloatingInteger< ScalarType >( 1 ) + fElement * cosineTrueLongitude
469  + gElement * sineTrueLongitude;
470  ScalarType radius = semiLatusRectum / wParameter;
471 
472  // Computing position and storing (using code archive Bart Rmgens and unnamed pdf).
473  convertedCartesianElements( xCartesianPositionIndex )
474  = radius / sSquaredParameter *
475  ( cosineTrueLongitude + aSquaredParameter
476  * cosineTrueLongitude + getFloatingInteger< ScalarType >( 2 ) * hElement * kElement
477  * sineTrueLongitude );
478  convertedCartesianElements( yCartesianPositionIndex )
479  = radius / sSquaredParameter *
480  ( sineTrueLongitude - aSquaredParameter
481  * sineTrueLongitude + getFloatingInteger< ScalarType >( 2 ) * hElement * kElement
482  * cosineTrueLongitude );
483  convertedCartesianElements( zCartesianPositionIndex ) = getFloatingInteger< ScalarType >( 2 ) * radius / sSquaredParameter
484  * ( hElement * sineTrueLongitude - kElement * cosineTrueLongitude );
485 
486  // Computing velocities (these can probably use more optimal computation).
487  convertedCartesianElements( xCartesianVelocityIndex ) = -getFloatingInteger< ScalarType >( 1 ) / sSquaredParameter
488  * squareRootOfGravitationalParameterOverSemiLatusRectum
489  * ( sineTrueLongitude + aSquaredParameter * sineTrueLongitude
490  - getFloatingInteger< ScalarType >( 2 ) * hElement * kElement * cosineTrueLongitude + gElement
491  - getFloatingInteger< ScalarType >( 2 ) * fElement * hElement * kElement + aSquaredParameter * gElement );
492  convertedCartesianElements( yCartesianVelocityIndex ) = -getFloatingInteger< ScalarType >( 1 ) / sSquaredParameter
493  * squareRootOfGravitationalParameterOverSemiLatusRectum
494  * ( -cosineTrueLongitude + aSquaredParameter * cosineTrueLongitude
495  + getFloatingInteger< ScalarType >( 2 ) * hElement * kElement * sineTrueLongitude - fElement
496  + getFloatingInteger< ScalarType >( 2 ) * gElement * hElement * kElement + aSquaredParameter * fElement );
497  convertedCartesianElements( zCartesianVelocityIndex ) = getFloatingInteger< ScalarType >( 2 ) / sSquaredParameter
498  * squareRootOfGravitationalParameterOverSemiLatusRectum
499  * ( hElement * cosineTrueLongitude + kElement * sineTrueLongitude
500  + fElement * hElement + gElement * kElement );
501  }
502  else // use the indirect transformation via an intermediate Kepler state.
503  {
504  // Use the Keplerian state as an intermediary step, as there is no direct transformation
505  // available in literature (personal derivation forgone for now due to time constraints).
506  convertedCartesianElements =
507  convertKeplerianToCartesianElements(
508  convertModifiedEquinoctialToKeplerianElements(
509  modifiedEquinoctialElements, flipSingularityToZeroInclination ),
510  centralBodyGravitationalParameter );
511  }
512 
513  // Return converted set of elements.
514  return convertedCartesianElements;
515 }

543 template< typename ScalarType = double >
544 Eigen::Matrix< ScalarType, 6, 1 > convertModifiedEquinoctialToCartesianElements(
545  const Eigen::Matrix< ScalarType, 6, 1 >& modifiedEquinoctialElements,
546  const ScalarType centralBodyGravitationalParameter,
547  const bool flipSingularityToZeroInclination )
548 {
549  if( flipSingularityToZeroInclination )
550  {
551  throw std::runtime_error(
552  "Error in direct conversion from MEE to Cartesian elements, implementation with singularity at 0 degrees not yet implemented" );
553  }
554  using mathematical_constants::getFloatingInteger;
555 
556  // Retrieve MEE components and compute intermediate quantities
557  ScalarType semiLatusRectrum = modifiedEquinoctialElements( semiParameterIndex );
558  ScalarType angularMomentumPerUnitGravitationalParameter =
559  std::sqrt( semiLatusRectrum / centralBodyGravitationalParameter );
560 
561  ScalarType sineTrueLongitude = std::sin( modifiedEquinoctialElements( trueLongitudeIndex ) );
562  ScalarType cosineTrueLongitude = std::cos( modifiedEquinoctialElements( trueLongitudeIndex ) );
563 
564  ScalarType parameterF = modifiedEquinoctialElements( fElementIndex );
565  ScalarType parameterG = modifiedEquinoctialElements( gElementIndex );
566  ScalarType parameterH = modifiedEquinoctialElements( hElementIndex );
567  ScalarType parameterK = modifiedEquinoctialElements( kElementIndex );
568 
569  ScalarType parameterW =
570  getFloatingInteger< ScalarType >( 1 ) + parameterF * cosineTrueLongitude + parameterG * sineTrueLongitude;
571  ScalarType parameterSSquared = getFloatingInteger< ScalarType >( 1 ) + parameterH * parameterH + parameterK * parameterK;
572  ScalarType parameterAlphaSquared = parameterH * parameterH - parameterK * parameterK;
573 
574  // Compute Cartesian elements
575  Eigen::Matrix< ScalarType, 6, 1 > cartesianElements;
576  cartesianElements( 0 ) = cosineTrueLongitude + parameterAlphaSquared * cosineTrueLongitude +
577  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * sineTrueLongitude;
578  cartesianElements( 1 ) = ( sineTrueLongitude - parameterAlphaSquared * sineTrueLongitude +
579  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * cosineTrueLongitude );
580  cartesianElements( 2 ) =
581  getFloatingInteger< ScalarType >( 2 ) * ( parameterH * sineTrueLongitude - parameterK * cosineTrueLongitude );
582  cartesianElements.segment( 0, 3 ) *= semiLatusRectrum / ( parameterW * parameterSSquared );
583 
584  cartesianElements( 3 ) =
585  -( sineTrueLongitude + parameterAlphaSquared * sineTrueLongitude -
586  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * cosineTrueLongitude + parameterG -
587  getFloatingInteger< ScalarType >( 2 ) * parameterF * parameterH * parameterK +
588  parameterAlphaSquared * parameterG );
589  cartesianElements( 4 ) = -( -cosineTrueLongitude + parameterAlphaSquared * cosineTrueLongitude +
590  getFloatingInteger< ScalarType >( 2 ) * parameterH * parameterK * sineTrueLongitude -
591  parameterF + getFloatingInteger< ScalarType >( 2 ) * parameterG * parameterH * parameterK +
592  parameterAlphaSquared * parameterF );
593  cartesianElements( 5 ) =
594  getFloatingInteger< ScalarType >( 2 ) * ( parameterH * cosineTrueLongitude + parameterK * sineTrueLongitude +
595  parameterF * parameterH + parameterG * parameterK );
596  cartesianElements.segment( 3, 3 ) *=
597  getFloatingInteger< ScalarType >( 1 ) / ( parameterSSquared * angularMomentumPerUnitGravitationalParameter );
598 
599  return cartesianElements;
600
