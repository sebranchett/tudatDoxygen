126 std::pair< int, std::pair< int, int > > TidalLoveNumberPartialInterface::setParameterPartialFunction(
127  const std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > parameter,
128  const int maximumDegree,
129  const int maximumOrder )
130 {
131  using namespace estimatable_parameters;
132 
133  // Define default values for return values
134  int maximumUsedDegree = -1;
135  int maximumUsedOrder = -1;
136  int numberOfRows = 0;
137 
138  // Check if parameter is tidal property of deformed body
139  if( parameter->getParameterName( ).second.first == deformedBody_ )
140  {
141  switch( parameter->getParameterName( ).first )
142  {
143  case full_degree_tidal_love_number:
144  {
145  // Cast parameter object to required type.
146  std::shared_ptr< FullDegreeTidalLoveNumber > coefficientsParameter =
147  std::dynamic_pointer_cast< FullDegreeTidalLoveNumber >( parameter );
148  if( coefficientsParameter == nullptr )
149  {
150  throw std::runtime_error(
151  "Error when setting partil function of full_degree_tidal_love_number in TidalLoveNumberPartialInterface, input is inconsistent" );
152  }
153 
154  // Retrieve the maximum degree and order that are to be used for this model
155  getMaximumUsedDegreeAndOrder(
156  maximumDegree, maximumOrder, coefficientsParameter->getDegree( ), maximumUsedDegree, maximumUsedOrder );
157 
158  // Set parameter partial function if relevant terms are found
159  if( maximumUsedDegree > 0 )
160  {
161  std::vector< int > selectedDeformingBodies = getSelectedDeformingBodyIds(
162  coefficientsParameter->getDeformingBodies( ) );
163 
164  // Check if deforming bodies correspond to bodies in model.
165  if( selectedDeformingBodies.size( ) == coefficientsParameter->getDeformingBodies( ).size( ) &&
166  deformingBodies_.size( ) == coefficientsParameter->getDeformingBodies( ).size( ) )
167  {
168  // Add partial function if it is not yet set.
169  if( parameterVectorPartialFunctions_.count(
170  std::make_pair( parameter, std::make_pair( maximumUsedDegree, maximumUsedOrder ) ) ) == 0 )
171  {
172  if( coefficientsParameter->useComplexComponents( ) )
173  {
174  // Calculate partials for complex love number
175  parameterVectorPartialFunctions_[ std::make_pair(
176  parameter, std::make_pair( maximumUsedDegree, maximumUsedOrder ) ) ] =
177  std::bind( &TidalLoveNumberPartialInterface::
178  calculateSphericalHarmonicCoefficientsPartialWrtComplexTidalLoveNumber, this,
179  coefficientsParameter->getDegree( ), selectedDeformingBodies,
180  maximumUsedDegree, maximumUsedOrder );
181  }
182  else
183  {
184  // Calculate partial for real love number
185  parameterVectorPartialFunctions_[ std::make_pair(
186  parameter, std::make_pair( maximumUsedDegree, maximumUsedOrder ) ) ] =
187  std::bind( &TidalLoveNumberPartialInterface::
188  calculateSphericalHarmonicCoefficientsPartialWrtRealTidalLoveNumber, this,
189  coefficientsParameter->getDegree( ), selectedDeformingBodies,
190  maximumUsedDegree, maximumUsedOrder );
191  }
192  }
193 
194  numberOfRows = coefficientsParameter->getParameterSize( );
195  }
196  }
197 
198  break;
199  }
200  case single_degree_variable_tidal_love_number:
201  {
202  // Cast parameter object to required type.
203  std::shared_ptr< SingleDegreeVariableTidalLoveNumber > coefficientsParameter =
204  std::dynamic_pointer_cast< SingleDegreeVariableTidalLoveNumber >( parameter );
205  if( coefficientsParameter == nullptr )
206  {
207  throw std::runtime_error(
208  "Error when setting partil function of single_degree_variable_tidal_love_number in TidalLoveNumberPartialInterface, input is inconsistent" );
209  }
210 
211  // Retrieve the maximum degree and order that are to be used for this model
212  getMaximumUsedDegreeAndOrder(
213  maximumDegree, maximumOrder, coefficientsParameter->getDegree( ), maximumUsedDegree, maximumUsedOrder );
214 
215  // Set parameter partial function if relevant terms are found
216  if( maximumUsedDegree > 0 )
217  {
218  // Check if deforming bodies correspond to bodies in model.
219  std::vector< int > selectedDeformingBodies = getSelectedDeformingBodyIds(
220  coefficientsParameter->getDeformingBodies( ) );
221  if( selectedDeformingBodies.size( ) == coefficientsParameter->getDeformingBodies( ).size( ) &&
222  coefficientsParameter->getDeformingBodies( ).size( ) == deformingBodies_.size( ) )
223  {
224  // Add partial function if it is not yet set.
225  if( parameterVectorPartialFunctions_.count(
226  std::make_pair( parameter, std::make_pair( maximumUsedDegree, maximumUsedOrder ) ) ) == 0 )
227  {
228  if( coefficientsParameter->useComplexComponents( ) )
229  {
230  // Calculate partials for complex love number
231  parameterVectorPartialFunctions_[ std::make_pair(
232  parameter, std::make_pair( maximumUsedDegree, maximumUsedOrder ) ) ] =
233  std::bind( &TidalLoveNumberPartialInterface::
234  calculateSphericalHarmonicCoefficientsPartialWrtComplexTidalLoveNumbers, this,
235  coefficientsParameter->getDegree( ), coefficientsParameter->getOrders( ),
236  selectedDeformingBodies, maximumUsedDegree, maximumUsedOrder );
237  }
238  else
239  {
240  // Calculate partial for real love number
241  parameterVectorPartialFunctions_[ std::make_pair(
242  parameter, std::make_pair( maximumUsedDegree, maximumUsedOrder ) ) ] =
243  std::bind( &TidalLoveNumberPartialInterface::
244  calculateSphericalHarmonicCoefficientsPartialWrtRealTidalLoveNumbers, this,
245  coefficientsParameter->getDegree( ), coefficientsParameter->getOrders( ),
246  selectedDeformingBodies, maximumUsedDegree, maximumUsedOrder );
247  }
248  }
249 
250  numberOfRows = coefficientsParameter->getParameterSize( );
251  }
252  }
253 
254  break;
255  }
256  default:
257  break;
258  }
259  }
260  return std::make_pair( numberOfRows, std::make_pair( maximumUsedDegree, maximumUsedOrder ) );

278 void TidalLoveNumberPartialInterface::setCurrentTidalBodyStates( const int degree, const int order, const int body )
279 {
280  {
281  // Calculate properties of currently considered body
282  massRatio_ = deformingBodyGravitationalParameters_[ body ]( ) / deformedBodyGravitationalParameter_;
283  relativeDeformingBodyPosition_ = rotationToTidallyDeformedBody_ * (
284  positionsOfDeformingBodies_[ body ] - positionOfTidallyDeformedBody_ );
285  relativeDeformingBodySphericalPosition_ =coordinate_conversions::
286  convertCartesianToSpherical( relativeDeformingBodyPosition_ );
287  radiusRatio_ = deformedBodyReferenceRadius_ / relativeDeformingBodySphericalPosition_.x( );
288  iLongitude_ = std::complex< double >( 0.0, relativeDeformingBodySphericalPosition_.z( ) );
289  sineOfLatitude_ = std::sin( mathematical_constants::PI / 2.0 - relativeDeformingBodySphericalPosition_.y( ) );

294 std::vector< Eigen::Matrix< double, 2, Eigen::Dynamic > > TidalLoveNumberPartialInterface::
295 calculateSphericalHarmonicCoefficientPartialMatrix(
296  const std::vector< Eigen::Vector2d >& coefficientPartialsPerOrder,
297  const std::pair< Eigen::Matrix< double, 2, Eigen::Dynamic >, Eigen::Matrix< double, 2, Eigen::Dynamic > >&
298  coefficientPartialScalers )
299 {
300  std::vector< Eigen::Matrix< double, 2, Eigen::Dynamic > > coefficientsPartial;
301 
302  // Check input consistency
303  if( coefficientPartialScalers.first.rows( ) != coefficientPartialScalers.second.rows( ) )
304  {
305  throw std::runtime_error(
306  "Error when making spherical harmonic coefficient partial from coefficient partial in TidalLoveNumberPartialInterface, scale matrices not of same size." );
307  }
308 
309  // COmpute scaled partials
310  for( unsigned int m = 0; m < coefficientPartialsPerOrder.size( ); m++ )
311  {
312  coefficientsPartial.push_back(
313  ( coefficientPartialScalers.first * coefficientPartialsPerOrder[ m ].x( ) +
314  coefficientPartialScalers.second * coefficientPartialsPerOrder[ m ].y( ) ) );
315  }
316 
317  return coefficientsPartial;
318 }

321 std::vector< Eigen::Vector2d > TidalLoveNumberPartialInterface::calculateCoefficientPartialWrtRealTidalLoveNumber(
322  const int degree,
323  const std::vector< int >& orders,
324  const std::vector< int >& deformingBodyIndices,
325  const int maximumDegree,
326  const int maximumOrder )
327 {
328  // Set default values of return variable
329  std::vector< Eigen::Vector2d > realCoefficientPartials;
330  realCoefficientPartials.resize( orders.size( ) );
331  for( unsigned int i = 0; i < orders.size( ); i ++ )
332  {
333  realCoefficientPartials[ i ] = Eigen::Vector2d::Zero( );
334  }
335 
336  // Check if any dependency exists
337  if( degree <= maximumDegree )
338  {
339  std::complex< double > unitLoveNumberCoefficientVariations;
340 
341  // Iterate over all bodies causing deformation and calculate and add associated corrections
342  for( unsigned int i = 0; i < deformingBodyIndices.size( ); i++ )
343  {
344  for( unsigned int m = 0; m < orders.size( ); m++ )
345  {
346  if( orders.at( m ) <= maximumOrder )
347  {
348  // Update member variables to current body, degree and order
349  setCurrentTidalBodyStates( degree, orders.at( m ), deformingBodyIndices.at( i ) );
350 
351  // Calculate and add coefficients.
352  unitLoveNumberCoefficientVariations =
353  gravitation::calculateSolidBodyTideSingleCoefficientSetCorrectionFromAmplitude(
354  std::complex< double >( 1.0, 0.0 ), massRatio_,
355  basic_mathematics::raiseToIntegerPower( radiusRatio_, degree + 1 ) ,
356  basic_mathematics::computeLegendrePolynomialExplicit(
357  degree, orders.at( m ), sineOfLatitude_ ),
358  static_cast< double >( orders.at( m ) ) * iLongitude_, degree, orders.at( m ) );
359  realCoefficientPartials[ m ].x( ) += unitLoveNumberCoefficientVariations.real( );
360  realCoefficientPartials[ m ].y( ) -= unitLoveNumberCoefficientVariations.imag( );
361  }
362  }
363  }
364  }
365  return realCoefficientPartials;
366 }
