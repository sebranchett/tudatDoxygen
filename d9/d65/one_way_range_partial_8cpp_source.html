21 void OneWayRangeScaling::update( const std::vector< Eigen::Vector6d >& linkEndStates,
22  const std::vector< double >& times,
23  const observation_models::LinkEndType fixedLinkEnd,
24  const Eigen::VectorXd currentObservation )
25 {
26  // Compute Euclidean distance vector
27  Eigen::Vector3d rangeVector = linkEndStates[ 1 ].segment( 0, 3 ) - linkEndStates[ 0 ].segment( 0, 3 );
28  Eigen::Matrix< double, 1, 3 > rangeVectorNormalized = rangeVector.transpose( ) / rangeVector.norm( );
29 
30  // Compute scaling for receiver reference
31  if( fixedLinkEnd == observation_models::receiver )
32  {
33  referenceLightTimeCorrectionScaling_ = 1.0 / ( 1.0 - rangeVectorNormalized.transpose( ).dot( linkEndStates[ 0 ].segment( 3, 3 ) ) /
34  physical_constants::SPEED_OF_LIGHT );
35  referenceScalingFactor_ = rangeVectorNormalized * referenceLightTimeCorrectionScaling_;
36  }
37 
38  // Compute scaling for transmitter reference
39  else if( fixedLinkEnd == observation_models::transmitter )
40  {
41  referenceLightTimeCorrectionScaling_ =
42  1.0 / ( 1.0 - rangeVectorNormalized.transpose( ).dot( linkEndStates[ 1 ].segment( 3, 3 ) ) /
43  physical_constants::SPEED_OF_LIGHT );
44  referenceScalingFactor_ = rangeVectorNormalized * referenceLightTimeCorrectionScaling_;
45  }
46 
47  currentLinkEndType_ = fixedLinkEnd;
48 }

51 OneWayRangePartial::OneWayRangePartialReturnType OneWayRangePartial::calculatePartial(
52  const std::vector< Eigen::Vector6d >& states,
53  const std::vector< double >& times,
54  const observation_models::LinkEndType linkEndOfFixedTime,
55  const Eigen::Vector1d& currentObservation )
56 {
57  if( linkEndOfFixedTime != oneWayRangeScaler_->getCurrentLinkEndType( ) )
58  {
59  std::cout << linkEndOfFixedTime << " " << oneWayRangeScaler_->getCurrentLinkEndType( ) << std::endl;
60  throw std::runtime_error( "Error one-way range partial and scaling are inconsistent" );
61  }
62 
63  OneWayRangePartialReturnType returnPartial;
64 
65  // Iterate over all link ends
66  for( positionPartialIterator_ = positionPartialList_.begin( ); positionPartialIterator_ != positionPartialList_.end( );
67  positionPartialIterator_++ )
68  {
69  if( positionPartialIterator_->first == observation_models::transmitter )
70  {
71  currentState_ = states[ 0 ];
72  currentTime_ = times[ 0 ];
73  }
74  else if( positionPartialIterator_->first == observation_models::receiver )
75  {
76  currentState_ = states[ 1 ];
77  currentTime_ = times[ 1 ];
78  }
79 
80  // Scale position partials
81  returnPartial.push_back(
82  std::make_pair(
83  oneWayRangeScaler_->getScalingFactor( positionPartialIterator_->first ) *
84  ( positionPartialIterator_->second->calculatePartialOfPosition(
85  currentState_ , currentTime_ ) ), currentTime_ ) );
86  }
87 
88  // Add scaled light-time correcion partials.
89  for( unsigned int i = 0; i < lighTimeCorrectionPartialsFunctions_.size( ); i++ )
90  {
91  returnPartial.push_back( lighTimeCorrectionPartialsFunctions_.at( i )( states, times ) );
92  returnPartial[ returnPartial.size( ) - 1 ].first *=
93  physical_constants::SPEED_OF_LIGHT * oneWayRangeScaler_->getLightTimePartialScalingFactor( );
94  }
95 
96  return returnPartial;
97 }
