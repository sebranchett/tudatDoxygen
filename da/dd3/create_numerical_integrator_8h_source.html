564 template< typename IndependentVariableType, typename DependentVariableType,
565  typename IndependentVariableStepType = IndependentVariableType >
566 std::shared_ptr< numerical_integrators::NumericalIntegrator< IndependentVariableType, DependentVariableType,
567 DependentVariableType, IndependentVariableStepType > > createIntegrator(
568  std::function< DependentVariableType(
569  const IndependentVariableType, const DependentVariableType& ) > stateDerivativeFunction,
570  const DependentVariableType initialState,
571  std::shared_ptr< IntegratorSettings< IndependentVariableType > > integratorSettings )
572 {
573  // Declare eventual output
574  std::shared_ptr< NumericalIntegrator
575  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > > integrator;
576 
577  // Retrieve requested type of integrator
578  switch( integratorSettings->integratorType_ )
579  {
580  case euler:
581  {
582  // Create Euler integrator
583  integrator = std::make_shared< EulerIntegrator
584  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
585  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
586  break;
587  }
588  case rungeKutta4:
589  {
590  // Create Runge-Kutta 4 integrator
591  integrator = std::make_shared< RungeKutta4Integrator
592  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
593  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
594  break;
595  }
596  case rungeKuttaVariableStepSize:
597  {
598  // Cast integrator
599  std::shared_ptr< RungeKuttaVariableStepSizeBaseSettings< IndependentVariableType > >
600  variableStepIntegratorSettings = std::dynamic_pointer_cast< RungeKuttaVariableStepSizeBaseSettings<
601  IndependentVariableType > >( integratorSettings );
602 
603  // Check input consistency
604  if ( variableStepIntegratorSettings == nullptr )
605  {
606  throw std::runtime_error( "Error, type of integrator settings (rungeKuttaVariableStepSize) not compatible with "
607  "selected integrator (derived class of IntegratorSettings must be "
608  "RungeKuttaVariableStepSizeBaseSettings for this type)." );
609  }
610 
611  // Get requested RK coefficients
612  RungeKuttaCoefficients coefficients = RungeKuttaCoefficients::get( variableStepIntegratorSettings->coefficientSet_ );
613 
614  // Check which constructor is being used
615  if ( variableStepIntegratorSettings->areTolerancesDefinedAsScalar_ )
616  {
617  // Settings with scalar tolerances
618  std::shared_ptr< RungeKuttaVariableStepSizeSettingsScalarTolerances< IndependentVariableType > >
619  scalarTolerancesIntegratorSettings = std::dynamic_pointer_cast<
620  RungeKuttaVariableStepSizeSettingsScalarTolerances< IndependentVariableType > >( variableStepIntegratorSettings );
621 
622  // Check input consistency
623  if ( scalarTolerancesIntegratorSettings == nullptr )
624  {
625  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. Input class must be of "
626  "RungeKuttaVariableStepSizeSettingsScalarTolerances type." );
627  }
628 
629  // Create Runge-Kutta integrator with scalar tolerances
630  integrator = std::make_shared< RungeKuttaVariableStepSizeIntegrator
631  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
632  ( coefficients, stateDerivativeFunction, integratorSettings->initialTime_, initialState,
633  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->minimumStepSize_ ),
634  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->maximumStepSize_ ),
635  static_cast< typename DependentVariableType::Scalar >( scalarTolerancesIntegratorSettings->relativeErrorTolerance_ ),
636  static_cast< typename DependentVariableType::Scalar >( scalarTolerancesIntegratorSettings->absoluteErrorTolerance_ ),
637  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->safetyFactorForNextStepSize_ ),
638  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
639  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
640  }
641  else
642  {
643  // Settings with vector tolerances
644  std::shared_ptr< RungeKuttaVariableStepSizeSettingsVectorTolerances< IndependentVariableType, DependentVariableType > >
645  vectorTolerancesIntegratorSettings = std::dynamic_pointer_cast<
646  RungeKuttaVariableStepSizeSettingsVectorTolerances< IndependentVariableType, DependentVariableType > >(
647  variableStepIntegratorSettings );
648 
649  // Check input consistency
650  if ( vectorTolerancesIntegratorSettings == nullptr )
651  {
652  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. Input class must be of "
653  "RungeKuttaVariableStepSizeSettingsVectorTolerances type." );
654  }
655 
656  // Check that sizes of tolerances and initial state match
657  DependentVariableType relativeErrorTolerance = vectorTolerancesIntegratorSettings->relativeErrorTolerance_.template cast<
658  typename DependentVariableType::Scalar >( );
659  DependentVariableType absoluteErrorTolerance = vectorTolerancesIntegratorSettings->absoluteErrorTolerance_.template cast<
660  typename DependentVariableType::Scalar >( );
661  if ( ( relativeErrorTolerance.rows( ) != initialState.rows( ) ) ||
662  ( relativeErrorTolerance.cols( ) != initialState.cols( ) ) ||
663  ( absoluteErrorTolerance.rows( ) != initialState.rows( ) ) ||
664  ( absoluteErrorTolerance.cols( ) != initialState.cols( ) ) )
665  {
666  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. The sizes of the "
667  "relative and absolute tolerance vectors do not match the size of the initial state. "
668  "This could be the case if you are propagating more than just one state, e.g., translational "
669  "and/or rotational dynamics and mass, or more than one body." );
670  }
671 
672  // Create Runge-Kutta integrator with vector tolerances
673  integrator = std::make_shared< RungeKuttaVariableStepSizeIntegrator
674  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
675  ( coefficients, stateDerivativeFunction, integratorSettings->initialTime_, initialState,
676  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->minimumStepSize_ ),
677  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->maximumStepSize_ ),
678  relativeErrorTolerance, absoluteErrorTolerance,
679  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->safetyFactorForNextStepSize_ ),
680  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
681  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
682  }
683  break;
684  }
685  case bulirschStoer:
686  {
687  // Check input consistency
688  std::shared_ptr< BulirschStoerIntegratorSettings< IndependentVariableType > > bulirschStoerIntegratorSettings =
689  std::dynamic_pointer_cast< BulirschStoerIntegratorSettings< IndependentVariableType > >(
690  integratorSettings );
691 
692  // Check that integrator type has been cast properly
693  if ( bulirschStoerIntegratorSettings == nullptr )
694  {
695  throw std::runtime_error( "Error, type of integrator settings (bulirschStoer) not compatible with "
696  "selected integrator (derived class of IntegratorSettings must be BulirschStoerIntegratorSettings "
697  "for this type)." );
698  }
699  else
700  {
701  integrator = std::make_shared< BulirschStoerVariableStepSizeIntegrator
702  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
703  ( getBulirschStoerStepSequence( bulirschStoerIntegratorSettings->extrapolationSequence_,
704  bulirschStoerIntegratorSettings->maximumNumberOfSteps_ ),
705  stateDerivativeFunction, integratorSettings->initialTime_, initialState,
706  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->minimumStepSize_ ),
707  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->maximumStepSize_ ),
708  bulirschStoerIntegratorSettings->relativeErrorTolerance_,
709  bulirschStoerIntegratorSettings->absoluteErrorTolerance_,
710  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->safetyFactorForNextStepSize_ ),
711  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
712  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
713  }
714  break;
715  }
716  case adamsBashforthMoulton:
717  {
718  // Check input consistency
719  std::shared_ptr< AdamsBashforthMoultonSettings< IndependentVariableType > > adamsBashforthMoultonIntegratorSettings =
720  std::dynamic_pointer_cast< AdamsBashforthMoultonSettings< IndependentVariableType > >(
721  integratorSettings );
722 
723  // Check that integrator type has been cast properly
724  if ( adamsBashforthMoultonIntegratorSettings == nullptr )
725  {
726  throw std::runtime_error( "Error, type of integrator settings (AdamsBashforthMoultonSettings) not compatible with "
727  "selected integrator (derived class of IntegratorSettings must be AdamsBashforthMoultonSettings "
728  "for this type)." );
729  }
730  else
731  {
732  // Create integrator
733  integrator = std::make_shared< AdamsBashforthMoultonIntegrator
734  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
735  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState,
736  static_cast< IndependentVariableStepType >( adamsBashforthMoultonIntegratorSettings->minimumStepSize_ ),
737  static_cast< IndependentVariableStepType >( adamsBashforthMoultonIntegratorSettings->maximumStepSize_ ),
738  adamsBashforthMoultonIntegratorSettings->relativeErrorTolerance_,
739  adamsBashforthMoultonIntegratorSettings->absoluteErrorTolerance_ ,
740  static_cast< IndependentVariableStepType >( adamsBashforthMoultonIntegratorSettings->bandwidth_ ) );
741 
742  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
743  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
744  integrator )->setMinimumOrder( adamsBashforthMoultonIntegratorSettings->minimumOrder_ );
745  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
746  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
747  integrator )->setMaximumOrder( adamsBashforthMoultonIntegratorSettings->maximumOrder_ );
748 
749  if ( adamsBashforthMoultonIntegratorSettings->minimumOrder_ == adamsBashforthMoultonIntegratorSettings->maximumOrder_ )
750  {
751  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
752  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
753  integrator )->setFixedOrder( true );
754  }
755 
756  if ( adamsBashforthMoultonIntegratorSettings->minimumStepSize_ ==
757  adamsBashforthMoultonIntegratorSettings->maximumStepSize_ )
758  {
759  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
760  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
761  integrator )->setFixedStepSize( true );
762  }
763  }
764  break;
765  }
766  default:
767  throw std::runtime_error( "Error, integrator " + std::to_string( integratorSettings->integratorType_ ) + " not found." );
768  }
769 
770  // Check that assignment of integrator went well
771  if ( integrator == nullptr )
772  {
773  throw std::runtime_error( "Error while creating integrator. The resulting integrator pointer is null." );
774  }
775 
776  // Give back integrator
777  return integrator;
778 }
