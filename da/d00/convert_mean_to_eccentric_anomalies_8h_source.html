157 template< typename ScalarType = double >
158 ScalarType convertMeanAnomalyToEccentricAnomaly(
159  const ScalarType eccentricity, const ScalarType aMeanAnomaly,
160  const bool useDefaultInitialGuess = true,
161  const ScalarType userSpecifiedInitialGuess = TUDAT_NAN,
162  boost::shared_ptr< root_finders::RootFinderCore< ScalarType > > rootFinder =
163  boost::shared_ptr< root_finders::RootFinderCore< ScalarType > >( ) )
164 {
165  using namespace mathematical_constants;
166  using namespace root_finders;
167  using namespace root_finders::termination_conditions;
168 
169  // Set mean anomaly to region between 0 and 2 PI.
170  ScalarType meanAnomaly = basic_mathematics::computeModulo< ScalarType >(
171  aMeanAnomaly, getFloatingInteger< ScalarType >( 2 ) *
172  getPi< ScalarType >( ) );
173 
174  // Required because the make_shared in the function definition gives problems for MSVC.
175  if ( !rootFinder.get( ) )
176  {
177  ScalarType tolerance = 200.0 * std::numeric_limits< ScalarType >::epsilon( );
178 
179  // Loosen tolerance for near-parabolic orbits
180  if( std::fabs( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) <
181  1.0E5 * std::numeric_limits< ScalarType >::epsilon( ) )
182  {
183  tolerance *= 2.5;
184  }
185 
186  rootFinder = boost::make_shared< NewtonRaphsonCore< ScalarType > >(
187  boost::bind(
188  &RootAbsoluteToleranceTerminationCondition< ScalarType >::
189  checkTerminationCondition,
190  boost::make_shared< RootAbsoluteToleranceTerminationCondition
191  < ScalarType > >( tolerance, 1000 ), _1, _2, _3, _4, _5 ) );
192  }
193 
194  // Declare eccentric anomaly.
195  ScalarType eccentricAnomaly = TUDAT_NAN;
196 
197  // Check if orbit is elliptical, and not near-parabolic.
198  if ( eccentricity < getFloatingInteger< ScalarType >( 1 ) &&
199  eccentricity >= getFloatingInteger< ScalarType >( 0 ) )
200  {
201  // Create an object containing the function of which we whish to obtain the root from.
202  boost::shared_ptr< basic_mathematics::FunctionProxy< ScalarType, ScalarType > > rootFunction
203  = boost::make_shared< basic_mathematics::FunctionProxy< ScalarType, ScalarType > >(
204  boost::bind( &computeKeplersFunctionForEllipticalOrbits< ScalarType >, _1,
205  eccentricity, meanAnomaly ) );
206 
207  // Add the first derivative of the root function.
208  rootFunction->addBinding(
209  -1, boost::bind(
210  &computeFirstDerivativeKeplersFunctionForEllipticalOrbits< ScalarType >, _1,
211  eccentricity ) );
212 
213  // Declare initial guess.
214  ScalarType initialGuess = TUDAT_NAN;
215 
216  // Set the initial guess. Check if the default scheme is to be used or a user specified
217  // value should be used.
218  // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
219  // If this scheme is changed, please run a very extensive test suite. The root finder
220  // function tends to be chaotic for some very specific combinations of mean anomaly and
221  // eccentricity. Various random tests of 100.000.000 samples were done to verify the
222  // functionality of this one, and of another option for the starter: PI. [Musegaas,2012]
223  if ( useDefaultInitialGuess )
224  {
225  if ( meanAnomaly > getPi< ScalarType >( ) )
226  {
227  initialGuess = meanAnomaly - eccentricity;
228  }
229  else
230  {
231  initialGuess = meanAnomaly + eccentricity;
232  }
233  }
234  else
235  {
236  initialGuess = userSpecifiedInitialGuess;
237  }
238 
239  try
240  {
241  // Set eccentric anomaly based on result of Newton-Raphson root-finding algorithm.
242  eccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
243  }
244  // Use bisection algorithm if root finder fails
245  catch( std::runtime_error )
246  {
247  // Set tolerance
248  ScalarType tolerance = 100.0 * std::numeric_limits< ScalarType >::epsilon( );
249 
250  // Set upper/lower bounds (loosely)
251  ScalarType lowerBound, upperBound;
252  if ( meanAnomaly > getPi< ScalarType >( ) )
253  {
254  lowerBound = 0.5 * getPi< ScalarType >( ) + 10.0 * std::numeric_limits< ScalarType >::epsilon( );
255  upperBound = mathematical_constants::getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( );
256  }
257  else
258  {
259  lowerBound = 0.0;
260  upperBound = 1.5 * getPi< ScalarType >( );
261  }
262 
263  // Create root finder
264  boost::shared_ptr< RootFinderCore< ScalarType > > bisectionRootfinder =
265  boost::make_shared< BisectionCore< ScalarType > >(
266  boost::bind( &checkRootFunctionValueCondition< ScalarType >, _1, _2, _3, _4, _5, tolerance ),
267  lowerBound, upperBound );
268 
269  // Set eccentric anomaly based on result of Newton-Raphson root-finding algorithm.
270  initialGuess = meanAnomaly;
271  eccentricAnomaly = bisectionRootfinder->execute( rootFunction, initialGuess );
272  }
273  }
274 
275  // Eccentricity is invalid: eccentricity < 0.0 or eccentricity >= 1.0.
276  else
277  {
278  throw std::runtime_error( "Invalid eccentricity. Valid range is 0.0 <= e < 1.0. Eccentricity was: " +
279  std::to_string( eccentricity ) );
280  }
281 
282  // Return eccentric anomaly.
283  return eccentricAnomaly;
284 }

304 template< typename ScalarType = double >
305 ScalarType convertMeanAnomalyToHyperbolicEccentricAnomaly(
306  const ScalarType eccentricity, const ScalarType hyperbolicMeanAnomaly,
307  const bool useDefaultInitialGuess = true,
308  const ScalarType userSpecifiedInitialGuess = TUDAT_NAN,
309  boost::shared_ptr< root_finders::RootFinderCore< ScalarType > > aRootFinder =
310  boost::shared_ptr< root_finders::RootFinderCore< ScalarType > >( ) )
311 {
312  using namespace mathematical_constants;
313  using namespace root_finders;
314  using namespace root_finders::termination_conditions;
315 
316  boost::shared_ptr< RootFinderCore< ScalarType > > rootFinder = aRootFinder;
317 
318  // Required because the make_shared in the function definition gives problems for MSVC.
319  if ( !rootFinder.get( ) )
320  {
321  rootFinder = boost::make_shared< NewtonRaphsonCore< ScalarType > >(
322  boost::bind(
323  &RootAbsoluteToleranceTerminationCondition< ScalarType >::
324  checkTerminationCondition,
325  boost::make_shared<
326  RootAbsoluteToleranceTerminationCondition< ScalarType > >(
327  25.0 * std::numeric_limits< ScalarType >::epsilon( ), 1000 ),
328  _1, _2, _3, _4, _5 ) );
329  }
330  // Declare hyperbolic eccentric anomaly.
331  ScalarType hyperbolicEccentricAnomaly = TUDAT_NAN;
332 
333  // Check if orbit is hyperbolic.
334  if ( eccentricity > getFloatingInteger< ScalarType >( 1 ) )
335  {
336  // Create an object containing the function of which we whish to obtain the root from.
337  boost::shared_ptr< basic_mathematics::FunctionProxy< ScalarType, ScalarType > > rootFunction
338  = boost::make_shared< basic_mathematics::FunctionProxy< ScalarType, ScalarType > >(
339  boost::bind( &computeKeplersFunctionForHyperbolicOrbits< ScalarType >, _1,
340  eccentricity, hyperbolicMeanAnomaly ) );
341 
342  // Add the first derivative of the root function.
343  rootFunction->addBinding(
344  -1, boost::bind(
345  &computeFirstDerivativeKeplersFunctionForHyperbolicOrbits< ScalarType >,
346  _1, eccentricity ) );
347 
348  // Declare initial guess.
349  ScalarType initialGuess = TUDAT_NAN;
350 
351  // Set the initial guess. Check if the default scheme is to be used or a user specified
352  // value should be used. See [Wakker, 2007] for derivations of the default values.
353  // Note that an error was detected in these starter values, as is discussed in
354  // [Musegaas,2012].
355  // !!!!!!!!!!!!! IMPORTANT !!!!!!!!!!!!!
356  // If this scheme is changed, please run a very extensive test suite. The root finder
357  // function tends to be chaotic for some very specific combinations of mean anomaly and
358  // eccentricity. Various random tests of 100.000.000 samples were done to verify the
359  // functionality of this one. [Musegaas,2012]
360  if ( useDefaultInitialGuess )
361  {
362  if ( std::abs( hyperbolicMeanAnomaly ) <
363  getFloatingInteger< ScalarType >( 6 ) * eccentricity )
364  {
365  initialGuess =
366  std::sqrt( getFloatingInteger< ScalarType >( 8 ) *
367  ( eccentricity - getFloatingInteger< ScalarType >( 1 ) ) /
368  eccentricity ) *
369  std::sinh( getFloatingFraction< ScalarType >( 1, 3 ) * boost::math::asinh(
370  getFloatingInteger< ScalarType >( 3 ) *
371  hyperbolicMeanAnomaly /
372  ( std::sqrt( getFloatingInteger< ScalarType >( 8 ) *
373  ( eccentricity -
374  getFloatingInteger< ScalarType >( 1 ) ) /
375  eccentricity ) *
376  ( eccentricity - getFloatingInteger< ScalarType >( 1 )
377  ) ) ) );
378  }
379  else if ( hyperbolicMeanAnomaly > getFloatingInteger< ScalarType >( 6 ) * eccentricity )
380  {
381  initialGuess = ( std::log( getFloatingInteger< ScalarType >( 2 ) *
382  hyperbolicMeanAnomaly / eccentricity ) );
383  }
384  else
385  {
386  initialGuess = ( - std::log( -getFloatingInteger< ScalarType >( 2 ) *
387  hyperbolicMeanAnomaly / eccentricity ) );
388  }
389  }
390  else
391  {
392  initialGuess = userSpecifiedInitialGuess;
393  }
394 
395  // Set hyperbolic eccentric anomaly based on result of Newton-Raphson root-finding
396  // algorithm.
397  try
398  {
399  hyperbolicEccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
400 
401  }
402  catch( std::runtime_error )
403  {
404  rootFinder = boost::make_shared< BisectionCore< ScalarType > >(
405  boost::bind(
406  &RootAbsoluteToleranceTerminationCondition< ScalarType >::
407  checkTerminationCondition,
408  boost::make_shared<
409  RootAbsoluteToleranceTerminationCondition< ScalarType > >(
410  20.0 * std::numeric_limits< ScalarType >::epsilon( ), 1000 ),
411  _1, _2, _3, _4, _5 ), getFloatingInteger< ScalarType >( 0 ),
412  getFloatingInteger< ScalarType >( 2 ) * getPi< ScalarType >( ) );
413 
414  hyperbolicEccentricAnomaly = rootFinder->execute( rootFunction, initialGuess );
415  }
416 
417  }
418 
419  // In this case the orbit is not hyperbolic.
420  else
421  {
422  throw std::runtime_error( "Invalid eccentricity. Valid range is e > 1.0. Eccentricity was: " +
423  std::to_string( eccentricity ) );
424  }
425 
426  // Return hyperbolic eccentric anomaly.
427  return hyperbolicEccentricAnomaly;
428 }
