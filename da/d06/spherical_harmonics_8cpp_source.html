43 Eigen::Vector3d computePotentialGradient(
44  const double distance,
45  const double radiusPowerTerm,
46  const double cosineOfOrderLongitude,
47  const double sineOfOrderLongitude,
48  const double cosineOfLatitude,
49  const double preMultiplier,
50  const int degree,
51  const int order,
52  const double cosineHarmonicCoefficient,
53  const double sineHarmonicCoefficient,
54  const double legendrePolynomial,
55  const double legendrePolynomialDerivative )
56 {
57  // Return result.
58  return ( Eigen::Vector3d( ) <<
59  - preMultiplier / distance
60  * radiusPowerTerm
61  * ( static_cast< double >( degree ) + 1.0 ) * legendrePolynomial
62  * ( cosineHarmonicCoefficient * cosineOfOrderLongitude
63  + sineHarmonicCoefficient * sineOfOrderLongitude ),
64  preMultiplier * radiusPowerTerm
65  * legendrePolynomialDerivative * cosineOfLatitude * (
66  cosineHarmonicCoefficient * cosineOfOrderLongitude
67  + sineHarmonicCoefficient * sineOfOrderLongitude ),
68  preMultiplier * radiusPowerTerm
69  * static_cast< double >( order ) * legendrePolynomial
70  * ( sineHarmonicCoefficient * cosineOfOrderLongitude
71  - cosineHarmonicCoefficient * sineOfOrderLongitude ) ).finished( );
72 }

75 Eigen::Vector3d computePotentialGradient(
76  const Eigen::Vector3d& sphericalPosition,
77  const double referenceRadius,
78  const double preMultiplier,
79  const int degree,
80  const int order,
81  const double cosineHarmonicCoefficient,
82  const double sineHarmonicCoefficient,
83  const double legendrePolynomial,
84  const double legendrePolynomialDerivative )
85 {
86  return computePotentialGradient(
87  sphericalPosition( radiusIndex ),
88  basic_mathematics::raiseToIntegerPower
89  ( referenceRadius / sphericalPosition( radiusIndex ), static_cast< double >( degree ) + 1.0 ),
90  std::cos( static_cast< double >( order ) * sphericalPosition( longitudeIndex ) ),
91  std::sin( static_cast< double >( order ) * sphericalPosition( longitudeIndex ) ),
92  std::cos( sphericalPosition( latitudeIndex ) ), preMultiplier, degree, order,
93  cosineHarmonicCoefficient, sineHarmonicCoefficient, legendrePolynomial,legendrePolynomialDerivative );
94 }
