144 template< typename StateScalarType >
145 std::map< propagators::EnvironmentModelsToUpdate,
146 std::vector< std::string > > createEnvironmentUpdaterSettings(
147  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
148  const simulation_setup::NamedBodyMap& bodyMap )
149 {
150  std::map< propagators::EnvironmentModelsToUpdate,
151  std::vector< std::string > > environmentModelsToUpdate;
152 
153  // Check dynamics type
154  switch( propagatorSettings->getStateType( ) )
155  {
156  case hybrid:
157  {
158  // Cast to derived type
159  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
160  std::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
161 
162  // Iterate over all propagation settings in hybrid model
163  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > singleAccelerationUpdateNeeds;
164 
165  for( typename std::map< IntegratedStateType,
166  std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >::const_iterator
167  typeIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
168  typeIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); typeIterator++ )
169  {
170  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
171  {
172  // Create current propagation model from settings list (must not be hybrid).
173  if( typeIterator->first != hybrid )
174  {
175  singleAccelerationUpdateNeeds = createEnvironmentUpdaterSettings< StateScalarType >(
176  typeIterator->second.at( i ), bodyMap );
177 
178  // Add single model environment model update settings to full list
179  checkValidityOfRequiredEnvironmentUpdates( singleAccelerationUpdateNeeds, bodyMap );
180  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
181  }
182  else
183  {
184  throw std::runtime_error( "Error when making environment updater type list, cannot handle hybrid "
185  "propagator inside hybrid propagator" );
186  }
187  }
188  }
189  break;
190  }
191  // Retrieve environment model settings for translational dynamics
192  case translational_state:
193  {
194  environmentModelsToUpdate = createTranslationalEquationsOfMotionEnvironmentUpdaterSettings(
195  std::dynamic_pointer_cast<
196  TranslationalStatePropagatorSettings< StateScalarType > >(
197  propagatorSettings )->getAccelerationsMap( ),
198  bodyMap );
199  break;
200  }
201  case rotational_state:
202  {
203  environmentModelsToUpdate = createRotationalEquationsOfMotionEnvironmentUpdaterSettings(
204  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >( propagatorSettings )->getTorqueModelsMap( ),
205  bodyMap );
206  break;
207  }
208  // Retrieve environment model settings for mass rate model
209  case body_mass_state:
210  {
211  environmentModelsToUpdate = createMassPropagationEnvironmentUpdaterSettings(
212  std::dynamic_pointer_cast<
213  MassPropagatorSettings< StateScalarType > >(
214  propagatorSettings )->getMassRateModelsMap( ), bodyMap );
215  break;
216  }
217  case custom_state:
218  {
219  break;
220  }
221  default:
222  {
223  throw std::runtime_error( "Error, cannot create environment updates for type " +
224  std::to_string( propagatorSettings->getStateType( ) ) );
225  }
226  }
227 
228  std::map< propagators::EnvironmentModelsToUpdate,
229  std::vector< std::string > > environmentModelsToUpdateForDependentVariables =
230  createEnvironmentUpdaterSettings( propagatorSettings->getDependentVariablesToSave( ), bodyMap );
231  addEnvironmentUpdates( environmentModelsToUpdate, environmentModelsToUpdateForDependentVariables );
232 
233  std::map< propagators::EnvironmentModelsToUpdate,
234  std::vector< std::string > > environmentModelsToUpdateForTerminationConditions =
235  createEnvironmentUpdaterSettings( propagatorSettings->getTerminationSettings( ), bodyMap );
236  addEnvironmentUpdates( environmentModelsToUpdate, environmentModelsToUpdateForTerminationConditions );
237 
238  // Remove variables from environment updates that are numerically propagated.
239  removePropagatedStatesFomEnvironmentUpdates(
240  environmentModelsToUpdate, getIntegratedTypeAndBodyList( propagatorSettings ) );
241  checkValidityOfRequiredEnvironmentUpdates( environmentModelsToUpdate, bodyMap );
242 
243 
244  return environmentModelsToUpdate;
245 
246 }

269 template< typename StateScalarType, typename TimeType >
270 std::shared_ptr< propagators::EnvironmentUpdater< StateScalarType, TimeType > >
271 createEnvironmentUpdaterForDynamicalEquations(
272  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
273  const simulation_setup::NamedBodyMap& bodyMap )
274 {
275  // Create environment update settings.
276  std::map< IntegratedStateType,
277  std::vector< std::pair< std::string, std::string > > > integratedTypeAndBodyList =
278  getIntegratedTypeAndBodyList< StateScalarType >( propagatorSettings );
279 
280  std::map< propagators::EnvironmentModelsToUpdate,
281  std::vector< std::string > > environmentModelsToUpdate =
282  createEnvironmentUpdaterSettings< StateScalarType >( propagatorSettings, bodyMap );
283 
284  // Create and return environment updater object.
285  return std::make_shared< EnvironmentUpdater< StateScalarType, TimeType > >(
286  bodyMap, environmentModelsToUpdate, integratedTypeAndBodyList );
287 }
