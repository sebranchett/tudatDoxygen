60 bool HybridPropagationTerminationCondition::checkStopCondition( const double time, const double cpuTime )
61 {
62  // Check if single condition is fulfilled.
63  bool stopPropagation = -1;
64  unsigned int stopIndex = 0;
65  if( fulfillSingleCondition_ )
66  {
67  stopPropagation = false;
68  for( unsigned int i = 0; i < propagationTerminationCondition_.size( ); i++ )
69  {
70  if( propagationTerminationCondition_.at( i )->checkStopCondition( time, cpuTime ) )
71  {
72  stopIndex = i;
73  stopPropagation = true;
74  isConditionMetWhenStopping_[ i ] = true;
75  break;
76  }
77  else
78  {
79  isConditionMetWhenStopping_[ i ] = false;
80  }
81  }
82  }
83  // Check all conditions are fulfilled.
84  else
85  {
86  stopPropagation = true;
87  for( unsigned int i = 0; i < propagationTerminationCondition_.size( ); i++ )
88  {
89  if( !propagationTerminationCondition_.at( i )->checkStopCondition( time, cpuTime ) )
90  {
91  stopIndex = i;
92  stopPropagation = false;
93  isConditionMetWhenStopping_[ i ] = false;
94  break;
95  }
96  else
97  {
98  isConditionMetWhenStopping_[ i ] = true;
99  }
100  }
101  }
102 
103  // Save if conditions were met
104  if( stopPropagation )
105  {
106  for( unsigned int i = ( stopIndex + 1 ); i < propagationTerminationCondition_.size( ); i++ )
107  {
108  isConditionMetWhenStopping_[ i ] = propagationTerminationCondition_.at( i )->checkStopCondition( time, cpuTime );
109  }
110  }
111 
112  return stopPropagation;
113 }

117 std::shared_ptr< PropagationTerminationCondition > createPropagationTerminationConditions(
118  const std::shared_ptr< PropagationTerminationSettings > terminationSettings,
119  const simulation_setup::NamedBodyMap& bodyMap,
120  const double initialTimeStep )
121 {
122  std::shared_ptr< PropagationTerminationCondition > propagationTerminationCondition;
123 
124  // Check termination type.
125  switch( terminationSettings->terminationType_ )
126  {
127  case time_stopping_condition:
128  {
129  // Create stopping time termination condition.
130  std::shared_ptr< PropagationTimeTerminationSettings > timeTerminationSettings =
131  std::dynamic_pointer_cast< PropagationTimeTerminationSettings >( terminationSettings );
132  propagationTerminationCondition = std::make_shared< FixedTimePropagationTerminationCondition >(
133  timeTerminationSettings->terminationTime_, ( initialTimeStep > 0 ),
134  timeTerminationSettings->terminateExactlyOnFinalCondition_ );
135  break;
136  }
137  case cpu_time_stopping_condition:
138  {
139  // Create stopping time termination condition.
140  std::shared_ptr< PropagationCPUTimeTerminationSettings > cpuTimeTerminationSettings =
141  std::dynamic_pointer_cast< PropagationCPUTimeTerminationSettings >( terminationSettings );
142  propagationTerminationCondition = std::make_shared< FixedCPUTimePropagationTerminationCondition >(
143  cpuTimeTerminationSettings->cpuTerminationTime_ );
144  break;
145  }
146  case dependent_variable_stopping_condition:
147  {
148  std::shared_ptr< PropagationDependentVariableTerminationSettings > dependentVariableTerminationSettings =
149  std::dynamic_pointer_cast< PropagationDependentVariableTerminationSettings >( terminationSettings );
150 
151  // Get dependent variable function
152  std::function< double( ) > dependentVariableFunction;
153  if( getDependentVariableSaveSize( dependentVariableTerminationSettings->dependentVariableSettings_ ) == 1 )
154  {
155  dependentVariableFunction =
156  getDoubleDependentVariableFunction(
157  dependentVariableTerminationSettings->dependentVariableSettings_, bodyMap );
158  }
159  else
160  {
161  throw std::runtime_error( "Error, cannot make stopping condition from vector dependent variable" );
162  }
163 
164  // Create dependent variable termination condition.
165  propagationTerminationCondition = std::make_shared< SingleVariableLimitPropagationTerminationCondition >(
166  dependentVariableTerminationSettings->dependentVariableSettings_,
167  dependentVariableFunction, dependentVariableTerminationSettings->limitValue_,
168  dependentVariableTerminationSettings->useAsLowerLimit_,
169  dependentVariableTerminationSettings->terminateExactlyOnFinalCondition_,
170  dependentVariableTerminationSettings->terminationRootFinderSettings_ );
171  break;
172  }
173  case custom_stopping_condition:
174  {
175  std::shared_ptr< PropagationCustomTerminationSettings > customTerminationSettings =
176  std::dynamic_pointer_cast< PropagationCustomTerminationSettings >( terminationSettings );
177 
178  // Create dependent variable termination condition.
179  propagationTerminationCondition = std::make_shared< CustomTerminationCondition >(
180  customTerminationSettings->checkStopCondition_,
181  customTerminationSettings->terminateExactlyOnFinalCondition_ );
182  break;
183  }
184  case hybrid_stopping_condition:
185  {
186  std::shared_ptr< PropagationHybridTerminationSettings > hybridTerminationSettings =
187  std::dynamic_pointer_cast< PropagationHybridTerminationSettings >( terminationSettings );
188 
189  // Recursively create termination condition list.
190  std::vector< std::shared_ptr< PropagationTerminationCondition > > propagationTerminationConditionList;
191  for( unsigned int i = 0; i < hybridTerminationSettings->terminationSettings_.size( ); i++ )
192  {
193  propagationTerminationConditionList.push_back(
194  createPropagationTerminationConditions(
195  hybridTerminationSettings->terminationSettings_.at( i ),
196  bodyMap, initialTimeStep ) );
197  }
198  propagationTerminationCondition = std::make_shared< HybridPropagationTerminationCondition >(
199  propagationTerminationConditionList, hybridTerminationSettings->fulfillSingleCondition_,
200  hybridTerminationSettings->terminateExactlyOnFinalCondition_ );
201  break;
202  }
203  default:
204  std::string errorMessage = "Error, stopping condition type " + std::to_string(
205  terminationSettings->terminationType_ ) + " not recognized when making stopping conditions object";
206  throw std::runtime_error( errorMessage );
207  }
208  return propagationTerminationCondition;
