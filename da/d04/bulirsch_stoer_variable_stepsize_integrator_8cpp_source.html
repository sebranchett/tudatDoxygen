30 std::vector< unsigned int > getBulirschStoerStepSequence(
31  const ExtrapolationMethodStepSequences& extrapolationMethodStepSequenceType,
32  const unsigned int lengthOfSequence )
33 {
34  std::vector< unsigned int > stepSequence;
35  switch ( extrapolationMethodStepSequenceType )
36  {
37  case bulirsch_stoer_sequence:
38  using namespace boost::assign;
39  stepSequence += 2, 4, 6;
40  for ( unsigned int i = 3; i < lengthOfSequence; i++ )
41  {
42  stepSequence.push_back(
43  2 * stepSequence.at( i - 2 ) );
44  }
45  break;
46  case deufelhard_sequence:
47  for ( unsigned int i = 0; i < lengthOfSequence; i++ )
48  {
49  stepSequence.push_back( 2 * ( i + 1 ) );
50  }
51  break;
52  default: // The default case will never occur because sequence is an enum
53  throw std::runtime_error( "Error, did not recognize step sequence" );
54  }
55 
56  return stepSequence;
57 }
