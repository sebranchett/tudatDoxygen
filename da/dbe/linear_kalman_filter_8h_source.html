127  void updateFilter( const DependentVector& currentMeasurementVector )
128  {
129  // Compute variables for current step
130  DependentMatrix currentSystemMatrix = stateTransitionMatrixFunction_( this->currentTime_, this->aPosterioriStateEstimate_ );
131  DependentMatrix currentMeasurementMatrix = measurementMatrixFunction_( this->currentTime_, this->aPosterioriStateEstimate_ );
132 
133  // Prediction step
134  DependentVector aPrioriStateEstimate = this->predictState( );
135  DependentVector measurementEstimate = this->measurementFunction_( this->currentTime_, aPrioriStateEstimate );
136  DependentMatrix aPrioriCovarianceEstimate = currentSystemMatrix * this->aPosterioriCovarianceEstimate_ *
137  currentSystemMatrix.transpose( ) + this->systemUncertainty_;
138 
139  // Compute Kalman gain
140  DependentMatrix kalmanGain = aPrioriCovarianceEstimate * currentMeasurementMatrix.transpose( ) * (
141  currentMeasurementMatrix * aPrioriCovarianceEstimate * currentMeasurementMatrix.transpose( ) +
142  this->measurementUncertainty_ ).inverse( );
143 
144  // Correction step
145  this->currentTime_ += this->filteringStepSize_;
146  this->correctState( aPrioriStateEstimate, currentMeasurementVector, measurementEstimate, kalmanGain );
147  this->correctCovariance( aPrioriCovarianceEstimate, currentMeasurementMatrix, kalmanGain );
148  }
