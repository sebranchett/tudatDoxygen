118  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess )
119  {
120  // Set the root function.
121  this->rootFunction = rootFunction;
122 
123  // Start at initial guess, and compute the function value, its first and second derivative.
124  DataType currentRootValue = TUDAT_NAN;
125  DataType nextRootValue = initialGuess;
126  DataType currentFunctionValue = TUDAT_NAN;
127  DataType nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
128  DataType currentFirstDerivativeValue = TUDAT_NAN;
129  DataType nextFirstDerivativeValue = this->rootFunction->
130  computeDerivative( 1, nextRootValue );
131  DataType currentSecondDerivativeValue = TUDAT_NAN;
132  DataType nextSecondDerivativeValue = this->rootFunction->
133  computeDerivative( 2, nextRootValue );
134 
135  // Loop counter.
136  unsigned int counter = 1;
137 
138  // Loop until we have a solution with sufficient accuracy.
139  do
140  {
141  // Save the old values.
142  currentRootValue = nextRootValue;
143  currentFunctionValue = nextFunctionValue;
144  currentFirstDerivativeValue = nextFirstDerivativeValue;
145  currentSecondDerivativeValue = nextSecondDerivativeValue;
146 
147  // Compute next value of root using the following algorithm (see class documentation):
148  nextRootValue = currentRootValue
149  - ( ( 2.0 * currentFunctionValue * currentFirstDerivativeValue )
150  / ( 2.0 * currentFirstDerivativeValue * currentFirstDerivativeValue
151  - currentFunctionValue * currentSecondDerivativeValue ) );
152  nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
153  nextFirstDerivativeValue = this->rootFunction->computeDerivative( 1, nextRootValue );
154  nextSecondDerivativeValue = this->rootFunction->computeDerivative( 2, nextRootValue );
155 
156  // Update the counter.
157  counter++;
158  }
159  while( !this->terminationFunction( nextRootValue, currentRootValue, nextFunctionValue,
160  currentFunctionValue, counter ) );
161 
162  return nextRootValue;
163  }
