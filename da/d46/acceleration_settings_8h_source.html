272  {
273  updateThrust( time );
274  return currentThrust_.normalized( );
275  }

284  {
285  rotationFunction_ = rotationFunction;
286  }

289  {
290  currentTime_ = currentTime;
291  }
292 
294 
297  std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::Vector3d > > getThrustInterpolator( )
298  {
299  return thrustInterpolator_;
300  }
301 
302 private:
303 
305 
309  void updateThrust( const double time )
310  {
311  if( !( time == currentTime_ ) )
312  {
313  currentThrust_ = rotationFunction_( ) * thrustInterpolator_->interpolate( time );
314  currentTime_ = time;
315  }
316  }

320
