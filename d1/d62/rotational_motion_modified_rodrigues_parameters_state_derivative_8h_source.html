158  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
159  {
160  // Loop over each body
161  Eigen::Matrix< StateScalarType, 3, 1 > modifiedRodriguesParametersVector;
162  StateScalarType modifiedRodriguesParametersMagnitude;
163  for( unsigned int i = 0; i < this->bodiesToPropagate_.size( ); i++ )
164  {
165  // Convert to/from shadow modifed Rodrigues parameters (SMRP) (transformation is the same either way)
166  modifiedRodriguesParametersVector = unprocessedState.block( i * 7, 0, 3, 1 );
167  modifiedRodriguesParametersMagnitude = modifiedRodriguesParametersVector.norm( );
168  if ( modifiedRodriguesParametersMagnitude >= 1.0 )
169  {
170  // Invert flag
171  unprocessedState.segment( i * 7 + 3, 1 ) = ( unprocessedState.block( i * 7 + 3, 0, 1, 1 ) -
172  Eigen::Matrix< StateScalarType, 1, 1 >::Ones( ) ).cwiseAbs( );
173 
174  // Convert to MRP/SMRP
175  modifiedRodriguesParametersVector /= - modifiedRodriguesParametersMagnitude *
176  modifiedRodriguesParametersMagnitude;
177 
178  // Replace MRP with SMPR, or vice-versa
179  unprocessedState.segment( i * 7, 3 ) = modifiedRodriguesParametersVector;
180  }
181  }
