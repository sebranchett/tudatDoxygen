23 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< EphemerisSettings >& ephemerisSettings )
24 {
25  if ( ! ephemerisSettings )
26  {
27  return;
28  }
29  using namespace json_interface;
30  using K = Keys::Body::Ephemeris;
31 
32  // Common keys
33  const EphemerisType ephemerisType = ephemerisSettings->getEphemerisType( );
34  jsonObject[ K::type ] = ephemerisType;
35  jsonObject[ K::frameOrigin ] = ephemerisSettings->getFrameOrigin( );
36  jsonObject[ K::frameOrientation ] = ephemerisSettings->getFrameOrientation( );
37  // jsonObject[ K::makeMultiArc ] = ephemerisSettings->getMakeMultiArcEphemeris( );
38 
39  switch ( ephemerisType )
40  {
41  case approximate_planet_positions:
42  {
43  std::shared_ptr< ApproximatePlanetPositionSettings > approximatePlanetPositionSettings =
44  std::dynamic_pointer_cast< ApproximatePlanetPositionSettings >( ephemerisSettings );
45  assertNonnullptrPointer( approximatePlanetPositionSettings );
46  jsonObject[ K::bodyIdentifier ] = approximatePlanetPositionSettings->getBodyIdentifier( );
47  jsonObject[ K::useCircularCoplanarApproximation ] =
48  approximatePlanetPositionSettings->getUseCircularCoplanarApproximation( );
49  return;
50  }
51  case direct_spice_ephemeris:
52  {
53  std::shared_ptr< DirectSpiceEphemerisSettings > directSpiceEphemerisSettings =
54  std::dynamic_pointer_cast< DirectSpiceEphemerisSettings >( ephemerisSettings );
55  assertNonnullptrPointer( directSpiceEphemerisSettings );
56  jsonObject[ K::correctForStellarAberration ] =
57  directSpiceEphemerisSettings->getCorrectForStellarAberration( );
58  jsonObject[ K::correctForLightTimeAberration ] =
59  directSpiceEphemerisSettings->getCorrectForLightTimeAberration( );
60  jsonObject[ K::convergeLighTimeAberration ] =
61  directSpiceEphemerisSettings->getConvergeLighTimeAberration( );
62  return;
63  }
64  case interpolated_spice:
65  {
66  std::shared_ptr< InterpolatedSpiceEphemerisSettings > interpolatedSpiceEphemerisSettings =
67  std::dynamic_pointer_cast< InterpolatedSpiceEphemerisSettings >( ephemerisSettings );
68  assertNonnullptrPointer( interpolatedSpiceEphemerisSettings );
69  jsonObject[ K::initialTime ] = interpolatedSpiceEphemerisSettings->getInitialTime( );
70  jsonObject[ K::finalTime ] = interpolatedSpiceEphemerisSettings->getFinalTime( );
71  jsonObject[ K::timeStep ] = interpolatedSpiceEphemerisSettings->getTimeStep( );
72  jsonObject[ K::interpolator ] = interpolatedSpiceEphemerisSettings->getInterpolatorSettings( );
73  jsonObject[ K::useLongDoubleStates ] = interpolatedSpiceEphemerisSettings->getUseLongDoubleStates( );
74  return;
75  }
76  case tabulated_ephemeris:
77  {
78  std::shared_ptr< TabulatedEphemerisSettings > tabulatedEphemerisSettings =
79  std::dynamic_pointer_cast< TabulatedEphemerisSettings >( ephemerisSettings );

81  jsonObject[ K::bodyStateHistory ] = tabulatedEphemerisSettings->getBodyStateHistory( );
82  jsonObject[ K::useLongDoubleStates ] = tabulatedEphemerisSettings->getUseLongDoubleStates( );
83  return;
84  }
85  case constant_ephemeris:
86  {
87  std::shared_ptr< ConstantEphemerisSettings > constantEphemerisSettings =
88  std::dynamic_pointer_cast< ConstantEphemerisSettings >( ephemerisSettings );
89  assertNonnullptrPointer( constantEphemerisSettings );
90  jsonObject[ K::constantState ] = constantEphemerisSettings->getConstantState( );
91  return;
92  }
93  case kepler_ephemeris:
94  {
95  std::shared_ptr< KeplerEphemerisSettings > keplerEphemerisSettings =
96  std::dynamic_pointer_cast< KeplerEphemerisSettings >( ephemerisSettings );
97  assertNonnullptrPointer( keplerEphemerisSettings );
98  jsonObject[ K::initialStateInKeplerianElements ] =
99  keplerEphemerisSettings->getInitialStateInKeplerianElements( );
100  jsonObject[ K::epochOfInitialState ] =
101  keplerEphemerisSettings->getEpochOfInitialState( );
102  jsonObject[ K::centralBodyGravitationalParameter ] =
103  keplerEphemerisSettings->getCentralBodyGravitationalParameter( );
104  jsonObject[ K::rootFinderAbsoluteTolerance ] =
105  keplerEphemerisSettings->getRootFinderAbsoluteTolerance( );
106  jsonObject[ K::rootFinderMaximumNumberOfIterations ] =
107  keplerEphemerisSettings->getRootFinderMaximumNumberOfIterations( );
108  return;
109  }
110  default:
111  handleUnimplementedEnumValue( ephemerisType, ephemerisTypes, unsupportedEphemerisTypes );
112  }

116 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< EphemerisSettings >& ephemerisSettings )
117 {
118  using namespace ephemerides;
119  using namespace json_interface;
120  using K = Keys::Body::Ephemeris;
121 
122  // Get atmosphere model type
123  const EphemerisType ephemerisType = getValue< EphemerisType >( jsonObject, K::type );
124 
125  switch ( ephemerisType ) {
126  case approximate_planet_positions:
127  {
128  ephemerisSettings = std::make_shared< ApproximatePlanetPositionSettings >(
129  getValue< ApproximatePlanetPositionsBase::BodiesWithEphemerisData >(
130  jsonObject, K::bodyIdentifier ),
131  getValue< bool >( jsonObject, K::useCircularCoplanarApproximation ) );
132  break;
133  }
134  case direct_spice_ephemeris:
135  {
136  DirectSpiceEphemerisSettings defaults;
137  ephemerisSettings = std::make_shared< DirectSpiceEphemerisSettings >(
138  defaults.getFrameOrigin( ),
139  defaults.getFrameOrientation( ),
140  getValue( jsonObject, K::correctForStellarAberration,
141  defaults.getCorrectForStellarAberration( ) ),
142  getValue( jsonObject, K::correctForLightTimeAberration,
143  defaults.getCorrectForLightTimeAberration( ) ),
144  getValue( jsonObject, K::convergeLighTimeAberration,
145  defaults.getConvergeLighTimeAberration( ) ) );
146  break;
147  }
148  case tabulated_ephemeris:
149  {
150  TabulatedEphemerisSettings defaults( ( std::map< double, Eigen::Vector6d >( ) ) );
151  TabulatedEphemerisSettings tabulatedEphemerisSettings(
152  getValue< std::map< double, Eigen::Vector6d > >( jsonObject, K::bodyStateHistory ) );
153  tabulatedEphemerisSettings.setUseLongDoubleStates(
154  getValue( jsonObject, K::useLongDoubleStates, defaults.getUseLongDoubleStates( ) ) );
155  ephemerisSettings = std::make_shared< TabulatedEphemerisSettings >( tabulatedEphemerisSettings );
156  break;
157  }
158  case interpolated_spice:
159  {
160  InterpolatedSpiceEphemerisSettings defaults( TUDAT_NAN, TUDAT_NAN, TUDAT_NAN );
161  InterpolatedSpiceEphemerisSettings interpolatedSpiceEphemerisSettings(
162  getValue< double >( jsonObject, K::initialTime ),
163  getValue< double >( jsonObject, K::finalTime ),
164  getValue< double >( jsonObject, K::timeStep ),
165  defaults.getFrameOrigin( ),
166  defaults.getFrameOrientation( ),
167  getValue( jsonObject, K::interpolator, defaults.getInterpolatorSettings( ) ) );
168  interpolatedSpiceEphemerisSettings.setUseLongDoubleStates(
169  getValue( jsonObject, K::useLongDoubleStates, defaults.getUseLongDoubleStates( ) ) );
170  ephemerisSettings = std::make_shared< InterpolatedSpiceEphemerisSettings >(
171  interpolatedSpiceEphemerisSettings );
172  break;
173  }
174  case constant_ephemeris:
175  {
176  ConstantEphemerisSettings defaults( ( Eigen::Vector6d( ) ) );
177  ephemerisSettings = std::make_shared< ConstantEphemerisSettings >(
178  getValue< Eigen::Vector6d >( jsonObject, K::constantState ) );
179  break;
180  }
181  case kepler_ephemeris:
182  {
183  KeplerEphemerisSettings defaults( Eigen::Vector6d( ), TUDAT_NAN, TUDAT_NAN );
184  ephemerisSettings = std::make_shared< KeplerEphemerisSettings >(
185  getValue< Eigen::Vector6d >( jsonObject, K::initialStateInKeplerianElements ),
186  getValue< double >( jsonObject, K::epochOfInitialState ),
187  getValue< double >( jsonObject, K::centralBodyGravitationalParameter ),
188  defaults.getFrameOrigin( ),
189  defaults.getFrameOrientation( ),
190  getValue( jsonObject, K::rootFinderAbsoluteTolerance,
191  defaults.getRootFinderAbsoluteTolerance( ) ),
192  getValue( jsonObject, K::rootFinderMaximumNumberOfIterations,
193  defaults.getRootFinderMaximumNumberOfIterations( ) ) );
194  break;
195  }
196  default:
197  handleUnimplementedEnumValue( ephemerisType, ephemerisTypes, unsupportedEphemerisTypes );
198  }
199 
200  if ( isDefined( jsonObject, K::frameOrigin ) )
201  {
202  ephemerisSettings->resetFrameOrigin( getValue< std::string >( jsonObject, K::frameOrigin ) );
203  }
204 
205  if ( isDefined( jsonObject, K::frameOrientation ) )
206  {
207  ephemerisSettings->resetFrameOrientation( getValue< std::string >( jsonObject, K::frameOrientation ) );
208  }
209 
210  /*
211  if ( isDefined( jsonObject, K::makeMultiArc ) )
212  {
213  ephemerisSettings->resetMakeMultiArcEphemeris( getValue< bool >( jsonObject, K::makeMultiArc ) );
214  }
215  */
216 }
