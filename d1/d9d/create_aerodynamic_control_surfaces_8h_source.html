247 template< unsigned int NumberOfIndependentVariables >
248 boost::shared_ptr< ControlSurfaceIncrementAerodynamicCoefficientSettings >
249 readGivenSizeTabulatedControlIncrementAerodynamicCoefficientsFromFiles(
250  const std::map< int, std::string > forceCoefficientFiles,
251  const std::map< int, std::string > momentCoefficientFiles,
252  const std::vector< aerodynamics::AerodynamicCoefficientsIndependentVariables > independentVariableNames )
253 {
254  std::pair< boost::multi_array< Eigen::Vector3d, NumberOfIndependentVariables >, std::vector< std::vector< double > > >
255  aerodynamicForceCoefficients = input_output::readAerodynamicCoefficients< NumberOfIndependentVariables >(
256  forceCoefficientFiles );
257  std::pair< boost::multi_array< Eigen::Vector3d, NumberOfIndependentVariables >, std::vector< std::vector< double > > >
258  aerodynamicMomentCoefficients = input_output::readAerodynamicCoefficients< NumberOfIndependentVariables >(
259  momentCoefficientFiles );
260 
261  if( !input_output::compareIndependentVariables(
262  aerodynamicForceCoefficients.second, aerodynamicMomentCoefficients.second ) )
263  {
264  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, force and moment independent variables are inconsistent" );
265  }
266 
267  if( independentVariableNames.size( ) != NumberOfIndependentVariables )
268  {
269  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, input sizes are inconsistent" );
270  }
271 
272  return boost::make_shared< TabulatedControlSurfaceIncrementAerodynamicCoefficientSettings<
273  NumberOfIndependentVariables > >(
274  aerodynamicForceCoefficients.second, aerodynamicForceCoefficients.first, aerodynamicMomentCoefficients.first,
275  independentVariableNames );
276 }

364 createTabulatedControlSurfaceIncrementAerodynamicCoefficientInterface(
365  const std::vector< std::vector< double > > independentVariables,
366  const boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) > forceCoefficients,
367  const boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) > momentCoefficients,
368  const std::vector< aerodynamics::AerodynamicCoefficientsIndependentVariables >
369  independentVariableNames )
370 {
371  // Check input consistency.
372  if( independentVariables.size( ) != NumberOfDimensions )
373  {
374  throw std::runtime_error( "Error when creating tabulated control surface increment aerodynamic coefficient interface, inconsistent variable vector dimensioning" );
375  }
376 
377  if( independentVariableNames.size( ) != NumberOfDimensions )
378  {
379  throw std::runtime_error( "Error when creating tabulated control surface increment aerodynamic coefficient interface, inconsistent variable name vector dimensioning" );
380 
381  }
382 
383  // Create interpolators for coefficients.
384  boost::shared_ptr< interpolators::MultiLinearInterpolator
385  < double, Eigen::Vector3d, NumberOfDimensions > > forceInterpolator =
386  boost::make_shared< interpolators::MultiLinearInterpolator
387  < double, Eigen::Vector3d, NumberOfDimensions > >(
388  independentVariables, forceCoefficients );
389  boost::shared_ptr< interpolators::MultiLinearInterpolator
390  < double, Eigen::Vector3d, NumberOfDimensions > > momentInterpolator =

392  < double, Eigen::Vector3d, NumberOfDimensions > >(
393  independentVariables, momentCoefficients );
394 
395  // Create aerodynamic coefficient interface.
396  return boost::make_shared< aerodynamics::CustomControlSurfaceIncrementAerodynamicInterface >(
397  boost::bind( &interpolators::MultiLinearInterpolator
398  < double, Eigen::Vector3d, NumberOfDimensions >::interpolate,
399  forceInterpolator, _1 ),
400  boost::bind( &interpolators::MultiLinearInterpolator
401  < double, Eigen::Vector3d, NumberOfDimensions >::interpolate,
402  momentInterpolator, _1 ),
403  independentVariableNames );
