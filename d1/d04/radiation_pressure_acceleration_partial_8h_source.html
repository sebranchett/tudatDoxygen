234  void update( const double currentTime = 0.0 )
235  {
236  if( !( currentTime_ == currentTime ) )
237  {
238  // Compute helper quantities.
239  Eigen::Vector3d rangeVector = ( acceleratedBodyState_( ) - sourceBodyState_( ) );
240  double range = rangeVector.norm( );
241  double rangeInverse = 1.0 / ( range );
242 
243  // Compute position partial.
244  currentPartialWrtPosition_ =
245  ( radiationPressureCoefficientFunction_( ) * areaFunction_( ) * radiationPressureFunction_( ) /
246  acceleratedBodyMassFunction_( ) ) * ( Eigen::Matrix3d::Identity( ) * rangeInverse - 3.0 *
247  rangeVector * rangeVector.transpose( ) * rangeInverse / (
248  range * range ) );
249  currentTime_ = currentTime;
250  }
