124  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess = 0.0 )
125  {
126  // The value of the initialGuess is not used.
127  TUDAT_UNUSED_PARAMETER( initialGuess );
128 
129  // Set the root function.
130  this->rootFunction = rootFunction;
131 
132  // Initialize previous values.
133  DataType previousRootValue = TUDAT_NAN;
134  DataType previousRootFunctionValue = TUDAT_NAN;
135 
136  // Duplicate the interval and use this duplicate to shrink the interval.
137  // Compute the midpoint of the interval, and take this as the current root.
138  DataType currentLowerBound = lowerBound_;
139  DataType currentUpperBound = upperBound_;
140  DataType rootValue = ( currentLowerBound + currentUpperBound ) / 2.0;
141 
142  // Find the corresponding function values at the important interval points (lower bound,
143  // upper bound and midpoint).
144  DataType currentLowerBoundFunctionValue = this->rootFunction->evaluate( currentLowerBound );
145  DataType currentUpperBoundFunctionValue = this->rootFunction->evaluate( currentUpperBound );
146  DataType rootFunctionValue = this->rootFunction->evaluate( rootValue );
147 
148  // Validate that upperbound and lowerbound function values have different signs
149  // (requirement).
150  if( currentLowerBoundFunctionValue * currentUpperBoundFunctionValue > 0.0 )
151  {
152  throw std::runtime_error( "The Bisection algorithm requires that the values at the upper, "
153  "and lower bounds have a different sign." );
154  }
155 
156  // Loop counter.
157  unsigned int counter = 1;
158 
159  // Loop until we have a solution with sufficient accuracy.
160  do
161  {
162  // Sanity check.
163  if( currentLowerBoundFunctionValue * currentUpperBoundFunctionValue > 0.0 )
164  {
165  throw std::runtime_error( "The Bisection algorithm requires that the values at the upper, "
166  "and lower bounds have a different sign, error during iteration." );
167  }
168 
169  // Save old values.
170  previousRootValue = rootValue;
171  previousRootFunctionValue = rootFunctionValue;
172 
173  // Check which subinterval to keep, by maintaining endpoints with opposite function
174  // value signs.
175  if( rootFunctionValue * currentLowerBoundFunctionValue < 0.0 )
176  {
177  // Different sign, hence the upper bound is replaced.
178  currentUpperBound = rootValue;
179  currentUpperBoundFunctionValue = rootFunctionValue;
180  }
181  else
182  {
183  // Same sign, hence the lower bound is replaced.
184  currentLowerBound = rootValue;
185  currentLowerBoundFunctionValue = rootFunctionValue;
186  }
187 
188  // Compute the new midpoint of the interval and its function value.
189  rootValue = ( currentLowerBound + currentUpperBound ) / 2.0;
190  rootFunctionValue = this->rootFunction->evaluate( rootValue );
191 
192  counter++;
193  }
194  while( !this->terminationFunction( rootValue, previousRootValue, rootFunctionValue,
195  previousRootFunctionValue, counter ) );
196 
197  return rootValue;
198  }
