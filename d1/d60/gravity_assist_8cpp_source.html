60 double gravityAssist( const double centralBodyGravitationalParameter,
61  const Eigen::Vector3d& centralBodyVelocity,
62  const Eigen::Vector3d& incomingVelocity,
63  const Eigen::Vector3d& outgoingVelocity,
64  const double smallestPeriapsisDistance,
65  const bool useEccentricityInsteadOfPericenter,
66  const double speedTolerance,
67  RootFinderPointer rootFinder )
68 {
69  using basic_mathematics::UnivariateProxyPointer;
70  using basic_mathematics::UnivariateProxy;
71 
72  // Compute incoming and outgoing hyperbolic excess velocity.
73  const Eigen::Vector3d incomingHyperbolicExcessVelocity
74  = incomingVelocity - centralBodyVelocity;
75  const Eigen::Vector3d outgoingHyperbolicExcessVelocity
76  = outgoingVelocity - centralBodyVelocity;
77 
78  // Compute absolute values of the hyperbolic excess velocities.
79  const double absoluteIncomingExcessVelocity = incomingHyperbolicExcessVelocity.norm( );
80  const double absoluteOutgoingExcessVelocity = outgoingHyperbolicExcessVelocity.norm( );
81 
82  // Compute bending angle.
83  double bendingAngle = linear_algebra::computeAngleBetweenVectors(
84  incomingHyperbolicExcessVelocity, outgoingHyperbolicExcessVelocity );
85 
86  // Compute maximum achievable bending angle.
87  const double maximumBendingAngle =
88  std::asin( 1.0 / ( 1.0 + ( smallestPeriapsisDistance *
89  absoluteIncomingExcessVelocity *
90  absoluteIncomingExcessVelocity /
91  centralBodyGravitationalParameter ) ) ) +
92  std::asin( 1.0 / ( 1.0 + ( smallestPeriapsisDistance *
93  absoluteOutgoingExcessVelocity *
94  absoluteOutgoingExcessVelocity /
95  centralBodyGravitationalParameter ) ) );
96 
97  // Initialize bending effect deltaV, which is zero, unless extra bending angle is required.
98  double bendingEffectDeltaV = 0.0;
99 
100  // Initialize velocity effect delta V parameter.
101  double velocityEffectDeltaV = 0.0;
102 
103  // Check if an additional bending angle is required. If so, the additional bending angle
104  // maneuver has to be performed. Also the pericenter radius will be the minimum pericenter
105  // radius to obtain the largest possible bending angle 'for free'. Hence no root finding is
106  // required for this case. As noted above, this may not be ideal for all cases. (for cases
107  // in which the excess velocities are relatively small)
108  if ( bendingAngle > maximumBendingAngle )
109  {
110  // Compute required extra bending angle that cannot be delivered by an unpowered swing-by.
111  const double extraBendingAngle = bendingAngle - maximumBendingAngle;
112 
113  // Compute necessary delta-V due to bending-effect.
114  bendingEffectDeltaV = 2.0 * std::min( absoluteIncomingExcessVelocity,
115  absoluteOutgoingExcessVelocity ) *
116  std::sin( extraBendingAngle / 2.0 );
117 
118  // This means the pericenter radius is now equal to the smallest pericenter radius, to
119  // ensure the largest possible bending angle.
120  const double pericenterRadius = smallestPeriapsisDistance;
121 
122  // Compute semi-major axis of hyperbolic legs.
123  const double incomingSemiMajorAxis = -1.0 * centralBodyGravitationalParameter /
124  absoluteIncomingExcessVelocity /
125  absoluteIncomingExcessVelocity;
126  const double outgoingSemiMajorAxis = -1.0 * centralBodyGravitationalParameter /
127  absoluteOutgoingExcessVelocity /
128  absoluteOutgoingExcessVelocity;
129 
130  // Compute incoming hyperbolic leg eccentricity.
131  const double incomingEccentricity = 1 - pericenterRadius / incomingSemiMajorAxis;
132 
133  // Compute outgoing hyperbolic leg eccentricity.
134  const double outgoingEccentricity = 1 - pericenterRadius / outgoingSemiMajorAxis;
135 
136  // Compute incoming and outgoing velocities at periapsis.
137  const double incomingVelocityAtPeriapsis = absoluteIncomingExcessVelocity *
138  std::sqrt( ( incomingEccentricity + 1.0 ) / ( incomingEccentricity - 1.0 ) );
139  const double outgoingVelocityAtPeriapsis = absoluteOutgoingExcessVelocity *
140  std::sqrt( ( outgoingEccentricity + 1.0 ) / ( outgoingEccentricity - 1.0 ) );
141 
142  // Compute necessary delta-V due to velocity-effect.
143  velocityEffectDeltaV = std::fabs( incomingVelocityAtPeriapsis -
144  outgoingVelocityAtPeriapsis );
145  }
146 
147  else if ( ( std::fabs( absoluteIncomingExcessVelocity - absoluteOutgoingExcessVelocity )
148  <= speedTolerance ) )
149  {
150  // In this case no maneuver has to be performed. Hence no iteration is performed, and the
151  // delta V is simply kept at 0.0.
152  }
153 
154  // Here the required maneuver to patch the incoming and outgoing excess velocities is
155  // calculated. In this implementation, the eccentricity will be used as iteration parameter.
156  else if ( useEccentricityInsteadOfPericenter )
157  {
158  // Compute semi-major axis of hyperbolic legs.
159  const double incomingSemiMajorAxis = -1.0 * centralBodyGravitationalParameter /
160  absoluteIncomingExcessVelocity /
161  absoluteIncomingExcessVelocity;
162  const double outgoingSemiMajorAxis = -1.0 * centralBodyGravitationalParameter /
163  absoluteOutgoingExcessVelocity /
164  absoluteOutgoingExcessVelocity;
165 
166  // Set the gravity assist function with the variables to perform root finder calculations.
167  EccentricityFindingFunctions eccentricityFindingFunctions( incomingSemiMajorAxis,
168  outgoingSemiMajorAxis,
169  bendingAngle );
170 
171  // Create an object containing the function of which we whish to obtain the root from.
172  UnivariateProxyPointer rootFunction = std::make_shared< UnivariateProxy >(
173  std::bind( &EccentricityFindingFunctions::
174  computeIncomingEccentricityFunction,
175  eccentricityFindingFunctions, std::placeholders::_1 ) );
176 
177  // Add the first derivative of the root function.
178  rootFunction->addBinding( -1, std::bind(
179  &EccentricityFindingFunctions::
180  computeFirstDerivativeIncomingEccentricityFunction,
181  eccentricityFindingFunctions, std::placeholders::_1 ) );
182 
183  // Initialize incoming eccentricity.
184  double incomingEccentricity = TUDAT_NAN;
185 
186  // Set initial guess of the variable computed in Newton-Rapshon method.
187  if ( ( absoluteOutgoingExcessVelocity / absoluteIncomingExcessVelocity ) < 100.0 )
188  {
189  // In these cases the very low estimate (which is given under else) may in some cases
190  // result in no convergence. Hence a higher value of 1.01 is necessary. This will not
191  // result in 'going through' 1.0 as mentioned below, because the eccentricity in these
192  // cases is always high!
193  try
194  {
195  incomingEccentricity = rootFinder->execute( rootFunction, 1.0 + 1.0e-2 );
196  }
197  catch(std::runtime_error)
198  {
199  root_finders::RootFinderPointer rootFinder_temp
200  = std::make_shared< root_finders::Bisection >( 1.0e-12, 1000 ) ;
201  incomingEccentricity = rootFinder_temp->execute( rootFunction, 1.0 + 1.0e-2 );
202 
203  }
204  }
205 
206  else
207  {
208  // This is set to a value that is close to 1.0. This is more robust than higher values,
209  // because for those higher values Newton Raphson sometimes 'goes through' 1.0. This
210  // results in NaN values for the derivative of the eccentricity finding function.
211  try
212  {
213  incomingEccentricity = rootFinder->execute( rootFunction, 1.0 + 1.0e-10 );
214  }
215  catch(std::runtime_error)
216  {
217  root_finders::RootFinderPointer rootFinder_temp
218  = std::make_shared< root_finders::Bisection >( 1.0e-12, 1000 ) ;
219  incomingEccentricity = rootFinder_temp->execute( rootFunction, 1.0 + 1.0e-10 );
220 
221  }
222  }
223 
224  // Compute outgoing hyperbolic leg eccentricity.
225  const double outgoingEccentricity = 1.0 - ( incomingSemiMajorAxis /
226  outgoingSemiMajorAxis ) *
227  ( 1.0 - incomingEccentricity );
228 
229  // Compute incoming and outgoing velocities at periapsis.
230  const double incomingVelocityAtPeriapsis = absoluteIncomingExcessVelocity *
231  std::sqrt( ( incomingEccentricity + 1.0 ) / ( incomingEccentricity - 1.0 ) );
232  const double outgoingVelocityAtPeriapsis = absoluteOutgoingExcessVelocity *
233  std::sqrt( ( outgoingEccentricity + 1.0 ) / ( outgoingEccentricity - 1.0 ) );
234 
235  // Compute necessary delta-V due to velocity-effect.
236  velocityEffectDeltaV = std::fabs( incomingVelocityAtPeriapsis -
237  outgoingVelocityAtPeriapsis );
238  }
239 
240  // Here the required maneuver to patch the incoming and outgoing excess velocities is
241  // calculated. In this implementation, the pericenter radius will be used as iteration
242  // parameter.
243  else
244  {
245  // Compute semi-major axis of hyperbolic legs. This is the absolute semi-major axis, because
246  // it will otherwisely result in the root of a negative function for various cases during
247  // the rootfinding process.
248  const double absoluteIncomingSemiMajorAxis = 1.0 * centralBodyGravitationalParameter /
249  absoluteIncomingExcessVelocity /
250  absoluteIncomingExcessVelocity;
251  const double absoluteOutgoingSemiMajorAxis = 1.0 * centralBodyGravitationalParameter /
252  absoluteOutgoingExcessVelocity /
253  absoluteOutgoingExcessVelocity;
254 
255  // Set the gravity assist function with the variables to perform root finder calculations.
256  PericenterFindingFunctions pericenterFindingFunctions( absoluteIncomingSemiMajorAxis,
257  absoluteOutgoingSemiMajorAxis,
258  bendingAngle);
259 
260  // Create an object containing the function of which we whish to obtain the root from.
261  UnivariateProxyPointer rootFunction = std::make_shared< UnivariateProxy >(
262  std::bind( &PericenterFindingFunctions::computePericenterRadiusFunction,
263  pericenterFindingFunctions, std::placeholders::_1 ) );
264 
265  // Add the first derivative of the root function.
266  rootFunction->addBinding( -1, std::bind( &PericenterFindingFunctions::
267  computeFirstDerivativePericenterRadiusFunction,
268  pericenterFindingFunctions, std::placeholders::_1 ) );
269 
270  // Set pericenter radius based on result of Newton-Raphson root-finding algorithm.
271  std::cout<<"here3"<<std::endl;
272  const double pericenterRadius = rootFinder->execute( rootFunction,
273  smallestPeriapsisDistance );
274 
275  // Compute incoming hyperbolic leg eccentricity.
276  const double incomingEccentricity = 1.0 + pericenterRadius / absoluteIncomingSemiMajorAxis;
277 
278  // Compute outgoing hyperbolic leg eccentricity.
279  const double outgoingEccentricity = 1.0 + pericenterRadius / absoluteOutgoingSemiMajorAxis;
280 
281  // Compute incoming and outgoing velocities at periapsis.
282  const double incomingVelocityAtPeriapsis = absoluteIncomingExcessVelocity *
283  std::sqrt( ( incomingEccentricity + 1.0 ) / ( incomingEccentricity - 1.0 ) );
284  const double outgoingVelocityAtPeriapsis = absoluteOutgoingExcessVelocity *
285  std::sqrt( ( outgoingEccentricity + 1.0 ) / ( outgoingEccentricity - 1.0 ) );
286 
287  // Compute necessary delta-V due to velocity-effect.
288  velocityEffectDeltaV = std::fabs( incomingVelocityAtPeriapsis -
289  outgoingVelocityAtPeriapsis );
290  }
291 
292  // Compute and return the total delta-V.
293  return bendingEffectDeltaV + velocityEffectDeltaV;
294 }

297 Eigen::Vector3d gravityAssist( const double centralBodyGravitationalParameter,
298  const Eigen::Vector3d& centralBodyVelocity,
299  const Eigen::Vector3d& incomingVelocity,
300  const double rotationAngle,
301  const double pericenterRadius )
302 {
303  // Calculate the incoming velocity.
304  const Eigen::Vector3d relativeIncomingVelocity = incomingVelocity - centralBodyVelocity;
305  const double absoluteRelativeIncomingVelocity = relativeIncomingVelocity.norm( );
306 
307  // Calculate the eccentricity and bending angle.
308  const double eccentricity = 1.0 + pericenterRadius / centralBodyGravitationalParameter *
309  absoluteRelativeIncomingVelocity * absoluteRelativeIncomingVelocity;
310  const double bendingAngle = 2.0 * std::asin ( 1.0 / eccentricity );
311 
312  // Calculate the unit vectors.
313  const Eigen::Vector3d unitVector1 = relativeIncomingVelocity /
314  absoluteRelativeIncomingVelocity;
315  const Eigen::Vector3d unitVector2 = unitVector1.cross( centralBodyVelocity ).normalized( );
316  const Eigen::Vector3d unitVector3 = unitVector1.cross( unitVector2 );
317 
318  // Calculate the relative outgoing velocity.
319  const Eigen::Vector3d relativeOutgoingVelocity = absoluteRelativeIncomingVelocity *
320  ( std::cos( bendingAngle ) * unitVector1 + std::sin( bendingAngle ) *
321  std::cos( rotationAngle ) * unitVector2 + std::sin( bendingAngle ) *
322  std::sin( rotationAngle ) * unitVector3 );
323 
324  // Add the relative outgoing velocity to the swing-by body velocity and return it.
325  return centralBodyVelocity + relativeOutgoingVelocity;
326 }

329 Eigen::Vector3d gravityAssist( const double centralBodyGravitationalParameter,
330  const Eigen::Vector3d& centralBodyVelocity,
331  const Eigen::Vector3d& incomingVelocity,
332  const double rotationAngle,
333  const double pericenterRadius,
334  const double deltaV )
335 {
336  // Calculate the incoming velocity.
337  const Eigen::Vector3d relativeIncomingVelocity = incomingVelocity - centralBodyVelocity;
338  const double absoluteRelativeIncomingVelocity = relativeIncomingVelocity.norm( );
339 
340  // Calculate the incoming eccentricity and bending angle.
341  const double incomingEccentricity = 1.0 + pericenterRadius /
342  centralBodyGravitationalParameter *
343  absoluteRelativeIncomingVelocity *
344  absoluteRelativeIncomingVelocity;
345  const double incomingBendingAngle = std::asin ( 1.0 / incomingEccentricity );
346 
347  // Calculate the pericenter velocities.
348  const double incomingPericenterVelocity = std::sqrt( absoluteRelativeIncomingVelocity *
349  absoluteRelativeIncomingVelocity *
350  ( incomingEccentricity + 1.0 ) /
351  ( incomingEccentricity - 1.0 ) );
352  const double outgoingPericenterVelocity = incomingPericenterVelocity + deltaV;
353 
354  // Calculate magnitude of the absolute relative outgoing velocity.
355  const double absoluteRelativeOutgoingVelocity =
356  std::sqrt( outgoingPericenterVelocity * outgoingPericenterVelocity -
357  2.0 * centralBodyGravitationalParameter / pericenterRadius );
358 
359  // Calculate the remaining bending angles.
360  const double outgoingBendingAngle =
361  std::asin ( 1.0 / ( 1.0 + absoluteRelativeOutgoingVelocity *
362  absoluteRelativeOutgoingVelocity * pericenterRadius /
363  centralBodyGravitationalParameter ) );
364  const double bendingAngle = incomingBendingAngle + outgoingBendingAngle;
365 
366  // Calculate the unit vectors.
367  const Eigen::Vector3d unitVector1 = relativeIncomingVelocity /
368  absoluteRelativeIncomingVelocity;
369  const Eigen::Vector3d unitVector2 = unitVector1.cross( centralBodyVelocity ).normalized( );
370  const Eigen::Vector3d unitVector3 = unitVector1.cross( unitVector2 );
371 
372  // Calculate the relative outgoing velocity.
373  const Eigen::Vector3d relativeOutgoingVelocity = absoluteRelativeOutgoingVelocity *
374  ( std::cos( bendingAngle ) * unitVector1 + std::sin( bendingAngle ) *
375  std::cos( rotationAngle ) * unitVector2 + std::sin( bendingAngle ) *
376  std::sin( rotationAngle ) * unitVector3 );
377 
378  // Add the relative outgoing velocity to the swing-by body velocity and return it.
379  return centralBodyVelocity + relativeOutgoingVelocity;
380 }

392 double PericenterFindingFunctions::computeFirstDerivativePericenterRadiusFunction(
393  const double pericenterRadius )
394 {
395  return -absoluteIncomingSemiMajorAxis_ / ( absoluteIncomingSemiMajorAxis_ + pericenterRadius )
396  / std::sqrt( ( pericenterRadius + 2.0 * absoluteIncomingSemiMajorAxis_ )
397  * pericenterRadius ) -
398  absoluteOutgoingSemiMajorAxis_ / ( absoluteOutgoingSemiMajorAxis_ + pericenterRadius )
399  / std::sqrt( ( pericenterRadius + 2.0 * absoluteOutgoingSemiMajorAxis_ ) *

404 double EccentricityFindingFunctions::computeIncomingEccentricityFunction(
405  const double incomingEccentricity )
406 {
407  return std::asin( 1.0 / incomingEccentricity )
408  + std::asin( 1.0 / ( 1.0 - incomingSemiMajorAxis_ / outgoingSemiMajorAxis_ *
409  ( 1.0 - incomingEccentricity ) ) ) - bendingAngle_;
410 }

413 double EccentricityFindingFunctions::computeFirstDerivativeIncomingEccentricityFunction(
414  const double incomingEccentricity )
415 {
416  const double eccentricitySquareMinusOne_ = incomingEccentricity * incomingEccentricity - 1.0;
417  const double semiMajorAxisRatio_ = incomingSemiMajorAxis_ / outgoingSemiMajorAxis_ ;
418  const double bParameter_ = 1.0 - semiMajorAxisRatio_ * ( 1.0 - incomingEccentricity );
419 
420  return -1.0 / ( incomingEccentricity * std::sqrt( eccentricitySquareMinusOne_ ) ) -
421  semiMajorAxisRatio_ / ( bParameter_ * std::sqrt( bParameter_ * bParameter_ - 1.0 ) );
422 }
