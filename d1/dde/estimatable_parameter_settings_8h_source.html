72 inline std::vector< std::pair< int, int > > getSphericalHarmonicBlockIndices(
73  const int minimumDegree,
74  const int minimumOrder,
75  const int maximumDegree,
76  const int maximumOrder )
77 {
78  std::vector< std::pair< int, int > > blockIndices;
79  for( int i = minimumDegree; i <= maximumDegree; i++ )
80  {
81  for( int j = minimumOrder; ( ( j <= i ) && ( j <= maximumOrder ) ); j++ )
82  {
83  blockIndices.push_back( std::make_pair( i, j ) );
84  }
85  }
86  return blockIndices;
87 }
