104 boost::gregorian::date convertJulianDayToCalendarDate( const double julianDay )
105 {
106  // Declare temporary variables.
107  int L, M, N, P, Q;
108 
109  // Declare date variables.
110  int day, month, year;
111 
112  // Execute algorithm.
113  double shiftedJulianDay = julianDay + 0.5;
114  if( shiftedJulianDay > 2299160 ) // after Oct 4, 1582
115  {
116  L = shiftedJulianDay + 68569;
117  M = (4 * L) / 146097;
118  L = L - ((146097 * M + 3) / 4);
119  N = (4000 * (L + 1)) / 1461001;
120  L = L - ((1461 * N) / 4) + 31;
121  P = (80 * L) / 2447;
122  day = int(L - (2447 * P) / 80);
123  L = P / 11;
124  month = int(P + 2 - 12 * L);
125  year = int(100 * (M - 49) + N + L);
126  }
127  else
128  {
129  P = shiftedJulianDay + 1402;
130  Q = (P - 1) / 1461;
131  L = P - 1461 * Q;
132  M = (L - 1) / 365 - L / 1461;
133  N = L - 365 * M + 30;
134  P = (80 * N) / 2447;
135  day = int(N - (2447 * P) / 80);
136  N = P / 11;
137  month = int( P + 2 - 12 * N );
138  year = int( 4 * Q + M + N - 4716 );
139  if(year <= 0)
140  {
141  --year;
142  }
143  }
144  // catch century/non-400 non-leap years
145  if( year > 1599 && !( year % 100 )
146  && ( year % 400 ) && month == 2 && day == 29 )
147  {
148  month = 3;
149  day = 1;
150  }
151 
152  // Create and return date object
153 
154  return boost::gregorian::date( year, month, static_cast< double >( day ) );
155 }

170 int getDaysInMonth( const int month,
171  const int year )
172 {
173  // Declare number of days per month
174  static const int daysPerMonth[ 12 ] =
175  { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
176  int numberOfDays = 0;
177 
178  // Check input consistency
179  if( month < 1 || month > 12 )
180  {
181  throw std::runtime_error( "Error, month number " + std::to_string( month ) +
182  " does not exist, value must be gretaer than 0 and smaller than 13" );
183  }
184  else
185  {
186  numberOfDays = daysPerMonth[ month - 1 ];
187 
188  // Check for leap year.
189  if( month == 2 && isLeapYear( year ) )
190  {
191  numberOfDays++;
192  }
193  }
194  return numberOfDays;
195 }

220 boost::gregorian::date convertYearAndDaysInYearToDate( const int year, const int daysInYear )
221 {
222  // Go to day 1 at 01-01 convention
223  int daysLeft = daysInYear + 1;
224 
225  // Loop over all months (starting at January until month in which daysinYear is situated is found)
226  int currentMonth = 1;
227  int daysInCurrentMonth;
228  bool isConverged = 0;
229  while( !isConverged )
230  {
231  // Determine days in current month
232  daysInCurrentMonth = getDaysInMonth( currentMonth, year );
233  if( daysInCurrentMonth < daysLeft )
234  {
235  // Subtract days in current month from days left.
236  daysLeft -= daysInCurrentMonth;
237 
238  // Update month and check consistency
239  currentMonth++;
240  if( currentMonth > 12 )
241  {
242  throw std::runtime_error(
243  "Error when converting year and days in year to date, month number has exceeded 12" );
244  }
245  }
246  else
247  {
248  isConverged = 1;
249  }
250  }
251 
252  // Create date and return
253  boost::gregorian::date date( year, currentMonth, daysLeft );
254 
255  if( date.day_of_year( ) != daysInYear + 1 )
256  {
257  throw std::runtime_error(
258  "Error when converting year and days in year to date, inconsistent output" );
259  }
260 
261  return date;
262 }
