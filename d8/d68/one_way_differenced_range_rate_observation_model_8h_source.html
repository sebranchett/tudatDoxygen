85  Eigen::Matrix< ObservationScalarType, 1, 1 > computeIdealObservations(
86  const TimeType time,
87  const LinkEndType linkEndAssociatedWithTime )
88 
89  {
90  ObservationScalarType lightTimeAtStartInterval;
91  ObservationScalarType lightTimeAtEndInterval;
92  TimeType currentIntegrationTime = integrationTimeFunction_( time );
93 
94  if ( linkEndAssociatedWithTime == receiver )
95  {
96  // Calculate light times at the start of the reception interval
97  lightTimeAtStartInterval = arcStartLightTimeCalculator_->calculateLightTime(
98  static_cast< double >( time ) - currentIntegrationTime, 1 );
99 
100  // Calculate light times at the end of the reception interval
101  lightTimeAtEndInterval = arcEndLightTimeCalculator_->calculateLightTime(
102  static_cast< double >( time ) , 1 );
103 
104  }
105  else if ( linkEndAssociatedWithTime == transmitter )
106  {
107 
108  // Calculate light times at the start of the reception interval
109  lightTimeAtEndInterval = arcEndLightTimeCalculator_->calculateLightTime(
110  static_cast< double >( time ) - currentIntegrationTime, 0 );
111 
112  // Calculate light times at the end of the reception interval
113  lightTimeAtStartInterval = arcStartLightTimeCalculator_->calculateLightTime(
114  static_cast< double >( time ), 0 );
115 
116  }
117  else
118  {
119  throw std::runtime_error( "Error in differenced range rate observation model, reference link end not recognized" );
120  }
121 
122  return ( Eigen::Matrix< ObservationScalarType, 1, 1 >( ) << ( lightTimeAtEndInterval - lightTimeAtStartInterval ) *
123  physical_constants::getSpeedOfLight< ObservationScalarType >( ) /
124  static_cast< ObservationScalarType >( currentIntegrationTime ) ).finished( );
125  }

164

167  receiverStateAtArcEnd, transmitterStateAtArcEnd, linkEndTimes[ 3 ] , 1 );
168

170  linkEndTimes[ 2 ] = linkEndTimes[ 3 ] - static_cast< double >( lightTimeAtEndInterval );
171 
172  }
173  else if ( linkEndAssociatedWithTime == transmitter )
174  {
175  //Calculate reception time at ground station at the start and end of the count interval at reception time.
176  linkEndTimes[ 0 ] = static_cast< double >( time ) - currentIntegrationTime;
177  linkEndTimes[ 2 ] = static_cast< double >( time );
178

181  receiverStateAtArcEnd, transmitterStateAtArcEnd, linkEndTimes[ 2 ], 0 );
182 
183  linkEndTimes[ 3 ] = linkEndTimes[ 2 ] + static_cast< double >( lightTimeAtEndInterval );
184 
185  // Calculate light times at the end of the reception interval
186  lightTimeAtStartInterval = arcStartLightTimeCalculator_->calculateLightTimeWithLinkEndsStates(
187  receiverStateAtArcStart, transmitterStateAtArcStart, linkEndTimes[ 0 ], 0 );
188 
189  linkEndTimes[ 1 ] = linkEndTimes[ 0 ] + static_cast< double >( lightTimeAtStartInterval );
190  }
191  else
192  {
193  throw std::runtime_error( "Error in differenced range rate observation model, reference link end not recognized" );
194  }
195

197  linkEndStates[ 1 ] = receiverStateAtArcStart.template cast< double >( );
198  linkEndStates[ 2 ] = transmitterStateAtArcEnd.template cast< double >( );
199  linkEndStates[ 3 ] = receiverStateAtArcEnd.template cast< double >( );

202  physical_constants::getSpeedOfLight< ObservationScalarType >( ) /
