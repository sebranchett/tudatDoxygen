42 ObservationViabilitySettingsList filterObservationViabilitySettings(
43  const ObservationViabilitySettingsList& observationViabilitySettings,
44  const LinkEnds& linkEnds )
45 {
46  ObservationViabilitySettingsList filteredViabilitySettings;
47 
48  // Iterate over all viability settings
49  for( unsigned int i = 0; i < observationViabilitySettings.size( ); i++ )
50  {
51  // Iterate over all link ends
52  for( LinkEnds::const_iterator linkEndIterator = linkEnds.begin( ); linkEndIterator != linkEnds.end( ); linkEndIterator++ )
53  {
54  // Check if present viabilitytt setting is relevant
55  if( linkEndIterator->second == observationViabilitySettings.at( i )->getAssociatedLinkEnd( ) ||
56  ( ( observationViabilitySettings.at( i )->getAssociatedLinkEnd( ).second == "" ) &&
57  ( observationViabilitySettings.at( i )->getAssociatedLinkEnd( ).first == linkEndIterator->second.first ) ) )
58  {
59  filteredViabilitySettings.push_back( observationViabilitySettings.at( i ) );
60  break;
61  }
62  }
63  }
64 
65  return filteredViabilitySettings;

69 std::vector< std::pair< int, int > > getLinkEndIndicesForObservationViability(
70  const LinkEnds& linkEnds, const ObservableType observableType, const LinkEndId linkEndToCheck )
71 {
72  std::vector< std::pair< int, int > > linkEndIndices;
73 
74  switch( observableType )
75  {
76  case one_way_range:
77  if( ( linkEnds.at( transmitter ) == linkEndToCheck ) ||
78  ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) && ( linkEndToCheck.second == "" ) ) )
79  {
80  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
81  }
82  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
83  linkEndToCheck.second == "" ) )
84  {
85  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
86  }
87  else
88  {
89  throw std::runtime_error( "Error, parsed irrelevant 1-way range viability indices" );
90  }
91  break;
92  case one_way_doppler:
93  if( ( linkEnds.at( transmitter ) == linkEndToCheck ) || ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) &&
94  ( linkEndToCheck.second == "" ) ) )
95  {
96  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
97  }
98  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
99  linkEndToCheck.second == "" ) )
100  {
101  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
102  }
103  else
104  {
105  throw std::runtime_error( "Error, parsed irrelevant 1-way doppler viability indices" );
106  }
107  break;
108  case one_way_differenced_range:
109  if( linkEnds.at( transmitter ) == linkEndToCheck || ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) &&
110  linkEndToCheck.second == "" ) )
111  {
112  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
113  linkEndIndices.push_back( std::make_pair( 2, 3 ) );
114  }
115  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
116  linkEndToCheck.second == "" ) )
117  {
118  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
119  linkEndIndices.push_back( std::make_pair( 3, 2 ) );
120  }
121  else
122  {
123  throw std::runtime_error( "Error, parsed irrelevant 1-way differenced range viability indices" );
124  }
125  break;
126  case n_way_range:
127  {
128  std::vector< int > matchingLinkEndIndices = getNWayLinkEndIndicesFromLinkEndId( linkEndToCheck, linkEnds );
129  if( matchingLinkEndIndices.size( ) > 0 )
130  {
131  for( unsigned int i = 0; i < matchingLinkEndIndices.size( ); i++ )
132  {
133  if( matchingLinkEndIndices.at( i ) == 0 )
134  {
135  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
136  }
137  else if( matchingLinkEndIndices.at( i ) == static_cast< int >( linkEnds.size( ) ) - 1 )
138  {
139  linkEndIndices.push_back( std::make_pair( 2 * ( linkEnds.size( ) - 1 ) - 1,
140  2 * ( linkEnds.size( ) - 1 ) - 2 ) );
141  }
142  else
143  {
144  linkEndIndices.push_back(
145  std::make_pair( 2 * matchingLinkEndIndices.at( i ),
146  2 * matchingLinkEndIndices.at( i ) + 1 ) );
147  linkEndIndices.push_back(
148  std::make_pair( 2 * matchingLinkEndIndices.at( i ) - 1,
149  2 * matchingLinkEndIndices.at( i ) - 2 ) );
150  }
151  }
152  }
153  else
154  {
155  throw std::runtime_error( "Error, parsed irrelevant n-way range viability indices" );
156  }
157  break;
158  }
159  case angular_position:
160  if( ( linkEnds.at( transmitter ) == linkEndToCheck ) || ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) &&
161  ( linkEndToCheck.second == "" ) ) )
162  {
163  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
164  }
165  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
166  linkEndToCheck.second == "" ) )
167  {
168  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
169  }
170  else
171  {
172  throw std::runtime_error( "Error, parsed irrelevant angular position viability indices" );
173  }
174  break;
175  case position_observable:
176 
177  throw std::runtime_error( "Error, parsed irrelevant position observable viability indices" );
178  break;
179  default:
180  throw std::runtime_error( "Error, observable type " + boost::lexical_cast< std::string >(
181  observableType ) + " not recognized when making viability link ends" );
182 
183  }
184 
185  return linkEndIndices;
186 }

190  const simulation_setup::NamedBodyMap& bodyMap,
191  const LinkEnds linkEnds,
192  const ObservableType observationType,
193  const boost::shared_ptr< ObservationViabilitySettings > observationViabilitySettings,
194  const std::string& stationName )
195 {
196  if( observationViabilitySettings->observationViabilityType_ != minimum_elevation_angle )
197  {
198  throw std::runtime_error( "Error when making minimum elevation angle calculator, inconsistent input" );
199  }
200 
201  // If specific link end is specified
202  std::string groundStationNameToUse;
203  if( observationViabilitySettings->getAssociatedLinkEnd( ).second != "" )
204  {
205  if( groundStationNameToUse != stationName )
206  {
207  throw std::runtime_error( "Error when making minimum elevation angle calculator, inconsistent station input" );
208  }
209  groundStationNameToUse = observationViabilitySettings->getAssociatedLinkEnd( ).second;
210  }
211  else
212  {
213  groundStationNameToUse = stationName;
214  }
215 
216  if( bodyMap.count( observationViabilitySettings->getAssociatedLinkEnd( ).first ) == 0 )
217  {
218  throw std::runtime_error( "Error when making minimum elevation angle calculator, body " +
219  observationViabilitySettings->getAssociatedLinkEnd( ).first + " not found." );
220  }
221 
222  // Retrieve pointing angles calculator
223  boost::shared_ptr< ground_stations::PointingAnglesCalculator > pointingAngleCalculator =
224  bodyMap.at( observationViabilitySettings->getAssociatedLinkEnd( ).first )->
225  getGroundStation( groundStationNameToUse )->getPointingAnglesCalculator( );
226 
227  // Create check object
228  double minimumElevationAngle = observationViabilitySettings->getDoubleParameter( );
229  return boost::make_shared< MinimumElevationAngleCalculator >(
230  getLinkEndIndicesForObservationViability(
231  linkEnds,observationType, observationViabilitySettings->getAssociatedLinkEnd( ) ),
232  minimumElevationAngle, pointingAngleCalculator );

299 std::vector< boost::shared_ptr< ObservationViabilityCalculator > > createObservationViabilityCalculators(
300  const simulation_setup::NamedBodyMap& bodyMap,
301  const LinkEnds linkEnds,
302  const ObservableType observationType,
303  const std::vector< boost::shared_ptr< ObservationViabilitySettings > >& observationViabilitySettings )
304 {
305  std::vector< boost::shared_ptr< ObservationViabilityCalculator > > linkViabilityCalculators;
306 
307  std::vector< boost::shared_ptr< ObservationViabilitySettings > > relevantObservationViabilitySettings =
308  filterObservationViabilitySettings( observationViabilitySettings, linkEnds );
309 
310  for( unsigned int i = 0; i < relevantObservationViabilitySettings.size( ); i++ )
311  {
312  switch( relevantObservationViabilitySettings.at( i )->observationViabilityType_ )
313  {
314  case minimum_elevation_angle:
315  {
316  // Create list of ground stations for which elevation angle check is to be made.
317  std::vector< std::string > listOfGroundStations;
318  for( LinkEnds::const_iterator linkEndIterator = linkEnds.begin( );
319  linkEndIterator != linkEnds.end( ); linkEndIterator++ )
320  {
321  if( linkEndIterator->second.first == relevantObservationViabilitySettings.at( i )->getAssociatedLinkEnd( ).first )
322  {
323  if( std::find( listOfGroundStations.begin( ), listOfGroundStations.end( ), linkEndIterator->second.second ) ==
324  listOfGroundStations.end( ) )
325  {
326  listOfGroundStations.push_back( linkEndIterator->second.second );
327  }
328  }
329  }
330 
331  // Create elevation angle check separately for eah ground station: check requires different pointing angles calculator
332  for( unsigned int j = 0; j < listOfGroundStations.size( ); j++ )
333  {
334  linkViabilityCalculators.push_back(
335  createMinimumElevationAngleCalculator(
336  bodyMap, linkEnds, observationType, relevantObservationViabilitySettings.at( i ),
337  listOfGroundStations.at( j ) ) );
338  }
339  break;
340  }
341  case body_avoidance_angle:
342 
343  linkViabilityCalculators.push_back(
344  createBodyAvoidanceAngleCalculator(
345  bodyMap, linkEnds, observationType, relevantObservationViabilitySettings.at( i ) ) );
346  break;
347  case body_occultation:
348 
349  linkViabilityCalculators.push_back(
350  createOccultationCalculator(
351  bodyMap, linkEnds, observationType, relevantObservationViabilitySettings.at( i ) ) );
352  break;
353  default:
354  throw std::runtime_error(
355  "Error when making observation viability calculator, type not recognized " +
356  boost::lexical_cast< std::string >(
357  relevantObservationViabilitySettings.at( i )->observationViabilityType_ ) );
358  }
359 
360  }
361 
362  return linkViabilityCalculators;
363 }
