95  Eigen::MatrixXd wrtParameter(
96  const std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
97  {
98  // Initialize partial
99  Eigen::MatrixXd partial = Eigen::MatrixXd( accelerationSize_, 1 );
100 
101  // Get partial computation function.
102  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunction =
103  getParameterPartialFunction( parameter );
104 
105  // If parameter dependency exists, compute it, otherwise set partial to zero.
106  if( partialFunction.second > 0 )
107  {
108  partialFunction.first( partial );
109  }
110  else
111  {
112  partial = Eigen::MatrixXd::Zero(
113  propagators::getSingleIntegrationSize( integratedStateType_ ) /
114  propagators::getSingleIntegrationDifferentialEquationOrder( integratedStateType_ ), 1 );
115  }
116 
117  return partial;

235  void getCurrentParameterPartial(
236  const std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter,
237  Eigen::Block< Eigen::MatrixXd > partialMatrix )
238  {
239  // Check if dependecy is computed
240  if( currentDoubleParameterPartials_.count( parameter ) == 0 )
241  {
242  // Check if dependecy exists at all
243  if( parameterDoublePartialFunctions_.count( parameter ) == 0 )
244  {
245  throw std::runtime_error(
246  "Parameter of type " +
247  std::to_string( parameter->getParameterName( ).first ) + ", " +
248  std::string( parameter->getParameterName( ).second.first ) + ", " +
249  std::string( parameter->getParameterName( ).second.second ) + ", " +
250  " not found in list of existing partials" );
251  }
252  else
253  {
254  std::cerr << "Warning, double partial should already be calculated" << std::endl;
255  parameterDoublePartialFunctions_.at( parameter )( currentDoubleParameterPartials_[ parameter ] );
256  }
257  }
258  partialMatrix = currentDoubleParameterPartials_[ parameter ];
259  }

332  virtual int setParameterPartialUpdateFunction(
333  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
334  {
335  // Get partial function.
336  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > parameterPartialFunction =
337  getParameterPartialFunction( parameter );
338 
339  // If partial function found, add function to list of computations to be performed when calling
340  // updateParameterPartials.
341  if( parameterPartialFunction.second > 0 && parameterDoublePartialFunctions_.count( parameter ) == 0 )
342  {
343  if( parameterDoublePartialFunctions_.count( parameter ) == 0 )
344  {
345  parameterDoublePartialFunctions_[ parameter ] = parameterPartialFunction.first;
346  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
347  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
348  }
349  }
350 
351  return parameterPartialFunction.second;

389  {
