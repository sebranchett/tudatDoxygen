66 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< GravityFieldVariationSettings >& variationSettings )
67 {
68  using namespace gravitation;
69  using namespace interpolators;
70  using namespace json_interface;
71  using K = Keys::Body::GravityFieldVariation;
72 
73  // Body deformation type
74  const BodyDeformationTypes bodyDeformationType =
75  getValue< BodyDeformationTypes >( jsonObject, K::bodyDeformationType );
76 
77  switch ( bodyDeformationType ) {
78  case basic_solid_body:
79  {
80  BasicSolidBodyGravityFieldVariationSettings defaults( { }, { }, TUDAT_NAN );
81  variationSettings = std::make_shared< BasicSolidBodyGravityFieldVariationSettings >(
82  getValue< std::vector< std::string > >( jsonObject, K::deformingBodies ),
83  getValue< std::vector< std::vector< std::complex< double > > > >( jsonObject, K::loveNumbers ),
84  getValue< double >( jsonObject, K::referenceRadius ),
85  getValue( jsonObject, K::modelInterpolation, defaults.getInterpolatorSettings( ) ) );
86  return;
87  }
88  case tabulated_variation:
89  variationSettings = std::make_shared< TabulatedGravityFieldVariationSettings >(
90  getValue< std::map< double, Eigen::MatrixXd > >( jsonObject, K::cosineCoefficientCorrections ),
91  getValue< std::map< double, Eigen::MatrixXd > >( jsonObject, K::sineCoefficientCorrections ),
92  getValue< int >( jsonObject, K::minimumDegree ),
93  getValue< int >( jsonObject, K::minimumOrder ),
94  getValue< std::shared_ptr< InterpolatorSettings > >( jsonObject, K::interpolator ) );
95  return;
96  default:
97  handleUnimplementedEnumValue( bodyDeformationType, bodyDeformationTypes, unsupportedBodyDeformationTypes );
