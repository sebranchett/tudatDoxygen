58  template< typename ParameterType >
59  VariationalEquations(
60  const std::map< IntegratedStateType, orbit_determination::StateDerivativePartialsMap >
61  stateDerivativePartialList,
62  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
63  const std::map< IntegratedStateType, int >& stateTypeStartIndices ):
64  stateDerivativePartialList_( stateDerivativePartialList ), stateTypeStartIndices_( stateTypeStartIndices )
65  {
66  dynamicalStatesToEstimate_ =
67  estimatable_parameters::getListOfInitialDynamicalStateParametersEstimate< ParameterType >(
68  parametersToEstimate );
69 
70  // Get size of dynamical state to estimate
71  numberOfParameterValues_ = estimatable_parameters::getSingleArcParameterSetSize( parametersToEstimate );
72  totalDynamicalStateSize_ = 0;
73  for( std::map< IntegratedStateType, orbit_determination::StateDerivativePartialsMap >::iterator
74  partialTypeIterator = stateDerivativePartialList_.begin( );
75  partialTypeIterator != stateDerivativePartialList_.end( ); partialTypeIterator++ )
76  {
77 
78  if( dynamicalStatesToEstimate_.count( partialTypeIterator->first ) == 0 )
79  {
80  std::string errorMessage = "Error when making variational equations object, found no state to estimate of type " +
81  boost::lexical_cast< std::string >( partialTypeIterator->first );
82  throw std::runtime_error( errorMessage );
83  }
84  else if( dynamicalStatesToEstimate_.at( partialTypeIterator->first ).size( ) !=
85  partialTypeIterator->second.size( ) )
86  {
87  throw std::runtime_error( "Error when making variational equations object, input partial list size is inconsistent" );
88  }
89 
90  totalDynamicalStateSize_ +=
91  getSingleIntegrationSize( partialTypeIterator->first ) * partialTypeIterator->second.size( );
92  }
93 
94  // Initialize matrices.
95  variationalMatrix_ = Eigen::MatrixXd::Zero( totalDynamicalStateSize_, totalDynamicalStateSize_ );
96  variationalParameterMatrix_ =
97  Eigen::MatrixXd::Zero( totalDynamicalStateSize_, numberOfParameterValues_ - totalDynamicalStateSize_ );
98 
99  // Set parameter partial functions.
100  setStatePartialFunctionList( );
101  setTranslationalStatePartialFrameScalingFunctions( parametersToEstimate );
102  setParameterPartialFunctionList( parametersToEstimate );
103  }

140  template< typename StateScalarType >
141  void getParameterPartialMatrix(
142  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > currentMatrixDerivative )
143  {
144  // Initialize matrix to zeros
145  variationalParameterMatrix_.setZero( );
146 
147 
148  // Iterate over all bodies undergoing accelerations for which initial condition is to be estimated.
149  for( std::map< IntegratedStateType, std::vector< std::multimap< std::pair< int, int >,
150  boost::function< void( Eigen::Block< Eigen::MatrixXd > ) > > > >::iterator typeIterator =
151  parameterPartialList_.begin( ); typeIterator != parameterPartialList_.end( ); typeIterator++ )
152  {
153  int startIndex = stateTypeStartIndices_.at( typeIterator->first );
154  int currentStateSize = getSingleIntegrationSize( typeIterator->first );
155  int entriesToSkipPerEntry = currentStateSize -
156  currentStateSize / getSingleIntegrationDifferentialEquationOrder( typeIterator->first );
157 
158  // Iterate over all bodies being estimated.
159  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
160  {
161  // Iterate over all parameter partial functions determined by setParameterPartialFunctionList( )
162  for( functionIterator = typeIterator->second[ i ].begin( );
163  functionIterator != typeIterator->second[ i ].end( );
164  functionIterator++ )
165  {
166  functionIterator->second(
167  variationalParameterMatrix_.block(
168  startIndex + entriesToSkipPerEntry + currentStateSize * i,
169  functionIterator->first.first - totalDynamicalStateSize_,
170  currentStateSize - entriesToSkipPerEntry,
171  functionIterator->first.second ) );
172  }
173  }
174 
175  }
176 
177  currentMatrixDerivative.block( 0, totalDynamicalStateSize_, totalDynamicalStateSize_,
178  numberOfParameterValues_ - totalDynamicalStateSize_ ) +=
179  variationalParameterMatrix_.template cast< StateScalarType >( );
180 
181  }

304  template< typename ParameterType >
305  void setParameterPartialFunctionList(
306  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > >
307  parametersToEstimate )
308  {
309  // Get double parameters.
310  std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >
311  doubleParametersToEstimate =
312  parametersToEstimate->getDoubleParameters( );
313 
314  // Get vector parameters.
315  std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
316  vectorParametersToEstimate =
317  parametersToEstimate->getVectorParameters( );
318 
319  int totalParameterVectorIndicesToSubtract = parametersToEstimate->getInitialDynamicalStateParameterSize( ) -
320  estimatable_parameters::getSingleArcInitialDynamicalStateParameterSetSize( parametersToEstimate );
321 
322  for( std::map< propagators::IntegratedStateType,
323  orbit_determination::StateDerivativePartialsMap >::iterator
324  stateDerivativeTypeIterator = stateDerivativePartialList_.begin( );
325  stateDerivativeTypeIterator != stateDerivativePartialList_.end( );
326  stateDerivativeTypeIterator++ )
327  {
328 
329  // Initialize vector of lists to correct size.
330  parameterPartialList_[ stateDerivativeTypeIterator->first ].resize(
331  stateDerivativeTypeIterator->second.size( ) );
332 
333  // Iterate over all bodies of which initial position is being estimated.
334  for( unsigned int i = 0; i < stateDerivativeTypeIterator->second.size( ); i++ )
335  {
336  // Initialize list of parameter partial functions for single body.
337  std::multimap< std::pair< int, int >, boost::function< void( Eigen::Block< Eigen::MatrixXd > ) > >
338  functionListOfBody;
339 
340  // Iterate over all accelerations due to this body on current body.
341  for( unsigned int j = 0; j < stateDerivativeTypeIterator->second.at( i ).size( ); j++ )
342  {
343  addParameterPartialToList< double >(
344  doubleParametersToEstimate, stateDerivativeTypeIterator->second.at( i ).at( j ),
345  functionListOfBody, totalParameterVectorIndicesToSubtract );
346  addParameterPartialToList< Eigen::VectorXd >(
347  vectorParametersToEstimate, stateDerivativeTypeIterator->second.at( i ).at( j ),
348  functionListOfBody, totalParameterVectorIndicesToSubtract );
349  }
350 
351  // Add generated parameter partial list of current body.
352  parameterPartialList_[ stateDerivativeTypeIterator->first ][ i ] = functionListOfBody;
353  }
354  }

364  template< typename ParameterType >
365  void setTranslationalStatePartialFrameScalingFunctions(
366  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > >
367  parametersToEstimate )
368  {
369  std::vector< boost::shared_ptr< estimatable_parameters::EstimatableParameter<
370  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
371  parametersToEstimate->getEstimatedInitialStateParameters( );
372 
373  std::vector< std::string > propagatedBodies;
374  std::vector< std::string > centralBodies;
375 
376  // Retrieve propagated bodies and central bodies of estimation.
377  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
378  {
379  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state )
380  {
381  propagatedBodies.push_back(
382  initialDynamicalParameters.at( i )->getParameterName( ).second.first );
383  centralBodies.push_back( boost::dynamic_pointer_cast
384  < estimatable_parameters::InitialTranslationalStateParameter< ParameterType > >(
385  initialDynamicalParameters.at( i ) )->getCentralBody( ) );
386  }
387  else if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state )
388  {
389  propagatedBodies.push_back(
390  initialDynamicalParameters.at( i )->getParameterName( ).second.first );
391  centralBodies.push_back( boost::dynamic_pointer_cast< estimatable_parameters::ArcWiseInitialTranslationalStateParameter< ParameterType > >(
392  initialDynamicalParameters.at( i ) )->getCentralBody( ) );
393  }
394  else
395  {
396  throw std::runtime_error( "Error when settinf up variational equations, did not recognize initial state type" );
397  }
398  }
399 
400  // Get order in which ephemerides were to be updated.
401  std::vector< std::string > updateOrder = determineEphemerisUpdateorder(
402  propagatedBodies, centralBodies, centralBodies );
403 
404  // Iterate over central bodies and propagated bodies and check for dependencies
405  for( int i = updateOrder.size( ) - 1; i >= 0 ; i-- )
406  {
407  int currentBodyIndex = std::distance(
408  propagatedBodies.begin( ),
409  std::find( propagatedBodies.begin( ), propagatedBodies.end( ), updateOrder.at( i ) ) );
410  for( unsigned int j = 0; j < propagatedBodies.size( ); j++ )
411  {
412  if( centralBodies.at( currentBodyIndex ) == propagatedBodies.at( j ) )
413  {
414 
415  statePartialAdditionIndices_.push_back(
416  std::make_pair( stateTypeStartIndices_[ propagators::transational_state ] +
417  currentBodyIndex * propagators::getSingleIntegrationSize( propagators::transational_state ),
418  stateTypeStartIndices_[ propagators::transational_state ] +
419  j * propagators::getSingleIntegrationSize( propagators::transational_state ) ) );
420  }
421  }
422  }
423  }
