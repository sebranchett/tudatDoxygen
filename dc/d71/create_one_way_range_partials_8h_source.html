119 template< typename ParameterType >
120 std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > createOneWayRangePartials(
121  const observation_models::LinkEnds oneWayRangeLinkEnds,
122  const simulation_setup::NamedBodyMap& bodyMap,
123  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
124  const std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >& lightTimeCorrections =
125  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >( ),
126  const bool useBiasPartials = true )
127 {
128 
129  std::vector< std::shared_ptr< observation_partials::LightTimeCorrectionPartial > > lightTimeCorrectionPartialObjects;
130  if( lightTimeCorrections.size( ) > 0 )
131  {
132  lightTimeCorrectionPartialObjects = observation_partials::createLightTimeCorrectionPartials( lightTimeCorrections );
133  }
134 
135  // Create scaling object, to be used for all one-way range partials in current link end.
136  std::shared_ptr< OneWayRangeScaling > oneWayRangeScaling = std::make_shared< OneWayRangeScaling >( );
137 
138  // Initialize vector index variables.
139  int currentIndex = 0;
140  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
141 
142  SingleLinkObservationPartialList rangePartials;
143 
144  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
145  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
146  parametersToEstimate->getEstimatedInitialStateParameters( );
147 
148  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
149  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
150  {
151  std::string acceleratedBody;
152  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state )
153  {
154  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
155  }
156  else if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state )
157  {
158  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
159  }
160  else
161  {
162  throw std::runtime_error( "Error when making one way range partials, could not identify parameter " +
163  std::to_string(
164  initialDynamicalParameters.at( i )->getParameterName( ).first ) );
165  }
166 
167 
168  // Create position one-way range partial for current body
169  std::shared_ptr< OneWayRangePartial > currentRangePartial = createOneWayRangePartialWrtBodyPosition(
170  oneWayRangeLinkEnds, bodyMap, acceleratedBody, oneWayRangeScaling, lightTimeCorrectionPartialObjects );
171 
172  // Check if partial is non-nullptr (i.e. whether dependency exists between current range and current body)
173  if( currentRangePartial != nullptr )
174  {
175  // Add partial to the list.
176  currentPair = std::pair< int, int >( currentIndex, 6 );
177  rangePartials[ currentPair ] = currentRangePartial;
178  }
179 
180  // Increment current index by size of body initial state (6).
181  currentIndex += 6;
182  }
183 
184  // Iterate over all double parameters that are to be estimated.
185  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParametersToEstimate =
186  parametersToEstimate->getDoubleParameters( );
187  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
188  parameterIterator =
189  doubleParametersToEstimate.begin( ); parameterIterator != doubleParametersToEstimate.end( ); parameterIterator++ )
190  {
191  // Create position one-way range partial for current parameter
192  std::shared_ptr< ObservationPartial< 1 > > currentRangePartial = createOneWayRangePartialWrtParameter(
193  oneWayRangeLinkEnds, bodyMap, parameterIterator->second,
194  oneWayRangeScaling, lightTimeCorrectionPartialObjects );
195 
196  // Check if partial is non-nullptr (i.e. whether dependency exists between current range and current parameter)
197  if( currentRangePartial != nullptr )
198  {
199  // Add partial to the list.
200  currentPair = std::pair< int, int >( parameterIterator->first, 1 );
201  rangePartials[ currentPair ] = currentRangePartial;
202  }
203  }
204 
205  // Iterate over all vector parameters that are to be estimated.
206  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
207  vectorParametersToEstimate =
208  parametersToEstimate->getVectorParameters( );
209  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
210  parameterIterator =
211  vectorParametersToEstimate.begin( ); parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
212  {
213  // Create position one-way range partial for current parameter
214  std::shared_ptr< ObservationPartial< 1 > > currentRangePartial;
215 
216  if( !isParameterObservationLinkProperty( parameterIterator->second->getParameterName( ).first ) )
217  {
218  currentRangePartial = createOneWayRangePartialWrtParameter(
219  oneWayRangeLinkEnds, bodyMap, parameterIterator->second, oneWayRangeScaling,
220  lightTimeCorrectionPartialObjects );
221  }
222  else
223  {
224  currentRangePartial = createObservationPartialWrtLinkProperty< 1 >(
225  oneWayRangeLinkEnds, observation_models::one_way_range, parameterIterator->second, useBiasPartials );
226  }
227 
228  // Check if partial is non-nullptr (i.e. whether dependency exists between current range and current parameter)
229  if( currentRangePartial != nullptr )
230  {
231  // Add partial to the list.
232  currentPair = std::pair< int, int >( parameterIterator->first,
233  parameterIterator->second->getParameterSize( ) );
234  rangePartials[ currentPair ] = currentRangePartial;
235  }
236  }
237 
238  // Return complete set of partials and scaling object.
239  return std::make_pair( rangePartials, oneWayRangeScaling );
240 }

258 template< typename ParameterType >
259 std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
260 std::shared_ptr< PositionPartialScaling > > > createOneWayRangePartials(
261  const std::vector< observation_models::LinkEnds >& linkEnds,
262  const simulation_setup::NamedBodyMap& bodyMap,
263  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
264  const std::map< observation_models::LinkEnds,
265  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >& lightTimeCorrections =
266  std::map< observation_models::LinkEnds,
267  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >( ),
268  const bool useBiasPartials = true )
269 {
270  // Declare return list.
271  std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
272  std::shared_ptr< PositionPartialScaling > > > rangePartials;
273 
274  // Iterate over all link ends.
275  for( unsigned int i = 0; i < linkEnds.size( ); i++ )
276  {
277  // Check if required link end types are present
278  if( ( linkEnds[ i ].count( observation_models::receiver ) == 0 ) ||
279  ( linkEnds[ i ].count( observation_models::transmitter ) == 0 ) )
280  {
281  throw std::runtime_error( "Error when making 1-way range partials, did not find both receiver and transmitter in link ends" );
282 
283  }
284 
285  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > singleLinkLightTimeCorrections;
286  if( lightTimeCorrections.count( linkEnds.at( i ) ) > 0 )
287  {
288  if( lightTimeCorrections.at( linkEnds.at( i ) ).size( ) > 1 )
289  {
290  std::cerr << "Error when making one way range partials, light time corrections for "
291  << lightTimeCorrections.at( linkEnds.at( i ) ).size( ) << " links found" << std::endl;
292  }
293  else if( lightTimeCorrections.at( linkEnds.at( i ) ).size( ) == 1 )
294  {
295  singleLinkLightTimeCorrections = lightTimeCorrections.at( linkEnds.at( i ) ).at( 0 );
296  }
297  }
298 
299  // Create range partials for current link ends
300  rangePartials[ linkEnds[ i ] ] = createOneWayRangePartials< ParameterType >(
301  linkEnds[ i ], bodyMap, parametersToEstimate, singleLinkLightTimeCorrections, useBiasPartials );
302  }
303 
304  // Return complete set of link ends.
305  return rangePartials;
306 }
