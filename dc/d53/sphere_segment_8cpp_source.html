63 Eigen::VectorXd SphereSegment::getSurfaceDerivative( const double azimuthAngle,
64  const double zenithAngle,
65  const int powerOfAzimuthAngleDerivative,
66  const int powerOfZenithAngleDerivative )
67 {
68  // Declare and set size of derivative vector.
69  Eigen::VectorXd derivative_ = Eigen::VectorXd( 3 );
70 
71  // Go through the different possibilities for the values of the power
72  // of the derivative.
73  if ( powerOfAzimuthAngleDerivative < 0 || powerOfZenithAngleDerivative < 0 )
74  {
75  derivative_( 0 ) = 0.0;
76  derivative_( 1 ) = 0.0;
77  derivative_( 2 ) = 0.0;
78 
79  throw std::runtime_error( "No negative power of derivatives allowed in sphere segment." );
80  }
81 
82  // When requesting the zeroth derivative with respect to the two
83  // independent variables, the surface point is returned. Note that this
84  // does include the offset.
85  else if ( powerOfAzimuthAngleDerivative == 0 && powerOfZenithAngleDerivative == 0 )
86  {
87  derivative_ = getSurfacePoint( azimuthAngle, zenithAngle );
88  }
89 
90  // The contributions of the two parameterizing variables are determined
91  // and then multiplied component-wise.
92  else
93  {
94  // Declare and set sizes contribution vectors.
95  Eigen::VectorXd derivative1Contribution_ = Eigen::VectorXd( 3 );
96  Eigen::VectorXd derivative2Contribution_ = Eigen::VectorXd( 3 );
97 
98  // Since this derivative is "cyclical", as it is
99  // only dependent on sines and cosines, only the "modulo 4"th
100  // derivatives need to be determined. Derivatives are determined
101  // from the form of the spherical coordinates, see
102  // basic_mathematics::coordinateConversions::convertSphericalToCartesian
103  switch( powerOfAzimuthAngleDerivative % 4 )
104  {
105  case( 0 ):
106 
107  derivative1Contribution_( 0 ) = cos( azimuthAngle );
108  derivative1Contribution_( 1 ) = sin( azimuthAngle );
109  derivative1Contribution_( 2 ) = 0.0;
110  break;
111 
112  case( 1 ):
113 
114  derivative1Contribution_( 0 ) = -sin( azimuthAngle );
115  derivative1Contribution_( 1 ) = cos( azimuthAngle );
116  derivative1Contribution_( 2 ) = 0.0;
117  break;
118 
119  case( 2 ):
120 
121  derivative1Contribution_( 0 ) = -cos( azimuthAngle );
122  derivative1Contribution_( 1 ) = -sin( azimuthAngle );
123  derivative1Contribution_( 2 ) = 0.0;
124  break;
125 
126  case( 3 ):
127 
128  derivative1Contribution_( 0 ) = sin( azimuthAngle );
129  derivative1Contribution_( 1 ) = -cos( azimuthAngle );
130  derivative1Contribution_( 2 ) = 0.0;
131  break;
132 
133  default:
134 
135  throw std::runtime_error( " Bad value for powerOfAzimuthAngleDerivative ( mod 4 ) of value is not 0, 1, 2 or 3 in sphere segment." );
136 
137  }
138 
139  // This derivative is "cyclical" in the same manner as the derivative
140  // with respect to the 1st independent variable.
141  switch( powerOfZenithAngleDerivative %4 )
142  {
143  case( 0 ):
144 
145  derivative2Contribution_( 0 ) = sin( zenithAngle );
146  derivative2Contribution_( 1 ) = sin( zenithAngle );
147  derivative2Contribution_( 2 ) = cos( zenithAngle );
148  break;
149 
150  case( 1 ):
151 
152  derivative2Contribution_( 0 ) = cos( zenithAngle );
153  derivative2Contribution_( 1 ) = cos( zenithAngle );
154  derivative2Contribution_( 2 ) = -sin( zenithAngle );
155  break;
156 
157  case( 2 ):
158 
159  derivative2Contribution_( 0 ) = -sin( zenithAngle );
160  derivative2Contribution_( 1 ) = -sin( zenithAngle );
161  derivative2Contribution_( 2 ) = -cos( zenithAngle );
162  break;
163 
164  case( 3 ):
165 
166  derivative2Contribution_( 0 ) = -cos( zenithAngle );
167  derivative2Contribution_( 1 ) = -cos( zenithAngle );
168  derivative2Contribution_( 2 ) = sin( zenithAngle );
169  break;
170 
171  default:
172 
173  throw std::runtime_error( " Bad value for powerOfZenithAngleDerivative ( mod 4 ) of value is not 0, 1, 2 or 3 in sphere segment." );
174  }
175 
176  // Construct the full derivative.
177  derivative_( 0 ) = derivative1Contribution_( 0 ) * derivative2Contribution_( 0 );
178  derivative_( 1 ) = derivative1Contribution_( 1 ) * derivative2Contribution_( 1 );
179  derivative_( 2 ) = derivative1Contribution_( 2 ) * derivative2Contribution_( 2 );
180 
181  // Scale vector by radius.
182  derivative_ = derivative_ * radius_;
183 
184  // Rotate derivatives to take into account rotation of sphere.
185  derivative_ = rotationMatrix_ * scalingMatrix_ * derivative_;
186  }
187 
188  return derivative_;
189 }
