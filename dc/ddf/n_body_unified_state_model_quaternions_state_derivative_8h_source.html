139  void calculateSystemStateDerivative(
140  const TimeType time, const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
141  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
142  {
143  // Get total inertial accelerations acting on bodies
144  Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > currentAccelerationInIntertialFrame;
145  currentAccelerationInIntertialFrame.resizeLike( currentCartesianLocalSolution_ );
146  this->sumStateDerivativeContributions( stateOfSystemToBeIntegrated, currentAccelerationInIntertialFrame, false );
147 
148  // Compute RSW accelerations for each body, and evaluate equations of motion for USM7 elements.
149  stateDerivative.setZero( );
150  Eigen::Vector3d currentAccelerationInRswFrame;
151  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
152  {
153  currentAccelerationInRswFrame = reference_frames::getInertialToRswSatelliteCenteredFrameRotationMatrix(
154  currentCartesianLocalSolution_.segment( i * 6, 6 ).template cast< double >( ) ) *
155  currentAccelerationInIntertialFrame.block( i * 6 + 3, 0, 3, 1 ).template cast< double >( );
156 
157  stateDerivative.block( i * 7, 0, 7, 1 ) = computeStateDerivativeForUnifiedStateModelQuaternions(
158  stateOfSystemToBeIntegrated.block( i * 7, 0, 7, 1 ).template cast< double >( ), currentAccelerationInRswFrame,
159  centralBodyGravitationalParameters_.at( i )( ) ).template cast< StateScalarType >( );
160  }

244  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
245  {
246  // Loop over each body
247  const double tolerance = 20.0 * std::numeric_limits< double >::epsilon( );
248  Eigen::Matrix< StateScalarType, 4, 1 > quaternionsVector;
249  StateScalarType quaternionsMagnitude;
250  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
251  {
252  // Normalize quaternions
253  quaternionsVector = unprocessedState.block( i * 7 + 3, 0, 4, 1 );
254  quaternionsMagnitude = quaternionsVector.norm( );
255  if ( std::fabs( quaternionsMagnitude - 1.0 ) >= tolerance )
256  {
257  // Normalize
258  quaternionsVector /= quaternionsMagnitude;
259 
260  // Replace old quaternions with normalized quaternions
261  unprocessedState.segment( i * 7 + 3, 4 ) = quaternionsVector;
262  }
263  }
264  }
