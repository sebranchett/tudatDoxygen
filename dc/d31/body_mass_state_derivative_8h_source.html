214  double getTotalMassRateForBody( const std::string bodyName )
215  {
216  // Check if body is propagated.
217  double totalMassRate = 0.0;
218 
219  if( std::find( bodiesToIntegrate_.begin( ),
220  bodiesToIntegrate_.end( ),
221  bodyName ) == bodiesToIntegrate_.end( ) )
222  {
223  std::string errorMessage = "Error when getting total mass rate for body " + bodyName +
224  ", no such acceleration is found";
225  throw std::runtime_error( errorMessage );
226  }
227  else
228  {
229  if( massRateModels_.count( bodyName ) != 0 )
230  {
231  for( unsigned int i = 0; i < massRateModels_.at( bodyName ).size( ); i++ )
232  {
233  totalMassRate += massRateModels_.at( bodyName ).at ( i )->getMassRate( );
234  }
235  }
236  }
