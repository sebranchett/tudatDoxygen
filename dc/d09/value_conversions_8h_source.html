44 template< template< typename ... > class MapType, typename KeyType, typename ValueType >
45 MapType< KeyType, ValueType > mapFromJson( const nlohmann::json& jsonObject )
46 {
47  MapType< KeyType, ValueType > map;
48  for ( nlohmann::json::const_iterator it = jsonObject.begin( ); it != jsonObject.end( ); ++it )
49  {
50  const std::string key = it.key( );
51  if ( ! contains( SpecialKeys::all, key ) )
52  {
53  map[ boost::lexical_cast< KeyType >( key ) ] = getValue< ValueType >( jsonObject, key );
54  }
55  }
56  return map;
57 }

177 template< typename ScalarType, int rows, int cols >
178 void from_json( const nlohmann::json& jsonObject, Matrix< ScalarType, rows, cols >& matrix )
179 {
180  nlohmann::json jsonArray;
181  if ( jsonObject.is_array( ) )
182  {
183  if ( jsonObject.empty( ) )
184  {
185  return;
186  }
187  jsonArray = jsonObject;
188  }
189  else if ( jsonObject.is_number( ) )
190  {
191  jsonArray.push_back( jsonObject );
192  }
193  else
194  {
195  throw nlohmann::detail::type_error::create( 0, "" );
196  }
197 
198  nlohmann::json jsonArrayOfArrays;
199  if ( jsonArray.front( ).is_array( ) ) // provided matrix
200  {
201  jsonArrayOfArrays = jsonArray;
202  }
203  else // provided vector
204  {
205  if ( rows == 1 ) // expected row vector
206  {
207  jsonArrayOfArrays.push_back( jsonArray );
208  }
209  else if ( cols == 1 ) // expected column vector
210  {
211  for ( unsigned int i = 0; i < jsonArray.size( ); ++i )
212  {
213  jsonArrayOfArrays.push_back( { jsonArray.at( i ) } );
214  }
215  }
216  else // expected matrix
217  {
218  std::cerr << "Expected a matrix, received a vector." << std::endl;
219  throw nlohmann::detail::type_error::create( 0, "" );
220  }
221  }
222 
223  const unsigned int providedRows = jsonArrayOfArrays.size( );
224  const unsigned int providedCols = jsonArrayOfArrays.front( ).size( );
225  if ( ( rows >= 0 && int( providedRows ) != rows ) || ( cols >= 0 && int( providedCols ) != cols ) )
226  {
227  std::cerr << "Expected matrix of size " << rows << "x" << cols
228  << ", received matrix of size " << providedRows << "x" << providedCols << "." << std::endl;
229  throw nlohmann::detail::type_error::create( 0, "" );
230  }
231 
232  matrix.resize( providedRows, providedCols );
233  for ( unsigned int r = 0; r < providedRows; ++r )
234  {
235  if ( jsonArrayOfArrays.at( r ).size( ) != providedCols )
236  {
237  std::cerr << "Unconsistent matrix size: some rows have different number of columns." << std::endl;
238  throw nlohmann::detail::type_error::create( 0, "" );
239  }
240  for ( unsigned int c = 0; c < providedCols; ++c )
241  {
242  matrix( r, c ) = jsonArrayOfArrays.at( r ).at( c );
243  }
244  }
245 }
