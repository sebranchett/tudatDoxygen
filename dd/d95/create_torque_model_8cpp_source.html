21 std::shared_ptr< aerodynamics::AerodynamicTorque > createAerodynamicTorqueModel(
22  const std::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
23  const std::shared_ptr< simulation_setup::Body > bodyExertingTorque,
24  const std::string& nameOfBodyUndergoingTorque,
25  const std::string& nameOfBodyExertingTorque )
26 {
27  // Check existence of required environment models
28  if( bodyUndergoingTorque->getAerodynamicCoefficientInterface( ) == nullptr )
29  {
30  throw std::runtime_error( "Error when making aerodynamic torque, body " +
31  nameOfBodyUndergoingTorque +
32  "has no aerodynamic coefficients." );
33  }
34 
35  if( bodyExertingTorque->getAtmosphereModel( ) == nullptr )
36  {
37  throw std::runtime_error( "Error when making aerodynamic torque, central body " +
38  nameOfBodyExertingTorque + " has no atmosphere model.");
39  }
40 
41  if( bodyExertingTorque->getShapeModel( ) == nullptr )
42  {
43  throw std::runtime_error( "Error when making aerodynamic torque, central body " +
44  nameOfBodyExertingTorque + " has no shape model." );
45  }
46 
47  // Retrieve flight conditions; create object if not yet extant.
48  std::shared_ptr< aerodynamics::AtmosphericFlightConditions > bodyFlightConditions =
49  std::dynamic_pointer_cast< aerodynamics::AtmosphericFlightConditions >(
50  bodyUndergoingTorque->getFlightConditions( ) );
51 
52  if( bodyFlightConditions == nullptr && bodyUndergoingTorque->getFlightConditions( ) == nullptr )
53  {
54  bodyFlightConditions = createAtmosphericFlightConditions( bodyUndergoingTorque,
55  bodyExertingTorque,
56  nameOfBodyUndergoingTorque,
57  nameOfBodyExertingTorque ) ;
58  bodyUndergoingTorque->setFlightConditions(
59  bodyFlightConditions );
60  }
61  else if( bodyFlightConditions == nullptr && bodyUndergoingTorque->getFlightConditions( ) != nullptr )
62  {
63  throw std::runtime_error( "Error when making aerodynamic torque, found flight conditions that are not atmospheric." );
64  }
65 
66  // Retrieve frame in which aerodynamic coefficients are defined.
67  std::shared_ptr< aerodynamics::AerodynamicCoefficientInterface > aerodynamicCoefficients =
68  bodyUndergoingTorque->getAerodynamicCoefficientInterface( );
69  reference_frames::AerodynamicsReferenceFrames torqueFrame;
70  if( aerodynamicCoefficients->getAreCoefficientsInAerodynamicFrame( ) )
71  {
72  torqueFrame = reference_frames::aerodynamic_frame;
73  }
74  else
75  {
76  torqueFrame = reference_frames::body_frame;
77  }
78 
79  // Create function to transform from frame of aerodynamic coefficienrs to that of propagation.
80  std::function< Eigen::Vector3d( const Eigen::Vector3d& ) > toPropagationFrameTransformation;
81  toPropagationFrameTransformation =
82  reference_frames::getAerodynamicForceTransformationFunction(
83  bodyFlightConditions->getAerodynamicAngleCalculator( ),
84  torqueFrame,
85  std::bind( &Body::getCurrentRotationToGlobalFrame, bodyExertingTorque ),
86  reference_frames::body_frame );
87 
88 
89  std::function< Eigen::Vector3d( ) > coefficientFunction =
90  std::bind( &aerodynamics::AerodynamicCoefficientInterface::getCurrentMomentCoefficients,
91  aerodynamicCoefficients );
92  std::function< Eigen::Vector3d( ) > coefficientInPropagationFrameFunction =
93  std::bind( &reference_frames::transformVectorFunctionFromVectorFunctions,
94  coefficientFunction, toPropagationFrameTransformation );
95 
96  // Create torque model.
97  return std::make_shared< aerodynamics::AerodynamicTorque >(
98  coefficientInPropagationFrameFunction,
99  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentDensity, bodyFlightConditions ),
100  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentAirspeed, bodyFlightConditions ),
101  std::bind( &aerodynamics::AerodynamicCoefficientInterface::getReferenceArea, aerodynamicCoefficients ),
102  std::bind( &aerodynamics::AerodynamicCoefficientInterface::getReferenceLengths, aerodynamicCoefficients ),
103  aerodynamicCoefficients->getAreCoefficientsInNegativeAxisDirection( ) );
104 }

107 std::shared_ptr< gravitation::SecondDegreeGravitationalTorqueModel > createSecondDegreeGravitationalTorqueModel(
108  const std::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
109  const std::shared_ptr< simulation_setup::Body > bodyExertingTorque,
110  const std::string& nameOfBodyUndergoingTorque,
111  const std::string& nameOfBodyExertingTorque )
112 {
113  // Retrieve state functions
114  std::function< Eigen::Vector3d( ) > positionOfBodySubjectToTorqueFunction =
115  std::bind( &simulation_setup::Body::getPosition, bodyUndergoingTorque );
116  std::function< Eigen::Vector3d( ) > positionOfBodyExertingTorqueFunction =
117  std::bind( &simulation_setup::Body::getPosition, bodyExertingTorque );
118 
119  // Check model availability
120  std::shared_ptr< gravitation::GravityFieldModel > gravityFieldModel = bodyExertingTorque->getGravityFieldModel( );
121  std::function< double( ) > gravitationalParameterOfAttractingBodyFunction;
122  if( gravityFieldModel == nullptr )
123  {
124  throw std::runtime_error( "Error when making second degree gravitational torque, " + nameOfBodyExertingTorque +
125  " does not possess a gravity field" );
126  }
127  else
128  {
129  gravitationalParameterOfAttractingBodyFunction = std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
130  gravityFieldModel );
131  }
132 
133  // Retrieve environment parameters
134  std::function< Eigen::Matrix3d( ) > inertiaTensorOfRotatingBodyFunction =
135  std::bind( &simulation_setup::Body::getBodyInertiaTensor, bodyUndergoingTorque );
136  std::function< Eigen::Quaterniond( ) > rotationToBodyFixedFrameFunction =
137  std::bind( &simulation_setup::Body::getCurrentRotationToLocalFrame, bodyUndergoingTorque );
138 
139  return std::make_shared<gravitation::SecondDegreeGravitationalTorqueModel >(
140  positionOfBodySubjectToTorqueFunction, gravitationalParameterOfAttractingBodyFunction,
141  inertiaTensorOfRotatingBodyFunction, positionOfBodyExertingTorqueFunction, rotationToBodyFixedFrameFunction );

146 std::shared_ptr< basic_astrodynamics::TorqueModel > createTorqueModel(
147  const std::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
148  const std::shared_ptr< simulation_setup::Body > bodyExertingTorque,
149  const std::shared_ptr< TorqueSettings > torqueSettings,
150  const std::string& nameOfBodyUndergoingTorque,
151  const std::string& nameOfBodyExertingTorque )
152 {
153  std::shared_ptr< basic_astrodynamics::TorqueModel > torqueModel;
154 
155  switch( torqueSettings->torqueType_ )
156  {
157  case basic_astrodynamics::second_order_gravitational_torque:
158  {
159  torqueModel = createSecondDegreeGravitationalTorqueModel(
160  bodyUndergoingTorque, bodyExertingTorque, nameOfBodyUndergoingTorque, nameOfBodyExertingTorque );
161  break;
162  }
163  case basic_astrodynamics::aerodynamic_torque:
164  {
165  torqueModel = createAerodynamicTorqueModel(
166  bodyUndergoingTorque, bodyExertingTorque, nameOfBodyUndergoingTorque, nameOfBodyExertingTorque );
167  break;
168  }
169  default:
170  throw std::runtime_error(
171  "Error, did not recognize type " + std::to_string( torqueSettings->torqueType_ ) +
172  " when making torque model" );
173  }
174 
175  return torqueModel;
176 }

180 basic_astrodynamics::TorqueModelMap createTorqueModelsMap(
181  const NamedBodyMap& bodyMap,
182  const SelectedTorqueMap& selectedTorquePerBody )
183 {
184  basic_astrodynamics::TorqueModelMap torqueModelMap;
185 
186  for( SelectedTorqueMap::const_iterator acceleratedBodyIterator = selectedTorquePerBody.begin( );
187  acceleratedBodyIterator != selectedTorquePerBody.end( ); acceleratedBodyIterator++ )
188  {
189  if( bodyMap.count( acceleratedBodyIterator->first ) == 0 )
190  {
191  throw std::runtime_error(
192  "Error, could not find body " + acceleratedBodyIterator->first + " when making torque model map." );
193  }
194  else
195  {
196  for( std::map< std::string, std::vector< std::shared_ptr< TorqueSettings > > >::const_iterator
197  acceleratingBodyIterator = acceleratedBodyIterator->second.begin( );
198  acceleratingBodyIterator != acceleratedBodyIterator->second.end( ); acceleratingBodyIterator++ )
199  {
200  if( bodyMap.count( acceleratingBodyIterator->first ) == 0 )
201  {
202  throw std::runtime_error(
203  "Error, could not find body " + acceleratingBodyIterator->first + " when making torque model map." );
204  }
205  for( unsigned int i = 0; i < acceleratingBodyIterator->second.size( ); i++ )
206  {
207  torqueModelMap[ acceleratedBodyIterator->first ][ acceleratingBodyIterator->first ].push_back(
208  createTorqueModel(
209  bodyMap.at( acceleratedBodyIterator->first ), bodyMap.at( acceleratingBodyIterator->first ),
210  acceleratingBodyIterator->second.at( i ),
211  acceleratedBodyIterator->first, acceleratingBodyIterator->first ) );
212  }
213  }
214  }
215  }
216 
217  return torqueModelMap;
218 }
