402 int indexFromKey( const std::string& key )
403 {
404  boost::cmatch groups;
405  boost::regex_match( key.c_str( ), groups, boost::regex( R"(\@(\d+))" ) );
406  if ( groups[ 1 ].matched )
407  {
408  try
409  {
410  return std::stoi( groups[ 1 ] );
411  }
412  catch ( ... ) { }
413  }
414  return -1;
415 }

418 KeyPath::KeyPath( const std::string& keyPathStringRepresentation ) : std::vector< std::string >( )
419 {
420  const std::vector< std::string > keys = split( keyPathStringRepresentation, SpecialKeys::dot );
421  for ( const std::string key : keys )
422  {
423  boost::cmatch groups;
424  boost::regex_match( key.c_str( ), groups, boost::regex( R"((.+?)\[(\d+?)\])" ) );
425  if ( groups[ 1 ].matched && groups[ 2 ].matched )
426  {
427  const std::string arrayKey( groups[ 1 ] );
428  const std::string arrayIndex( groups[ 2 ] );
429  push_back( arrayKey );
430  push_back( "@" + arrayIndex );
431  }
432  else
433  {
434  push_back( key );
435  }
436  }
437 }

440 std::ostream& operator << ( std::ostream& stringRepresentation, const KeyPath& keyPath )
441 {
442  bool somethingAdded = false;
443  for ( unsigned int i = 0; i < keyPath.size( ); ++i )
444  {
445  const std::string key = keyPath.at( i );
