52 std::pair< Eigen::Vector2d, double > PrecessionNutationCalculator::getPositionOfCipInGcrs(
53  const double terrestrialTime,
54  const double utc )
55 {
56  // Calculate nominal precession-nutation values.
57  std::pair< Eigen::Vector2d, double > nominalCipPosition = nominalCipPositionFunction_( terrestrialTime );
58 
59  // Retrieve measured corrections to model.
60  Eigen::Vector2d iersCorrections = dailyCorrectionInterpolator_->interpolate( utc );
61 
62  // Add nominal values and corrections and return.
