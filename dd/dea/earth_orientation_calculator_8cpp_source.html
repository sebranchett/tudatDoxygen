28 Eigen::Quaterniond calculateRotationFromCirsToGcrs(
29  const double celestialPoleXPosition, const double celestialPoleYPosition, const double cioLocator )
30 {
31  // Pre-compute reused parameters
32  double xParameterSquared =celestialPoleXPosition * celestialPoleXPosition;
33  double yParameterSquared =celestialPoleYPosition * celestialPoleYPosition;
34  double xyCrossTerm =celestialPoleXPosition * celestialPoleYPosition;
35  double parameterA = 0.5 + 0.125 * ( xParameterSquared + yParameterSquared );
36 
37  // Set up rotation.
38  Eigen::Matrix3d rotationMatrix;
39  rotationMatrix << 1.0 - parameterA * xParameterSquared, -parameterA * xyCrossTerm, celestialPoleXPosition,
40  - parameterA * xyCrossTerm, 1.0 - parameterA * yParameterSquared, celestialPoleYPosition,
41  -celestialPoleXPosition, -celestialPoleYPosition, 1.0 - parameterA * ( xParameterSquared + yParameterSquared );
42  return Eigen::Quaterniond( rotationMatrix ) *
43  Eigen::Quaterniond( Eigen::AngleAxisd( -cioLocator, Eigen::Vector3d::UnitZ( ) ) );
44 }

62 std::shared_ptr< EarthOrientationAnglesCalculator > createStandardEarthOrientationCalculator(
63  const std::shared_ptr< EOPReader > eopReader )
64 {
65  // Load polar motion corrections
66  std::shared_ptr< interpolators::LinearInterpolator< double, Eigen::Vector2d > > cipInItrsInterpolator =
67  std::make_shared< interpolators::LinearInterpolator< double, Eigen::Vector2d > >(
68  eopReader->getCipInItrsMapInSecondsSinceJ2000( ) );
69 
70  // Load nutation corrections
71  std::shared_ptr< interpolators::LinearInterpolator< double, Eigen::Vector2d > > cipInGcrsCorrectionInterpolator =
72  std::make_shared< interpolators::LinearInterpolator< double, Eigen::Vector2d > >(
73  eopReader->getCipInGcrsCorrectionMapInSecondsSinceJ2000( ) );
74 
75  // Load default polar motion correction (sub-diural frequencies) object
76  std::shared_ptr< ShortPeriodEarthOrientationCorrectionCalculator< Eigen::Vector2d > > shortPeriodPolarMotionCalculator =
77  getDefaultPolarMotionCorrectionCalculator( );
78 
79  // Create full polar motion calculator
80  std::shared_ptr< PolarMotionCalculator > polarMotionCalculator = std::make_shared< PolarMotionCalculator >
81  ( cipInItrsInterpolator, shortPeriodPolarMotionCalculator );
82 
83  // Create IAU 2006 precession/nutation calculator
84  std::shared_ptr< PrecessionNutationCalculator > precessionNutationCalculator =
85  std::make_shared< PrecessionNutationCalculator >( basic_astrodynamics::iau_2006, cipInGcrsCorrectionInterpolator );
86 
87  // Create default time scale converter
88  std::shared_ptr< TerrestrialTimeScaleConverter > terrestrialTimeScaleConverter =
89  createDefaultTimeConverter( eopReader );
90 
91  // Create EarthOrientationAnglesCalculator object
92  return std::make_shared< EarthOrientationAnglesCalculator >(
93  polarMotionCalculator, precessionNutationCalculator, terrestrialTimeScaleConverter );
94 }
