80 int getNWayLinkIndexFromLinkEndType( const LinkEndType linkEndType, const int numberOfLinkEnds )
81 {
82  int linkEndIndex;
83  // If index is first or last, set 0 or numberOfLinkEnds - 1, respectively
84  if( linkEndType == transmitter )
85  {
86  linkEndIndex = 0;
87  }
88  else if( linkEndType == receiver )
89  {
90  linkEndIndex = numberOfLinkEnds - 1;
91  }
92  else
93  {
94  linkEndIndex = static_cast< int >( linkEndType ) - static_cast< int >( reflector1 ) + 1;
95  if( linkEndIndex > numberOfLinkEnds - 2 )
96  {
97  throw std::runtime_error( "Error when getting n-way link end index; value too large." );
98  }
99  }
100  return linkEndIndex;
101 }

104 LinkEndType getNWayLinkEnumFromIndex( const int linkEndIndex, const int numberOfLinkEnds )
105 {
106  LinkEndType linkEndType;
107 
108  // If index is first or last, set transmitter or receiver, respectively
109  if( linkEndIndex == 0 )
110  {
111  linkEndType = transmitter;
112  }
113  else if( linkEndIndex == numberOfLinkEnds - 1 )
114  {
115  linkEndType = receiver;
116  }
117  // Check feasibility of inner link end
118  else if( linkEndIndex >= numberOfLinkEnds )
119  {
120  throw std::runtime_error(
121  "Error, found link end index " + std::to_string( linkEndIndex ) +
122  " when getting n-way link end index for " + std::to_string(
123  numberOfLinkEnds ) + " link end total." );
124  }
125  else
126  {
127  linkEndType = static_cast< LinkEndType >( static_cast< int >( reflector1 ) + ( linkEndIndex - 1 ) );
128  }
129 
130  return linkEndType;
131 }
