51 {

131 double computeModifiedDahlemBuckPressureCoefficient(
132  double inclinationAngle, double machNumber )
133 {
134  // Declare local variables.
135  double checkAngle_ = 22.5 * PI / 180.0;
136  double factor1_;
137  double factor2_;
138  double exponent_;
139  double pressureCoefficient_;
140 
141  // Check if inclination angle is greater than check angle. If so, use
142  // Newtonian approximation.
143  if ( inclinationAngle > checkAngle_ )
144  {
145  pressureCoefficient_
146  = computeNewtonianPressureCoefficient( inclinationAngle );
147  }
148 
149  // Else use Dahlem-Buck method.
150  else
151  {
152  pressureCoefficient_
153  = ( 1.0 + sin( 4.0 * pow( inclinationAngle , 0.75 ) ) )
154  / ( pow( 4.0 * cos( inclinationAngle )
155  * cos( 2.0 * inclinationAngle ), 0.75 ) )
156  * pow( sin( inclinationAngle ), 1.25 );
157  }
158 
159  // For mach < 20, a correction term should be applied.
160  if ( machNumber > 20.0 )
161  {
162  factor2_ = 1.0;
163  }
164  else
165  {
166  // Determine correction term.
167  factor1_ = ( 6.0 - 0.3 * machNumber ) + sin( PI * ( log( machNumber ) - 0.588 ) / 1.20 );
168 
169  exponent_ = 1.15 + 0.5 * sin( PI * ( log( machNumber ) - 0.916 ) / 3.29 );
170 
171  factor2_ = 1.0 + factor1_ * pow( inclinationAngle * 180.0 / PI, -1.0 * exponent_ );
172  }
173 
174  // Return pressure coefficient.
175  return pressureCoefficient_ * factor2_;
176 }

236 double computeVanDykeUnifiedPressureCoefficient(
237  double inclinationAngle, double machNumber,
238  double ratioOfSpecificHeats, int type )
239 {
240  // Declare and initialize local variables and pre-compute for efficiency.
241  double ratioOfSpecificHeatsTerm_ = ( ratioOfSpecificHeats + 1.0 ) / 2.0;
242  double machNumberTerm_ = sqrt( pow( machNumber , 2.0 ) - 1.0 );
243  double exponent_ = 2.0 * ratioOfSpecificHeats / ( ratioOfSpecificHeats - 1.0 );
244 
245  // Declare and initialize value.
246  double pressureCoefficient_ = 0.0;
247 
248  // Calculate compression pressure coefficient.
249  if ( inclinationAngle >= 0.0 && type == 1 )
250  {
251  pressureCoefficient_ = pow( inclinationAngle, 2.0 )
252  * ( ratioOfSpecificHeatsTerm_
253  + sqrt( pow( ratioOfSpecificHeatsTerm_, 2.0 )
254  + 4.0 / ( pow( inclinationAngle * machNumberTerm_, 2.0 ) ) ) );
255  }
256 
257  // Calculate expansion pressure coefficient.
258  else if ( inclinationAngle < 0.0 && type == -1 )
259  {
260  // Calculate vacuum pressure coefficient.
261  double vacuumPressureCoefficient_ = computeVacuumPressureCoefficient(
262  machNumber, ratioOfSpecificHeats );
263 
264  // Check to see if pressure coefficient will be lower than vacuum case,
265  // set to vacuum if so.
266  if ( -1.0 * inclinationAngle * machNumberTerm_
267  > 2.0 / ( ratioOfSpecificHeats - 1.0 ) )
268  {
269  pressureCoefficient_ = vacuumPressureCoefficient_;
270  }
271  else
272  {
273  pressureCoefficient_
274  = 2.0 / ( ratioOfSpecificHeats * pow( machNumberTerm_, 2.0 ) )
275  * ( pow( 1.0 - ( ratioOfSpecificHeats - 1.0 ) / 2.0
276  * - 1.0 * inclinationAngle * machNumberTerm_, exponent_ ) - 1.0 );
277 
278  if ( pressureCoefficient_ < vacuumPressureCoefficient_ )
279  {
280  pressureCoefficient_ = vacuumPressureCoefficient_;
281  }
282  }
283  }
284 
285  // Return pressure coefficient.
286  return pressureCoefficient_;
287 }
