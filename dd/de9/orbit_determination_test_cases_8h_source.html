54  //Define setting for total number of bodies and those which need to be integrated numerically.

73  getDefaultBodySettings( bodyNames,initialEphemerisTime - buffer, finalEphemerisTime + buffer );
74  bodySettings[ "Moon" ]->ephemerisSettings->resetFrameOrigin( "Sun" );

76  // Create bodies needed in simulation
77  NamedBodyMap bodyMap = createBodies( bodySettings );
78 
79  setGlobalFrameBodyEphemerides( bodyMap, "SSB", "ECLIPJ2000" );

82  // Set accelerations between bodies that are to be taken into account.
83  SelectedAccelerationMap accelerationMap;
84  std::map< std::string, std::vector< std::shared_ptr< AccelerationSettings > > > accelerationsOfEarth;

87  accelerationsOfEarth[ "Mars" ].push_back( std::make_shared< AccelerationSettings >( central_gravity ) );
88  accelerationsOfEarth[ "Jupiter" ].push_back( std::make_shared< AccelerationSettings >( central_gravity ) );

95  std::vector< std::string > bodiesToEstimate;
96  bodiesToEstimate.push_back( "Earth" );
97  std::vector< std::string > bodiesToIntegrate;
98  bodiesToIntegrate.push_back( "Earth" );

102  std::vector< std::string > centralBodies;
103  std::map< std::string, std::string > centralBodyMap;

106  for( unsigned int i = 0; i < numberOfNumericalBodies; i++ )

206  // Define observation simulation settings

241  typedef Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > ObservationVectorType;
242  typedef std::map< LinkEnds, std::pair< ObservationVectorType, std::pair< std::vector< TimeType >, LinkEndType > > > SingleObservablePodInputType;
243  typedef std::map< ObservableType, SingleObservablePodInputType > PodInputDataType;

245  // Simulate observations
246  PodInputDataType observationsAndTimes = simulateObservations< StateScalarType, TimeType >(
247  measurementSimulationInput, orbitDeterminationManager.getObservationSimulators( ) );
248 
249  // Perturb parameter estimate
250  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialParameterEstimate =
251  parametersToEstimate->template getFullParameterValues< StateScalarType >( );
252  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > truthParameters = initialParameterEstimate;
253  if( parameterPerturbation.rows( ) == 0 )
254  {
255  parameterPerturbation = Eigen::VectorXd::Zero( 7 );
256  }
257  for( unsigned int i = 0; i < 7; i++ )
258  {
259  initialParameterEstimate( i ) += parameterPerturbation( i );
260  }
261 
262  // Define estimation input
263  std::shared_ptr< PodInput< StateScalarType, TimeType > > podInput =
264  std::make_shared< PodInput< StateScalarType, TimeType > >(
265  observationsAndTimes, initialParameterEstimate.rows( ), inverseAPrioriCovariance,
266  initialParameterEstimate - truthParameters );
267  if( observableType == 4 )
268  {
269  std::map< observation_models::ObservableType, double > weightPerObservable;
270  weightPerObservable[ one_way_range ] = 1.0 / ( 1.0 * 1.0 );
271  weightPerObservable[ angular_position ] = 1.0 / ( 1.0E-9 * 1.0E-9 );
272  weightPerObservable[ one_way_doppler ] = 1.0 / ( 1.0E-12 * 1.0E-12 );
273 
274  podInput->setConstantPerObservableWeightsMatrix( weightPerObservable );
275  }
276  else
277  {
278  podInput->setConstantWeightsMatrix( weight );
279  }
280  podInput->defineEstimationSettings( true, true, false, false, false );
281 
282  // Perform estimation
283  std::shared_ptr< PodOutput< StateScalarType > > podOutput = orbitDeterminationManager.estimateParameters(
284  podInput, std::make_shared< EstimationConvergenceChecker >( ) );
285 
286  return std::make_pair( podOutput,
287  ( podOutput->parameterEstimate_.template cast< double >( ) -
288  truthParameters .template cast< double >( ) ) );
289 }
