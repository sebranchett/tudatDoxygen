24 KeplerEphemeris::KeplerEphemeris(
25  const Eigen::Vector6d& initialStateInKeplerianElements,
26  const double epochOfInitialState,
27  const double centralBodyGravitationalParameter,
28  const std::string& referenceFrameOrigin,
29  const std::string& referenceFrameOrientation,
30  const double rootFinderAbsoluteTolerance,
31  const double rootFinderMaximumNumberOfIterations ):
32  Ephemeris( referenceFrameOrigin, referenceFrameOrientation ),
33  initialStateInKeplerianElements_( initialStateInKeplerianElements ),
34  epochOfInitialState_( epochOfInitialState ),
35  centralBodyGravitationalParameter_( centralBodyGravitationalParameter )
36 {
37  using namespace tudat::orbital_element_conversions;
38  using namespace tudat::root_finders;
39  using namespace root_finders::termination_conditions;
40 
41  // Check whether orbit is elliptical or hyperbolic (parabola not supported).
42  if( initialStateInKeplerianElements( eccentricityIndex ) < 1.0 )
43  {
44  isOrbitHyperbolic_ = 0;
45  }
46  else if( initialStateInKeplerianElements( eccentricityIndex ) > 1.0 )
47  {
48  isOrbitHyperbolic_ = 1;
49  }
50  else
51  {
52  throw std::runtime_error(
53  "Error, Kepler ephemeris cannot handle parabolic orbit" );
54  }
55 
56  // Convert initial true anomaly to mean anomaly.
57  if( !isOrbitHyperbolic_ )
58  {
59  initialMeanAnomaly_ = convertEccentricAnomalyToMeanAnomaly(
60  convertTrueAnomalyToEccentricAnomaly(
61  initialStateInKeplerianElements_( trueAnomalyIndex ),
62  initialStateInKeplerianElements_( eccentricityIndex ) ),
63  initialStateInKeplerianElements_( eccentricityIndex ) );
64  }
65  else
66  {
67  initialMeanAnomaly_ = convertHyperbolicEccentricAnomalyToMeanAnomaly(
68  convertTrueAnomalyToHyperbolicEccentricAnomaly(
69  initialStateInKeplerianElements_( trueAnomalyIndex ),
70  initialStateInKeplerianElements_( eccentricityIndex ) ),
71  initialStateInKeplerianElements_( eccentricityIndex ) );
72  }
73 
74 
75  // Calculate ancilliary variables for conversion to Cartesian elements
76  eccentricity_ = initialStateInKeplerianElements_( eccentricityIndex );
77  semiMajorAxis_ = initialStateInKeplerianElements_( semiMajorAxisIndex );
78  semiLatusRectum_ = initialStateInKeplerianElements_( semiMajorAxisIndex ) *
79  ( 1.0 - pow( initialStateInKeplerianElements_( eccentricityIndex ), 2 ) );
80 
81  rotationFromOrbitalPlane_ =
82  Eigen::AngleAxisd(
83  initialStateInKeplerianElements_(
84  longitudeOfAscendingNodeIndex ), Eigen::Vector3d::UnitZ( ) ) *
85  Eigen::AngleAxisd(
86  initialStateInKeplerianElements_(
87  inclinationIndex ), Eigen::Vector3d::UnitX( ) ) *
88  Eigen::AngleAxisd(
89  initialStateInKeplerianElements_(
90  argumentOfPeriapsisIndex ), Eigen::Vector3d::UnitZ( ) ) ;
91 
92  // Create root finder to be used for converting mean to eccentric anomaly
93  rootFinder_ = std::make_shared< NewtonRaphsonCore< double > >(
94  std::bind( &RootAbsoluteToleranceTerminationCondition< >::checkTerminationCondition,
95  std::make_shared< RootAbsoluteToleranceTerminationCondition< > >(

97  std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, std::placeholders::_4, std::placeholders::_5 ) );

101 Eigen::Vector6d KeplerEphemeris::getCartesianState(
102  const double secondsSinceEpoch )
103 {
104  using namespace tudat::orbital_element_conversions;
105 
106  Eigen::Vector6d currentCartesianState = Eigen::Vector6d::Zero( );
107 
108  double propagationTime = secondsSinceEpoch - epochOfInitialState_;
109 
110  // Calculate eccentric anomaly at epoch.
111  double eccentricAnomaly = TUDAT_NAN;
112 
113  // Check if the orbit is hyperbolic.
114  if( !isOrbitHyperbolic_ )
115  {
116 
117  // Compute change of mean anomaly between start and end of propagation.
118  const double meanAnomalyChange =
119  convertElapsedTimeToEllipticalMeanAnomalyChange(
120  propagationTime, centralBodyGravitationalParameter_, semiMajorAxis_ );
121 
122  // Compute eccentric anomaly for mean anomaly.
123  eccentricAnomaly =
124  convertMeanAnomalyToEccentricAnomaly(
125  eccentricity_,
126  initialMeanAnomaly_ + meanAnomalyChange );
127  }
128  else
129  {
130  // Compute change of mean anomaly because of the propagation time.
131  const double hyperbolicMeanAnomalyChange =
132  convertElapsedTimeToHyperbolicMeanAnomalyChange(
133  propagationTime, centralBodyGravitationalParameter_, semiMajorAxis_ );
134 
135  // Compute hyperbolic eccentric anomaly for mean anomaly.
136  eccentricAnomaly =
137  convertMeanAnomalyToHyperbolicEccentricAnomaly(
138  eccentricity_,
139  initialMeanAnomaly_ + hyperbolicMeanAnomalyChange );
140  }
141 
142  // Calculate true anomaly.
143  double trueAnomaly = convertEccentricAnomalyToTrueAnomaly(
144  eccentricAnomaly, eccentricity_ );
145  double cosineOfTrueAnomaly = cos( trueAnomaly );
146  double sineOfTrueAnomaly = sin( trueAnomaly );
147 
148  // Definition of position in the perifocal coordinate system.
149  currentCartesianState( 0 ) = semiLatusRectum_ * cosineOfTrueAnomaly
150  / ( 1.0 + eccentricity_ * cosineOfTrueAnomaly );
151  currentCartesianState( 1 ) = semiLatusRectum_ * sineOfTrueAnomaly
152  / ( 1.0 + eccentricity_ * cosineOfTrueAnomaly );
153 
154  // Definition of velocity in the perifocal coordinate system.
155  currentCartesianState( 3 ) =
156  -sqrt( centralBodyGravitationalParameter_ / semiLatusRectum_ ) * sineOfTrueAnomaly;
157  currentCartesianState( 4 ) =
158  sqrt( centralBodyGravitationalParameter_ / semiLatusRectum_ )
159  * ( eccentricity_ + cosineOfTrueAnomaly );
160 
161  // Rotate orbital plane to correct orientation.
162  currentCartesianState.segment( 0, 3 ) = rotationFromOrbitalPlane_ *
163  currentCartesianState.segment( 0, 3 );
164  currentCartesianState.segment( 3, 3 ) = rotationFromOrbitalPlane_ *
165  currentCartesianState.segment( 3, 3 );
166 
167  return currentCartesianState;
168 }
