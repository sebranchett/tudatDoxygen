81 {

84  double orbitalPeriod = 2 * mathematical_constants::PI / meanMotion;
85  double numberOfSteps = std::ceil( orbitalPeriod / maximumTimeStep );
86  double timeStep = orbitalPeriod / numberOfSteps;

90

95  for ( int counter = 0; counter < numberOfSteps + 1; counter++ )
96  {
97  timeVector[ counter ] = time + startingTime;
98  positionVector[ counter ] << semiMajorAxis * std::cos( time * meanMotion ),
