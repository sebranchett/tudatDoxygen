27 double GaussianCopulaDistributionXd::evaluatePdf(
28  const Eigen::VectorXd& independentVariables )
29 {
30  double probabilityDensity = 0.0 ;
31 
32  // Check if vector independentVariables is inside [0,1]
33  int inBound = 0 ;
34  for( int i = 0 ; i < dimension_ ; i++ )
35  {
36  if( independentVariables(i) > 0.0 && independentVariables(i) < 1.0 )
37  {
38  inBound++;
39  }
40  }
41 
42  // If data is in bounds
43  if( inBound == dimension_ )
44  {
45  // Convert U[0,1] to N[0,1] using inverse CDF of standard normal distribution
46  Eigen::VectorXd gaussianQuantiles( dimension_ ) ;
47  boost::math::normal distribution( 0.0 , 1.0 );
48 
49  for( int i = 0 ; i < dimension_ ; i++ )
50  {
51  gaussianQuantiles( i ) = boost::math::quantile( distribution , independentVariables( i ) ); // Inverse cdf
52  }
53 
54  // Calculate probability density
55  Eigen::MatrixXd location = - 0.5 * ( gaussianQuantiles.transpose() *
56  ( inverseCorrelationMatrix_ - Eigen::MatrixXd::Identity( dimension_ , dimension_ ) ) *
57  gaussianQuantiles );
58 
59  probabilityDensity = ( ( 1.0 / ( std::sqrt( determinant_ ) ) ) * std::exp( location( 0, 0 ) ) ) ;
60  }
61 
62  return probabilityDensity ;
63 }
