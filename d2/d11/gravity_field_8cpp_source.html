21 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< GravityFieldSettings >& gravityFieldSettings )
22 {
23  if ( ! gravityFieldSettings )
24  {
25  return;
26  }
27  using namespace json_interface;
28  using K = Keys::Body::GravityField;
29 
30  // Type
31  const GravityFieldType gravityFieldType = gravityFieldSettings->getGravityFieldType( );
32  jsonObject[ K::type ] = gravityFieldType;
33 
34  switch ( gravityFieldType )
35  {
36  case central:
37  {
38  std::shared_ptr< CentralGravityFieldSettings > centralGravityFieldSettings =
39  std::dynamic_pointer_cast< CentralGravityFieldSettings >( gravityFieldSettings );
40  assertNonnullptrPointer( centralGravityFieldSettings );
41  jsonObject[ K::gravitationalParameter ] = centralGravityFieldSettings->getGravitationalParameter( );
42  return;
43  }
44  case central_spice:
45  return;
46  case spherical_harmonic:
47  {
48  std::shared_ptr< SphericalHarmonicsGravityFieldSettings > shGravityFieldSettings =
49  std::dynamic_pointer_cast< SphericalHarmonicsGravityFieldSettings >( gravityFieldSettings );
50  assertNonnullptrPointer( shGravityFieldSettings );
51 
52  // FromFileSphericalHarmonicsGravityFieldSettings
53  std::shared_ptr< FromFileSphericalHarmonicsGravityFieldSettings > shModelGravityFieldSettings =
54  std::dynamic_pointer_cast< FromFileSphericalHarmonicsGravityFieldSettings >( gravityFieldSettings );
55  if ( shModelGravityFieldSettings )
56  {
57  const SphericalHarmonicsModel model = shModelGravityFieldSettings->getSphericalHarmonicsModel( );
58  if ( model == customModel )
59  {
60  jsonObject[ K::file ] = boost::filesystem::path( shModelGravityFieldSettings->getFilePath( ) );
61  jsonObject[ K::associatedReferenceFrame ] = shModelGravityFieldSettings->getAssociatedReferenceFrame( );
62  jsonObject[ K::maximumDegree ] = shModelGravityFieldSettings->getMaximumDegree( );
63  jsonObject[ K::maximumOrder ] = shModelGravityFieldSettings->getMaximumOrder( );
64 
65  // Gravitational parameter (index)
66  const int gmIndex = shModelGravityFieldSettings->getGravitationalParameterIndex( );
67  jsonObject[ K::gravitationalParameterIndex ] = gmIndex;
68  if ( gmIndex < 0 )
69  {
70  jsonObject[ K::gravitationalParameter ] =
71  shModelGravityFieldSettings->getGravitationalParameter( );
72  }
73 
74  // Reference radius (index)
75  const int rIndex = shModelGravityFieldSettings->getReferenceRadiusIndex( );
76  jsonObject[ K::referenceRadiusIndex ] = rIndex;
77  if ( rIndex < 0 )
78  {
79  jsonObject[ K::referenceRadius ] =
80  shModelGravityFieldSettings->getReferenceRadius( );
81  }
82  }
83  else
84  {
85  jsonObject[ K::model ] = model;
86  }
87  }
88  else
89  {
90  jsonObject[ K::gravitationalParameter ] = shGravityFieldSettings->getGravitationalParameter( );
91  jsonObject[ K::referenceRadius ] = shGravityFieldSettings->getReferenceRadius( );
92  jsonObject[ K::cosineCoefficients ] = shGravityFieldSettings->getCosineCoefficients( );
93  jsonObject[ K::sineCoefficients ] = shGravityFieldSettings->getSineCoefficients( );
94  jsonObject[ K::associatedReferenceFrame ] = shGravityFieldSettings->getAssociatedReferenceFrame( );
95  }
96  return;
97  }
98  default:
99  handleUnimplementedEnumValue( gravityFieldType, gravityFieldTypes, unsupportedGravityFieldTypes );
100  }

104 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< GravityFieldSettings >& gravityFieldSettings )
105 {
106  using namespace json_interface;
107  using K = Keys::Body::GravityField;
108 
109  // Get atmosphere model type
110  const GravityFieldType gravityFieldType = getValue< GravityFieldType >( jsonObject, K::type );
111 
112  switch ( gravityFieldType ) {
113  case central:
114  {
115  gravityFieldSettings = std::make_shared< CentralGravityFieldSettings >(
116  getValue< double >( jsonObject, K::gravitationalParameter ) );
117  return;
118  }
119  case central_spice:
120  {
121  gravityFieldSettings = std::make_shared< GravityFieldSettings >( central_spice );
122  return;
123  }
124  case spherical_harmonic:
125  {
126  // load coefficients from custom file
127  if ( isDefined( jsonObject, K::file ) )
128  {
129  const int gmIndex = getValue( jsonObject, K::gravitationalParameterIndex, 0 );
130  const int radiusIndex = getValue( jsonObject, K::referenceRadiusIndex, 1 );
131  gravityFieldSettings = std::make_shared< FromFileSphericalHarmonicsGravityFieldSettings >(
132  getValue< boost::filesystem::path >( jsonObject, K::file ).string( ),
133  getValue< std::string >( jsonObject, K::associatedReferenceFrame ),
134  getValue< int >( jsonObject, K::maximumDegree ),
135  getValue< int >( jsonObject, K::maximumOrder ),
136  gmIndex,
137  radiusIndex,
138  getValue< double >( jsonObject, K::gravitationalParameter, TUDAT_NAN ),
139  getValue< double >( jsonObject, K::referenceRadius, TUDAT_NAN ) );
140  return;
141  }
142 
143  // load coefficients from model included in Tudat
144  if ( isDefined( jsonObject, K::model ) )
145  {
146  gravityFieldSettings = std::make_shared< FromFileSphericalHarmonicsGravityFieldSettings >(
147  getValue< SphericalHarmonicsModel >( jsonObject, K::model ) );
148  return;
149  }
150 
151  // user-provided coefficients in JSON object
152  gravityFieldSettings = std::make_shared< SphericalHarmonicsGravityFieldSettings >(
153  getValue< double >( jsonObject, K::gravitationalParameter ),
154  getValue< double >( jsonObject, K::referenceRadius ),
155  getValue< Eigen::MatrixXd >( jsonObject, K::cosineCoefficients ),
156  getValue< Eigen::MatrixXd >( jsonObject, K::sineCoefficients ),
157  getValue< std::string >( jsonObject, K::associatedReferenceFrame ) );
158  return;
159  }
160  default:
161  handleUnimplementedEnumValue( gravityFieldType, gravityFieldTypes, unsupportedGravityFieldTypes );
162  }
163 }
