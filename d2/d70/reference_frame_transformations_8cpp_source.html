45 Eigen::Vector3d calculateInertialToPlanetFixedRotationAnglesFromMatrix(
46  const Eigen::Matrix3d& rotationMatrixFromInertialToPlanetFixedFrame )
47 {
48  Eigen::Vector3d rotationAngles;
49  rotationAngles.x( ) = basic_mathematics::computeModulo(
50  std::atan2( rotationMatrixFromInertialToPlanetFixedFrame( 2, 0 ),
51  -rotationMatrixFromInertialToPlanetFixedFrame( 2, 1 ) ) - mathematical_constants::PI / 2.0,
52  2.0 * mathematical_constants::PI );//right ascension
53  rotationAngles.y( ) = -std::acos( rotationMatrixFromInertialToPlanetFixedFrame( 2, 2 ) ) + mathematical_constants::PI / 2.0 ; // declination
54  rotationAngles.z( ) = std::atan2( rotationMatrixFromInertialToPlanetFixedFrame( 0, 2 ),
55  rotationMatrixFromInertialToPlanetFixedFrame( 1, 2 ) ); // longitude of prime meridian
56  return rotationAngles;
57 }

122 Eigen::Quaterniond getRotatingPlanetocentricToInertialFrameTransformationQuaternion(
123  const double declinationOfPole,
124  const double rightAscensionOfPole,
125  const double longitudeOfPrimeMeridian )
126 {
127  // Compute transformation quaternion.
128  // Note the sign change, because how angleAxisd is defined.
129  Eigen::AngleAxisd firstRotationAroundZaxis =
130  Eigen::AngleAxisd( longitudeOfPrimeMeridian, Eigen::Vector3d::UnitZ( ) );
131  Eigen::AngleAxisd rotationAroundXaxis =
132  Eigen::AngleAxisd(
133  ( mathematical_constants::PI / 2.0 - declinationOfPole ),
134  Eigen::Vector3d::UnitX( ) );

136  rightAscensionOfPole

138  Eigen::Quaterniond frameTransformationQuaternion = Eigen::Quaterniond(
139  ( secondRotationAroundZaxis * rotationAroundXaxis * firstRotationAroundZaxis ) );

157 Eigen::Matrix3d getVelocityBasedLvlhToInertialRotation(
158  const Eigen::Vector6d& vehicleState,
159  const Eigen::Vector6d& centralBodyState,
160  const bool doesNaxisPointAwayFromCentralBody )
161 {
162  Eigen::Vector3d vehicleVelocity, vehicleRadius;
163  vehicleRadius = vehicleState.segment( 0, 3 ) - centralBodyState.segment( 0, 3 );
164  vehicleVelocity = vehicleState.segment( 3, 3 ) - centralBodyState.segment( 3, 3 );
165 
166  Eigen::Vector3d unitT = vehicleVelocity.normalized( );// / vehicleVelocity.norm( );
167  if ( vehicleRadius.cross( vehicleVelocity ).norm( ) == 0.0 )
168  {
169  std::string errorMessage = "Division by zero: radius and velocity are in the same direction.";
170  throw std::runtime_error( errorMessage );
171  }
172 
173  Eigen::Vector3d unitW = ( ( ( doesNaxisPointAwayFromCentralBody ) ? -1.0 : 1.0 ) *
174  ( vehicleRadius.cross( vehicleVelocity ) ).normalized( ) );
175 
176  Eigen::Vector3d unitN = ( unitW.cross( unitT ) ).normalized( );
177 
178  Eigen::Matrix3d transformationMatrix;
179  transformationMatrix << unitT( 0 ), unitN( 0 ), unitW( 0 ),
180  unitT( 1 ), unitN( 1 ), unitW( 1 ),
181  unitT( 2 ), unitN( 2 ), unitW( 2 );
182 
183  return transformationMatrix;
184 }

197 Eigen::Quaterniond getVelocityBasedLvlhToPlanetocentricRotationKeplerian(
198  const Eigen::Matrix< double, 6, 1 > spacecraftKeplerianState )
199 {
200  double eccentricity = spacecraftKeplerianState( 1 );
201  double inclination = spacecraftKeplerianState( 2 );
202  double argumentOfPeriapsis = spacecraftKeplerianState( 3 );
203  double rightAscensionOfAscendingNode = spacecraftKeplerianState( 4 );
204  double trueAnomaly = spacecraftKeplerianState( 5 );
205 
206  double flightPathAngle = std::atan( ( eccentricity * std::sin( trueAnomaly ) ) /
207  ( 1.0 + eccentricity * std::cos( trueAnomaly ) ) );
208 
209  // Compute first rotation around Z axis.
210  Eigen::AngleAxisd firstRotationAroundZaxis(
211  -( -mathematical_constants::PI * 0.5 + flightPathAngle - ( trueAnomaly + argumentOfPeriapsis ) ),
212  Eigen::Vector3d::UnitZ( ) );
213 
214  // Compute rotation around X axis.
215  Eigen::AngleAxisd rotationAroundXaxis( inclination, Eigen::Vector3d::UnitX( ) );
216 
217  // Compute second rotation around Z axis.
218  Eigen::AngleAxisd secondRotationAroundZaxis( rightAscensionOfAscendingNode, Eigen::Vector3d::UnitZ( ) );
219 
220  Eigen::Quaterniond frameTransformationQuaternion = Eigen::Quaterniond(
221  ( secondRotationAroundZaxis * rotationAroundXaxis * firstRotationAroundZaxis) );
222 
223  // Return transformation quaternion.
224  return frameTransformationQuaternion;
225 }

228 Eigen::Matrix3d getInertialToRswSatelliteCenteredFrameRotationMatrix(
229  const Eigen::Vector6d bodyState )
230 {
231  Eigen::Vector3d vehicleVelocity, vehicleRadius;
232  vehicleRadius = bodyState.segment( 0, 3 );
233  vehicleVelocity = bodyState.segment( 3, 3 );
234 
235  Eigen::Vector3d unitR = vehicleRadius.normalized( );// / vehicleVelocity.norm( );
236  if ( vehicleRadius.cross( vehicleVelocity ).norm( ) == 0.0 )
237  {
238  std::string errorMessage = "Division by zero: radius and velocity are in the same direction in RSW frame.";
239  throw std::runtime_error( errorMessage );
240  }
241 
242  Eigen::Vector3d unitW = ( vehicleRadius.cross( vehicleVelocity ) ).normalized( );
243 
244  Eigen::Vector3d unitS = ( unitW.cross( unitR ) ).normalized( );
245 
246  Eigen::Matrix3d transformationMatrix;
247  transformationMatrix << unitR( 0 ), unitR( 1 ), unitR( 2 ),
248  unitS( 0 ), unitS( 1 ), unitS( 2 ),
249  unitW( 0 ), unitW( 1 ), unitW( 2 );
250  return transformationMatrix;
