202 boost::shared_ptr< CentralGravitationalAccelerationModel3d > createCentralGravityAcceleratioModel(
203  const boost::shared_ptr< Body > bodyUndergoingAcceleration,
204  const boost::shared_ptr< Body > bodyExertingAcceleration,
205  const std::string& nameOfBodyUndergoingAcceleration,
206  const std::string& nameOfBodyExertingAcceleration,
207  const bool useCentralBodyFixedFrame )
208 {
209  // Declare pointer to return object.
210  boost::shared_ptr< CentralGravitationalAccelerationModel3d > accelerationModelPointer;
211 
212  // Check if body is endowed with a gravity field model (i.e. is capable of exerting
213  // gravitation acceleration).
214  if( bodyExertingAcceleration->getGravityFieldModel( ) == NULL )
215  {
216  throw std::runtime_error(
217  std::string( "Error, gravity field model not set when making central ") +
218  " gravitational acceleration of " + nameOfBodyExertingAcceleration + " on " +
219  nameOfBodyUndergoingAcceleration );
220  }
221  else
222  {
223  boost::function< double( ) > gravitationalParameterFunction;
224 
225  // Set correct value for gravitational parameter.
226  if( useCentralBodyFixedFrame == 0 ||
227  bodyUndergoingAcceleration->getGravityFieldModel( ) == NULL )
228  {
229  gravitationalParameterFunction =
230  boost::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
231  bodyExertingAcceleration->getGravityFieldModel( ) );
232  }
233  else
234  {
235  boost::function< double( ) > gravitationalParameterOfBodyExertingAcceleration =
236  boost::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
237  bodyExertingAcceleration->getGravityFieldModel( ) );
238  boost::function< double( ) > gravitationalParameterOfBodyUndergoingAcceleration =
239  boost::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
240  bodyUndergoingAcceleration->getGravityFieldModel( ) );
241  gravitationalParameterFunction =
242  boost::bind( &utilities::sumFunctionReturn< double >,
243  gravitationalParameterOfBodyExertingAcceleration,
244  gravitationalParameterOfBodyUndergoingAcceleration );
245  }
246 
247  // Create acceleration object.
248  accelerationModelPointer =
249  boost::make_shared< CentralGravitationalAccelerationModel3d >(
250  boost::bind( &Body::getPosition, bodyUndergoingAcceleration ),
251  gravitationalParameterFunction,
252  boost::bind( &Body::getPosition, bodyExertingAcceleration ),
253  useCentralBodyFixedFrame );
254  }
255 
256 
257  return accelerationModelPointer;
258 }

267  const boost::shared_ptr< AccelerationSettings > accelerationSettings,
268  const bool useCentralBodyFixedFrame )
269 {

271  boost::shared_ptr< SphericalHarmonicsGravitationalAccelerationModel > accelerationModel;
272 
273  // Dynamic cast acceleration settings to required type and check consistency.
274  boost::shared_ptr< SphericalHarmonicAccelerationSettings > sphericalHarmonicsSettings =
275  boost::dynamic_pointer_cast< SphericalHarmonicAccelerationSettings >(
276  accelerationSettings );
277  if( sphericalHarmonicsSettings == NULL )
278  {
279  throw std::runtime_error(
280  std::string( "Error, acceleration settings inconsistent ") +
281  " making sh gravitational acceleration of " + nameOfBodyExertingAcceleration +
282  " on " + nameOfBodyUndergoingAcceleration );
283  }
284  else
285  {
286  // Get pointer to gravity field of central body and cast to required type.
287  boost::shared_ptr< SphericalHarmonicsGravityField > sphericalHarmonicsGravityField =
288  boost::dynamic_pointer_cast< SphericalHarmonicsGravityField >(
289  bodyExertingAcceleration->getGravityFieldModel( ) );
290 
291  boost::shared_ptr< RotationalEphemeris> rotationalEphemeris =
292  bodyExertingAcceleration->getRotationalEphemeris( );
293  if( sphericalHarmonicsGravityField == NULL )
294  {
295  throw std::runtime_error(
296  std::string( "Error, spherical harmonic gravity field model not set when ")
297  + " making sh gravitational acceleration of " +
298  nameOfBodyExertingAcceleration +
299  " on " + nameOfBodyUndergoingAcceleration );
300  }
301  else
302  {
303  if( rotationalEphemeris == NULL )
304  {
305  throw std::runtime_error( "Warning when making spherical harmonic acceleration on body " +
306  nameOfBodyUndergoingAcceleration + ", no rotation model found for " +
307  nameOfBodyExertingAcceleration );
308  }
309 
310  if( rotationalEphemeris->getTargetFrameOrientation( ) !=
311  sphericalHarmonicsGravityField->getFixedReferenceFrame( ) )
312  {
313  throw std::runtime_error( "Warning when making spherical harmonic acceleration on body " +
314  nameOfBodyUndergoingAcceleration + ", rotation model found for " +
315  nameOfBodyExertingAcceleration + " is incompatible, frames are: " +
316  rotationalEphemeris->getTargetFrameOrientation( ) + " and " +
317  sphericalHarmonicsGravityField->getFixedReferenceFrame( ) );
318  }
319 
320  boost::function< double( ) > gravitationalParameterFunction;
321 
322  // Check if mutual acceleration is to be used.
323  if( useCentralBodyFixedFrame == false ||
324  bodyUndergoingAcceleration->getGravityFieldModel( ) == NULL )
325  {
326  gravitationalParameterFunction =
327  boost::bind( &SphericalHarmonicsGravityField::getGravitationalParameter,
328  sphericalHarmonicsGravityField );
329  }
330  else
331  {
332  // Create function returning summed gravitational parameter of the two bodies.
333  boost::function< double( ) > gravitationalParameterOfBodyExertingAcceleration =
334  boost::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
335  sphericalHarmonicsGravityField );
336  boost::function< double( ) > gravitationalParameterOfBodyUndergoingAcceleration =
337  boost::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
338  bodyUndergoingAcceleration->getGravityFieldModel( ) );
339  gravitationalParameterFunction =
340  boost::bind( &utilities::sumFunctionReturn< double >,
341  gravitationalParameterOfBodyExertingAcceleration,
342  gravitationalParameterOfBodyUndergoingAcceleration );
343  }
344 
345  // Create acceleration object.
346  accelerationModel =
347  boost::make_shared< SphericalHarmonicsGravitationalAccelerationModel >
348  ( boost::bind( &Body::getPosition, bodyUndergoingAcceleration ),
349  gravitationalParameterFunction,
350  sphericalHarmonicsGravityField->getReferenceRadius( ),
351  boost::bind( &SphericalHarmonicsGravityField::getCosineCoefficients,
352  sphericalHarmonicsGravityField,
353  sphericalHarmonicsSettings->maximumDegree_,
354  sphericalHarmonicsSettings->maximumOrder_ ),
355  boost::bind( &SphericalHarmonicsGravityField::getSineCoefficients,
356  sphericalHarmonicsGravityField,
357  sphericalHarmonicsSettings->maximumDegree_,
358  sphericalHarmonicsSettings->maximumOrder_ ),
359  boost::bind( &Body::getPosition, bodyExertingAcceleration ),
360  boost::bind( &Body::getCurrentRotationToGlobalFrame,
361  bodyExertingAcceleration ), useCentralBodyFixedFrame );
362  }
363  }
364  return accelerationModel;
365 }
