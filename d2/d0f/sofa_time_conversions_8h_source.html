62 template< typename TimeType >
63 TimeType convertTAItoUTC( const TimeType taiSeconds )
64 {
65  // Retrieve number of leap seconds from Sofa, assuming TAI=UTC
66  double deltaAt = getDeltaAtFromUtc( static_cast< double >( taiSeconds ) / physical_constants::JULIAN_DAY );
67 
68  // Update correction in case conversion is close to leap second introduction.
69  TimeType utc = taiSeconds - static_cast< TimeType >( deltaAt );
70  deltaAt = getDeltaAtFromUtc( static_cast< double >( utc ) / physical_constants::JULIAN_DAY );
71 
72  // Return converted time
73  return taiSeconds - static_cast< TimeType >( deltaAt );
74 }
