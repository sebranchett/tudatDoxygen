227  Eigen::Vector3d getTotalTorqueForBody( const std::string& bodyName )
228  {
229  // Check if body is propagated.
230  Eigen::Vector3d totalTorque = Eigen::Vector3d::Zero( );
231  if( std::find( bodiesToPropagate_.begin( ),
232  bodiesToPropagate_.end( ),
233  bodyName ) == bodiesToPropagate_.end( ) )
234  {
235  std::string errorMessage = "Error when getting total torque for body " + bodyName +
236  ", no such torque is found";
237  throw std::runtime_error( errorMessage );
238  }
239  else
240  {
241  if( torqueModelsPerBody_.count( bodyName ) != 0 )
242  {
243  basic_astrodynamics::SingleBodyTorqueModelMap torquesOnBody =
244  torqueModelsPerBody_.at( bodyName );
245 
246  // Iterate over all torques acting on body
247  for( innerTorqueIterator = torquesOnBody.begin( );
248  innerTorqueIterator != torquesOnBody.end( );
249  innerTorqueIterator++ )
250  {
251  for( unsigned int j = 0; j < innerTorqueIterator->second.size( ); j++ )
252  {
253  // Calculate torque and add to state derivative.
254  totalTorque += innerTorqueIterator->second[ j ]->getTorque( );
255  }
256  }
257  }
258  }
259  return totalTorque;
260  }
