123  const std::vector< int >& numberOfLines,

296 Vector6d HypersonicLocalInclinationAnalysis::determinePartCoefficients(
297  const int partNumber, const boost::array< int, 3 > independentVariableIndices )
298 {
299  // Declare and determine angles of attack and sideslip for analysis.
300  double angleOfAttack = dataPointsOfIndependentVariables_[ 1 ]
301  [ independentVariableIndices[ 1 ] ];
302 
303  double angleOfSideslip = dataPointsOfIndependentVariables_[ 2 ]
304  [ independentVariableIndices[ 2 ] ];
305 
306  // Declare partCoefficient vector.
307  Vector6d partCoefficients = Vector6d::Zero( );
308 
309  // Check whether the inclinations of the vehicle part have already been computed.
310  if ( previouslyComputedInclinations_.count( std::pair< double, double >(
311  angleOfAttack, angleOfSideslip ) ) == 0 )
312  {
313  // Determine panel inclinations for part.
314  determineInclinations( angleOfAttack, angleOfSideslip );
315 
316  // Add panel inclinations to container
317  previouslyComputedInclinations_[ std::pair< double, double >(
318  angleOfAttack, angleOfSideslip ) ] = inclination_;
319  }
320 
321  else
322  {
323  // Fetch inclinations from container
324  inclination_ = previouslyComputedInclinations_[ std::pair< double, double >(
325  angleOfAttack, angleOfSideslip ) ];
326  }
327 
328  // Set pressureCoefficient_ array for given independent variables.
329  determinePressureCoefficients( partNumber, independentVariableIndices );
330 
331  // Calculate force coefficients from pressure coefficients.
332  partCoefficients.segment( 0, 3 ) = calculateForceCoefficients( partNumber );
333 
334  // Calculate moment coefficients from pressure coefficients.
335  partCoefficients.segment( 3, 3 ) = calculateMomentCoefficients( partNumber );
336 
337  return partCoefficients;
338 }

358 Eigen::Vector3d HypersonicLocalInclinationAnalysis::calculateForceCoefficients(
359  const int partNumber )
360 {
361  // Declare force coefficient vector and intialize to zeros.
362  Eigen::Vector3d forceCoefficients = Eigen::Vector3d::Zero( );
363 
364  // Loop over all panels and add pressures, scaled by panel area, to force
365  // coefficients.
366  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
367  {
368  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++)
369  {
370  forceCoefficients -=
371  pressureCoefficient_[ partNumber ][ i ][ j ] *
372  vehicleParts_[ partNumber ]->getPanelArea( i, j ) *
373  vehicleParts_[ partNumber ]->getPanelSurfaceNormal( i, j );
374  }
375  }
376 
377  // Normalize result by reference area.
378  forceCoefficients /= referenceArea_;
379 
380  return forceCoefficients;
381 }

384 Eigen::Vector3d HypersonicLocalInclinationAnalysis::calculateMomentCoefficients(
385  const int partNumber )
386 {
387  // Declare moment coefficient vector and intialize to zeros.
388  Eigen::Vector3d momentCoefficients = Eigen::Vector3d::Zero( );
389 
390  // Declare moment arm for panel moment determination.
391  Eigen::Vector3d referenceDistance ;
392 
393  // Loop over all panels and add moments due pressures.
394  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
395  {
396  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++ )
397  {
398  // Determine moment arm for given panel centroid.
399  referenceDistance = ( vehicleParts_[ partNumber ]->
400  getPanelCentroid( i, j ) - momentReferencePoint_ );
401 
402  momentCoefficients -=
403  pressureCoefficient_[ partNumber ][ i ][ j ] *
404  vehicleParts_[ partNumber ]->getPanelArea( i, j ) *
405  ( referenceDistance.cross( vehicleParts_[ partNumber ]->
406  getPanelSurfaceNormal( i, j ) ) );
407  }
408  }
409 
410  // Scale result by reference length and area.
411  momentCoefficients /= ( referenceLength_ * referenceArea_ );
412 
413  return momentCoefficients;
414 }

417 void HypersonicLocalInclinationAnalysis::determineInclinations( const double angleOfAttack,
418  const double angleOfSideslip )
419 {
420  // Declare free-stream velocity vector.
421  Eigen::Vector3d freestreamVelocityDirection;
422 
423  // Set freestream velocity vector in body frame.
424  double freestreamVelocityDirectionX = cos( angleOfAttack )* cos( angleOfSideslip );
425  double freestreamVelocityDirectionY = sin( angleOfSideslip );
426  double freestreamVelocityDirectionZ = sin( angleOfAttack ) * cos( angleOfSideslip );
427  freestreamVelocityDirection( 0 ) = freestreamVelocityDirectionX;
428  freestreamVelocityDirection( 1 ) = freestreamVelocityDirectionY;
429  freestreamVelocityDirection( 2 ) = freestreamVelocityDirectionZ;
430 
431  // Declare cosine of inclination angle.
432  double cosineOfInclination;
433 
434  // Loop over all panels of given vehicle part and set inclination angles.
435  for( unsigned int k = 0; k < vehicleParts_.size( ); k++ )
436  {
437  for ( int i = 0 ; i < vehicleParts_[ k ]->getNumberOfLines( ) - 1 ; i++ )
438  {
439  for ( int j = 0 ; j < vehicleParts_[ k ]->getNumberOfPoints( ) - 1 ; j++ )
440  {
441 
442  // Determine cosine of inclination angle from inner product between
443  // surface normal and free-stream direction.
444  cosineOfInclination = vehicleParts_[ k ]->
445  getPanelSurfaceNormal( i, j ).
446  dot( freestreamVelocityDirection );
447 
448  // Set inclination angle.
449  inclination_[ k ][ i ][ j ] = PI / 2.0 - acos( cosineOfInclination );
450  }
451  }
452  }
453 }

470 
471  case 1:

475  break;
476 
477  case 2:
478  // Method currently disabled.
479  break;
480 
481  case 3:
482  // Method currently disabled.
483  break;
484 
485  case 4:
486  pressureFunction =
487  boost::bind( aerodynamics::computeEmpiricalTangentWedgePressureCoefficient, _1,
488  machNumber );
489  break;
490 
491  case 5:
492  pressureFunction =
493  boost::bind( aerodynamics::computeEmpiricalTangentConePressureCoefficient, _1,
494  machNumber );
495  break;
496 
497  case 6:
498  pressureFunction =
499  boost::bind( aerodynamics::computeModifiedDahlemBuckPressureCoefficient, _1,
500  machNumber );
501  break;
502 
503  case 7:
504  pressureFunction =
505  boost::bind( aerodynamics::computeVanDykeUnifiedPressureCoefficient, _1,
506  machNumber, ratioOfSpecificHeats, 1 );
507  break;

513  break;
514 
515  case 9:
516  pressureFunction =
517  boost::bind( aerodynamics::computeHankeyFlatSurfacePressureCoefficient, _1,

525  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1; i++ )

540 void HypersonicLocalInclinationAnalysis::updateExpansionPressures( const double machNumber,
541  const int partNumber )
542 {
543  // Get analysis method of part to analyze.
544  int method = selectedMethods_[ 1 ][ partNumber ];
545 
546  if ( method == 0 || method == 1 || method == 4 )
547  {
548  boost::function< double( ) > pressureFunction;
549  switch( method )
550  {
551  case 0:
552  pressureFunction = boost::bind( &aerodynamics::computeVacuumPressureCoefficient,
553  machNumber, ratioOfSpecificHeats );
554  break;
555 
556  case 1:
557  pressureFunction = boost::lambda::constant( 0.0 );
558  break;
559 
560  case 4:
561  pressureFunction = boost::bind( &aerodynamics::computeHighMachBasePressure,
562  machNumber );
563  break;
564 
565  }
566 
567  // Iterate over all panels on part.
568  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
569  {
570  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++ )
571  {
572  if ( inclination_[ partNumber ][ i ][ j ] <= 0 )
573  {
574  // If panel inclination is negative, calculate pressure using
575  // Van Dyke unified method.
576  pressureCoefficient_[ partNumber ][ i ][ j ] =
577  pressureFunction( );
578  }
579  }
580  }
581  }
582 
583  else if( method == 3 || method == 5 || method == 6 )
584  {
585 
586  boost::function< double( double ) > pressureFunction;
587 
588  // Declare local variable.
589  double freestreamPrandtlMeyerFunction;
590 
591  // Switch to analyze part using correct method.
592  switch( method )
593  {
594  case 3:
595  // Calculate freestream Prandtl-Meyer function.
596  freestreamPrandtlMeyerFunction = aerodynamics::computePrandtlMeyerFunction(
597  machNumber, ratioOfSpecificHeats );
598  pressureFunction =
599  boost::bind( &aerodynamics::computePrandtlMeyerFreestreamPressureCoefficient,
600  _1, machNumber, ratioOfSpecificHeats,
601  freestreamPrandtlMeyerFunction );
602  break;
603 
604  case 5:
605  pressureFunction =
606  boost::bind( &aerodynamics::computePrandtlMeyerFreestreamPressureCoefficient,
607  _1, machNumber, ratioOfSpecificHeats, -1 );
608  break;
609 
610  case 6:
611  pressureFunction = boost::bind( &aerodynamics::computeAcmEmpiricalPressureCoefficient,
612  _1, machNumber );
613  break;
614  }
615 
616  // Iterate over all panels on part.
617  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
618  {
619  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++ )
620  {
621  if ( inclination_[ partNumber ][ i ][ j ] <= 0 )
622  {
623  // If panel inclination is negative, calculate pressure using
624  // Van Dyke unified method.
625  pressureCoefficient_[ partNumber ][ i ][ j ] =
626  pressureFunction( inclination_[ partNumber ][ i ][ j ] );
627  }
628  }
629  }
630  }
631 
632  else
633  {
634  std::string errorMessage = "Error, expansion local inclination method number "
635  + boost::lexical_cast< std::string >( method ) + " not recognized";
636  throw std::runtime_error( errorMessage );
637  }
638 }
