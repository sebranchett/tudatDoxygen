66 void LegendreCache::update( const double polynomialParameter )
67 {
68  // Check if cache needs update
69  if( !( polynomialParameter == currentPolynomialParameter_ ) )
70  {
71  currentPolynomialParameter_ = polynomialParameter;
72 
73  // Set complement of argument (assuming it to be sine of latitude) cosine of latitude is always positive.
74  currentPolynomialParameterComplement_ = std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
75 
76  LegendreCache& thisReference = *this;
77 
78  int jMax = -1;
79  for( int i = 0; i <= maximumDegree_; i++ )
80  {
81  jMax = std::min( i, maximumOrder_ );
82  for( int j = 0; j <= jMax ; j++ )
83  {
84  // Compute legendre polynomial
85  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ] = legendrePolynomialFunction_( i, j, thisReference );
86 
87  if( j != 0 )
88  {
89  // Compute legendre polynomial derivative
90  if( useGeodesyNormalization_ )
91  {
92  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
93  computeGeodesyLegendrePolynomialDerivative(
94  i, j - 1, currentPolynomialParameter_,
95  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
96  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ],
97  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] );
98  }
99  else
100  {
101  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
102  computeLegendrePolynomialDerivative(
103  j - 1, currentPolynomialParameter_,
104  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
105  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ] );
106  }
107 
108  }
109  }
110 
111  // Compute legendre polynomial derivative for i = j (if needed)
112  if( jMax == i )
113  {
114  if( useGeodesyNormalization_ )
115  {
116  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
117  computeGeodesyLegendrePolynomialDerivative(
118  i, jMax, currentPolynomialParameter_,
119  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
120  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + jMax ] );
121  }
122  else
123  {
124  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
125  computeLegendrePolynomialDerivative(
126  jMax, currentPolynomialParameter_,
127  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0 );
128  }
129  }
130  }
131 
132  // Compute second derivatives of Legendre polynomials if needed
133  if( computeSecondDerivatives_ )
134  {
135  for( int i = 0; i <= maximumDegree_; i++ )
136  {
137  jMax = std::min( i, maximumOrder_ );
138  for( int j = 0; j <= jMax ; j++ )
139  {
140  if( j != 0 )
141  {
142  // Compute legendre polynomial second derivatives
143  if( useGeodesyNormalization_ )
144  {
145  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
146  computeGeodesyLegendrePolynomialSecondDerivative(
147  i, j - 1, currentPolynomialParameter_,
148  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
149  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ],
150  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
151  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + j ],
152  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] );
153  }
154  else
155  {
156  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
157  computeGeodesyLegendrePolynomialSecondDerivative(
158  i, j - 1, currentPolynomialParameter_,
159  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
160  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ],
161  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
162  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + j ], 1.0 );
163  }
164 
165  }
166  }
167  // Compute legendre polynomial second derivative for i = j (if needed)
168  if( jMax == i )
169  {
170  if( useGeodesyNormalization_ )
171  {
172  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
173  computeGeodesyLegendrePolynomialSecondDerivative(
174  i, jMax, currentPolynomialParameter_,
175  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
176  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
177  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + jMax ] );
178  }
179  else
180  {
181  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
182  computeGeodesyLegendrePolynomialSecondDerivative(
183  i, jMax, currentPolynomialParameter_,
184  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
185  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
186  1.0 );
187  }
188  }

196 void LegendreCache::resetMaximumDegreeAndOrder( const int maximumDegree, const int maximumOrder )
197 {
198  maximumDegree_ = maximumDegree;
199  maximumOrder_ = maximumOrder;
200 
201  if( maximumOrder_ > maximumDegree_ )
202  {
203  maximumOrder_ = maximumDegree_;
204  }
205 
206  legendreValues_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
207  legendreDerivatives_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
208  legendreSecondDerivatives_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
209 
210  derivativeNormalizations_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
211 
212  for( int i = 0; i <= maximumDegree_; i++ )
213  {
214  for( int j = 0; ( ( j <= i ) && ( j <= maximumOrder_ ) ) ; j++ )
215  {
216  // Compute normalization correction factor.
217  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + j ] = std::sqrt(
218  ( static_cast< double >( i + j + 1 ) )
219  * ( static_cast< double >( i - j ) ) );
220 
221  // If order is zero apply multiplication factor.
222  if ( j == 0 )
223  {
224  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + j ] *= std::sqrt( 0.5 );
225  }
226  }
227  }
228 
229  currentPolynomialParameter_ = TUDAT_NAN;
230  currentPolynomialParameterComplement_ = TUDAT_NAN;
231 }

311 double computeLegendrePolynomialFromCache( const int degree,
312  const int order,
313  LegendreCache& legendreCache )
314 {
315  if( legendreCache.getUseGeodesyNormalization( ) )
316  {
317  throw std::runtime_error( "Error when computing Legendre polynomial, input uses normalization" );
318  }
319 
320  // If degree or order is negative...
321  if ( degree < 0 || order < 0 )
322  {
323  // Set error message.
324  std::stringstream errorMessage;
325  errorMessage << "Error: the Legendre polynomial of = " << degree << " and order = "
326  << order << " is undefined." << std::endl;
327 
328  // Throw a run-time error.
329  throw std::runtime_error( errorMessage.str( ) );
330  }
331 
332  // Else if order is greater than degree...
333  else if ( order > degree && degree >= 0 )
334  {
335  // Return zero.
336  return 0.0;
337  }
338 
339  // Else if order and degree are lower than 2...
340  else if ( degree <= 1 && order <= 1 )
341  {
342  // Compute polynomial explicitly.
343  return computeLegendrePolynomialExplicit( degree, order, legendreCache.getCurrentPolynomialParameter( ) );
344  }
345 
346  // Else if degree and order are sectoral...
347  else if ( degree == order )
348  {
349  // Obtain polynomial of degree one and order one.
350  const double degreeOneOrderOnePolynomial = legendreCache.getLegendrePolynomial(
351  1, 1 );
352 
353  // Obtain prior sectoral polynomial.
354  const double priorSectoralPolynomial = legendreCache.getLegendrePolynomial(
355  degree - 1, order - 1 );
356 
357  // Compute polynomial.
358  return computeLegendrePolynomialDiagonal(
359  degree, degreeOneOrderOnePolynomial, priorSectoralPolynomial );
360  }
361 
362  // Else degree and order are zonal/tessoral...
363  else
364  {
365  // Obtain prior degree polynomial.
366  const double oneDegreePriorPolynomial = legendreCache.getLegendrePolynomial(
367  degree - 1, order );
368 
369  // Obtain two degrees prior polynomial.
370  const double twoDegreesPriorPolynomial = legendreCache.getLegendrePolynomial(
371  degree - 2, order );
372 
373  // Compute polynomial.
374  return computeLegendrePolynomialVertical( degree,
375  order,
376  legendreCache.getCurrentPolynomialParameter( ),
377  oneDegreePriorPolynomial,
378  twoDegreesPriorPolynomial );
379  }
380 }

508  const int order,

522  }

531 double computeGeodesyLegendrePolynomialSecondDerivative( const int degree,
532  const int order,
533  const double polynomialParameter,
534  const double currentLegendrePolynomial,
535  const double incrementedLegendrePolynomial,
536  const double currentLegendrePolynomialDerivative,
537  const double incrementedLegendrePolynomialDerivative,
538  const double normalizationCorrection )
539 {
540  double polynomialParameterSquare = polynomialParameter * polynomialParameter;
541 
542  // Return polynomial derivative.
543  return normalizationCorrection * (
544  incrementedLegendrePolynomialDerivative / std::sqrt( 1.0 - polynomialParameterSquare ) +
545  polynomialParameter * std::pow( 1.0 - polynomialParameterSquare, -1.5 ) * incrementedLegendrePolynomial ) -
546  static_cast< double >( order ) *
547  ( polynomialParameter / ( 1.0 - polynomialParameter * polynomialParameter ) * currentLegendrePolynomialDerivative +
548  ( 1.0 + polynomialParameterSquare ) / ( ( 1.0 - polynomialParameterSquare ) * ( 1.0 - polynomialParameterSquare ) ) * currentLegendrePolynomial );
549 }

553 double computeLegendrePolynomialExplicit( const int degree,
554  const int order,
555  const double polynomialParameter )
556 {
557  // Check which order is required for Legendre polynomial.
558  switch( degree )
559  {
560  case 0:
561  switch( order )
562  {
563  case 0:
564  return 1.0;
565  default:
566  {
567  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
568  boost::lexical_cast< std::string >( degree ) + ", " +
569  boost::lexical_cast< std::string >( order );
570  throw std::runtime_error( errorMessage );
571  }
572  }
573  break;
574  case 1:
575  switch( order )
576  {
577  case 0:
578  return polynomialParameter;
579  case 1:
580  return std::sqrt( 1 - polynomialParameter * polynomialParameter );
581  default:
582  {
583  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
584  boost::lexical_cast< std::string >( degree ) + ", " +
585  boost::lexical_cast< std::string >( order );
586  throw std::runtime_error( errorMessage );
587  }
588  }
589  break;
590  case 2:
591  switch( order )
592  {
593  case 0:
594  return 0.5 * ( 3.0 * polynomialParameter * polynomialParameter - 1.0 );
595  case 1:
596  return 3.0 * polynomialParameter
597  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
598  case 2:
599  return 3.0 * ( 1.0 - polynomialParameter * polynomialParameter );
600  default:
601  {
602  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
603  boost::lexical_cast< std::string >( degree ) + ", " +
604  boost::lexical_cast< std::string >( order );
605  throw std::runtime_error( errorMessage );
606  }
607  }
608  break;
609  case 3:
610  switch( order )
611  {
612  case 0:
613  return 0.5 * polynomialParameter
614  * ( 5.0 * polynomialParameter * polynomialParameter - 3.0 );
615  case 1:
616  return 1.5 * ( 5.0 * polynomialParameter * polynomialParameter - 1.0 )
617  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
618  case 2:
619  return 15.0 * polynomialParameter * ( 1.0 - polynomialParameter * polynomialParameter );
620  case 3:
621  return 15.0 * ( 1.0 - polynomialParameter * polynomialParameter )
622  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
623  default:
624  {
625  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
626  boost::lexical_cast< std::string >( degree ) + ", " +
627  boost::lexical_cast< std::string >( order );
628  throw std::runtime_error( errorMessage );
629  }
630  }
631  break;
632  case 4:
633  switch( order )
634  {
635  case 0:
636  return ( 35.0 * polynomialParameter * polynomialParameter
637  * polynomialParameter * polynomialParameter
638  - 30.0 * polynomialParameter * polynomialParameter + 3.0 ) / 8.0;
639  case 1:
640  return -2.5 * ( 7.0 * polynomialParameter * polynomialParameter * polynomialParameter
641  - 3.0 * polynomialParameter )
642  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
643  case 2:
644  return 15.0 / 2.0 * ( - 1.0 + 7.0 * polynomialParameter * polynomialParameter )
645  * ( 1.0 - polynomialParameter * polynomialParameter );
646  case 3:
647  return -105.0 * polynomialParameter * ( 1.0 - polynomialParameter * polynomialParameter )
648  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
649  case 4:
650  return 105.0 * ( 1.0 - polynomialParameter * polynomialParameter )
651  * ( 1.0 - polynomialParameter * polynomialParameter );
652 
653  default:
654  {
655  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
656  boost::lexical_cast< std::string >( degree ) + ", " +
657  boost::lexical_cast< std::string >( order );
658  throw std::runtime_error( errorMessage );
659  }
660  }
661  break;
662  default:
663  {
664  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
665  boost::lexical_cast< std::string >( degree ) + ", " +
666  boost::lexical_cast< std::string >( order );
667  throw std::runtime_error( errorMessage );
668  }
669  }
670  return TUDAT_NAN;
671 }

674 double computeGeodesyLegendrePolynomialExplicit( const int degree,
675  const int order,
676  const double polynomialParameter )
677 {
678  // If 0,0 term is requested return Legendre polynomial value.
679  if ( degree == 0 && order == 0 )
680  {
681  return 1.0;
682  }
683 
684  // Else if 1,0 term is requested return polynomial value.
685  else if ( degree == 1 && order == 0 )
686  {
687  return std::sqrt( 3.0 ) * polynomialParameter;
688  }
689 
690  // Else if 1,1 term is requested return polynomial value.
691  else if ( degree == 1 && order == 1 )
692  {

736  const double twoDegreesPriorPolynomial )

755  - std::sqrt( ( static_cast< double >( degree + order ) - 1.0 )

763 {
764 
765  double deltaFunction = 0.0;

767  {
768  deltaFunction = 1.0;
769  }
770

772  boost::math::factorial< double >( static_cast< double >( degree + order ) )
773  / ( ( 2.0 - deltaFunction ) * ( 2.0 * static_cast< double >( degree ) + 1.0 )
774  * boost::math::factorial< double >( static_cast< double >( degree - order ) ) ) );
