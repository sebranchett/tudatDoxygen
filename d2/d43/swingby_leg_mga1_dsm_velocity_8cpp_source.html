20  double& deltaV )
21 {
22  // Calculate the DSM time of application from the time of flight fraction.
23  dsmTime_ = dsmTimeOfFlightFraction_ * timeOfFlight_;
24 
25  // Prepare the gravity assist propagator module.
26  velocityAfterDeparture_ = mission_segments::gravityAssist( swingbyBodyGravitationalParameter_,
27  departureBodyVelocity_,
28  ( *velocityBeforeDepartureBodyPtr_ ),
29  rotationAngle_, pericenterRadius_,

32  // Transfer the initial position and velocity into a vectorXd object with Cartesian
33  // coordinates.
34  Eigen::Vector6d cartesianElements ( 6 ), keplerianElements ( 6 );
35  cartesianElements.segment( 0, 3 ) = departureBodyPosition_;
36  cartesianElements.segment( 3, 3 ) = velocityAfterDeparture_;

39  keplerianElements = orbital_element_conversions::convertCartesianToKeplerianElements(
40  cartesianElements, centralBodyGravitationalParameter_ );
41 
42  // Propagate the keplerian elements until the moment of application of the DSM.
43  keplerianElements = orbital_element_conversions::propagateKeplerOrbit( keplerianElements,
44  dsmTime_, centralBodyGravitationalParameter_ );
45 
46  // Convert the keplerian elements back into Cartesian elements.
47  cartesianElements = orbital_element_conversions::convertKeplerianToCartesianElements(
48  keplerianElements, centralBodyGravitationalParameter_ );
49 
50  // Set the corresponding position and velocity vectors.
51  dsmLocation_ = cartesianElements.segment( 0, 3 );
52  velocityBeforeDsm_ = cartesianElements.segment( 3, 3 );
53

56  dsmTime_, centralBodyGravitationalParameter_,
57  velocityAfterDsm_, velocityBeforeArrivalBody );

60  deltaVDsm_ = ( velocityAfterDsm_ - velocityBeforeDsm_ ).norm( );
61

63  deltaV_ = swingbyDeltaV_ + deltaVDsm_;
