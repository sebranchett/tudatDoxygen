36 template< typename StateScalarType, typename TimeType >
37 std::map< propagators::IntegratedStateType, orbit_determination::StateDerivativePartialsMap > createStateDerivativePartials(
38  const std::unordered_map< propagators::IntegratedStateType,
39  std::vector< std::shared_ptr< propagators::SingleStateTypeDerivative< StateScalarType, TimeType > > > >
40  stateDerivativeModels,
41  const simulation_setup::NamedBodyMap& bodyMap,
42  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< StateScalarType > >
43  parametersToEstimate )
44 {
45  std::map< propagators::IntegratedStateType, orbit_determination::StateDerivativePartialsMap > stateDerivativePartials;
46 
47  // Iterate over all state types
48  for( typename std::unordered_map< propagators::IntegratedStateType,
49  std::vector< std::shared_ptr< propagators::SingleStateTypeDerivative< StateScalarType, TimeType > > > >::
50  const_iterator stateDerivativeIterator = stateDerivativeModels.begin( );
51  stateDerivativeIterator != stateDerivativeModels.end( );
52  stateDerivativeIterator++ )
53  {
54  // Identify state types
55  switch( stateDerivativeIterator->first )
56  {
57  case propagators::translational_state:
58  {
59  if( stateDerivativeIterator->second.size( ) > 1 )
60  {
61  throw std::runtime_error(
62  "Error, cannot yet process multiple separate same type propagators when making partial "
63  "derivatives of translational state." );
64  }
65  else
66  {
67  // Retrieve acceleration models and create partials
68  basic_astrodynamics::AccelerationMap accelerationModelList =
69  std::dynamic_pointer_cast< propagators::NBodyStateDerivative< StateScalarType, TimeType > >(
70  stateDerivativeIterator->second.at( 0 ) )->getFullAccelerationsMap( );
71  stateDerivativePartials[ propagators::translational_state ] =
72  createAccelerationPartialsMap< StateScalarType >(
73  accelerationModelList, bodyMap, parametersToEstimate );
74  }
75  break;
76  }
77  default:
78  std::string errorMessage = "Cannot yet create state derivative partial models for type " +
79  std::to_string( stateDerivativeIterator->first );
80  throw std::runtime_error( errorMessage );
81  }
82  }
83  return stateDerivativePartials;
84 }
