22 Eigen::Vector7d computeStateDerivativeForUnifiedStateModelModifiedRodriguesParameters(
23  const Eigen::Vector7d& currentUnifiedStateModelElements,
24  const Eigen::Vector3d& accelerationsInRswFrame,
25  const double sineLambda,
26  const double cosineLambda,
27  const double gammaParameter,
28  const Eigen::Vector3d& rotationalVelocityVector,
29  const Eigen::Vector3d& pAuxiliaryVector )
30 {
31  // Compute supporting parameters
32  Eigen::Matrix3d hodographMatrix = Eigen::Matrix3d::Zero( );
33  hodographMatrix( 0, 1 ) = - pAuxiliaryVector( 0 );
34  hodographMatrix( 1, 0 ) = cosineLambda;
35  hodographMatrix( 1, 1 ) = - ( 1.0 + pAuxiliaryVector( 0 ) ) * sineLambda;
36  hodographMatrix( 1, 2 ) = - gammaParameter * pAuxiliaryVector( 1 );
37  hodographMatrix( 2, 0 ) = sineLambda;
38  hodographMatrix( 2, 1 ) = ( 1.0 + pAuxiliaryVector( 0 ) ) * cosineLambda;
39  hodographMatrix( 2, 2 ) = gammaParameter * pAuxiliaryVector( 2 );
40 
41  // Evaluate USM6 equations.
42  Eigen::Vector7d stateDerivative;
43  stateDerivative.segment( 0, 3 ) = hodographMatrix * accelerationsInRswFrame;
44  stateDerivative.segment( 3, 4 ) = calculateModifiedRodriguesParametersDerivative(
45  currentUnifiedStateModelElements.segment( 3, 4 ), rotationalVelocityVector );
46 
47  // Give output
48  return stateDerivative;
49 }

52 Eigen::Vector7d computeStateDerivativeForUnifiedStateModelModifiedRodriguesParameters(
53  const Eigen::Vector7d& currentUnifiedStateModelElements,
54  const Eigen::Vector3d& accelerationsInRswFrame,
55  const double centralBodyGravitationalParameter )
56 {
57  using namespace orbital_element_conversions;
58 
59  // Compute auxiliary parameters
60  Eigen::Vector3d modifiedRodriguesParametersVector = currentUnifiedStateModelElements.segment( sigma1USM6Index, 3 );
61  double modifiedRodriguesParametersMagnitude = modifiedRodriguesParametersVector.norm( );
62  // magnitude of modified rodrigues parameters, also called sigma
63 
64  // Precompute often used variables
65  // Note the for SMRP some variable names do not match their definitions
66  double modifiedRodriguesParametersMagnitudeSquared =
67  modifiedRodriguesParametersMagnitude * modifiedRodriguesParametersMagnitude;
68  double oneMinusModifiedRodriguesParametersMagnitudeSquaredSquared =
69  std::pow( 1.0 - modifiedRodriguesParametersMagnitudeSquared, 2 );
70  double sigma1ModifiedRodriguesParametersSquared = std::pow( modifiedRodriguesParametersVector( 0 ), 2 );
71  double sigma2ModifiedRodriguesParametersSquared = std::pow( modifiedRodriguesParametersVector( 1 ), 2 );
72  double sigma3ModifiedRodriguesParametersSquared = std::pow( modifiedRodriguesParametersVector( 2 ), 2 );
73 
74  // Compute supporting parameters
75  double denominator = 4.0 * sigma3ModifiedRodriguesParametersSquared +
76  oneMinusModifiedRodriguesParametersMagnitudeSquaredSquared; // denominator is never null
77  double sineLambda = 4.0 * modifiedRodriguesParametersVector( 2 ) *
78  ( 1.0 - modifiedRodriguesParametersMagnitudeSquared ) / denominator;
79  double cosineLambda = ( oneMinusModifiedRodriguesParametersMagnitudeSquaredSquared -
80  4.0 * sigma3ModifiedRodriguesParametersSquared ) / denominator;
81  double gammaParameter = 2.0 * ( modifiedRodriguesParametersVector( 1 ) * ( modifiedRodriguesParametersMagnitudeSquared - 1.0 ) +
82  2.0 * modifiedRodriguesParametersVector( 0 ) * modifiedRodriguesParametersVector( 2 ) ) /
83  ( std::pow( sigma1ModifiedRodriguesParametersSquared, 2 ) + std::pow( sigma2ModifiedRodriguesParametersSquared, 2 ) +
84  std::pow( sigma3ModifiedRodriguesParametersSquared, 2 ) + 2.0 *
85  ( sigma1ModifiedRodriguesParametersSquared * sigma2ModifiedRodriguesParametersSquared +
86  sigma1ModifiedRodriguesParametersSquared * sigma3ModifiedRodriguesParametersSquared +
87  sigma2ModifiedRodriguesParametersSquared * sigma3ModifiedRodriguesParametersSquared -
88  modifiedRodriguesParametersMagnitudeSquared + 2.0 * sigma3ModifiedRodriguesParametersSquared ) + 1.0 );
89 
90  double velocityHodographParameter = currentUnifiedStateModelElements( CHodographUSM6Index ) -
91  currentUnifiedStateModelElements( Rf1HodographUSM6Index ) * sineLambda +
92  currentUnifiedStateModelElements( Rf2HodographUSM6Index ) * cosineLambda;
93  Eigen::Vector3d rotationalVelocityVector = Eigen::Vector3d::Zero( );
94  rotationalVelocityVector( 0 ) = accelerationsInRswFrame( 2 ) / velocityHodographParameter;
95  rotationalVelocityVector( 2 ) = std::pow( velocityHodographParameter, 2 ) *
96  currentUnifiedStateModelElements( CHodographUSM6Index ) / centralBodyGravitationalParameter;
97 
98  Eigen::Vector3d pAuxiliaryVector = Eigen::Vector3d::Zero( );
99  pAuxiliaryVector( 0 ) = currentUnifiedStateModelElements( CHodographUSM6Index );
100  pAuxiliaryVector( 1 ) = currentUnifiedStateModelElements( Rf2HodographUSM6Index );
101  pAuxiliaryVector( 2 ) = currentUnifiedStateModelElements( Rf1HodographUSM6Index );
102  pAuxiliaryVector /= velocityHodographParameter;
103 
104  // Evaluate USM6 equations
105  return computeStateDerivativeForUnifiedStateModelModifiedRodriguesParameters(
106  currentUnifiedStateModelElements, accelerationsInRswFrame, sineLambda,
107  cosineLambda, gammaParameter, rotationalVelocityVector, pAuxiliaryVector );
108 }
