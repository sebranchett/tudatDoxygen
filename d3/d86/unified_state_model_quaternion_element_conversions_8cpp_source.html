31 Eigen::Vector7d convertKeplerianToUnifiedStateModelQuaternionsElements(
32  const Eigen::Vector6d& keplerianElements,
33  const double centralBodyGravitationalParameter )
34 {
35  using mathematical_constants::PI;
36 
37  // Declaring eventual output vector.
38  Eigen::Vector7d convertedUnifiedStateModelElements = Eigen::Vector7d::Zero( );
39 
40  // Define the tolerance of a singularity
41  const double singularityTolerance = 20.0 * std::numeric_limits< double >::epsilon( );
42 
43  // If eccentricity is outside range [0,inf)
44  if ( keplerianElements( eccentricityIndex ) < 0.0 )
45  {
46  //Define the error message
47  std::stringstream errorMessage;
48  errorMessage << "Eccentricity is expected in range [0,inf)\n"
49  << "Specified eccentricity: " << keplerianElements( eccentricityIndex ) << std::endl;
50 
51  // Throw exception
52  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
53  }
54 
55  // If inclination is outside range [0,PI]
56  if ( ( keplerianElements( inclinationIndex ) < 0.0 ) || ( keplerianElements( inclinationIndex ) > PI ) )
57  {
58  // Define the error message.
59  std::stringstream errorMessage;
60  errorMessage << "Inclination is expected in range [0," << PI << "]\n"
61  << "Specified inclination: " << keplerianElements( inclinationIndex ) << " rad." << std::endl;
62 
63  // Throw exception.
64  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
65  }
66 
67  // If argument of pericenter is outside range [0,2.0 * PI]
68  if ( ( keplerianElements( argumentOfPeriapsisIndex ) < 0.0 ) || ( keplerianElements( argumentOfPeriapsisIndex ) >
69  2.0 * PI ) )
70  {
71  // Define the error message.
72  std::stringstream errorMessage;
73  errorMessage << "Argument of periapsis is expected in range [0," << 2.0 * PI << "]\n"
74  << "Specified inclination: " << keplerianElements( argumentOfPeriapsisIndex ) << " rad." << std::endl;
75 
76  // Throw exception.
77  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
78  }
79 
80  // If right ascension of ascending node is outside range [0,2.0 * PI]
81  if ( ( keplerianElements( longitudeOfAscendingNodeIndex ) < 0.0 ) ||
82  ( keplerianElements( longitudeOfAscendingNodeIndex ) > 2.0 * PI ) )
83  {
84  // Define the error message.
85  std::stringstream errorMessage;
86  errorMessage << "RAAN is expected in range [0," << 2.0 * PI << "]\n"
87  << "Specified inclination: " << keplerianElements( longitudeOfAscendingNodeIndex ) << " rad."
88  << std::endl;
89 
90  // Throw exception.
91  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
92  }
93 
94  // If true anomaly is outside range [0,2.0 * PI]
95  if ( ( keplerianElements( trueAnomalyIndex ) < 0.0 ) || ( keplerianElements( trueAnomalyIndex ) > 2.0 * PI ) )
96  {
97  // Define the error message.
98  std::stringstream errorMessage;
99  errorMessage << "True anomaly is expected in range [0," << 2.0 * PI << "]\n"
100  << "Specified inclination: " << keplerianElements( trueAnomalyIndex ) << " rad." << std::endl;
101 
102  // Throw exception.
103  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
104  }
105 
106  // If inclination is zero and the right ascension of ascending node is non-zero
107  if ( ( std::fabs( keplerianElements( inclinationIndex ) ) < singularityTolerance ) &&
108  ( std::fabs( keplerianElements( longitudeOfAscendingNodeIndex ) ) > singularityTolerance ) )
109  {
110  // Define the error message.
111  std::stringstream errorMessage;
112  errorMessage << "When the inclination is zero, the right ascending node should be zero by definition\n"
113  << "Specified right ascension of ascending node: " <<
114  keplerianElements( longitudeOfAscendingNodeIndex ) << " rad." << std::endl;
115 
116  // Throw exception.
117  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
118  }
119 
120  // If eccentricity is zero and the argument of pericenter is non-zero
121  if ( ( std::fabs( keplerianElements( eccentricityIndex ) ) < singularityTolerance ) &&
122  ( std::fabs( keplerianElements( argumentOfPeriapsisIndex ) ) > singularityTolerance ) )
123  {
124  // Define the error message.
125  std::stringstream errorMessage;
126  errorMessage << "When the eccentricity is zero, the argument of pericenter should be zero by definition\n"
127  << "Specified argument of pericenter: " <<
128  keplerianElements( argumentOfPeriapsisIndex ) << " rad." << std::endl;
129 
130  // Throw exception.
131  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
132  }
133 
134  // If semi-major axis is negative and the eccentricity is smaller or equal to one
135  if ( ( keplerianElements( semiMajorAxisIndex ) < 0.0 ) && ( keplerianElements( eccentricityIndex ) <= 1.0 ) )
136  {
137  // Define the error message.
138  std::stringstream errorMessage;
139  errorMessage << "When the semi-major axis is negative, the eccentricity should be larger than one\n"
140  << "Specified semi-major axis: " << keplerianElements( semiMajorAxisIndex ) << " m.\n"
141  << "Specified eccentricity: " << keplerianElements( eccentricityIndex ) << " rad." << std::endl;
142 
143  // Throw exception.
144  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
145  }
146 
147  // If semi-major axis is positive and the eccentricity is larger than one
148  if ( ( keplerianElements( semiMajorAxisIndex ) > 0.0 ) && ( keplerianElements( eccentricityIndex ) > 1.0 ) )
149  {
150  // Define the error message.
151  std::stringstream errorMessage;
152  errorMessage << "When the semi-major axis is positive, the eccentricity should be smaller than or equal to one\n"
153  << "Specified semi-major axis: " << keplerianElements( semiMajorAxisIndex ) << " m.\n"
154  << "Specified eccentricity: " << keplerianElements( eccentricityIndex ) << " rad." << std::endl;
155 
156  // Throw exception.
157  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
158  }
159  // Else, nothing wrong and continue
160 
161  // Compute the C hodograph element of the unified state model
162  if ( std::fabs( keplerianElements( eccentricityIndex ) - 1.0) < singularityTolerance )
163  // parabolic orbit -> semi-major axis is not defined
164  {
165  convertedUnifiedStateModelElements( CHodographUSM7Index ) =
166  std::sqrt( centralBodyGravitationalParameter / keplerianElements( semiLatusRectumIndex ) );
167  }
168  else
169  {
170  convertedUnifiedStateModelElements( CHodographUSM7Index ) =
171  std::sqrt( centralBodyGravitationalParameter / ( keplerianElements( semiMajorAxisIndex )
172  * ( 1 - keplerianElements( eccentricityIndex ) *
173  keplerianElements( eccentricityIndex ) ) ) );
174  }
175 
176  // Calculate the additional R hodograph parameter
177  double RHodographElement = keplerianElements( eccentricityIndex ) *
178  convertedUnifiedStateModelElements( CHodographUSM7Index );
179 
180  // Compute the Rf1 hodograph element of the unified state model
181  convertedUnifiedStateModelElements( Rf1HodographUSM7Index ) =
182  - RHodographElement * std::sin( keplerianElements( longitudeOfAscendingNodeIndex )
183  + keplerianElements( argumentOfPeriapsisIndex ) );
184 
185  // Compute the Rf2 hodograph element of the unified state model
186  convertedUnifiedStateModelElements( Rf2HodographUSM7Index ) =
187  RHodographElement * std::cos( keplerianElements( longitudeOfAscendingNodeIndex )
188  + keplerianElements( argumentOfPeriapsisIndex ) );
189 
190  // Calculate the additional argument of longitude u
191  double argumentOfLatitude = keplerianElements( argumentOfPeriapsisIndex ) +
192  keplerianElements( trueAnomalyIndex );
193 
194  // Compute the eta quaternion of the unified state model
195  convertedUnifiedStateModelElements( etaUSM7Index ) =
196  std::cos( 0.5 * keplerianElements( inclinationIndex ) ) *
197  std::cos( 0.5 * ( keplerianElements( longitudeOfAscendingNodeIndex ) + argumentOfLatitude ) );
198 
199  // Compute the epsilon1 quaternion of the unified state model
200  convertedUnifiedStateModelElements( epsilon1USM7Index ) =
201  std::sin( 0.5 * keplerianElements( inclinationIndex ) ) *
202  std::cos( 0.5 * ( keplerianElements( longitudeOfAscendingNodeIndex ) - argumentOfLatitude ) );
203 
204  // Compute the epsilon2 quaternion of the unified state model
205  convertedUnifiedStateModelElements( epsilon2USM7Index ) =
206  std::sin( 0.5 * keplerianElements( inclinationIndex ) ) *
207  std::sin( 0.5 * ( keplerianElements( longitudeOfAscendingNodeIndex ) - argumentOfLatitude ) );
208 
209  // Compute the epsilon3 quaternion of the unified state model
210  convertedUnifiedStateModelElements( epsilon3USM7Index ) =
211  std::cos( 0.5 * keplerianElements( inclinationIndex ) ) *
212  std::sin( 0.5 * ( keplerianElements( longitudeOfAscendingNodeIndex ) + argumentOfLatitude ) );
213 
214  // Give back result
215  return convertedUnifiedStateModelElements;
216 }

219 Eigen::Vector6d convertUnifiedStateModelQuaternionsToKeplerianElements(
220  const Eigen::Vector7d& unifiedStateModelElements,
221  const double centralBodyGravitationalParameter,
222  const bool forceQuaternionNormalization )
223 {
224  using mathematical_constants::PI;
225 
226  // Declaring eventual output vector.
227  Eigen::Vector6d convertedKeplerianElements = Eigen::Vector6d::Zero( );
228 
229  // Define the tolerance of a singularity
230  const double singularityTolerance = 20.0 * std::numeric_limits< double >::epsilon( );
231 
232  // Check whether the unified state model elements are within expected limits
233  // If inclination is zero and the right ascension of ascending node is non-zero
234  Eigen::Vector4d quaternionsVector = unifiedStateModelElements.segment( etaUSM7Index, 4 );
235  const double normOfQuaternionElements = quaternionsVector.norm( );
236  if ( std::fabs( normOfQuaternionElements - 1.0 ) > singularityTolerance )
237  {
238  if ( forceQuaternionNormalization )
239  {
240  quaternionsVector /= normOfQuaternionElements;
241  }
242  else
243  {
244  // Define the error message.
245  std::stringstream errorMessage;
246  errorMessage << "The norm of the quaternion should be equal to one.\n"
247  << "Norm of the specified quaternion is: " << normOfQuaternionElements - 1.0 << " + 1." << std::endl;
248 
249  // Throw exception.
250  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
251  }
252  }
253  // Else, nothing wrong and continue
254 
255  // Extract quaternion elements
256  double etaQuaternionParameter = quaternionsVector( etaQuaternionIndex );
257  double epsilon1QuaternionParameter = quaternionsVector( epsilon1QuaternionIndex );
258  double epsilon2QuaternionParameter = quaternionsVector( epsilon2QuaternionIndex );
259  double epsilon3QuaternionParameter = quaternionsVector( epsilon3QuaternionIndex );
260 
261  // Check whether the orbit is pure-retrograde
262  if ( ( std::fabs( epsilon3QuaternionParameter ) < singularityTolerance ) &&
263  ( std::fabs( etaQuaternionParameter ) < singularityTolerance ) )
264  // pure-retrograde orbit -> inclination = PI
265  {
266  // Define the error message
267  std::stringstream errorMessage;
268  errorMessage << "Pure-retrograde orbit (inclination = PI).\n"
269  << "Unified state model elements cannot be transformed to Kepler elements." << std::endl;
270 
271  // Throw exception
272  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
273  }
274 
275  // Compute auxiliary parameters cosineLambda, sineLambda and lambda
276  double denominator = epsilon3QuaternionParameter * epsilon3QuaternionParameter +
277  etaQuaternionParameter * etaQuaternionParameter;
278  double cosineLambda = ( etaQuaternionParameter * etaQuaternionParameter -
279  epsilon3QuaternionParameter * epsilon3QuaternionParameter ) / denominator;
280  double sineLambda = ( 2.0 * epsilon3QuaternionParameter *
281  etaQuaternionParameter ) / denominator;
282  double rightAscensionOfLatitude = std::atan2( sineLambda, cosineLambda );
283 
284  // Compute auxiliary parameters auxiliaryParameter1 and auxiliaryParameter2
285  double auxiliaryParameter1 = unifiedStateModelElements( Rf1HodographUSM7Index ) * cosineLambda +
286  unifiedStateModelElements( Rf2HodographUSM7Index ) * sineLambda;
287  double auxiliaryParameter2 = unifiedStateModelElements( CHodographUSM7Index ) -
288  unifiedStateModelElements( Rf1HodographUSM7Index ) * sineLambda +
289  unifiedStateModelElements( Rf2HodographUSM7Index ) * cosineLambda;
290 
291  // Compute auxiliary R hodograph parameter
292  double RHodographElement = std::sqrt( unifiedStateModelElements( Rf1HodographUSM7Index ) *
293  unifiedStateModelElements( Rf1HodographUSM7Index ) +
294  unifiedStateModelElements( Rf2HodographUSM7Index ) *
295  unifiedStateModelElements( Rf2HodographUSM7Index ) );
296 
297  // Compute eccentricity
298  convertedKeplerianElements( eccentricityIndex ) =
299  RHodographElement / unifiedStateModelElements( CHodographUSM7Index );
300 
301  // Compute semi-major axis or, in case of a parabolic orbit, the semi-latus rectum.
302  if ( std::fabs( convertedKeplerianElements( eccentricityIndex ) - 1.0 ) < singularityTolerance )
303  // parabolic orbit -> semi-major axis is not defined. Use semi-latus rectum instead.
304  {
305  convertedKeplerianElements( semiLatusRectumIndex ) = centralBodyGravitationalParameter /
306  ( unifiedStateModelElements( CHodographUSM7Index ) * unifiedStateModelElements( CHodographUSM7Index ) );
307  }
308  else
309  {
310  convertedKeplerianElements( semiMajorAxisIndex ) =
311  centralBodyGravitationalParameter /
312  ( std::pow( unifiedStateModelElements( CHodographUSM7Index ), 2 ) *
313  ( 1 - std::pow( convertedKeplerianElements( eccentricityIndex ), 2 ) ) );
314  }
315 
316  // Compute inclination
317  convertedKeplerianElements( inclinationIndex ) =
318  std::acos( 1.0 - 2.0 * ( epsilon1QuaternionParameter * epsilon1QuaternionParameter +
319  epsilon2QuaternionParameter * epsilon2QuaternionParameter ) );
320  // This acos is always defined correctly because the inclination is always below pi rad.
321 
322  // Find sine and cosine of longitude of ascending node separately
323  double sineOmega = epsilon1QuaternionParameter * epsilon3QuaternionParameter +
324  epsilon2QuaternionParameter * etaQuaternionParameter;
325  double cosineOmega = epsilon1QuaternionParameter * etaQuaternionParameter -
326  epsilon2QuaternionParameter * epsilon3QuaternionParameter;
327  denominator = std::sqrt( cosineOmega * cosineOmega +
328  sineOmega * sineOmega ); // overwrite old denominator
329 
330  // Compute longitude of ascending node
331  if ( std::fabs( convertedKeplerianElements( inclinationIndex ) - PI ) < singularityTolerance )
332  // pure-retrograde orbit -> inclination = PI
333  {
334  // Define the error message
335  std::stringstream errorMessage;
336  errorMessage << "Pure-retrograde orbit (inclination = PI).\n"
337  << "Unified state model elements cannot be transformed to Kepler elements." << std::endl;
338 
339  // Throw exception
340  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
341  }
342  else if ( std::fabs( denominator ) < singularityTolerance )
343  // null denominator, find work-around
344  {
345  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) = 0.0; // by definition
346  }
347  else
348  {
349  // Compute longitude of ascending node
350  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) = std::atan2(
351  sineOmega / denominator, cosineOmega / denominator );
352 
353  // Round off small values of the right ascension of ascending node to zero
354  if ( std::fabs( convertedKeplerianElements( longitudeOfAscendingNodeIndex ) ) < singularityTolerance )
355  {
356  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) = 0.0;
357  }
358 
359  // Ensure the longitude of ascending node is positive
360  while ( convertedKeplerianElements( longitudeOfAscendingNodeIndex ) < 0.0 )
361  // Because of the previous if statement, if the longitude of ascending node is smaller than 0, it will
362  // always be smaller than -singularityTolerance
363  {
364  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) += 2.0 * PI;
365  }
366  }
367 
368  // Compute true anomaly and argument of periapsis
369  if ( std::fabs( RHodographElement ) < singularityTolerance ) // circular orbit
370  {
371  convertedKeplerianElements( argumentOfPeriapsisIndex ) = 0.0; // by definition
372  convertedKeplerianElements( trueAnomalyIndex ) =
373  rightAscensionOfLatitude - convertedKeplerianElements( longitudeOfAscendingNodeIndex );
374 
375  // Round off small theta to zero
376  if ( std::fabs( convertedKeplerianElements( trueAnomalyIndex ) ) < singularityTolerance )
377  {
378  convertedKeplerianElements( trueAnomalyIndex ) = 0.0;
379  }
380 
381  // Ensure the true anomaly is positive
382  while ( convertedKeplerianElements( trueAnomalyIndex ) < 0.0 )
383  // Because of the previous if statement, if the true anomaly is smaller than zero, it will always be smaller than
384  // -singularityTolerance
385  {
386  convertedKeplerianElements( trueAnomalyIndex ) += 2.0 * PI;
387  }
388  }
389  else
390  {
391  convertedKeplerianElements( trueAnomalyIndex ) =
392  std::atan2( ( auxiliaryParameter1 / RHodographElement ),
393  ( ( auxiliaryParameter2 - unifiedStateModelElements( CHodographUSM7Index ) )
394  / RHodographElement ) );
395 
396  // Round off small theta to zero
397  if ( std::fabs( convertedKeplerianElements( trueAnomalyIndex ) ) < singularityTolerance )
398  {
399  convertedKeplerianElements( trueAnomalyIndex ) = 0.0;
400  }
401 
402  // Ensure the true anomaly is positive
403  while ( convertedKeplerianElements( trueAnomalyIndex ) < 0.0 )
404  // Because of the previous if statement, if the true anomaly is smaller than zero, it will always
405  // be smaller than -singularityTolerance
406  {
407  convertedKeplerianElements( trueAnomalyIndex ) += 2.0 * PI;
408  }
409 
410  convertedKeplerianElements( argumentOfPeriapsisIndex ) =
411  rightAscensionOfLatitude -
412  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) -
413  convertedKeplerianElements( trueAnomalyIndex );
414 
415  // Round off small omega to zero
416  if ( std::fabs( convertedKeplerianElements( argumentOfPeriapsisIndex ) ) < singularityTolerance )
417  {
418  convertedKeplerianElements( argumentOfPeriapsisIndex ) = 0.0;
419  }
420 
421  // Ensure the argument of periapsis is positive
422  while ( convertedKeplerianElements( argumentOfPeriapsisIndex ) < 0.0 )
423  // Because of the previous if statement, if the argument of pericenter is smaller than zero,
424  // it will be smaller than -singularityTolerance
425  {
426  convertedKeplerianElements( argumentOfPeriapsisIndex ) += 2.0 * PI;
427  }
428  }
429 
430  // Give back result
431  return convertedKeplerianElements;
432 }

435 Eigen::Vector7d convertCartesianToUnifiedStateModelQuaternionsElements(
436  const Eigen::Vector6d& cartesianElements,
437  const double centralBodyGravitationalParameter )
438 {
439  using mathematical_constants::PI;
440 
441  // Declaring eventual output vector.
442  Eigen::Vector7d convertedUnifiedStateModelElements = Eigen::Vector7d::Zero( );
443 
444  // Define the tolerance of a singularity
445  const double singularityTolerance = 20.0 * std::numeric_limits< double >::epsilon( );
446 
447  // Find Cartesian position and velocity vectors and magnitudes
448  Eigen::Vector3d positionVector = cartesianElements.segment( xCartesianPositionIndex, 3 );
449  double positionMagnitude = positionVector.norm( );
450  Eigen::Vector3d velocityVector = cartesianElements.segment( xCartesianVelocityIndex, 3 );
451 
452  // Determine specific angular momentum vector and magnitude
453  Eigen::Vector3d angularMomentumVector = positionVector.cross( velocityVector );
454  double angularMomentumMagnitude = angularMomentumVector.norm( );
455 
456  // Check whether the orbit is pure-retrograde
457  double angularMomentumMagnitudeMinusZComponent = angularMomentumMagnitude + angularMomentumVector( 2 );
458  if ( std::fabs( angularMomentumMagnitudeMinusZComponent ) < singularityTolerance )
459  // pure-retrograde orbit -> inclination = PI
460  {
461  // Define the error message
462  std::stringstream errorMessage;
463  errorMessage << "Pure-retrograde orbit (inclination = PI).\n"
464  << "Unified state model elements cannot be transformed to Kepler elements." << std::endl;
465 
466  // Throw exception
467  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
468  }
469 
470  // Find C hodograph element of the unified state model
471  convertedUnifiedStateModelElements( CHodographUSM7Index ) = centralBodyGravitationalParameter /
472  angularMomentumMagnitude;
473 
474  // Find direction cosine matrix with position and angular momentum vectors
475  Eigen::Matrix3d directionCosineMatrix = Eigen::Matrix3d::Zero( );
476  directionCosineMatrix.block( 0, 0, 1, 3 ) = positionVector.normalized( ).transpose( );
477  directionCosineMatrix.block( 1, 0, 1, 3 ) = ( angularMomentumVector.normalized( ).cross(
478  positionVector.normalized( ) ) ).transpose( );
479  directionCosineMatrix.block( 2, 0, 1, 3 ) = angularMomentumVector.normalized( ).transpose( );
480 
481  // Compute square of quaternions
482  double traceDirectionCosineMatrix = directionCosineMatrix.trace( );
483  Eigen::Vector4d quaternionsSquaredVector;
484  for ( unsigned int i = 0; i < 3; i++ )
485  {
486  quaternionsSquaredVector( i ) = ( 1.0 - traceDirectionCosineMatrix + 2.0 *
487  directionCosineMatrix( i, i ) ) / 4.0;
488  }
489  quaternionsSquaredVector( 3 ) = ( 1.0 + traceDirectionCosineMatrix ) / 4.0;
490 
491  // Based on the maximum value, find the quaternion elements
492  Eigen::Vector4d::Index indexLargestQuaternionElement;
493  double valueLargestQuaternionElement = quaternionsSquaredVector.maxCoeff( &indexLargestQuaternionElement );
494  switch ( indexLargestQuaternionElement )
495  {
496  case 0:
497  {
498  // Find value of largest quaternion parameter
499  convertedUnifiedStateModelElements( epsilon1USM7Index ) = std::sqrt( valueLargestQuaternionElement );
500 
501  // Find other values
502  Eigen::Vector3d auxiliaryVector = Eigen::Vector3d::Zero( );
503  auxiliaryVector( 0 ) = directionCosineMatrix( 1, 0 ) + directionCosineMatrix( 0, 1 );
504  auxiliaryVector( 1 ) = directionCosineMatrix( 2, 0 ) + directionCosineMatrix( 0, 2 );
505  auxiliaryVector( 2 ) = directionCosineMatrix( 1, 2 ) - directionCosineMatrix( 2, 1 );
506  auxiliaryVector /= 4 * convertedUnifiedStateModelElements( epsilon1USM7Index );
507 
508  // Distribute to state vector
509  convertedUnifiedStateModelElements( epsilon2USM7Index ) = auxiliaryVector( 0 );
510  convertedUnifiedStateModelElements( epsilon3USM7Index ) = auxiliaryVector( 1 );
511  convertedUnifiedStateModelElements( etaUSM7Index ) = auxiliaryVector( 2 );
512  break;
513  }
514  case 1:
515  {
516  // Find value of largest quaternion parameter
517  convertedUnifiedStateModelElements( epsilon2USM7Index ) = std::sqrt( valueLargestQuaternionElement );
518 
519  // Find other values
520  Eigen::Vector3d auxiliaryVector = Eigen::Vector3d::Zero( );
521  auxiliaryVector( 0 ) = directionCosineMatrix( 0, 1 ) + directionCosineMatrix( 1, 0 );
522  auxiliaryVector( 1 ) = directionCosineMatrix( 2, 1 ) + directionCosineMatrix( 1, 2 );
523  auxiliaryVector( 2 ) = directionCosineMatrix( 2, 0 ) - directionCosineMatrix( 0, 2 );
524  auxiliaryVector /= 4 * convertedUnifiedStateModelElements( epsilon2USM7Index );
525 
526  // Distribute to state vector
527  convertedUnifiedStateModelElements( epsilon1USM7Index ) = auxiliaryVector( 0 );
528  convertedUnifiedStateModelElements( epsilon3USM7Index ) = auxiliaryVector( 1 );
529  convertedUnifiedStateModelElements( etaUSM7Index ) = auxiliaryVector( 2 );
530  break;
531  }
532  case 2:
533  {
534  // Find value of largest quaternion parameter
535  convertedUnifiedStateModelElements( epsilon3USM7Index ) = std::sqrt( valueLargestQuaternionElement );
536 
537  // Find other values
538  Eigen::Vector3d auxiliaryVector = Eigen::Vector3d::Zero( );
539  auxiliaryVector( 0 ) = directionCosineMatrix( 0, 2 ) + directionCosineMatrix( 2, 0 );
540  auxiliaryVector( 1 ) = directionCosineMatrix( 1, 2 ) + directionCosineMatrix( 2, 1 );
541  auxiliaryVector( 2 ) = directionCosineMatrix( 0, 1 ) - directionCosineMatrix( 1, 0 );
542  auxiliaryVector /= 4 * convertedUnifiedStateModelElements( epsilon3USM7Index );
543 
544  // Distribute to state vector
545  convertedUnifiedStateModelElements( epsilon1USM7Index ) = auxiliaryVector( 0 );
546  convertedUnifiedStateModelElements( epsilon2USM7Index ) = auxiliaryVector( 1 );
547  convertedUnifiedStateModelElements( etaUSM7Index ) = auxiliaryVector( 2 );
548  break;
549  }
550  case 3:
551  {
552  // Find value of largest quaternion parameter
553  convertedUnifiedStateModelElements( etaUSM7Index ) = std::sqrt( valueLargestQuaternionElement );
554 
555  // Find other values
556  Eigen::Vector3d auxiliaryVector = Eigen::Vector3d::Zero( );
557  auxiliaryVector( 0 ) = directionCosineMatrix( 1, 2 ) - directionCosineMatrix( 2, 1 );
558  auxiliaryVector( 1 ) = directionCosineMatrix( 2, 0 ) - directionCosineMatrix( 0, 2 );
559  auxiliaryVector( 2 ) = directionCosineMatrix( 0, 1 ) - directionCosineMatrix( 1, 0 );
560  auxiliaryVector /= 4 * convertedUnifiedStateModelElements( etaUSM7Index );
561 
562  // Distribute to state vector
563  convertedUnifiedStateModelElements( epsilon1USM7Index ) = auxiliaryVector( 0 );
564  convertedUnifiedStateModelElements( epsilon2USM7Index ) = auxiliaryVector( 1 );
565  convertedUnifiedStateModelElements( epsilon3USM7Index ) = auxiliaryVector( 2 );
566  break;
567  }
568  default:
569  {
570  // Define the error message.
571  std::stringstream errorMessage;
572  errorMessage << "Could not find the maximum value of the quaternion.\n"
573  << "Specified squared quaternion: " << quaternionsSquaredVector.transpose( ) << std::endl;
574 
575  // Throw exception.
576  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
577  }
578  }
579 
580  // Compute sine and cosine of right ascension of latitude (lambda)
581  double denominator = convertedUnifiedStateModelElements( epsilon3USM7Index ) *
582  convertedUnifiedStateModelElements( epsilon3USM7Index ) +
583  convertedUnifiedStateModelElements( etaUSM7Index ) *
584  convertedUnifiedStateModelElements( etaUSM7Index );
585  double cosineLambda = ( convertedUnifiedStateModelElements( etaUSM7Index ) *
586  convertedUnifiedStateModelElements( etaUSM7Index ) -
587  convertedUnifiedStateModelElements( epsilon3USM7Index ) *
588  convertedUnifiedStateModelElements( epsilon3USM7Index ) ) / denominator;
589  double sineLambda = ( 2.0 * convertedUnifiedStateModelElements( epsilon3USM7Index ) *
590  convertedUnifiedStateModelElements( etaUSM7Index ) ) / denominator;
591 
592  // Compute auxiliary parameters
593  double radialVelocity = positionVector.dot( velocityVector ) / positionMagnitude;
594  Eigen::Vector3d auxiliaryParameter3 = radialVelocity / positionMagnitude * positionVector;
595  double auxiliaryParameter2 = ( velocityVector - auxiliaryParameter3 ).norm( );
596  double auxiliaryParameter1 = ( ( radialVelocity >= 0 ) ? 1.0 : - 1.0 ) *
597  auxiliaryParameter3.norm( ); // take norm now that vector value has been used
598  // The sign of first velocity component depends on true anomaly (positive if < PI), and true anomaly can be
599  // related to the radial velocity
600 
601  // Compute Rf1 and Rf2 hodograph elements
602  convertedUnifiedStateModelElements( Rf1HodographUSM7Index ) = auxiliaryParameter1 * cosineLambda -
603  ( auxiliaryParameter2 - convertedUnifiedStateModelElements( CHodographUSM7Index ) ) * sineLambda;
604  convertedUnifiedStateModelElements( Rf2HodographUSM7Index ) = auxiliaryParameter1 * sineLambda +
605  ( auxiliaryParameter2 - convertedUnifiedStateModelElements( CHodographUSM7Index ) ) * cosineLambda;
606 
607  // Give back result
608  return convertedUnifiedStateModelElements;
609 }

612 Eigen::Vector6d convertUnifiedStateModelQuaternionsToCartesianElements(
613  const Eigen::Vector7d& unifiedStateModelElements,
614  const double centralBodyGravitationalParameter,
615  const bool forceQuaternionNormalization )
616 {
617  using mathematical_constants::PI;
618 
619  // Declaring eventual output vector.
620  Eigen::Vector6d convertedCartesianElements = Eigen::Vector6d::Zero( );
621 
622  // Define the tolerance of a singularity
623  const double singularityTolerance = 20.0 * std::numeric_limits< double >::epsilon( );
624 
625  // Extract quaternion elements
626  Eigen::Vector4d quaternionsVector = unifiedStateModelElements.segment( etaUSM7Index, 4 );
627  const double normOfQuaternionElements = quaternionsVector.norm( );
628  if ( std::fabs( normOfQuaternionElements - 1.0 ) > singularityTolerance )
629  {
630  if ( forceQuaternionNormalization )
631  {
632  quaternionsVector /= normOfQuaternionElements;
633  }
634  else
635  {
636  // Define the error message.
637  std::stringstream errorMessage;
638  errorMessage << "The norm of the quaternion should be equal to one.\n"
639  << "Norm of the specified quaternion is: " << normOfQuaternionElements - 1.0 << " + 1." << std::endl;
640 
641  // Throw exception.
642  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
643  }
644  }
645  // Else, nothing wrong and continue
646 
647  // Extract quaternion elements
648  double etaQuaternionParameter = quaternionsVector( etaQuaternionIndex );
649  double epsilon3QuaternionParameter = quaternionsVector( epsilon3QuaternionIndex );
650 
651  // Declare auxiliary parameters before using them in the if statement
652  double cosineLambda;
653  double sineLambda;
654  double rightAscensionOfLatitude;
655 
656  // Compute auxiliary parameters cosineLambda, sineLambda and lambda
657  if ( ( std::fabs( epsilon3QuaternionParameter ) < singularityTolerance ) &&
658  ( std::fabs( etaQuaternionParameter ) < singularityTolerance ) )
659  // pure-retrograde orbit -> inclination = PI
660  {
661  // Define the error message
662  std::stringstream errorMessage;
663  errorMessage << "Pure-retrograde orbit (inclination = PI).\n"
664  << "Unified state model elements cannot be transformed to Kepler elements." << std::endl;
665 
666  // Throw exception
667  throw std::runtime_error( std::runtime_error( errorMessage.str( ) ) );
668  }
669  else
670  {
671  double denominator = epsilon3QuaternionParameter * epsilon3QuaternionParameter +
672  etaQuaternionParameter * etaQuaternionParameter;
673  cosineLambda = ( etaQuaternionParameter * etaQuaternionParameter -
674  epsilon3QuaternionParameter * epsilon3QuaternionParameter ) / denominator;
675  sineLambda = ( 2.0 * epsilon3QuaternionParameter *
676  etaQuaternionParameter ) / denominator;
677  rightAscensionOfLatitude = std::atan2( sineLambda, cosineLambda );
678  }
679 
680  // Compute auxiliary parameters auxiliaryParameter1, auxiliaryParameter2 and auxiliaryVector1
681  double auxiliaryParameter1 = unifiedStateModelElements( Rf1HodographUSM7Index ) * cosineLambda +
682  unifiedStateModelElements( Rf2HodographUSM7Index ) * sineLambda;
683  double auxiliaryParameter2 = unifiedStateModelElements( CHodographUSM7Index ) -
684  unifiedStateModelElements( Rf1HodographUSM7Index ) * sineLambda +
685  unifiedStateModelElements( Rf2HodographUSM7Index ) * cosineLambda;
686  Eigen::Vector2d auxiliaryVector1;
687  auxiliaryVector1( 0 ) = auxiliaryParameter1;
688  auxiliaryVector1( 1 ) = auxiliaryParameter2;
689 
690  // Find direction cosine matrix in terms of quaternions
691  Eigen::Matrix3d inverseDirectionCosineMatrix = linear_algebra::convertVectorToQuaternionFormat(
692  quaternionsVector ).toRotationMatrix( );
693 
694  // Get Cartesian position vector
695  convertedCartesianElements.segment( xCartesianPositionIndex, 3 ) =
696  centralBodyGravitationalParameter / unifiedStateModelElements( CHodographUSM7Index ) /
697  auxiliaryParameter2 * inverseDirectionCosineMatrix.block( 0, 0, 3, 1 ); // take first column of matrix
698 
699  // Get Cartesian velocity vector
700  convertedCartesianElements.segment( xCartesianVelocityIndex, 3 ) =
701  inverseDirectionCosineMatrix.block( 0, 0, 3, 2 ) * auxiliaryVector1;
702 
703  // Give back result
704  return convertedCartesianElements;
705 }
