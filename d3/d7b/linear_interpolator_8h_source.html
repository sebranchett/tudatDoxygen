70  LinearInterpolator( const std::map< IndependentVariableType, DependentVariableType >& dataMap,
71  const AvailableLookupScheme selectedLookupScheme = huntingAlgorithm )
72  {
73  // Verify that the initialization variables are not empty.
74  if ( dataMap.size( ) == 0 )
75  {
76  throw std::runtime_error(
77  "The vectors used in the linear interpolator initialization are empty." );
78  }
79 
80  // Resize data vectors of independent/dependent values.
81  independentValues_.resize( dataMap.size( ) );
82  dependentValues_.resize( dataMap.size( ) );
83 
84  // Fill data vectors with data from map.
85  int counter = 0;
86  for ( typename std::map< IndependentVariableType, DependentVariableType >::const_iterator
87  mapIterator = dataMap.begin( ); mapIterator != dataMap.end( ); mapIterator++ )
88  {
89  independentValues_[ counter ] = mapIterator->first;
90  dependentValues_[ counter ] = mapIterator->second;
91  counter++;
92  }
93 
94  // Create lookup scheme from independent variable data points.
95  this->makeLookupScheme( selectedLookupScheme );
96  }

110  LinearInterpolator( const std::vector< IndependentVariableType >& independentValues,
111  const std::vector< DependentVariableType >& dependentValues,
112  const AvailableLookupScheme selectedLookupScheme = huntingAlgorithm )
113  {
114  // Verify that the initialization variables are not empty.
115  if ( independentValues.size( ) == 0 || dependentValues.size( ) == 0 )
116  {
117  throw std::runtime_error(
118  "The vectors used in the linear interpolator initialization are empty." );
119  }
120 
121  // Set data vectors.
122  independentValues_ = independentValues;
123  dependentValues_= dependentValues;
124 
125  // Check if data is in ascending order
126  if( !std::is_sorted( independentValues_.begin( ), independentValues_.end( ) ) )
127  {
128  throw std::runtime_error( "Error when making linear interpolator, input vector with independent variables should be in ascending order" );
129  }
130 
131  // Create lookup scheme from independent variable values
132  this->makeLookupScheme( selectedLookupScheme );
133  }

148  DependentVariableType interpolate( const IndependentVariableType independentVariableValue )
149  {
150  // Lookup nearest lower index.
151  int newNearestLowerIndex = lookUpScheme_->findNearestLowerNeighbour(
152  independentVariableValue );
153 
154  // Perform linear interpolation.
155  DependentVariableType interpolatedValue = dependentValues_[ newNearestLowerIndex ] +
156  ( independentVariableValue - independentValues_[ newNearestLowerIndex ] ) /
157  ( independentValues_[ newNearestLowerIndex + 1 ] -
158  independentValues_[ newNearestLowerIndex ] ) *
159  ( dependentValues_[ newNearestLowerIndex + 1 ] -
160  dependentValues_[ newNearestLowerIndex ] );
161 
162  return interpolatedValue;
163  }
