37 void TabulatedGravityFieldVariations::resetCoefficientInterpolator(
38  const std::map< double, Eigen::MatrixXd >& cosineCoefficientCorrections,
39  const std::map< double, Eigen::MatrixXd >& sineCoefficientCorrections )
40 {
41  // Set current coefficient tables.
42  cosineCoefficientCorrections_ = cosineCoefficientCorrections;
43  sineCoefficientCorrections_ = sineCoefficientCorrections;
44 
45  // Check consistency of map sizes.
46  if( cosineCoefficientCorrections_.size( ) != sineCoefficientCorrections_.size( ) )
47  {
48  throw std::runtime_error( "Error when resetting tabulated gravity field corrections, sine and cosine data size incompatible" );
49  }
50 
51  // Create iterators over maps.
52  std::map< double, Eigen::MatrixXd >::iterator cosineIterator =
53  cosineCoefficientCorrections_.begin( );
54  std::map< double, Eigen::MatrixXd >::iterator sineIterator =
55  sineCoefficientCorrections_.begin( );
56 
57  // Declare map to store concatenated (horizontally) [cosine|sine] coefficients for
58  // given discrete times.
59  std::map< double, Eigen::MatrixXd > sineCosinePairMap;
60 
61  // Declare matrix to set concatenated (horizontally) [cosine|sine] coefficients at
62  // current time in loop.
63  Eigen::MatrixXd currentCoefficients = Eigen::MatrixXd::Zero( numberOfDegrees_, 2 * numberOfOrders_ );
64 
65  // Iterate over all times in correction maps.
66  for( unsigned int i = 0; i < cosineCoefficientCorrections_.size( ); i++ )
67  {
68  // Check whether input times are consistent
69  if( cosineIterator->first != sineIterator->first )
70  {
71  std::string errorMessage = "Error when resetting tabulated gravity field corrections, sine and cosine data time differ by" +
72  std::to_string( cosineIterator->first - sineIterator->first );
73  throw std::runtime_error( errorMessage );
74  }
75  // Check whether matrix sizes are consistent.
76  if( ( cosineIterator->second.rows( ) != numberOfDegrees_ ) ||
77  ( sineIterator->second.rows( ) != numberOfDegrees_ ) ||
78  ( cosineIterator->second.cols( ) != numberOfOrders_ ) ||
79  ( sineIterator->second.cols( ) != numberOfOrders_ ) )
80  {
81  std::string errorMessage = "Error when resetting tabulated gravity field corrections, sine and cosine blocks of inconsistent size";
82  throw std::runtime_error( errorMessage );
83  }
84 
85  // Concatenate cosine and sine matrices
86  currentCoefficients.block( 0, 0, numberOfDegrees_, numberOfOrders_ ) =
87  cosineIterator->second;
88  currentCoefficients.block( 0, numberOfOrders_, numberOfDegrees_, numberOfOrders_ ) =
89  sineIterator->second;
90 
91  // Set as entry in map of concatenated matrices
92  sineCosinePairMap[ cosineIterator->first ] = currentCoefficients;
93 
94  // Increment iterators.
95  cosineIterator++;
96  sineIterator++;
97  }
98 
99  // Create interpolator
100  variationInterpolator_ =
101  interpolators::createOneDimensionalInterpolator< double, Eigen::MatrixXd >(
102  sineCosinePairMap, interpolatorType_ );
103 }
