37 Eigen::Vector6d convertCartesianToSphericalOrbitalState(
38  const Eigen::Vector6d& bodyFixedCartesianState )
39 {
40  Eigen::Vector6d sphericalOrbitalState;
41 
42  // Compute and set spherical position
43  Eigen::Vector3d sphericalPosition = coordinate_conversions::convertCartesianToSpherical< double >(
44  bodyFixedCartesianState.segment( 0, 3 ) );
45  sphericalOrbitalState( radiusIndex ) = sphericalPosition( 0 );
46  sphericalOrbitalState( latitudeIndex ) = mathematical_constants::PI / 2.0 - sphericalPosition( 1 );
47  sphericalOrbitalState( longitudeIndex ) = sphericalPosition( 2 );
48 
49  // Convert velocity to vertical frame.
50  Eigen::Vector3d verticalFrameVelocity =
51  reference_frames::getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
52  sphericalOrbitalState( longitudeIndex ), sphericalOrbitalState( latitudeIndex ) ) *
53  bodyFixedCartesianState.segment( 3, 3 );
54 
55  // Set velocity in spherical orbital state.
56  sphericalOrbitalState( speedIndex ) = verticalFrameVelocity.norm( );
57  sphericalOrbitalState( flightPathIndex ) = calculateFlightPathAngle( verticalFrameVelocity );
58  sphericalOrbitalState( headingAngleIndex ) = calculateHeadingAngle( verticalFrameVelocity );
59 
60  return sphericalOrbitalState;

65 Eigen::Vector6d convertSphericalOrbitalToCartesianState(
66  const Eigen::Vector6d& sphericalOrbitalState )
67 {
68  Eigen::Vector6d cartesianState;
69 
70  // Compute Cartesian position
71  Eigen::Vector3d sphericalPosition = sphericalOrbitalState.segment( 0, 3 );
72  sphericalPosition( 1 ) = mathematical_constants::PI / 2.0 - sphericalOrbitalState( 1 );
73  cartesianState.segment( 0, 3 ) = coordinate_conversions::convertSphericalToCartesian(
74  sphericalPosition );
75 
76  // Check whether flight path angle and heading angle are valid (corresponding velocity components are zero otherwise)
77  bool isFlightPathAngleValid = ( sphericalOrbitalState( flightPathIndex ) == sphericalOrbitalState( flightPathIndex ) );
78  bool isHeadingAngleValid = ( sphericalOrbitalState( headingAngleIndex ) == sphericalOrbitalState( headingAngleIndex ) );
79 
80  // Compute velocity in vertical frame.
81  Eigen::Vector3d velocityInVerticalFrame = Eigen::Vector3d::Zero( );
82  if( isFlightPathAngleValid && isHeadingAngleValid )
83  {
84  velocityInVerticalFrame( 0 ) = sphericalOrbitalState( speedIndex ) *
85  std::cos( sphericalOrbitalState( flightPathIndex ) ) *
86  std::cos( sphericalOrbitalState( headingAngleIndex ) );
87  velocityInVerticalFrame( 1 ) = sphericalOrbitalState( speedIndex ) *
88  std::cos( sphericalOrbitalState( flightPathIndex ) ) *
89  std::sin( sphericalOrbitalState( headingAngleIndex ) );
90  }
91 
92  if( isFlightPathAngleValid )
93  {
94  velocityInVerticalFrame( 2 ) = -sphericalOrbitalState( speedIndex ) *
95  std::sin( sphericalOrbitalState( flightPathIndex ) );
96  }
97 
98  // Set velocity in body-fixed frame.
99  cartesianState.segment( 3, 3 ) = reference_frames::getLocalVerticalToRotatingPlanetocentricFrameTransformationQuaternion(
100  sphericalOrbitalState( longitudeIndex ), sphericalOrbitalState( latitudeIndex ) ) * velocityInVerticalFrame;
101 
102  return cartesianState;
103 }
