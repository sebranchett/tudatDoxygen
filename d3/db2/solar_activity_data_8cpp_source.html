99 SolarActivityDataMap readSolarActivityData( std::string filePath )
100 {
101  // Data Vector container
102  tudat::input_output::parsed_data_vector_utilities::ParsedDataVectorPtr parsedDataVector;
103 
104  // datafile Parser and Extractor
105  tudat::input_output::solar_activity::ParseSolarActivityData solarActivityParser;
106  tudat::input_output::solar_activity::ExtractSolarActivityData solarActivityExtractor;
107 
108 
109  // Open dataFile and Parse
110  std::ifstream dataFile;
111  dataFile.open( filePath.c_str( ), std::ifstream::in );
112  parsedDataVector = solarActivityParser.parse( dataFile );
113  dataFile.close( );
114 
115  int numberOfLines = parsedDataVector->size( );
116  SolarActivityDataMap dataMap;
117  double julianDate = TUDAT_NAN;
118 
119  // Save each line to datamap
120  for(int i = 0 ; i < numberOfLines ; i++ ){
121  julianDate = tudat::basic_astrodynamics::convertCalendarDateToJulianDay(
122  solarActivityExtractor.extract( parsedDataVector->at( i ) )->year,
123  solarActivityExtractor.extract( parsedDataVector->at( i ) )->month,
124  solarActivityExtractor.extract( parsedDataVector->at( i ) )->day,
125  0, 0, 0.0 ) ;
126  dataMap[ julianDate ] = solarActivityExtractor.extract( parsedDataVector->at( i ) ) ;
127  }
128 
129  return dataMap;
130 
131 }
