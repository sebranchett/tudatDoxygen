34 double getConditionNumberOfDecomposedMatrix( const Eigen::JacobiSVD< Eigen::MatrixXd >& singularValueDecomposition )
35 {
36  Eigen::VectorXd singularValues = singularValueDecomposition.singularValues( );
37  return singularValues( 0 ) / singularValues( singularValues.rows( ) - 1 );

41 Eigen::VectorXd solveSystemOfEquationsWithSvd( const Eigen::MatrixXd matrixToInvert,
42  const Eigen::VectorXd rightHandSideVector,
43  const bool checkConditionNumber,
44  const double maximumAllowedConditionNumber )
45 {
46  Eigen::JacobiSVD< Eigen::MatrixXd > svdDecomposition = matrixToInvert.jacobiSvd(
47  Eigen::ComputeThinU | Eigen::ComputeThinV );
48  if( checkConditionNumber )
49  {
50  double conditionNumber = getConditionNumberOfDecomposedMatrix( svdDecomposition );
51 
52  if( conditionNumber > maximumAllowedConditionNumber )
53  {
54  std::cerr << "Warning when performing least squares, condition number is " << conditionNumber << std::endl;
55  }
56  }
57  return svdDecomposition.solve( rightHandSideVector );
58 }

61 Eigen::MatrixXd multiplyInformationMatrixByDiagonalWeightMatrix(
62  const Eigen::MatrixXd& informationMatrix,
63  const Eigen::VectorXd& diagonalOfWeightMatrix )
64 {
65  Eigen::MatrixXd weightedInformationMatrix = Eigen::MatrixXd::Zero( informationMatrix.rows( ), informationMatrix.cols( ) );
66 
67  for( int i = 0; i < informationMatrix.cols( ); i++ )
68  {
69  weightedInformationMatrix.block( 0, i, informationMatrix.rows( ), 1 ) =
70  informationMatrix.block( 0, i, informationMatrix.rows( ), 1 ).cwiseProduct( diagonalOfWeightMatrix );
71  }
72 
73  return weightedInformationMatrix;
74 }

98 std::pair< Eigen::VectorXd, Eigen::MatrixXd > performLeastSquaresAdjustmentFromInformationMatrix(
99  const Eigen::MatrixXd& informationMatrix,
100  const Eigen::VectorXd& observationResiduals,
101  const Eigen::VectorXd& diagonalOfWeightMatrix,
102  const Eigen::MatrixXd& inverseOfAPrioriCovarianceMatrix,
103  const bool checkConditionNumber,
104  const double maximumAllowedConditionNumber )
105 {
106  Eigen::VectorXd rightHandSide = informationMatrix.transpose( ) *
107  ( diagonalOfWeightMatrix.cwiseProduct( observationResiduals ) );
108  Eigen::MatrixXd inverseOfCovarianceMatrix = calculateInverseOfUpdatedCovarianceMatrix(
109  informationMatrix, diagonalOfWeightMatrix, inverseOfAPrioriCovarianceMatrix );
110  return std::make_pair( solveSystemOfEquationsWithSvd( inverseOfCovarianceMatrix, rightHandSide,
111  checkConditionNumber, maximumAllowedConditionNumber ),
112  inverseOfCovarianceMatrix );
113 }

142 Eigen::VectorXd getLeastSquaresPolynomialFit(
143  const Eigen::VectorXd& independentValues,
144  const Eigen::VectorXd& dependentValues,
145  const std::vector< double >& polynomialPowers )
146 {
147  if( independentValues.rows( ) != dependentValues.rows( ) )
148  {
149  throw std::runtime_error( "Error when doing least squares polynomial fit, size of dependent and independent "
150  "variable vectors is not equal." );
151  }
152 
153  Eigen::MatrixXd informationMatrix = Eigen::MatrixXd::Zero( dependentValues.rows( ), polynomialPowers.size( ) );
154 
155  // Compute information matrix
156  for( int i = 0; i < independentValues.rows( ); i++ )
157  {
158  for( unsigned int j = 0; j < polynomialPowers.size( ); j++ )
159  {
160  informationMatrix( i, j ) = std::pow( independentValues( i ), polynomialPowers.at( j ) );
161  }
162  }
163 
164  return performLeastSquaresAdjustmentFromInformationMatrix( informationMatrix, dependentValues ).first;
165 }

182 Eigen::VectorXd nonLinearLeastSquaresFit(
183  const std::function< std::pair< Eigen::VectorXd, Eigen::MatrixXd >( const Eigen::VectorXd& ) >& observationAndJacobianFunctions,
184  const Eigen::VectorXd& initialEstimate, const Eigen::VectorXd& actualObservations, const double initialScaling,
185  const double convergenceTolerance, const unsigned int maximumNumberOfIterations )
186 {
187  // Set current estimate to initial value
188  Eigen::VectorXd currentEstimate = initialEstimate;
189 
190  // Initialize variables
191  std::pair< Eigen::VectorXd, Eigen::MatrixXd > pairOfEstimatedObservationsAndDesignMatrix;
192  Eigen::MatrixXd designMatrix;
193  Eigen::VectorXd offsetInObservations;
194  Eigen::VectorXd updateInEstimate;
195 
196  // Initial parameters for Levenberg–Marquardt method
197  double levenbergMarquardtDampingParameter = 0.0;
198  double scalingParameterUpdate = 2.0;
199  double levenbergMarquardtGainRatio = 0.0;
200 
201  // Start iterative loop
202  unsigned int iteration = 0;
203  do
204  {
205  // Compute current system and jacobian functions
206  pairOfEstimatedObservationsAndDesignMatrix = observationAndJacobianFunctions( currentEstimate );
207  designMatrix = pairOfEstimatedObservationsAndDesignMatrix.second;
208 
209  // Offset in observation
210  offsetInObservations = actualObservations - pairOfEstimatedObservationsAndDesignMatrix.first;
211 
212  // Compute damping parameter for first iteration
213  if ( iteration == 0 )
214  {
215  levenbergMarquardtDampingParameter = initialScaling * ( designMatrix.transpose( ) * designMatrix ).diagonal( ).maxCoeff( );
216  }
217 
218  // Compute update in estimate
219  Eigen::VectorXd diagonalOfWeightMatrix = Eigen::VectorXd::Ones( offsetInObservations.rows( ) );
220  Eigen::MatrixXd inverseOfAPrioriCovarianceMatrix = levenbergMarquardtDampingParameter *
221 // Eigen::MatrixXd( ( designMatrix.transpose( ) * designMatrix ).diagonal( ).asDiagonal( ) ); // Marquardt’s update
222  Eigen::MatrixXd::Identity( currentEstimate.rows( ), currentEstimate.rows( ) );
223  updateInEstimate = linear_algebra::performLeastSquaresAdjustmentFromInformationMatrix(
224  designMatrix, offsetInObservations, diagonalOfWeightMatrix, inverseOfAPrioriCovarianceMatrix, false ).first;
225 
226  // Check that update is real
227  if ( ( !updateInEstimate.allFinite( ) ) || ( updateInEstimate.hasNaN( ) ) )
228  {
229  throw std::runtime_error( "Error in non-linear least squares estimation. Iterative process diverges." );
230  }
231 
232  // Compute gain ratio
233  levenbergMarquardtGainRatio =
234  ( offsetInObservations.squaredNorm( ) -
235  ( actualObservations - observationAndJacobianFunctions( currentEstimate + updateInEstimate ).first ).squaredNorm( ) ) /
236  ( updateInEstimate.transpose( ) * ( levenbergMarquardtDampingParameter * updateInEstimate +
237  designMatrix.transpose( ) * offsetInObservations ) );
238 
239  // Update damping parameter
240  if ( levenbergMarquardtGainRatio > 0 )
241  {
242  // Reduce damping parameter, since good approximation
243  levenbergMarquardtDampingParameter *= std::max( 1.0 / 3.0, 1.0 - std::pow( 2.0 * levenbergMarquardtGainRatio - 1.0, 3 ) );
244  scalingParameterUpdate = 2; // reset
245 
246  // Correct estimate
247  currentEstimate += updateInEstimate;
248  }
249  else
250  {
251  // Increase damping parameter, since bad approximation and reject step
252  levenbergMarquardtDampingParameter *= scalingParameterUpdate;
253  scalingParameterUpdate *= 2.0;
254  }
255 // std::cout << "Iter: " << iteration + 1 << ". Update: " << updateInEstimate.transpose( ) << ". "
256 // << ( ( levenbergMarquardtGainRatio > 0 ) ? "Accepted." : "Rejected." ) << std::endl;
257 
258  // Increase iteration counter
259  iteration++;
260  }
261  while ( ( updateInEstimate.norm( ) > convergenceTolerance ) && ( iteration <= maximumNumberOfIterations ) );
262 // std::cout << "Final: " << currentEstimate.transpose( ) << ". Iterations: " << iteration << std::endl;
263 
264  // Warn user of exceeded maximum number of iterations
265  if ( iteration > maximumNumberOfIterations )
266  {
267  std::cerr << "Warning in non-linear least squares estimation. Maximum number of iterations exceeded." << std::endl;
268  }
269 
270  // Give out new estimate in parameters
271  return currentEstimate;
272 }
