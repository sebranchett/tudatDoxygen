102  IndependentVariableArray getUniqueWeights( const unsigned int order )
103  {
104  if ( uniqueWeights_.count( order ) == 0 )
105  {
106  std::string errorMessage = "Error in Gaussian quadrature, weights not available for n=" +
107  std::to_string( order );
108  throw std::runtime_error( errorMessage );

110  return uniqueWeights_.at( order );

114 
119  IndependentVariableArray getNodes( const unsigned int order )
120  {
121  if ( nodes_.count( order ) == 0 )
122  {
123  IndependentVariableArray newNodes( order );
124 
125  // Include node 0.0 if order is odd
126  unsigned int i = 0;
127  if ( order % 2 == 1 )
128  {
129  newNodes.row( i++ ) = 0.0;
130  }
131 
132  // Include Â± nodes
133  IndependentVariableArray uniqueNodes_ = getUniqueNodes( order );
134  for ( int j = 0; j < uniqueNodes_.size( ); j++ )
135  {
136  newNodes.row( i++ ) = -uniqueNodes_[ j ];
137  newNodes.row( i++ ) = uniqueNodes_[ j ];
138  }
139 
140  nodes_[ order ] = newNodes;
141  }
142 
143  return nodes_.at( order );
144  }
145 
147  IndependentVariableArray getWeights( const unsigned int n )
148  {
149  if ( weights_.count( n ) == 0 )
150  {
151  IndependentVariableArray newWeights( n );
152  IndependentVariableArray orderNWeights = getUniqueWeights( n );
153 
154  // Include non-repeated weight factor if n is odd
155  unsigned int i = 0;
156  int j = 0;
157  if ( n % 2 == 1 )
158  {
159  newWeights.row( i++ ) = orderNWeights[ j++ ];
160  }
161 
162  // Include repeated weight factors
163  for ( ; j < orderNWeights.size( ); j++ )
164  {
165  newWeights.row( i++ ) = orderNWeights[ j ];
166  newWeights.row( i++ ) = orderNWeights[ j ];
167  }
168 
169  weights_[ n ] = newWeights;
170  }
171 
172  return weights_.at( n );
173  }

257 template< typename IndependentVariableType, typename DependentVariableType >
258 class GaussianQuadrature : public NumericalQuadrature< IndependentVariableType , DependentVariableType >
259 {

275  const IndependentVariableType lowerLimit, const IndependentVariableType upperLimit,

293  const IndependentVariableType lowerLimit, const IndependentVariableType upperLimit,
294  const unsigned int numberOfNodes )
295  {

298  upperLimit_ = upperLimit;

308  DependentVariableType getQuadrature( )
309  {
310  if ( ! quadratureHasBeenPerformed_ )
311  {
312  if ( integrand_ == nullptr )
313  {
314  throw std::runtime_error(
315  "The integrand for the Gaussian quadrature has not been set." );
316  }
317 
318  if ( lowerLimit_ > upperLimit_ )
319  {
320  throw std::runtime_error(
321  "The lower limit for the Gaussian quadrature is larger than the upper limit." );
322  }
323 
324  if ( numberOfNodes_ < 2 || numberOfNodes_ > 64 )
325  {
326  throw std::runtime_error(
327  "The number of nodes for the Gaussian quadrature must be between 2 and 64." );
328  }
329 
330  performQuadrature( );
331  quadratureHasBeenPerformed_ = true;
332  }
333 
334  return quadratureResult_;
335  }

338 protected:
339

345  void performQuadrature( )
346  {
347  // Determine the values of the auxiliary independent variable (nodes)
348  const IndependentVariableArray nodes = gaussQuadratureNodesAndWeights_->getNodes( numberOfNodes_ );
349 
350  // Determine the values of the weight factors
351  const IndependentVariableArray weights = gaussQuadratureNodesAndWeights_->getWeights( numberOfNodes_ );
352 
353  // Change of variable -> from range [-1, 1] to range [lowerLimit, upperLimit]
354  const IndependentVariableArray independentVariables =
355  0.5 * ( ( upperLimit_ - lowerLimit_ ) * nodes + upperLimit_ + lowerLimit_ );
356 
357  // Determine the value of the dependent variable
358  DependentVariableArray weighedIntegrands( numberOfNodes_ );
359  for ( unsigned int i = 0; i < numberOfNodes_; i++ )
360  {
361  weighedIntegrands( i ) = weights( i ) * integrand_( independentVariables( i ) );
362  }
363 
364  quadratureResult_ = 0.5 * ( upperLimit_ - lowerLimit_ ) * weighedIntegrands.sum( );
365  }
