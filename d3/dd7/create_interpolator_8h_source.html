447 template< typename IndependentVariableType, typename DependentVariableType >
448 std::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
449 createOneDimensionalInterpolator(
450  const std::map< IndependentVariableType, DependentVariableType > dataToInterpolate,
451  const std::shared_ptr< InterpolatorSettings > interpolatorSettings,
452  const std::pair< DependentVariableType, DependentVariableType >& defaultExtrapolationValue =
453  std::make_pair( IdentityElement::getAdditionIdentity< DependentVariableType >( ),
454  IdentityElement::getAdditionIdentity< DependentVariableType >( ) ),
455  const std::vector< DependentVariableType > firstDerivativeOfDependentVariables =
456  std::vector< DependentVariableType >( ) )
457 {
458  std::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
459  createdInterpolator;
460 
461  // Check that boundary handling is one-dimensional
462  if ( interpolatorSettings->getBoundaryHandling( ).size( ) != 1 )
463  {
464  throw std::runtime_error( "Error while creating interpolator of type: " +
465  std::to_string( interpolatorSettings->getInterpolatorType( ) ) +
466  ". The interpolator is one-dimensional, but more than one boundary "
467  "handling methods have been defined." );
468  }
469 
470  // Check type of interpolator.
471  switch( interpolatorSettings->getInterpolatorType( ) )
472  {
473  case linear_interpolator:
474  createdInterpolator = std::make_shared< LinearInterpolator
475  < IndependentVariableType, DependentVariableType > >(
476  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
477  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
478  break;
479  case cubic_spline_interpolator:
480  {
481  if( !interpolatorSettings->getUseLongDoubleTimeStep( ) )
482  {
483  createdInterpolator = std::make_shared< CubicSplineInterpolator
484  < IndependentVariableType, DependentVariableType > >(
485  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
486  interpolatorSettings->getBoundaryHandling( ).at( 0 ) );
487  }
488  else
489  {
490  createdInterpolator = std::make_shared< CubicSplineInterpolator
491  < IndependentVariableType, DependentVariableType, long double > >(
492  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
493  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
494  }
495  break;
496  }
497  case lagrange_interpolator:
498  {
499  // Check consistency of input
500  std::shared_ptr< LagrangeInterpolatorSettings > lagrangeInterpolatorSettings =
501  std::dynamic_pointer_cast< LagrangeInterpolatorSettings >( interpolatorSettings );
502  if( lagrangeInterpolatorSettings != nullptr )
503  {
504  // Create Lagrange interpolator with requested time step type
505  if( !lagrangeInterpolatorSettings->getUseLongDoubleTimeStep( ) )
506  {
507  createdInterpolator = std::make_shared< LagrangeInterpolator
508  < IndependentVariableType, DependentVariableType, double > >(
509  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
510  interpolatorSettings->getSelectedLookupScheme( ),
511  lagrangeInterpolatorSettings->getLagrangeBoundaryHandling( ),
512  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
513  }
514  else
515  {
516  createdInterpolator = std::make_shared< LagrangeInterpolator
517  < IndependentVariableType, DependentVariableType, long double > >(
518  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
519  interpolatorSettings->getSelectedLookupScheme( ),
520  lagrangeInterpolatorSettings->getLagrangeBoundaryHandling( ),
521  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
522  }
523  }
524  else
525  {
526  throw std::runtime_error( "Error, did not recognize lagrange interpolator settings" );
527 
528  }
529  break;
530  }
531  case hermite_spline_interpolator:
532  {
533  if( firstDerivativeOfDependentVariables.size( ) != dataToInterpolate.size( ) )
534  {
535  throw std::runtime_error(
536  "Error when creating hermite spline interpolator, derivative size is inconsistent" );
537  }
538  createdInterpolator = std::make_shared< HermiteCubicSplineInterpolator
539  < IndependentVariableType, DependentVariableType > >(
540  dataToInterpolate, firstDerivativeOfDependentVariables,
541  interpolatorSettings->getSelectedLookupScheme( ),
542  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
543  break;
544  }
545  case piecewise_constant_interpolator:
546  createdInterpolator = std::make_shared< PiecewiseConstantInterpolator
547  < IndependentVariableType, DependentVariableType > >(
548  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ),
549  interpolatorSettings->getBoundaryHandling( ).at( 0 ), defaultExtrapolationValue );
550  break;
551  default:
552  throw std::runtime_error( "Error when making interpolator, function cannot be used to create interplator of type " +
553  std::to_string( interpolatorSettings->getInterpolatorType( ) ) );
554  }
555  return createdInterpolator;
556 }

601 template< typename IndependentVariableType, typename DependentVariableType, unsigned int NumberOfDimensions >
602 std::shared_ptr< MultiDimensionalInterpolator< IndependentVariableType, DependentVariableType, NumberOfDimensions > >
603 createMultiDimensionalInterpolator(
604  const std::vector< std::vector< IndependentVariableType > >& independentValues,
605  const boost::multi_array< DependentVariableType, static_cast< size_t >( NumberOfDimensions ) >& dependentData,
606  const std::shared_ptr< InterpolatorSettings > interpolatorSettings,
607  const std::vector< std::pair< DependentVariableType, DependentVariableType > >& defaultExtrapolationValue =
608  std::vector< std::pair< DependentVariableType, DependentVariableType > >
609  ( NumberOfDimensions, std::make_pair( IdentityElement::getAdditionIdentity< DependentVariableType >( ),
610  IdentityElement::getAdditionIdentity< DependentVariableType >( ) ) ) )
611 {
612  std::shared_ptr< MultiDimensionalInterpolator< IndependentVariableType, DependentVariableType, NumberOfDimensions > >
613  createdInterpolator;
614 
615  // Assign default values
616  std::vector< BoundaryInterpolationType > boundaryHandlingVector = interpolatorSettings->getBoundaryHandling( );
617  if ( boundaryHandlingVector.empty( ) )
618  {
619  boundaryHandlingVector = std::vector< BoundaryInterpolationType >( NumberOfDimensions, extrapolate_at_boundary );
620  }
621  // Check size of boundary handling methods
622  else if ( boundaryHandlingVector.size( ) != NumberOfDimensions )
623  {
624  throw std::runtime_error( "Error while creating multi-dimensional interpolator. The number of boundary handling methods does not "
625  "match the number of dimensions." );
626  }
627 
628  // Check type of interpolator.
629  switch ( interpolatorSettings->getInterpolatorType( ) )
630  {
631  case multi_linear_interpolator:
632  {
633  createdInterpolator = std::make_shared< MultiLinearInterpolator
634  < IndependentVariableType, DependentVariableType, NumberOfDimensions > >(
635  independentValues, dependentData, interpolatorSettings->getSelectedLookupScheme( ),
636  interpolatorSettings->getBoundaryHandling( ), defaultExtrapolationValue );
637  break;
638  }
639  default:
640  throw std::runtime_error( "Error when making interpolator, function cannot be used to create interplator of type " +
641  std::to_string( interpolatorSettings->getInterpolatorType( ) ) );
642  }
643  return createdInterpolator;
644 }
