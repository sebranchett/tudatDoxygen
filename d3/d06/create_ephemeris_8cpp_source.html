35 boost::shared_ptr< ephemerides::Ephemeris > createBodyEphemeris(
36  const boost::shared_ptr< EphemerisSettings > ephemerisSettings,
37  const std::string& bodyName )
38 {
39  // Declare return object.
40  boost::shared_ptr< ephemerides::Ephemeris > ephemeris;
41 
42  if( ephemerisSettings->getMakeMultiArcEphemeris( ) )
43  {
44  std::map< double, boost::shared_ptr< Ephemeris > > singleArcEphemerides;
45  ephemerisSettings->resetMakeMultiArcEphemeris( false );
46 
47  singleArcEphemerides[ -std::numeric_limits< double >::lowest( ) ] = createBodyEphemeris(
48  ephemerisSettings, bodyName );
49 
50  ephemeris = boost::make_shared< MultiArcEphemeris >(
51  singleArcEphemerides, ephemerisSettings->getFrameOrigin( ), ephemerisSettings->getFrameOrientation( ) );
52  }
53  else
54  {
55 
56  // Check which type of ephemeris model is to be created.
57  switch( ephemerisSettings->getEphemerisType( ) )
58  {
59 #if USE_CSPICE
60  case direct_spice_ephemeris:
61  {
62  // Check consistency of type and class.
63  boost::shared_ptr< DirectSpiceEphemerisSettings > directEphemerisSettings =
64  boost::dynamic_pointer_cast< DirectSpiceEphemerisSettings >( ephemerisSettings );
65  if( directEphemerisSettings == NULL )
66  {
67  throw std::runtime_error(
68  "Error, expected direct spice ephemeris settings for body " + bodyName );
69  }
70  else
71  {
72  // Create corresponding ephemeris object.
73  ephemeris = boost::make_shared< SpiceEphemeris >(
74  bodyName,
75  directEphemerisSettings->getFrameOrigin( ),
76  directEphemerisSettings->getCorrectForStellarAberration( ),
77  directEphemerisSettings->getCorrectForLightTimeAberration( ),
78  directEphemerisSettings->getConvergeLighTimeAberration( ),
79  directEphemerisSettings->getFrameOrientation( ) );
80  }
81  break;
82  }
83  case interpolated_spice:
84  {
85  // Check consistency of type and class.
86  boost::shared_ptr< InterpolatedSpiceEphemerisSettings > interpolatedEphemerisSettings =
87  boost::dynamic_pointer_cast< InterpolatedSpiceEphemerisSettings >(
88  ephemerisSettings );
89  if( interpolatedEphemerisSettings == NULL )
90  {
91  throw std::runtime_error(
92  "Error, expected tabulated spice ephemeris settings for body " + bodyName );
93  }
94  else
95  {
96  // Since only the barycenters of planetary systems are included in the standard DE
97  // ephemerides, append 'Barycenter' to body name.
98  std::string inputName;
99  inputName = bodyName;
100  if( bodyName == "Mars" ||
101  bodyName == "Jupiter" || bodyName == "Saturn" ||
102  bodyName == "Uranus" || bodyName == "Neptune" )
103  {
104  inputName += " Barycenter";
105  std::cerr << "Warning, position of " << bodyName << " taken as barycenter of that body's "
106  << "planetary system." << std::endl;
107  }
108 
109  // Create corresponding ephemeris object.
110  if( !interpolatedEphemerisSettings->getUseLongDoubleStates( ) )
111  {
112  ephemeris = createTabulatedEphemerisFromSpice< double, double >(
113  inputName,
114  interpolatedEphemerisSettings->getInitialTime( ),
115  interpolatedEphemerisSettings->getFinalTime( ),
116  interpolatedEphemerisSettings->getTimeStep( ),
117  interpolatedEphemerisSettings->getFrameOrigin( ),
118  interpolatedEphemerisSettings->getFrameOrientation( ),
119  interpolatedEphemerisSettings->getInterpolatorSettings( ) );
120  }
121  else
122  {
123  ephemeris = createTabulatedEphemerisFromSpice< long double, double >(
124  inputName,
125  static_cast< long double >( interpolatedEphemerisSettings->getInitialTime( ) ),
126  static_cast< long double >( interpolatedEphemerisSettings->getFinalTime( ) ),
127  static_cast< long double >( interpolatedEphemerisSettings->getTimeStep( ) ),
128  interpolatedEphemerisSettings->getFrameOrigin( ),
129  interpolatedEphemerisSettings->getFrameOrientation( ),
130  interpolatedEphemerisSettings->getInterpolatorSettings( ) );
131  }
132  }
133  break;
134  }
135 #endif
136  case tabulated_ephemeris:
137  {
138  // Check consistency of type and class.
139  boost::shared_ptr< TabulatedEphemerisSettings > tabulatedEphemerisSettings =
140  boost::dynamic_pointer_cast< TabulatedEphemerisSettings >( ephemerisSettings );
141  if( tabulatedEphemerisSettings == NULL )
142  {
143  throw std::runtime_error(
144  "Error, expected tabulated ephemeris settings for body " + bodyName );
145  }
146  else
147  {
148  // Create corresponding ephemeris object.
149  if( !tabulatedEphemerisSettings->getUseLongDoubleStates( ) )
150  {
151  ephemeris = boost::make_shared< TabulatedCartesianEphemeris< > >(
152  boost::make_shared<
153  interpolators::LagrangeInterpolator< double, Eigen::Vector6d > >
154  ( tabulatedEphemerisSettings->getBodyStateHistory( ), 6,
155  interpolators::huntingAlgorithm,
156  interpolators::lagrange_cubic_spline_boundary_interpolation ),
157  tabulatedEphemerisSettings->getFrameOrigin( ),
158  tabulatedEphemerisSettings->getFrameOrientation( ) );
159  }
160  else
161  {
162  // Cast input history to required type.
163  std::map< double, Eigen::Vector6d > originalStateHistory =
164  tabulatedEphemerisSettings->getBodyStateHistory( );
165  std::map< double, Eigen::Matrix< long double, 6, 1 > > longStateHistory;
166 
167  for( std::map< double, Eigen::Vector6d >::const_iterator stateIterator =
168  originalStateHistory.begin( ); stateIterator != originalStateHistory.end( ); stateIterator++ )
169  {
170  longStateHistory[ stateIterator->first ] = stateIterator->second.cast< long double >( );
171  ephemeris =
172  boost::make_shared< TabulatedCartesianEphemeris< long double, double > >(
173  boost::make_shared< interpolators::LagrangeInterpolator<
174  double, Eigen::Matrix< long double, 6, 1 > > >
175  ( longStateHistory, 6,
176  interpolators::huntingAlgorithm,
177  interpolators::lagrange_cubic_spline_boundary_interpolation ),
178  tabulatedEphemerisSettings->getFrameOrigin( ),
179  tabulatedEphemerisSettings->getFrameOrientation( ) );
180  }
181  }
182  }
183  break;
184  }
185  case constant_ephemeris:
186  {
187  // Check consistency of type and class.
188  boost::shared_ptr< ConstantEphemerisSettings > constantEphemerisSettings =
189  boost::dynamic_pointer_cast< ConstantEphemerisSettings >( ephemerisSettings );
190  if( constantEphemerisSettings == NULL )
191  {
192  throw std::runtime_error( "Error, expected constant ephemeris settings for " + bodyName );
193  }
194  else
195  {
196  // Create ephemeris
197  ephemeris = boost::make_shared< ConstantEphemeris >(
198  boost::lambda::constant( constantEphemerisSettings->getConstantState( ) ),
199  constantEphemerisSettings->getFrameOrigin( ),
200  constantEphemerisSettings->getFrameOrientation( ) );
201  }
202  break;
203  }
204  case custom_ephemeris:
205  {
206  // Check consistency of type and class.
207  boost::shared_ptr< CustomEphemerisSettings > customEphemerisSettings =
208  boost::dynamic_pointer_cast< CustomEphemerisSettings >( ephemerisSettings );
209  if( customEphemerisSettings == NULL )
210  {
211  throw std::runtime_error( "Error, expected constant ephemeris settings for " + bodyName );
212  }
213  else
214  {
215  // Create ephemeris
216  ephemeris = boost::make_shared< CustomEphemeris >(
217  customEphemerisSettings->getCustomStateFunction( ),
218  customEphemerisSettings->getFrameOrigin( ),
219  customEphemerisSettings->getFrameOrientation( ) );
220  }
221  break;
222  }
223 
224  case kepler_ephemeris:
225  {
226  // Check consistency of type and class.
227  boost::shared_ptr< KeplerEphemerisSettings > keplerEphemerisSettings =
228  boost::dynamic_pointer_cast< KeplerEphemerisSettings >( ephemerisSettings );
229  if( keplerEphemerisSettings == NULL )
230  {
231  throw std::runtime_error( "Error, expected Kepler ephemeris settings for " + bodyName );
232  }
233  else
234  {
235  // Create ephemeris
236  ephemeris = boost::make_shared< KeplerEphemeris >(
237  keplerEphemerisSettings->getInitialStateInKeplerianElements( ),
238  keplerEphemerisSettings->getEpochOfInitialState( ),
239  keplerEphemerisSettings->getCentralBodyGravitationalParameter( ),
240  keplerEphemerisSettings->getFrameOrigin( ),
241  keplerEphemerisSettings->getFrameOrientation( ),
242  keplerEphemerisSettings->getRootFinderAbsoluteTolerance( ),
243  keplerEphemerisSettings->getRootFinderMaximumNumberOfIterations( ) );
244  }
245  break;
246  }
247  case approximate_planet_positions:
248  {
249  // Check consistency of type and class.
250  boost::shared_ptr< ApproximatePlanetPositionSettings > approximateEphemerisSettings =
251  boost::dynamic_pointer_cast< ApproximatePlanetPositionSettings >(
252  ephemerisSettings );
253  if( approximateEphemerisSettings == NULL )
254  {
255  throw std::runtime_error(
256  "Error, expected approximate ephemeris settings for body " + bodyName );
257  }
258  else
259  {
260  // Create corresponding ephemeris object.
261  if( approximateEphemerisSettings->getUseCircularCoplanarApproximation( ) )
262  {
263  ephemeris = boost::make_shared< ApproximatePlanetPositionsCircularCoplanar >(
264  approximateEphemerisSettings->getBodyIdentifier( ) );
265  }
266  else
267  {
268  ephemeris = boost::make_shared< ApproximatePlanetPositions >(
269  approximateEphemerisSettings->getBodyIdentifier( ) );
270  }
271  }
272  break;
273  }
274  default:
275  {
276  throw std::runtime_error(
277  "Error, did not recognize ephemeris model settings type " +
278  std::to_string( ephemerisSettings->getEphemerisType( ) ) );
279  }
280  }
281  }
282  return ephemeris;

287 std::pair< double, double > getSafeInterpolationInterval( const boost::shared_ptr< ephemerides::Ephemeris > ephemerisModel )
288 {
289  // Make default output pair
290  std::pair< double, double > safeInterval = std::make_pair(
291  std::numeric_limits< double >::lowest( ), std::numeric_limits< double >::max( ) );
292 
293  // Check if model is tabulated, and retrieve safe interval from model
294  if( isTabulatedEphemeris( ephemerisModel ) )
295  {
296  safeInterval = getTabulatedEphemerisSafeInterval( ephemerisModel );
297  }
298  // Check if model is multi-arc, and retrieve safe intervals from first and last arc.
299  else if( boost::dynamic_pointer_cast< ephemerides::MultiArcEphemeris >( ephemerisModel ) != NULL )
300  {
301  boost::shared_ptr< ephemerides::MultiArcEphemeris > multiArcEphemerisModel =
302  boost::dynamic_pointer_cast< ephemerides::MultiArcEphemeris >( ephemerisModel );
303  safeInterval.first = getSafeInterpolationInterval( multiArcEphemerisModel->getSingleArcEphemerides( ).at( 0 ) ).first;
304  safeInterval.second = getSafeInterpolationInterval(
305  multiArcEphemerisModel->getSingleArcEphemerides( ).at(
306  multiArcEphemerisModel->getSingleArcEphemerides( ).size( ) - 1 ) ).second;
307  }
308  return safeInterval;
309 }
