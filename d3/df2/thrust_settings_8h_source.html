610 class AccelerationLimitedThrottleGuidance: public ThrustInputParameterGuidance
611 {
612 public:
613 
615 
626  AccelerationLimitedThrottleGuidance(
627  const NamedBodyMap& bodyMap,
628  const std::string nameOfBodyWithGuidance,
629  const std::string nameOfCentralBody,
630  const std::vector< propulsion::ThrustIndependentVariables > independentVariables,
631  const std::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator,
632  const double maximumAcceleration ): ThrustInputParameterGuidance( 1, 0, true, 0 ),
633  bodyWithGuidance_( bodyMap.at( nameOfBodyWithGuidance ) ), thrustInterpolator_( thrustInterpolator ),
634  maximumAcceleration_( maximumAcceleration )
635  {
636  // Split independent variables into environmental/guidance.
637  int numberOfThrottles = 0;
638  std::vector< propulsion::ThrustIndependentVariables > guidanceFreeIndependentVariables;
639  for( unsigned int i = 0; i < independentVariables.size( ); i++ )
640  {
641  if( independentVariables.at( i ) == propulsion::throttle_dependent_thrust )
642  {
643  numberOfThrottles ++;
644  }
645  else if( independentVariables.at( i ) == propulsion::guidance_input_dependent_thrust )
646  {
647  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, guidance input not supported" );
648  }
649  else
650  {
651  guidanceFreeIndependentVariables.push_back( independentVariables.at( i ) );
652  }
653  }
654 
655  // Check input consistency.
656  if( numberOfThrottles > 1 )
657  {
658  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, multiple throttles detected" );
659  }
660 
661  // Create thrust input.
662  if( bodyWithGuidance_->getFlightConditions( ) == nullptr && nameOfCentralBody != "" )
663  {
664  bodyWithGuidance_->setFlightConditions(
665  createAtmosphericFlightConditions( bodyWithGuidance_,
666  bodyMap.at( nameOfCentralBody ),
667  nameOfBodyWithGuidance,
668  nameOfCentralBody ) );
669  }
670  thrustInputFunctions_ = getPropulsionInputVariables(
671  bodyWithGuidance_, guidanceFreeIndependentVariables );
672 
673  currentThrustInput_.resize( thrustInputFunctions_.size( ) );
674  }
675 
677 
681  void updateGuidanceParameters( )
682  {
683  // Compute environmental input variables for thrust.
684  for( unsigned int i = 0; i < thrustInputFunctions_.size( ); i++ )
685  {
686  currentThrustInput_[ i ] = thrustInputFunctions_.at( i )( );
687  }
688 
689  // Compute maximum thrust
690  double currentThrust = thrustInterpolator_->interpolate( currentThrustInput_ );
691  double currentMass = bodyWithGuidance_->getBodyMass( );
692 
693  // Set throttle (to < 1 if necessary)
694  if( currentThrust / currentMass < maximumAcceleration_ )
695  {
696  currentThrustGuidanceParameters_[ 0 ] = 1.0;
697  }
698  else
699  {
700  currentThrustGuidanceParameters_[ 0 ] = maximumAcceleration_ * currentMass / currentThrust;
701  }
702  }
703 
704 private:
705 
707  std::shared_ptr< simulation_setup::Body > bodyWithGuidance_;
708 
710  std::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator_;
711 
713  double maximumAcceleration_;
714 
716  std::vector< std::function< double( ) > > thrustInputFunctions_;
717 
719  std::vector< double > currentThrustInput_;
720 
721 };
