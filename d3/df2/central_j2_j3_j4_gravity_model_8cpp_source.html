25 Eigen::Vector3d computeGravitationalAccelerationDueToJ4(
26  const Eigen::Vector3d& positionOfBodySubjectToAcceleration,
27  const double gravitationalParameterOfBodyExertingAcceleration,
28  const double equatorialRadiusOfBodyExertingAcceleration,
29  const double j4CoefficientOfGravityField,
30  const Eigen::Vector3d& positionOfBodyExertingAcceleration )
31 {
32  // Set constant values reused for optimal computation of acceleration components.
33  const double distanceBetweenBodies = ( positionOfBodySubjectToAcceleration
34  - positionOfBodyExertingAcceleration ).norm( );
35 
36  const double preMultiplier = gravitationalParameterOfBodyExertingAcceleration
37  / std::pow( distanceBetweenBodies, 6.0 ) * 4.375 * j4CoefficientOfGravityField
38  * std::pow( equatorialRadiusOfBodyExertingAcceleration, 4.0 );
39 
40  const double scaledZCoordinate = ( positionOfBodySubjectToAcceleration.z( )
41  - positionOfBodyExertingAcceleration.z( ) )
42  / distanceBetweenBodies;
43 
44  const double scaledZCoordinateSquared = scaledZCoordinate * scaledZCoordinate;
45 
46  const double scaledZCoordinateToPower4 = scaledZCoordinateSquared * scaledZCoordinateSquared;
47 
48  const double factorForXAndYDirections = ( 3.0 / 7.0 - 6.0 * scaledZCoordinateSquared
49  + 9.0 * scaledZCoordinateToPower4 )
50  / distanceBetweenBodies;
51 
52  // Compute components of acceleration due to J4-effect.
53  Eigen::Vector3d gravitationalAccelerationDueToJ4 = Eigen::Vector3d::Constant( preMultiplier );
54 
55  gravitationalAccelerationDueToJ4( orbital_element_conversions::xCartesianPositionIndex )
56  *= ( positionOfBodySubjectToAcceleration.x( )
57  - positionOfBodyExertingAcceleration.x( ) ) * factorForXAndYDirections;
58 
59  gravitationalAccelerationDueToJ4( orbital_element_conversions::yCartesianPositionIndex )
60  *= ( positionOfBodySubjectToAcceleration.y( )
61  - positionOfBodyExertingAcceleration.y( ) ) * factorForXAndYDirections;
62 
63  gravitationalAccelerationDueToJ4( orbital_element_conversions::zCartesianPositionIndex )
64  *= ( 15.0 / 7.0 - 10.0 * scaledZCoordinateSquared + 9.0 * scaledZCoordinateToPower4 )
65  * scaledZCoordinate;
66 
67  return gravitationalAccelerationDueToJ4;
68 }

71 Eigen::Vector3d computeGravitationalAccelerationZonalSum(
72  const Eigen::Vector3d& positionOfBodySubjectToAcceleration,
73  const double gravitationalParameterOfBodyExertingAcceleration,
74  const double equatorialRadiusOfBodyExertingAcceleration,
75  const std::map< int, double > zonalCoefficientsOfGravityField,
76  const Eigen::Vector3d& positionOfBodyExertingAcceleration )
77 {
78  // Check that only coefficients for the gravity field are given up to J2 (i.e., size of
79  // vector is 3 at max), else throw an error.
80  if ( zonalCoefficientsOfGravityField.size( ) > 3 )
81  {
82  throw std::runtime_error(
83  "Currently, accelerations can only be computed up to J4." );
84  }
85 
86  // Check if position of body subject to acceleration falls within the effective radius of
87  // the central body. If so, throw an error.
88  if ( ( positionOfBodySubjectToAcceleration - positionOfBodyExertingAcceleration ).norm( )
89  < equatorialRadiusOfBodyExertingAcceleration )
90  {
91  throw std::runtime_error(
92  "Position of body subject to acceleration is within effective radius." );
93  }
94 
95  // Set gravitational acceleration sum equal to central term contribution.
96  Eigen::Vector3d gravitationalAccelerationSum
97  = computeGravitationalAcceleration(
98  positionOfBodySubjectToAcceleration,
99  gravitationalParameterOfBodyExertingAcceleration,
100  positionOfBodyExertingAcceleration );
101 
102  // Add contributions from zonal terms. The switch statement checks
103  for ( std::map< int, double >::const_iterator mapIterator
104  = zonalCoefficientsOfGravityField.begin( );
105  mapIterator != zonalCoefficientsOfGravityField.end( ); mapIterator++ )
106  {
107  switch ( mapIterator->first )
108  {
109  case 2:
110 
111  gravitationalAccelerationSum += computeGravitationalAccelerationDueToJ2(
112  positionOfBodySubjectToAcceleration,
113  gravitationalParameterOfBodyExertingAcceleration,
114  equatorialRadiusOfBodyExertingAcceleration,
115  mapIterator->second,
116  positionOfBodyExertingAcceleration );
117 
118  break;
119 
120  case 3:
121 
122  gravitationalAccelerationSum += computeGravitationalAccelerationDueToJ3(
123  positionOfBodySubjectToAcceleration,
124  gravitationalParameterOfBodyExertingAcceleration,
125  equatorialRadiusOfBodyExertingAcceleration,
126  mapIterator->second,
127  positionOfBodyExertingAcceleration );
128 
129  break;
130 
131  case 4:
132 
133  gravitationalAccelerationSum += computeGravitationalAccelerationDueToJ4(
134  positionOfBodySubjectToAcceleration,
135  gravitationalParameterOfBodyExertingAcceleration,
136  equatorialRadiusOfBodyExertingAcceleration,
137  mapIterator->second,
138  positionOfBodyExertingAcceleration );
139 
140  break;
141 
142  default:
143 
144  throw std::runtime_error(
145  "Degree must be 2, 3, or 4 in current implementation." );
146  };
147  }
148 
149  // Return total gravitational acceleration computed.
150  return gravitationalAccelerationSum;
151 }
