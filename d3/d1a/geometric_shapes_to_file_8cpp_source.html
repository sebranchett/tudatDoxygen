22 void writeSingleSurfaceGeometryPointsToFile(
23  geometric_shapes::SingleSurfaceGeometryPointer singleSurfaceGeometry,

26  const bool isIndependentVariableInverted )
27 {

29 
30  // Declare local variables.
31  // Declare grid sizes.
32  double gridSize1_;
33  double gridSize2_;
34 
35  // Declaration of vector which will be iteratively retrieved from geometry
36  // and written to file.
37  Eigen::VectorXd point = Eigen::VectorXd( 3 );
38 
39  // Open the file to which writing will take place if it is to overwrite any
40  // existing content.
41  if ( writeType == 0 )
42  {
43  // Open output file.
44  outputFile_.open( filename.c_str( ) );
45  }
46 
47  // Open the file to which writing will take place if it is to append to
48  // existing content.
49  else if ( writeType == 1 )
50  {
51  // Open output file with append option.
52  outputFile_.open( filename.c_str( ), std::ios::app );
53  }
54 
55 
56  // Sets the grid size in both directions.
57  if ( isIndependentVariableInverted == false )
58  {
59  // Set grid size 1.
60  gridSize1_ = ( singleSurfaceGeometry->getMaximumIndependentVariable( 1 )
61  - singleSurfaceGeometry->getMinimumIndependentVariable( 1 ) )
62  / ( numberOfLines - 1 );
63 
64  // Set grid size 2.
65  gridSize2_ = ( singleSurfaceGeometry->getMaximumIndependentVariable( 2 )
66  - singleSurfaceGeometry->getMinimumIndependentVariable( 2 ) )
67  / ( numberOfPoints - 1 );
68  }
69 
70  // Sets the grid size in both directions, inverted
71  // ( i.e., numberOfPoints corresponds to number of points in independent
72  // variable 2 direction and numberOfLines corresponds to number of points
73  // in indepedent variable 1 direction )
74  else
75  {
76  // Set grid size 1.
77  gridSize1_ = ( singleSurfaceGeometry->getMaximumIndependentVariable( 1 )
78  - singleSurfaceGeometry->getMinimumIndependentVariable( 1 ) )
79  / ( numberOfPoints - 1 );
80 
81  // Set grid size 1.
82  gridSize2_ = ( singleSurfaceGeometry->getMaximumIndependentVariable( 2 )
83  - singleSurfaceGeometry->getMinimumIndependentVariable( 2 ) )
84  / ( numberOfLines - 1 );
85  }
86 
87  // Iterate over all points, and retrieve the surface point from the
88  // surfaceGeometry object, and write it to the file.
89  for ( int i = 0; i < numberOfLines ; i++ )
90  {
91  for ( int j = 0; j < numberOfPoints ; j++ )
92  {
93  // If not inverted, all points from a single value of independent
94  // variable 2 are written first.
95  if ( isIndependentVariableInverted == false )
96  {
97  // Set point.
98  point = singleSurfaceGeometry->getSurfacePoint(
99  singleSurfaceGeometry->getMinimumIndependentVariable( 1 )
100  + i * gridSize1_,
101  singleSurfaceGeometry->getMinimumIndependentVariable( 2 )
102  + j * gridSize2_ );
103  }
104 
105  // If not inverted, all points from a single value of independent
106  // variable 2 are written first.
107  else
108  {
109  // Set point.
110  point = singleSurfaceGeometry->getSurfacePoint(
111  singleSurfaceGeometry->getMinimumIndependentVariable( 1 )
112  + j * gridSize1_,
113  singleSurfaceGeometry->getMinimumIndependentVariable( 2 )
114  + i * gridSize2_);
115  }
116 
117  // Write the x-, y- and z-coordinates, followed by a next line command.
118  outputFile_ << i+1 << " " << j+1 << " " << point( 0 ) << " " << point( 1 ) << " ";
119  outputFile_ << point( 2 ) << " " << std::endl;
120  }
121  }
122 
123  // Closing output file.
124  outputFile_.close( );
125 }
