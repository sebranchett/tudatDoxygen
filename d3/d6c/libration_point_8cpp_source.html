43 void LibrationPoint::computeLocationOfLibrationPoint(
44  LagrangeLibrationPoints lagrangeLibrationPoint )
45 {
46  using std::pow;
47  using std::sqrt;
48 
49  // Set functions for Newton-Raphson based on collinear libration point passed as input
50  // parameter, or computed locations directly of equilateral libration points.
51  switch( lagrangeLibrationPoint )
52  {
53  case l1:
54  {
55  // Create an object containing the function of which we whish to obtain the root from.
56  UnivariateProxyPointer rootFunction = boost::make_shared< UnivariateProxy >(
57  boost::bind( &LibrationPoint::computeL1LocationFunction, this, _1 ) );
58 
59  // Add the first derivative of the root function.
60  rootFunction->addBinding( -1, boost::bind( &LibrationPoint::
61  computeL1FirstDerivativeLocationFunction, this, _1 ) );
62 
63  // Set position vector of L1 in Cartesian elements based on result of Newton-Raphson
64  // root-finding algorithm.
65  positionOfLibrationPoint_ << rootFinder->execute( rootFunction, 1.0 ), 0.0, 0.0;
66  }
67  break;
68 
69  case l2:
70  {
71  // Create an object containing the function of which we whish to obtain the root from.
72  UnivariateProxyPointer rootFunction = boost::make_shared< UnivariateProxy >(
73  boost::bind( &LibrationPoint::computeL2LocationFunction, this, _1 ) );
74 
75  // Add the first derivative of the root function.
76  rootFunction->addBinding( -1, boost::bind( &LibrationPoint::
77  computeL2FirstDerivativeLocationFunction, this, _1 ) );
78 
79  // Set position vector of L1 in Cartesian elements based on result of Newton-Raphson
80  // root-finding algorithm.
81  positionOfLibrationPoint_ << rootFinder->execute( rootFunction, 1.0 ), 0.0, 0.0;
82  }
83  break;
84 
85  case l3:
86  {
87  // Create an object containing the function of which we whish to obtain the root from.
88  UnivariateProxyPointer rootFunction = boost::make_shared< UnivariateProxy >(
89  boost::bind( &LibrationPoint::computeL3LocationFunction, this, _1 ) );
90 
91  // Add the first derivative of the root function.
92  rootFunction->addBinding( -1, boost::bind( &LibrationPoint::
93  computeL3FirstDerivativeLocationFunction, this, _1 ) );
94 
95  // Set position vector of L1 in Cartesian elements based on result of Newton-Raphson
96  // root-finding algorithm.
97  positionOfLibrationPoint_ << rootFinder->execute( rootFunction, -1.0 ), 0.0, 0.0;
98  }
99  break;
100 
101  case l4:
102 
103  // Set position vector of L4 in Cartesian elements.
104  positionOfLibrationPoint_.x( ) = 0.5 - massParameter;
105  positionOfLibrationPoint_.y( ) = 0.5 * sqrt( 3.0 );
106  positionOfLibrationPoint_.z( ) = 0.0;
107 
108  break;
109 
110  case l5:
111 
112  // Set position vector of L5 in Cartesian elements.
113  positionOfLibrationPoint_.x( ) = 0.5 - massParameter;
114  positionOfLibrationPoint_.y( ) = -0.5 * sqrt( 3.0 );
115  positionOfLibrationPoint_.z( ) = 0.0;
116 
117  break;
118 
119  default:
120 
121  throw std::runtime_error(
122  "The Lagrange libration point requested does not exist." );
123  };
124 }
