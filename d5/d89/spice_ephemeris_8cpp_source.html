28 SpiceEphemeris::SpiceEphemeris( const std::string& targetBodyName,
29  const std::string& observerBodyName,
30  const bool correctForStellarAbberation,
31  const bool correctForLightTimeAbberation,
32  const bool convergeLighTimeAbberation,
33  const std::string& referenceFrameName,
34  const double referenceJulianDay )
35  : Ephemeris( observerBodyName, referenceFrameName ),
36  targetBodyName_( targetBodyName )
37 {
38  referenceDayOffSet_ = ( referenceJulianDay - basic_astrodynamics::JULIAN_DAY_ON_J2000 ) * physical_constants::JULIAN_DAY;
39 
40  // Check consistency of input.
41  if ( correctForLightTimeAbberation == 0 && convergeLighTimeAbberation == 1 )
42  {
43  throw std::runtime_error(
44  "Error, requested multiple iterations for light time correction, but not light time correction itself." );
45  }
46 
47  if ( correctForLightTimeAbberation == 0 && correctForStellarAbberation == 1 )
48  {
49 
50  throw std::runtime_error(
51  "Error, requested stellar aberration, but not light-time correction, this is not supprted by spice." );
52  }
53 
54  // Set aberration corrections variable.
55  abberationCorrections_ = "";
56  if ( correctForLightTimeAbberation && !convergeLighTimeAbberation )
57  {
58  abberationCorrections_.append( "LT" );
59  }
60 
61  else if ( correctForLightTimeAbberation && convergeLighTimeAbberation )
62  {
63  abberationCorrections_.append( "CN" );
64  }
65 
66  else if ( !correctForLightTimeAbberation && !correctForStellarAbberation )
67  {
68  abberationCorrections_.append( "NONE" );
69  }
70 
71  if ( correctForLightTimeAbberation && correctForStellarAbberation )
72  {
73  abberationCorrections_.append( "+S" );
74  }
