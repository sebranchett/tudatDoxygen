26 boost::filesystem::path getPathForJSONFile( const std::string& file, const boost::filesystem::path& basePath )
27 {
28  // Get absolute path to input file
29  boost::filesystem::path filePath = file;
30  if ( ! filePath.is_absolute( ) )
31  {
32  filePath = basePath / filePath;
33  }
34 
35  if ( boost::filesystem::exists( filePath ) )
36  {
37  if ( boost::filesystem::is_directory( filePath ) )
38  {
39  // If it's a directory, try to find a "main.json" file in that directory
40  return getPathForJSONFile( "main.json", filePath );
41  }
42  else
43  {
44  return filePath;
45  }
46  }
47  else
48  {
49  // Try appending extension
50  const boost::filesystem::path originalFilePath = filePath;
51  filePath += ".json";
52  if ( boost::filesystem::exists( filePath ) )
53  {
54  return filePath;
55  }
56  else
57  {
58  std::cerr << "The file " << originalFilePath << " does not exist." << std::endl;
59  throw;
60  }
61  }
62 }

72 std::pair< unsigned int, unsigned int > getLineAndCol( std::ifstream& stream, const std::streampos position )
73 {
74  std::streampos originalPos = stream.tellg( );
75  stream.seekg( 0, std::ifstream::beg );
76 
77  std::string lineContents;
78  unsigned int line = 1;
79  std::streampos col;
80  std::streampos lastPos = stream.tellg( );
81  while ( std::getline( stream, lineContents ) )
82  {
83  const std::streampos currentPos = stream.tellg( );
84  if ( currentPos > position )
85  {
86  col = position - lastPos;
87  break;
88  }
89  lastPos = currentPos;
90  line++;
91  }
92 
93  stream.seekg( originalPos );
94  return { line, col };
95 }

121 void updatePaths( nlohmann::json& jsonObject, const boost::filesystem::path& filePath, const boost::filesystem::path& parentFilePath, const boost::filesystem::path& rootFilePath )
122 {
123  if ( jsonObject.is_structured( ) )
124  {
125  for ( nlohmann::json::iterator it = jsonObject.begin( ); it != jsonObject.end( ); ++it )
126  {
127  nlohmann::json& subjson = it.value( );
128  updatePaths( subjson, filePath, parentFilePath, rootFilePath );
129  }
130  }
131  else if ( jsonObject.is_string( ) )
132  {
133  std::string str = jsonObject;
134 
135  // Replace: ${FILE_DIR}, ${FILE_STEM}, ${FILE_NAME}, etc.
136  str = regex_replace( str, boost::regex( R"(\$\{FILE_DIR\})" ), filePath.parent_path( ).string( ) );
137  str = regex_replace( str, boost::regex( R"(\$\{FILE_STEM\})" ), filePath.stem( ).string( ) );
138  str = regex_replace( str, boost::regex( R"(\$\{FILE_NAME\})" ), filePath.filename( ).string( ) );
139  str = regex_replace( str, boost::regex( R"(\$\{PARENT_FILE_DIR\})" ), parentFilePath.parent_path( ).string( ) );
140  str = regex_replace( str, boost::regex( R"(\$\{PARENT_FILE_STEM\})" ), parentFilePath.stem( ).string( ) );
141  str = regex_replace( str, boost::regex( R"(\$\{PARENT_FILE_NAME\})" ), parentFilePath.filename( ).string( ) );
142  str = regex_replace( str, boost::regex( R"(\$\{ROOT_FILE_DIR\})" ), rootFilePath.parent_path( ).string( ) );
143  str = regex_replace( str, boost::regex( R"(\$\{ROOT_FILE_STEM\})" ), rootFilePath.stem( ).string( ) );
144  str = regex_replace( str, boost::regex( R"(\$\{ROOT_FILE_NAME\})" ), rootFilePath.filename( ).string( ) );
145 
146  // Fix relative paths
147  boost::cmatch groups;
148  boost::regex_match( str.c_str( ), groups, boost::regex( R"(\@path\((.*)\))" ) );
149  if ( groups[ 1 ].matched )
150  {
151  const boost::filesystem::path providedPath = std::string( groups[ 1 ] );
152  if ( providedPath.is_relative( ) )
153  {
154  const boost::filesystem::path rel = boost::filesystem::relative( filePath.parent_path( ), rootFilePath.parent_path( ) );
155  str = ( *rel.filename( ).c_str( ) == '.' ? providedPath : rel / providedPath ).string( );
156  }
157  else
158  {
159  const boost::filesystem::path rel = boost::filesystem::relative( providedPath, rootFilePath.parent_path( ) );
160  str = ( ( ! rel.empty( ) && rel.size( ) < providedPath.size( ) ) ? rel : providedPath ).string( );
161  }
162  }
163 
164  jsonObject = str;
165  }
166 }

169 nlohmann::json readJSON( const boost::filesystem::path& filePath, const boost::filesystem::path& parentFilePath, const boost::filesystem::path& rootFilePath )
170 {
171  std::ifstream stream( filePath.string( ) );
172  nlohmann::json jsonObject;
173  try
174  {
175  jsonObject = nlohmann::json::parse( stream );
176  }
177  catch ( const nlohmann::detail::parse_error& error )
178  {
179  std::pair< unsigned int, unsigned int > errorLineCol = getLineAndCol( stream, error.byte );
180  std::cerr << "Parse error in file " << filePath
181  << " at line " << errorLineCol.first << ", col " << errorLineCol.second << "." << std::endl;
182  throw error;
183  }
184  updatePaths( jsonObject, filePath, parentFilePath.empty( ) ? filePath : parentFilePath,
185  rootFilePath.empty( ) ? filePath : rootFilePath );
186  return jsonObject;
187 }

202 void mergeJSON( nlohmann::json& jsonObject, const boost::filesystem::path& filePath )
203 {
204  if ( jsonObject.is_array( ) )
205  {
206  nlohmann::json jsonArray = jsonObject;
207  for ( nlohmann::json::iterator it = jsonArray.begin( ); it != jsonArray.end( ); ++it )
208  {
209  nlohmann::json subjson = it.value( );
210  mergeJSON( subjson, filePath );
211  if ( it == jsonArray.begin( ) )
212  {
213  jsonObject = subjson;
214  }
215  else
216  {
217  for ( nlohmann::json::iterator subit = subjson.begin( ); subit != subjson.end( ); ++subit )
218  {
219  const KeyPath keyPath( subit.key( ) );
220  try
221  {
222  nlohmann::json newValue = subit.value( );
223  for ( unsigned int j = 0; j < keyPath.size( ); ++j )
224  {
225  nlohmann::json updatedsonObject = jsonObject;
226  for ( unsigned int i = 0; i < keyPath.size( ) - j; ++i )
227  {
228  const std::string key = keyPath.at( i );
229  if ( i < keyPath.size( ) - 1 - j )
230  {
231  updatedsonObject = valueAt( updatedsonObject, key );
232  }
233  else
234  {
235  valueAt( updatedsonObject, key, true ) = newValue;
236  }
237  }
238  newValue = updatedsonObject;
239  }
240  jsonObject = newValue;
241  }
242  catch ( ... )
243  {
244  std::cerr << "Could not update value for " << keyPath << " referenced from file "
245  << filePath << std::endl;
246  throw;
247  }
248  }
249  }
250  }

285 void parseModularJSON( nlohmann::json& jsonObject, const boost::filesystem::path& filePath,
286  nlohmann::json parentObject = nlohmann::json( ), boost::filesystem::path rootFilePath = boost::filesystem::path( ) )
287 {
288  if ( parentObject.is_null( ) )
289  {
290  parentObject = jsonObject;
291  }
292  if ( rootFilePath.empty( ) )
293  {
294  rootFilePath = filePath;
295  }
296  if ( jsonObject.is_object( ) )
297  {
298  for ( nlohmann::json::iterator it = jsonObject.begin( ); it != jsonObject.end( ); ++it )
299  {
300  nlohmann::json& subjson = it.value( );
301  parseModularJSON( subjson, filePath, jsonObject, rootFilePath );
302  }
303  }
304  else if ( jsonObject.is_array( ) )
305  {
306  for ( unsigned int i = 0; i < jsonObject.size( ); ++i )
307  {
308  nlohmann::json& subjson = jsonObject.at( i );
309  parseModularJSON( subjson, filePath, jsonObject, rootFilePath );
310  }
311  }
312  else if ( jsonObject.is_string( ) )
313  {
314  std::string objectString = jsonObject.get< std::string >( );
315  if( objectString.size( ) > 3 )
316  {
317  if( objectString.at( 0 ) == '$' )
318  {
319  int fileStartIndex = -1;
320  int fileStringSize = 0;
321 
322  int variableStartIndex = -1;
323  int variableStringSize = 0;
324 
325  if( objectString.at( 1 ) == '(' )
326  {
327  fileStartIndex = 2;
328  }
329  else if( objectString.at( 1 ) == '{' )
330  {
331  variableStartIndex = 2;
332  }
333  for( unsigned int i = 2; i < objectString.size( ); i++ )
334  {
335  if( objectString.at( i ) == '{' )
336  {
337  if( variableStartIndex == -1 )
338  {
339  variableStartIndex = i + 1;
340  }
341  else
342  {
343  throw std::runtime_error( "Error in JSON file deserialization, found '{' twice" );
344  }
345  }
346  else if( objectString.at( i ) == '}' )
347  {
348  if( variableStartIndex != -1 )
349  {
350  variableStringSize = i - variableStartIndex;
351  }
352  else
353  {
354  throw std::runtime_error( "Error in JSON file deserialization, found '}' before '{'" );
355  }
356  }
357  else if( objectString.at( i ) == ')' )
358  {
359  if( fileStartIndex != -1 )
360  {
361  fileStringSize = i - fileStartIndex;
362  }
363  else
364  {
365  throw std::runtime_error( "Error in JSON file deserialization, found ')' before '('" );
366  }
367  }
368  }
369 
370  if( fileStringSize > 0 || variableStringSize > 0 )
371  {
372  std::string file = "";
373  std::string vars = "";
374 
375  if( fileStringSize > 0 )
376  {
377  file = objectString.substr( fileStartIndex, fileStringSize );
378  }
379 
380  if( variableStringSize > 0 )
381  {
382  vars = objectString.substr( variableStartIndex, variableStringSize );
383  }
384 
385  const boost::filesystem::path importPath =
386  file.empty( ) ? filePath : getPathForJSONFile( file, filePath.parent_path( ) );
387  const nlohmann::json importedJsonObject = readJSON( importPath, filePath, rootFilePath );
388  std::vector< std::string > keys;
389  std::vector< KeyPath > keyPaths;
390  if ( vars.empty( ) )
391  {
392  keyPaths = { KeyPath( ) };
393  }
394  else
395  {
396  for ( const std::string variable : split( vars, ',' ) )
397  {
398  const std::vector< std::string > keyVar = split( variable, ':' );
399  if ( keyVar.size( ) == 2 )
400  {
401  keys.push_back( keyVar.front( ) );
402  }
403  keyPaths.push_back( KeyPath( keyVar.back( ) ) );
404  }
405  }
406  nlohmann::json parsedJsonObject;
407  for ( unsigned int i = 0; i < keyPaths.size( ); ++i )
408  {
409  const KeyPath keyPath = keyPaths.at( i );
410  nlohmann::json subJsonObject = importedJsonObject;
411  if ( keyPath.empty( ) )
412  {
413  parseModularJSON( subJsonObject, importPath, parentObject, rootFilePath );
414  }
415  else
416  {
417  try
418  {
419  // Recursively update jsonObject for every key in keyPath
420  for ( const std::string key : keyPath )
421  {
422  subJsonObject = valueAt( subJsonObject, key );
423  parseModularJSON( subJsonObject, importPath, parentObject, rootFilePath );
424  }
425  }
426  catch ( ... )
427  {
428  std::cerr << "Could not load ";
429  if ( ! keyPath.empty( ) )
430  {
431  std::cerr << "value for " << keyPath << " from ";
432  }
433  std::cerr << "file " << importPath << " referenced from file " << filePath << std::endl;
434  throw;
435  }
436  }
437  if ( keys.size( ) > 0 )
438  {
439  parsedJsonObject[ keys.at( i ) ] = subJsonObject;
440  }
441  else
442  {
443  parsedJsonObject[ i ] = subJsonObject;
444  }
445  }
446  if ( keyPaths.size( ) == 1 && keys.size( ) == 0 && vars.find( ',' ) == std::string::npos )
447  {
448  jsonObject = parsedJsonObject.front( );
449  }
450  else
451  {
452  jsonObject = parsedJsonObject;
453  }
454  }
455  else
456  {
457  throw std::runtime_error( "Error in JSON file deserialization, found none of '(', ')' '{' or '}' in string starting with '$' " );
458  }
459  }
460  }
461  }
462 }
