158  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
159  {
160  // Loop over each body
161  Eigen::Matrix< StateScalarType, 3, 1 > exponentialMapVector;
162  StateScalarType exponentialMapMagnitude;
163  for( unsigned int i = 0; i < this->bodiesToPropagate_.size( ); i++ )
164  {
165  // Convert to/from shadow exponential map (SEM) (transformation is the same either way)
166  exponentialMapVector = unprocessedState.block( i * 7, 0, 3, 1 );
167  exponentialMapMagnitude = exponentialMapVector.norm( );
168  if ( exponentialMapMagnitude >= mathematical_constants::PI )
169  {
170  // Invert flag
171  unprocessedState.segment( i * 7 + 3, 1 ) = ( unprocessedState.block( i * 7 + 3, 0, 1, 1 ) -
172  Eigen::Matrix< StateScalarType, 1, 1 >::Ones( ) ).cwiseAbs( );
173 
174  // Convert to EM/SEM
175  exponentialMapVector *= ( 1.0 - ( 2.0 * mathematical_constants::PI / exponentialMapMagnitude ) );
176 
177  // Replace EM with SEM, or vice-versa
178  unprocessedState.segment( i * 7, 3 ) = exponentialMapVector;
179  }
180  }
181  }
