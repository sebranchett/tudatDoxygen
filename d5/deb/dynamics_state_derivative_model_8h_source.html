63  DynamicsStateDerivativeModel(
64  const std::vector< std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > >
65  stateDerivativeModels,
66  const std::function< void(
67  const TimeType, const std::unordered_map< IntegratedStateType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >&,
68  const std::vector< IntegratedStateType > ) > environmentUpdateFunction,
69  const std::shared_ptr< VariationalEquations > variationalEquations =
70  std::shared_ptr< VariationalEquations >( ) ):
71  environmentUpdateFunction_( environmentUpdateFunction ), variationalEquations_( variationalEquations ),
72  functionEvaluationCounter_( 0 )
73  {
74  std::vector< IntegratedStateType > stateTypeList;
75  totalConventionalStateSize_ = 0;
76  totalPropagatedStateSize_ = 0;
77 
78  // Iterate over vector of state derivative models, check validity, set member variable map
79  // stateDerivativeModels_ and size indices.
80  for( unsigned int i = 0; i < stateDerivativeModels.size( ); i++ )
81  {
82  if( i > 0 )
83  {
84  if( ( std::find( stateTypeList.begin( ), stateTypeList.end( ),
85  stateDerivativeModels.at( i )->getIntegratedStateType( ) )
86  != stateTypeList.end( ) )
87  && ( stateDerivativeModels.at( i )->getIntegratedStateType( )
88  != stateDerivativeModels.at( i - 1 )->getIntegratedStateType( ) ) )
89  {
90  throw std::runtime_error( "Warning when making hybrid state derivative models, state type " +
91  std::to_string( stateDerivativeModels.at( i )->getIntegratedStateType( ) )
92  + " entries are non-contiguous" );
93  }
94  }
95 
96  // Check uniqueness of state derivative type calculator in list
97  if( std::find( stateTypeList.begin( ), stateTypeList.end( ),
98  stateDerivativeModels.at( i )->getIntegratedStateType( ) ) == stateTypeList.end( ) )
99  {
100  conventionalStateTypeSize_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ] = 0;
101  conventionalStateTypeStartIndex_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ]
102  = totalConventionalStateSize_;
103 
104  propagatedStateTypeSize_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ] = 0;
105  propagatedStateTypeStartIndex_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ]
106  = totalPropagatedStateSize_;
107  }
108  stateTypeList.push_back( stateDerivativeModels.at( i )->getIntegratedStateType( ) );
109 
110  // Set state part sizes
111  conventionalStateIndices_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ].push_back(
112  std::make_pair( totalConventionalStateSize_, stateDerivativeModels.at( i )->getConventionalStateSize( ) ) );
113  totalConventionalStateSize_ += stateDerivativeModels.at( i )->getConventionalStateSize( );
114 
115  propagatedStateIndices_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ].push_back(
116  std::make_pair( totalPropagatedStateSize_, stateDerivativeModels.at( i )->getPropagatedStateSize( ) ) );
117  totalPropagatedStateSize_ += stateDerivativeModels.at( i )->getPropagatedStateSize( );
118 
119  conventionalStateTypeSize_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ] +=
120  stateDerivativeModels.at( i )->getConventionalStateSize( );
121  propagatedStateTypeSize_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ] +=
122  stateDerivativeModels.at( i )->getPropagatedStateSize( );
123 
124  // Set current model in member map.
125  stateDerivativeModels_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ].push_back(
126  stateDerivativeModels.at( i ) );
127 
128  currentStatesPerTypeInConventionalRepresentation_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ] =
129  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero(
130  conventionalStateTypeSize_.at( stateDerivativeModels.at( i )->getIntegratedStateType( ) ), 1 );
131  }
132  }

146  StateType computeStateDerivative( const TimeType time, const StateType& state )
147  {
148 // std::cout << "Computing state derivative: " <<time<<" "<<state.transpose( ) << std::endl;
149 
150  // Initialize state derivative
151  if( stateDerivative_.rows( ) != state.rows( ) || stateDerivative_.cols( ) != state.cols( ) )
152  {
153  stateDerivative_.resize( state.rows( ), state.cols( ) );
154  }
155 
156  // If dynamical equations are integrated, update the environment with the current state.
157  if( evaluateDynamicsEquations_ )
158  {
159  // Iterate over all types of equations.
160  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
161  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
162  stateDerivativeModelsIterator_++ )
163  {
164  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
165  {
166  stateDerivativeModelsIterator_->second.at( i )->clearStateDerivativeModel( );
167  }
168  }
169 
170  convertCurrentStateToGlobalRepresentationPerType( state, time, evaluateVariationalEquations_ );
171  environmentUpdateFunction_( time, currentStatesPerTypeInConventionalRepresentation_,
172  integratedStatesFromEnvironment_ );
173  }
174  else
175  {
176  environmentUpdateFunction_(
177  time, std::unordered_map<
178  IntegratedStateType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >( ),
179  integratedStatesFromEnvironment_ );
180  }
181 
182  if( evaluateVariationalEquations_ )
183  {
184  variationalEquations_->clearPartials( );
185  }
186 
187  // If dynamical equations are integrated, evaluate dynamics state derivatives.
188  std::pair< int, int > currentIndices;
189  if( evaluateDynamicsEquations_ )
190  {
191  // Iterate over all types of equations.
192  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
193  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
194  stateDerivativeModelsIterator_++ )
195  {
196  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
197  {
198  // Update state derivative models
199  stateDerivativeModelsIterator_->second.at( i )->updateStateDerivativeModel( time );
200  }
201  }
202 
203  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
204  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
205  stateDerivativeModelsIterator_++ )
206  {
207  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
208  {
209  // Evaluate and set current dynamical state derivative
210  currentIndices = propagatedStateIndices_.at( stateDerivativeModelsIterator_->first ).at( i );
211 
212  stateDerivativeModelsIterator_->second.at( i )->calculateSystemStateDerivative(
213  time, state.block( currentIndices.first, dynamicsStartColumn_, currentIndices.second, 1 ),
214  stateDerivative_.block( currentIndices.first, dynamicsStartColumn_, currentIndices.second, 1 ) );
215  }
216  }
217  }
218 
219  // If variational equations are to be integrated: evaluate and set.
220  if( evaluateVariationalEquations_ )
221  {
222  variationalEquations_->updatePartials( time );
223 
224  variationalEquations_->evaluateVariationalEquations< StateScalarType >(
225  time, state.block( 0, 0, totalConventionalStateSize_, variationalEquations_->getNumberOfParameterValues( ) ),
226  stateDerivative_.block( 0, 0, totalConventionalStateSize_, variationalEquations_->getNumberOfParameterValues( ) ) );
227  }
228 
229  // Update counters
230  functionEvaluationCounter_++;
231  cumulativeFunctionEvaluationCounter_[ time ] = functionEvaluationCounter_;
232 
233  return stateDerivative_;
234  }

261  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > convertFromOutputSolution(
262  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& outputState,
263  const TimeType& time )
264  {
265  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > internalState =
266  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( totalPropagatedStateSize_, 1 );
267 
268  // Iterate over all state derivative models and convert associated state entries
269  std::vector< std::pair< int, int > > currentConventionalStateIndices;
270  std::vector< std::pair< int, int > > currentPropagatedStateIndices;
271  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
272  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
273  stateDerivativeModelsIterator_++ )
274  {
275  currentConventionalStateIndices = conventionalStateIndices_.at( stateDerivativeModelsIterator_->first );
276  currentPropagatedStateIndices = propagatedStateIndices_.at( stateDerivativeModelsIterator_->first );
277  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
278  {
279  internalState.segment( currentPropagatedStateIndices.at( i ).first,
280  currentPropagatedStateIndices.at( i ).second ) =
281  stateDerivativeModelsIterator_->second.at( i )->convertFromOutputSolution(
282  outputState.segment( currentConventionalStateIndices.at( i ).first,
283  currentConventionalStateIndices.at( i ).second ), time );
284  }
285  }
286 
287  return internalState;
288  }

302  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > convertToOutputSolution(
303  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& internalSolution,
304  const TimeType& time )
305  {
306  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > outputState =
307  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( totalConventionalStateSize_, 1 );
308 
309  // Iterate over all state derivative models and convert associated state entries
310  std::vector< std::pair< int, int > > currentConventionalStateIndices;
311  std::vector< std::pair< int, int > > currentPropagatedStateIndices;
312  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
313  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
314  stateDerivativeModelsIterator_++ )
315  {
316  currentConventionalStateIndices = conventionalStateIndices_.at( stateDerivativeModelsIterator_->first );
317  currentPropagatedStateIndices = propagatedStateIndices_.at( stateDerivativeModelsIterator_->first );
318  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
319  {
320  stateDerivativeModelsIterator_->second.at( i )->convertToOutputSolution(
321  internalSolution.segment(
322  currentPropagatedStateIndices.at( i ).first,
323  currentPropagatedStateIndices.at( i ).second ), time,
324  outputState.block( currentConventionalStateIndices.at( i ).first, 0,
325  currentConventionalStateIndices.at( i ).second, 1 ) );
326  }
327  }
328 
329  return outputState;
330  }

459  void updateStateDerivativeModelSettings(
460  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialBodyStates )
461  {
462  // Iterate over all dynamics types
463  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
464  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
465  stateDerivativeModelsIterator_++ )
466  {
467  switch( stateDerivativeModelsIterator_->first )
468  {
469  case translational_state:
470  {
471  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
472  {
473  std::shared_ptr< NBodyStateDerivative< StateScalarType, TimeType > > currentTranslationalStateDerivative =
474  std::dynamic_pointer_cast< NBodyStateDerivative< StateScalarType, TimeType > >(
475  stateDerivativeModelsIterator_->second.at( i ) );
476  switch( currentTranslationalStateDerivative->getTranslationalPropagatorType( ) )
477  {
478  case cowell:
479  break;
480  case encke:
481  throw std::runtime_error( "Error, reference orbit not reset in Encke propagator" );
482  break;
483  case gauss_keplerian:
484  break;
485  case gauss_modified_equinoctial:
486  break;
487  case unified_state_model_quaternions:
488  break;
489  case unified_state_model_modified_rodrigues_parameters:
490  break;
491  case unified_state_model_exponential_map:
492  break;
493  default:
494  throw std::runtime_error( "Error when updating state derivative model settings, did not recognize translational propagator type" );
495  break;
496  }
497  }
498  }
499  case rotational_state:
500  break;
501  case body_mass_state:
502  break;
503  case custom_state:
504  break;
505  default:
506  throw std::runtime_error( "Error when updating state derivative model settings, did not recognize dynamics type" );
507  break;
508  }
509  }
510  }

593  void convertCurrentStateToGlobalRepresentationPerType(
594  const StateType& state, const TimeType& time, const bool stateIncludesVariationalState )
595  {
596  int startColumn = 0;
597  if( stateIncludesVariationalState )
598  {
599  startColumn = variationalEquations_->getNumberOfParameterValues( );
600  }
601  else
602  {
603  startColumn = 0;
604  }
605 
606  std::pair< int, int > currentPropagatedIndices, currentConventionalIndices;
607 
608  // Iterate over all state derivative models
609  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
610  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
611  stateDerivativeModelsIterator_++ )
612  {
613  int currentStateTypeSize = 0;
614 
615  // Iterate over all state derivative models of current type
616  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
617  {
618  // Get state block indices of current state derivative model
619  currentPropagatedIndices = propagatedStateIndices_.at( stateDerivativeModelsIterator_->first ).at( i );
620  currentConventionalIndices = conventionalStateIndices_.at( stateDerivativeModelsIterator_->first ).at( i );
621 
622  // Set current block in split state (in global form)
623  stateDerivativeModelsIterator_->second.at( i )->convertCurrentStateToGlobalRepresentation(
624  state.block( currentPropagatedIndices.first, startColumn, currentPropagatedIndices.second, 1 ), time,
625  currentStatesPerTypeInConventionalRepresentation_.at(
626  stateDerivativeModelsIterator_->first ).block(
627  currentStateTypeSize, 0, currentConventionalIndices.second, 1 ) );
628 // std::cout << "Computing Cart. state: " <<currentStatesPerTypeInConventionalRepresentation_.at(
629 // stateDerivativeModelsIterator_->first ).transpose( ) << std::endl;
630 
631  }
632  }

727 template< typename TimeType = double, typename StateScalarType = double >
728 std::vector< std::shared_ptr< basic_astrodynamics::AccelerationModel3d > > getAccelerationBetweenBodies(
729  const std::string bodyUndergoingAcceleration,
730  const std::string bodyExertingAcceleration,
731  const std::unordered_map< IntegratedStateType,
732  std::vector< std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > > >& stateDerivativeModels,
733  const basic_astrodynamics::AvailableAcceleration accelerationModelType )
734 {

736  listOfSuitableAccelerationModels;

740  {

742  std::dynamic_pointer_cast< NBodyStateDerivative< StateScalarType, TimeType > >(
743  stateDerivativeModels.at( propagators::translational_state ).at( 0 ) )->getAccelerationsMap( );
744 
745  if( accelerationModelList.count( bodyUndergoingAcceleration ) == 0 )
746  {
747

749  bodyUndergoingAcceleration + " are found";

752  else

756  {
757  std::string errorMessage = "Error when getting acceleration between bodies, no translational dynamics models by " +
758  bodyExertingAcceleration + " acting on " + bodyUndergoingAcceleration + " are found";
759  throw std::runtime_error( errorMessage );

761  else
762  {

764  listOfSuitableAccelerationModels = basic_astrodynamics::getAccelerationModelsOfType(
765  accelerationModelList.at( bodyUndergoingAcceleration ).at( bodyExertingAcceleration ), accelerationModelType );
766  }
