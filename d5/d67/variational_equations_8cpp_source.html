39 void VariationalEquations::setBodyStatePartialMatrix( )
40 {
41  // Initialize partial matrix
42  variationalMatrix_.setZero( );
43 
44  if( dynamicalStatesToEstimate_.count( propagators::translational_state ) > 0 )
45  {
46  int startIndex = stateTypeStartIndices_.at( propagators::translational_state );
47  for( unsigned int i = 0; i < dynamicalStatesToEstimate_.at( propagators::translational_state ).size( ); i++ )
48  {
49  variationalMatrix_.block( startIndex + i * 6, startIndex + i * 6 + 3, 3, 3 ).setIdentity( );
50  }
51  }
52 
53  // Iterate over all bodies undergoing accelerations for which initial condition is to be estimated.
54  for( std::map< IntegratedStateType, std::vector< std::multimap< std::pair< int, int >,
55  std::function< void( Eigen::Block< Eigen::MatrixXd > ) > > > >::iterator
56  typeIterator = statePartialList_.begin( ); typeIterator != statePartialList_.end( ); typeIterator++ )
57  {
58  int startIndex = stateTypeStartIndices_.at( typeIterator->first );
59  int currentStateSize = getSingleIntegrationSize( typeIterator->first );
60  int entriesToSkipPerEntry = currentStateSize - currentStateSize /
61  getSingleIntegrationDifferentialEquationOrder( typeIterator->first );
62  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
63  {
64  // Iterate over all bodies exerting an acceleration on this body.
65  for( statePartialIterator_ = typeIterator->second.at( i ).begin( );
66  statePartialIterator_ != typeIterator->second.at( i ).end( );
67  statePartialIterator_++ )
68  {
69  statePartialIterator_->second(
70  variationalMatrix_.block(
71  startIndex + entriesToSkipPerEntry + i* currentStateSize, statePartialIterator_->first.first,
72  currentStateSize - entriesToSkipPerEntry, statePartialIterator_->first.second ) );
73 
74  }
75  }
76  }
77 
78  // Correct partials for hierarchical dynamics
79  for( unsigned int i = 0; i < statePartialAdditionIndices_.size( ); i++ )
80  {
81  variationalMatrix_.block( 0, statePartialAdditionIndices_.at( i ).second, totalDynamicalStateSize_, 3 ) +=
82  variationalMatrix_.block( 0, statePartialAdditionIndices_.at( i ).first, totalDynamicalStateSize_, 3 );
83  }
84 }

139 void VariationalEquations::setStatePartialFunctionList( )
140 {
141  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int > currentDerivativeFunction;
142 
143 
144  // Iterate over all state types
145  for( std::map< propagators::IntegratedStateType,
146  orbit_determination::StateDerivativePartialsMap >::iterator
147  stateDerivativeTypeIterator_ = stateDerivativePartialList_.begin( );
148  stateDerivativeTypeIterator_ != stateDerivativePartialList_.end( );
149  stateDerivativeTypeIterator_++ )
150  {
151  // Iterate over all bodies undergoing 'accelerations' for which initial state is to be estimated.
152  for( unsigned int i = 0; i < stateDerivativeTypeIterator_->second.size( ); i++ )
153  {
154  std::multimap< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >
155  currentBodyPartialList;
156 
157  // Iterate over all 'accelerations' from single body on other single body
158  for( unsigned int j = 0; j < stateDerivativeTypeIterator_->second.at( i ).size( ); j++ )
159  {
160  for( std::map< propagators::IntegratedStateType,
161  std::vector< std::pair< std::string, std::string > > >::iterator
162  estimatedStateIterator = dynamicalStatesToEstimate_.begin( );
163  estimatedStateIterator != dynamicalStatesToEstimate_.end( );
164  estimatedStateIterator++ )
165  {
166  // Iterate over all bodies to see if body exerting acceleration is also to be estimated (cross-terms)
167  for( unsigned int k = 0; k < estimatedStateIterator->second.size( ); k++ )
168  {
169  currentDerivativeFunction = stateDerivativeTypeIterator_->second.at( i ).at( j )->
170  getDerivativeFunctionWrtStateOfIntegratedBody(
171  estimatedStateIterator->second.at( k ), estimatedStateIterator->first );
172 
173  // If function is not-empty: add to list.
174  if( currentDerivativeFunction.second != 0 )
175  {
176  currentBodyPartialList.insert(
177  std::make_pair(
178  std::make_pair( k * getSingleIntegrationSize( estimatedStateIterator->first ) +
179  stateTypeStartIndices_.at( estimatedStateIterator->first ),
180  getSingleIntegrationSize( estimatedStateIterator->first ) ),
181  currentDerivativeFunction.first ) );
182  }
183  }
184  }
185  }
186  statePartialList_[ stateDerivativeTypeIterator_->first ].push_back( currentBodyPartialList );
187  }
188  }
189 }
