23 Eigen::Matrix< double, 1, 6 > calculateNumericalPartialOfTrueAnomalyWrtState(
24  const Eigen::Vector6d& cartesianElements, const double gravitationalParameter,
25  const Eigen::Vector6d& cartesianStateElementPerturbations )
26 {
27  using namespace tudat::orbital_element_conversions;
28 
29  // Initialize partial to zero
30  Eigen::Matrix< double, 1, 6 > partial = Eigen::Matrix< double, 1, 6 >::Zero( );
31 
32  // Iterate over all six elements and calculate partials.
33  Eigen::Vector6d perturbedCartesianElements;
34  double upPerturbedTrueAnomaly, downPerturbedTrueAnomaly;
35  for( int i = 0; i < 6; i++ )
36  {
37  // Calculate true anomaly at up-perturbed cartesian element i
38  perturbedCartesianElements = cartesianElements;
39  perturbedCartesianElements( i ) += cartesianStateElementPerturbations( i );
40  upPerturbedTrueAnomaly = convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter )( trueAnomalyIndex );
41 
42  // Check validity of result
43  if( upPerturbedTrueAnomaly != upPerturbedTrueAnomaly )
44  {
45  std::cout << "Error 1 in partial of true anomaly wrt cartesian state, element" << i
46  << ", keplerian state: " << std::endl
47  << convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter ).transpose( ) << std::endl
48  << perturbedCartesianElements.transpose( ) << std::endl;
49  }
50 
51  // Calculate true anomaly at down-perturbed cartesian element i
52  perturbedCartesianElements = cartesianElements;
53  perturbedCartesianElements( i ) -= cartesianStateElementPerturbations( i );
54  downPerturbedTrueAnomaly = convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter )( trueAnomalyIndex );
55 
56  // Check validity of result
57  if( downPerturbedTrueAnomaly != downPerturbedTrueAnomaly )
58  {
59  std::cout << "Error 2 in partial of true anomaly wrt cartesian state, element" << i
60  << ", keplerian state: " << std::endl
61  << convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter ).transpose( ) << std::endl
62  << perturbedCartesianElements.transpose( ) << std::endl;
63  }
64 
65  // Calculate central difference of term i
66  partial( 0, i ) = ( upPerturbedTrueAnomaly - downPerturbedTrueAnomaly ) / ( 2.0 * cartesianStateElementPerturbations( i ) );
67 
68  // Check validity of result
69  if( partial( 0, i ) != partial( 0, i ) )
70  {
71  std::cout << "Error 2 in partial of true anomaly wrt cartesian state, element" << i << std::endl;
72  }
73  }
74 
75  return partial;
76 }

116 void EmpiricalAccelerationPartial::update( const double currentTime )
117 {
118  if( !( currentTime_ == currentTime ) )
119  {
120 
121  using namespace tudat::basic_mathematics;
122  using namespace tudat::linear_algebra;
123 
124  // Get current state and associated data.
125  Eigen::Vector6d currentState = empiricalAcceleration_->getCurrentState( );
126  Eigen::Vector3d angularMomentumVector = Eigen::Vector3d( currentState.segment( 0, 3 ) ).cross(
127  Eigen::Vector3d( currentState.segment( 3, 3 ) ) );
128  Eigen::Vector3d crossVector = angularMomentumVector.cross( Eigen::Vector3d( currentState.segment( 0, 3 ) ) );
129 
130  // Compute constituent geometric partials.
131  Eigen::Matrix3d normPositionWrtPosition = calculatePartialOfNormalizedVector( Eigen::Matrix3d::Identity( ), currentState.segment( 0, 3 ) );
132  Eigen::Matrix3d angularMomentumWrtPosition = -getCrossProductMatrix( currentState.segment( 3, 3 ) );
133  Eigen::Matrix3d angularMomentumWrtVelocity = getCrossProductMatrix( currentState.segment( 0, 3 ) );
134  Eigen::Matrix3d normAngularMomentumWrtPosition = calculatePartialOfNormalizedVector( angularMomentumWrtPosition, angularMomentumVector );
135  Eigen::Matrix3d normAngularMomentumWrtVelocity = calculatePartialOfNormalizedVector( angularMomentumWrtVelocity, angularMomentumVector );
136  Eigen::Matrix3d crossVectorWrtPosition = getCrossProductMatrix( angularMomentumVector ) -
137  angularMomentumWrtVelocity * angularMomentumWrtPosition;
138  Eigen::Matrix3d crossVectorWrtVelocity = -getCrossProductMatrix( currentState.segment( 0, 3 ) ) *
139  getCrossProductMatrix( currentState.segment( 0, 3 ) );
140  Eigen::Matrix3d normCrossVectorWrtPosition = calculatePartialOfNormalizedVector( crossVectorWrtPosition, crossVector );
141  Eigen::Matrix3d normCrossVectorWrtVelocity = calculatePartialOfNormalizedVector( crossVectorWrtVelocity, crossVector );
142 
143  // Retrieve current empirical acceleration
144  Eigen::Vector3d localAcceleration = empiricalAcceleration_->getCurrentLocalAcceleration( );
145 
146  // Compute partial derivative contribution of derivative of rotation matrix from RSW to inertial frame
147  currentPositionPartial_ = localAcceleration.x( ) * normPositionWrtPosition + localAcceleration.y( ) * normCrossVectorWrtPosition +
148  localAcceleration.z( ) * normAngularMomentumWrtPosition;
149  currentVelocityPartial_ = localAcceleration.y( ) * normCrossVectorWrtVelocity + localAcceleration.z( ) * normAngularMomentumWrtVelocity;
150 
151  // Compute partial derivative contribution of derivative of true anomaly
152  Eigen::Matrix< double, 1, 6 > localTrueAnomalyPartial = calculateNumericalPartialOfTrueAnomalyWrtState(
153  empiricalAcceleration_->getCurrentState( ),
154  empiricalAcceleration_->getCurrentGravitationalParameter( ), cartesianStateElementPerturbations );
155  Eigen::Matrix< double, 1, 6 > trueAnomalyPartial = localTrueAnomalyPartial;
156  currentPositionPartial_ += empiricalAcceleration_->getCurrentToInertialFrame( ) * (
157  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::sine_empirical ) *
158  std::cos( empiricalAcceleration_->getCurrentTrueAnomaly( ) ) -
159  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::cosine_empirical ) *
160  std::sin( empiricalAcceleration_->getCurrentTrueAnomaly( ) )
161  ) * trueAnomalyPartial.block( 0, 0, 1, 3 );
162  currentVelocityPartial_ +=
163  ( empiricalAcceleration_->getCurrentToInertialFrame( ) * (
164  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::sine_empirical ) *
165  std::cos( empiricalAcceleration_->getCurrentTrueAnomaly( ) ) -
166  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::cosine_empirical ) *
167  std::sin( empiricalAcceleration_->getCurrentTrueAnomaly( ) ) ) ) * trueAnomalyPartial.block( 0, 3, 1, 3 );
168  currentTime_ = currentTime;
169 
170  // Check output.
171  if( currentPositionPartial_ != currentPositionPartial_ )
172  {
173  throw std::runtime_error( "Error, empirical acceleration position partials are NaN" );
174  }
175  if( currentVelocityPartial_ != currentVelocityPartial_ )
176  {
177  throw std::runtime_error( "Error, empirical acceleration velocity partials are NaN" );
178  }
179  }
180 }

184  std::shared_ptr< estimatable_parameters::ArcWiseEmpiricalAccelerationCoefficientsParameter > parameter,
185  Eigen::MatrixXd& partialDerivativeMatrix )
186 {

188  int singleArcParameterSize = parameter->getSingleArcParameterSize( );
189  Eigen::MatrixXd partialWrtCurrentArcAccelerations;
190  wrtEmpiricalAccelerationCoefficientFromIndices(
191  singleArcParameterSize, parameter->getIndices( ), partialWrtCurrentArcAccelerations );
192 
193  // Retrieve arc of current time.
194  std::shared_ptr< interpolators::LookUpScheme< double > > currentArcIndexLookUp =
195  parameter->getArcTimeLookupScheme( );

197

200  partialDerivativeMatrix.block(
201  0, currentArc * singleArcParameterSize, 3, singleArcParameterSize ) =

207 void EmpiricalAccelerationPartial::wrtEmpiricalAccelerationCoefficientFromIndices(
208  const int numberOfAccelerationComponents,
209  const std::map< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes, std::vector< int > >& accelerationIndices,
210  Eigen::MatrixXd& partial )
211 {
212  // Retrieve rotation matrix to inertial frame
213  Eigen::Matrix3d rotationMatrix = Eigen::Matrix3d( empiricalAcceleration_->getCurrentToInertialFrame( ) );
214 
215  // Initialize partial derivatives
216  partial = Eigen::Matrix< double, 3, Eigen::Dynamic >::Zero( 3, numberOfAccelerationComponents );
217 
218  // Iterate over all terms, and set partial
219  int currentIndex = 0;
220  double multiplier = 0.0;
221  for( std::map< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes, std::vector< int > >::const_iterator
222  indexIterator = accelerationIndices.begin( );
223  indexIterator != accelerationIndices.end( ); indexIterator++ )
224  {
225  // Get multiplier associated with functional shape of empirical acceleration
226  switch( indexIterator->first )
227  {
228  case basic_astrodynamics::constant_empirical:
229  multiplier = 1.0;
230  break;
231  case basic_astrodynamics::sine_empirical:
232  multiplier = std::sin( empiricalAcceleration_->getCurrentTrueAnomaly( ) );
233  break;
234  case basic_astrodynamics::cosine_empirical:
235  multiplier = std::cos( empiricalAcceleration_->getCurrentTrueAnomaly( ) );
236  break;
237  default:
238  throw std::runtime_error(
239  "Error when calculating partial w.r.t. empirical accelerations, could not find functional shape " );
240  }
241 
242  // Set partial value for current component and shape
243  for( unsigned int i = 0; i < indexIterator->second.size( ); i++ )
244  {
245  partial.block( 0, currentIndex, 3, 1 ) = multiplier * rotationMatrix.block ( 0, indexIterator->second.at( i ), 3, 1 );
246  currentIndex++;
247  }
248 
249  }
250 }
251 
252 }
