22 TrimOrientationCalculator::TrimOrientationCalculator(
23  const std::shared_ptr< AerodynamicCoefficientInterface > coefficientInterface,
24  const std::shared_ptr< root_finders::RootFinderCore< double > > rootFinder ):
25  coefficientInterface_( coefficientInterface ), rootFinder_( rootFinder )
26 {
27  // Find index of angle of attack in aerodynamic coefficient interface (throw error if not found)
28  std::vector< AerodynamicCoefficientsIndependentVariables > independentVariables =
29  coefficientInterface->getIndependentVariableNames( );
30  std::vector< AerodynamicCoefficientsIndependentVariables >::iterator variableIterator =
31  std::find( independentVariables.begin( ), independentVariables.end( ), angle_of_attack_dependent );
32 
33  if( variableIterator == independentVariables.end( ) )
34  {
35  throw std::runtime_error( "Error when getting trim angle of attack, no angle of attack dependency is found" );
36  }
37  variableIndex_ = std::distance( independentVariables.begin( ), variableIterator );
38 
39  // Find indices of angle of attack in control surface interfaces.
40  std::map< std::string, std::vector< AerodynamicCoefficientsIndependentVariables > > controlSurfaceIndependentVariables =
41  coefficientInterface->getControlSurfaceIndependentVariables( );
42  for( std::map< std::string, std::vector< AerodynamicCoefficientsIndependentVariables > >::iterator
43  controlSurfaceIterator = controlSurfaceIndependentVariables.begin( );
44  controlSurfaceIterator != controlSurfaceIndependentVariables.end( ); controlSurfaceIterator++ )
45  {
46  std::vector< AerodynamicCoefficientsIndependentVariables > currentIndependentVariables =
47  controlSurfaceIterator->second;
48  variableIterator =
49  std::find(
50  currentIndependentVariables.begin( ), currentIndependentVariables.end( ), angle_of_attack_dependent );
51  if( ( variableIterator == currentIndependentVariables.end( ) ) )
52  {
53  std::cerr << "Warning when getting trim angle of attack, no angle of attack dependency is found for control surface "
54  + controlSurfaceIterator->first << std::endl;
55  }
56  else
57  {
58  controlSurfaceVariableIndex_[ controlSurfaceIterator->first ] =
59  std::distance( currentIndependentVariables.begin( ), variableIterator );
60  }
61  }
62 
63  // If no root finder provided, use default value.
64  if ( !rootFinder_.get( ) )
65  {
66  rootFinder_ = std::make_shared< root_finders::SecantRootFinderCore< double > >(
67  std::bind(
68  &root_finders::termination_conditions::RootAbsoluteToleranceTerminationCondition< double >::
69  checkTerminationCondition,
70  std::make_shared< root_finders::termination_conditions::RootAbsoluteToleranceTerminationCondition
71  < double > >( 1.0E-15, 1000 ), std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, std::placeholders::_4, std::placeholders::_5 ), 0.5 );
72  }

76 double TrimOrientationCalculator::findTrimAngleOfAttack(
77  const std::vector< double > untrimmedIndependentVariables,
78  const std::map< std::string, std::vector< double > > untrimmedControlSurfaceIndependentVariables )
79 {
80  // Determine function for which the root is to be determined.
81  std::function< double( const double ) > coefficientFunction =
82  std::bind( &TrimOrientationCalculator::getPerturbedMomentCoefficient,
83  this, std::placeholders::_1, untrimmedIndependentVariables, untrimmedControlSurfaceIndependentVariables );
84 
85  double trimmedAngleOfAttack = TUDAT_NAN;
86 
87  // Find root of pitch moment function
88  try
89  {
90  trimmedAngleOfAttack = rootFinder_->execute(
91  std::make_shared< basic_mathematics::FunctionProxy< double, double > >( coefficientFunction ),
92  untrimmedIndependentVariables.at( variableIndex_ ) );
93  }
94  // Throw error if not converged
95  catch( std::runtime_error )
96  {
97  throw std::runtime_error( "Error when getting trim angle of attack, root finder did not converge." );
98 
99  }
100 
101  return trimmedAngleOfAttack;
102 }
