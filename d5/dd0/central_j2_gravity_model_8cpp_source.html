23 Eigen::Vector3d computeGravitationalAccelerationDueToJ2(
24  const Eigen::Vector3d& positionOfBodySubjectToAcceleration,
25  const double gravitationalParameterOfBodyExertingAcceleration,
26  const double equatorialRadiusOfBodyExertingAcceleration,
27  const double j2CoefficientOfGravityField,
28  const Eigen::Vector3d& positionOfBodyExertingAcceleration )
29 {
30  // Set constant values reused for optimal computation of acceleration components.
31  const double distanceBetweenBodies = ( positionOfBodySubjectToAcceleration
32  - positionOfBodyExertingAcceleration ).norm( );
33 
34  const double preMultiplier = -gravitationalParameterOfBodyExertingAcceleration
35  / std::pow( distanceBetweenBodies, 4.0 ) * 1.5 * j2CoefficientOfGravityField
36  * equatorialRadiusOfBodyExertingAcceleration
37  * equatorialRadiusOfBodyExertingAcceleration;
38 
39  const double scaledZCoordinate = ( positionOfBodySubjectToAcceleration.z( )
40  - positionOfBodyExertingAcceleration.z( ) )
41  / distanceBetweenBodies;
42 
43  const double scaledZCoordinateSquared = scaledZCoordinate * scaledZCoordinate;
44 
45  const double factorForXAndYDirections = ( 1.0 - 5.0 * scaledZCoordinateSquared )
46  / distanceBetweenBodies;
47 
48  // Compute components of acceleration due to J2-effect.
49  Eigen::Vector3d gravitationalAccelerationDueToJ2 = Eigen::Vector3d::Constant( preMultiplier );
50 
51  gravitationalAccelerationDueToJ2( orbital_element_conversions::xCartesianPositionIndex )
52  *= ( positionOfBodySubjectToAcceleration.x( )
53  - positionOfBodyExertingAcceleration.x( ) ) * factorForXAndYDirections;
54 
55  gravitationalAccelerationDueToJ2( orbital_element_conversions::yCartesianPositionIndex )
56  *= ( positionOfBodySubjectToAcceleration.y( )
57  - positionOfBodyExertingAcceleration.y( ) ) * factorForXAndYDirections;
58 
59  gravitationalAccelerationDueToJ2( orbital_element_conversions::zCartesianPositionIndex )
60  *= ( 3.0 - 5.0 * scaledZCoordinateSquared ) * scaledZCoordinate;
61 
62  return gravitationalAccelerationDueToJ2;
63 }
