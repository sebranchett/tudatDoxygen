template<typename IndependentVariableType = double, typename StateType = Eigen::VectorXd, typename StateDerivativeType = Eigen::VectorXd, typename TimeStepType = double>
class tudat::numerical_integrators::BulirschStoerVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >

Class that implements the Bulirsch-Stoer variable step size integrator.

Template Parameters
StateTypeThe type of the state. This type should be an Eigen::Matrix derived type.
StateDerivativeTypeThe type of the state derivative. This type should be an Eigen::Matrix derived type.
IndependentVariableTypeThe type of the independent variable. This type should be either a float or double.
