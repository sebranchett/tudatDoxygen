21 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< RotationModelSettings >& rotationModelSettings )
22 {
23  if ( ! rotationModelSettings )
24  {
25  return;
26  }
27  using namespace json_interface;
28  using K = Keys::Body::RotationModel;
29 
30  const RotationModelType rotationModelType = rotationModelSettings->getRotationType( );
31  jsonObject[ K::type ] = rotationModelType;
32  jsonObject[ K::originalFrame ] = rotationModelSettings->getOriginalFrame( );
33  jsonObject[ K::targetFrame ] = rotationModelSettings->getTargetFrame( );
34 
35  switch ( rotationModelType )
36  {
37  case simple_rotation_model:
38  {
39  std::shared_ptr< SimpleRotationModelSettings > simpleRotationModelSettings =
40  std::dynamic_pointer_cast< SimpleRotationModelSettings >( rotationModelSettings );
41  assertNonnullptrPointer( simpleRotationModelSettings );
42  jsonObject[ K::initialOrientation ] = simpleRotationModelSettings->getInitialOrientation( );
43  jsonObject[ K::initialTime ] = simpleRotationModelSettings->getInitialTime( );
44  jsonObject[ K::rotationRate ] = simpleRotationModelSettings->getRotationRate( );
45  return;
46  }
47  case spice_rotation_model:
48  return;
49  case gcrs_to_itrs_rotation_model:
50  {
51  std::cerr<<"Warning when writing GCRS to ITRS rotation model to JSON, only precession-nutation theory and base frame are saved, rest is kept at default values"<<std::endl;
52 
53  std::shared_ptr< GcrsToItrsRotationModelSettings > simpleRotationModelSettings =
54  std::dynamic_pointer_cast< GcrsToItrsRotationModelSettings >( rotationModelSettings );
55  assertNonnullptrPointer( simpleRotationModelSettings );
56  jsonObject[ K::precessionNutationTheory ] = simpleRotationModelSettings->getNutationTheory( );
57  }
58  default:
59  handleUnimplementedEnumValue( rotationModelType, rotationModelTypes, unsupportedRotationModelTypes );
60  }
61 }

64 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< RotationModelSettings >& rotationModelSettings )
65 {
66  using namespace json_interface;
67  using K = Keys::Body::RotationModel;
68 
69  // Base class settings
70  const RotationModelType rotationModelType = getValue< RotationModelType >( jsonObject, K::type );
71  const std::string originalFrame = getValue< std::string >( jsonObject, K::originalFrame );
72 
73  std::string targetFrame = "";
74  if( rotationModelType != gcrs_to_itrs_rotation_model )
75  {
76  targetFrame = getValue< std::string >( jsonObject, K::targetFrame );
77  }
78  else
79  {
80  targetFrame = "GCRS";
81  }
82 
83  switch ( rotationModelType ) {
84  case simple_rotation_model:
85  {
86  const double initialTime = getValue< double >( jsonObject, K::initialTime );
87 
88  // Get JSON object for initialOrientation (or create it if not defined)
89  nlohmann::json jsonInitialOrientation;
90  if ( isDefined( jsonObject, K::initialOrientation ) )
91  {
92  jsonInitialOrientation = getValue< nlohmann::json >( jsonObject, K::initialOrientation );
93  }
94  else
95  {
96  jsonInitialOrientation[ K::originalFrame ] = originalFrame;
97  jsonInitialOrientation[ K::targetFrame ] = targetFrame;
98  jsonInitialOrientation[ K::initialTime ] = initialTime;
99  }
100 
101  rotationModelSettings = std::make_shared< SimpleRotationModelSettings >(
102  originalFrame,
103  targetFrame,
104  getAs< Eigen::Quaterniond >( jsonInitialOrientation ),
105  initialTime,
106  getValue< double >( jsonObject, K::rotationRate ) );
107  return;
108  }
109  case spice_rotation_model:
110  {
111  rotationModelSettings = std::make_shared< RotationModelSettings >(
112  rotationModelType, originalFrame, targetFrame );
113  return;
114  }
115  case gcrs_to_itrs_rotation_model:
116  {
117  rotationModelSettings = std::make_shared< GcrsToItrsRotationModelSettings >(
118  getValue< basic_astrodynamics::IAUConventions >(
119  jsonObject, K::precessionNutationTheory, basic_astrodynamics::iau_2006 ),
120  originalFrame );
121  return;
122  }
123  default:
124  handleUnimplementedEnumValue( rotationModelType, rotationModelTypes, unsupportedRotationModelTypes );
125  }
126 }
