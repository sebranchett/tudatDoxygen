77 template< typename TimeType >
78 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< IntegratorSettings< TimeType > >& integratorSettings )
79 {
80  if ( ! integratorSettings )
81  {
82  return;
83  }
84  using namespace json_interface;
85  using K = Keys::Integrator;
86 
87  // Common keys
88  const AvailableIntegrators integratorType = integratorSettings->integratorType_;
89  jsonObject[ K::type ] = integratorType;
90  jsonObject[ K::initialTime ] = integratorSettings->initialTime_;
91  jsonObject[ K::saveFrequency ] = integratorSettings->saveFrequency_;
92  jsonObject[ K::assessPropagationTerminationConditionDuringIntegrationSubsteps ] =
93  integratorSettings->assessPropagationTerminationConditionDuringIntegrationSubsteps_;
94 
95  switch ( integratorType )
96  {
97  case rungeKutta4:
98  case euler:
99  jsonObject[ K::stepSize ] = integratorSettings->initialTimeStep_;
100  return;
101  case rungeKuttaVariableStepSize:
102  {
103  // Create Runge-Kutta base object
104  std::shared_ptr< RungeKuttaVariableStepSizeBaseSettings< TimeType > > rungeKuttaVariableStepSizeSettings =
105  std::dynamic_pointer_cast< RungeKuttaVariableStepSizeBaseSettings< TimeType > >( integratorSettings );
106  assertNonnullptrPointer( rungeKuttaVariableStepSizeSettings );
107 
108  // Check which integrator settings is requested
109  if ( rungeKuttaVariableStepSizeSettings->areTolerancesDefinedAsScalar_ )
110  {
111  // Integrator with scalar tolerances
112  std::shared_ptr< RungeKuttaVariableStepSizeSettingsScalarTolerances< TimeType > > scalarTolerancesIntegratorSettings =
113  std::dynamic_pointer_cast< RungeKuttaVariableStepSizeSettingsScalarTolerances< TimeType > >( integratorSettings );
114 
115  jsonObject[ K::rungeKuttaCoefficientSet ] =
116  stringFromEnum( scalarTolerancesIntegratorSettings->coefficientSet_, rungeKuttaCoefficientSets );
117  jsonObject[ K::initialStepSize ] = scalarTolerancesIntegratorSettings->initialTimeStep_;
118  jsonObject[ K::minimumStepSize ] = scalarTolerancesIntegratorSettings->minimumStepSize_;
119  jsonObject[ K::maximumStepSize ] = scalarTolerancesIntegratorSettings->maximumStepSize_;
120  jsonObject[ K::relativeErrorTolerance ] = scalarTolerancesIntegratorSettings->relativeErrorTolerance_;
121  jsonObject[ K::absoluteErrorTolerance ] = scalarTolerancesIntegratorSettings->absoluteErrorTolerance_;
122  jsonObject[ K::areTolerancesDefinedAsScalar ] = scalarTolerancesIntegratorSettings->areTolerancesDefinedAsScalar_;
123  jsonObject[ K::safetyFactorForNextStepSize ] = scalarTolerancesIntegratorSettings->safetyFactorForNextStepSize_;
124  jsonObject[ K::maximumFactorIncreaseForNextStepSize ] =
125  scalarTolerancesIntegratorSettings->maximumFactorIncreaseForNextStepSize_;
126  jsonObject[ K::minimumFactorDecreaseForNextStepSize ] =
127  scalarTolerancesIntegratorSettings->minimumFactorDecreaseForNextStepSize_;
128  }
129  else
130  {
131  throw std::runtime_error( "Error while creating Runge-Kutta variable step-size integrator via JSON interface. RK "
132  "integrators with vector tolerances are not yet supported via JSON." );
133  }
134  return;
135  }
136  case adamsBashforthMoulton:
137  {
138  std::shared_ptr< AdamsBashforthMoultonSettings< TimeType > > adamsBashforthMoultonSettings =
139  std::dynamic_pointer_cast< AdamsBashforthMoultonSettings< TimeType > >( integratorSettings );
140  assertNonnullptrPointer( adamsBashforthMoultonSettings );
141  jsonObject[ K::initialStepSize ] = adamsBashforthMoultonSettings->initialTimeStep_;
142  jsonObject[ K::minimumStepSize ] = adamsBashforthMoultonSettings->minimumStepSize_;
143  jsonObject[ K::maximumStepSize ] = adamsBashforthMoultonSettings->maximumStepSize_;
144  jsonObject[ K::relativeErrorTolerance ] = adamsBashforthMoultonSettings->relativeErrorTolerance_;
145  jsonObject[ K::absoluteErrorTolerance ] = adamsBashforthMoultonSettings->absoluteErrorTolerance_;
146  jsonObject[ K::minimumOrder ] = adamsBashforthMoultonSettings->minimumOrder_;
147  jsonObject[ K::maximumOrder ] = adamsBashforthMoultonSettings->maximumOrder_;
148  jsonObject[ K::bandwidth ] = adamsBashforthMoultonSettings->bandwidth_;
149  return;
150  }
151  case bulirschStoer:
152  {
153  std::shared_ptr< BulirschStoerIntegratorSettings< TimeType > > bulirschStoerSettings =
154  std::dynamic_pointer_cast< BulirschStoerIntegratorSettings< TimeType > >( integratorSettings );
155  assertNonnullptrPointer( bulirschStoerSettings );
156  jsonObject[ K::initialStepSize ] = bulirschStoerSettings->initialTimeStep_;
157  jsonObject[ K::minimumStepSize ] = bulirschStoerSettings->minimumStepSize_;
158  jsonObject[ K::maximumStepSize ] = bulirschStoerSettings->maximumStepSize_;
159  jsonObject[ K::relativeErrorTolerance ] = bulirschStoerSettings->relativeErrorTolerance_;
160  jsonObject[ K::absoluteErrorTolerance ] = bulirschStoerSettings->absoluteErrorTolerance_;
161  jsonObject[ K::extrapolationSequence ] = bulirschStoerSettings->extrapolationSequence_;
162  jsonObject[ K::maximumNumberOfSteps ] = bulirschStoerSettings->maximumNumberOfSteps_;
163  jsonObject[ K::safetyFactorForNextStepSize ] =
164  bulirschStoerSettings->safetyFactorForNextStepSize_;
165  jsonObject[ K::maximumFactorIncreaseForNextStepSize ] =
166  bulirschStoerSettings->maximumFactorIncreaseForNextStepSize_;
167  jsonObject[ K::minimumFactorDecreaseForNextStepSize ] =
168  bulirschStoerSettings->minimumFactorDecreaseForNextStepSize_;
169 
170  return;
171  }
172  default:
173  handleUnimplementedEnumValue( integratorType, integratorTypes, unsupportedIntegratorTypes );
174  }

178 template< typename TimeType >
179 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< IntegratorSettings< TimeType > >& integratorSettings )
180 {
181  using namespace json_interface;
182  using RungeKuttaCoefficientSet = RungeKuttaCoefficients::CoefficientSets;
183  using K = Keys::Integrator;
184 
185  // Read JSON settings shared by all supported integrators
186  const AvailableIntegrators integratorType = getValue( jsonObject, K::type, rungeKutta4 );
187  const TimeType initialTime =
188  getValue< TimeType >( jsonObject, { K::initialTime, SpecialKeys::root / Keys::initialEpoch } );
189 
190  // Create IntegratorSettings pointer from JSON settings
191  switch ( integratorType )
192  {
193  case euler:
194  case rungeKutta4:
195  {
196  IntegratorSettings< TimeType > defaults( integratorType, 0.0, 0.0 );
197  integratorSettings = std::make_shared< IntegratorSettings< TimeType > >(
198  integratorType,
199  initialTime,
200  getValue< TimeType >( jsonObject, K::stepSize ),
201  getValue( jsonObject, K::saveFrequency, defaults.saveFrequency_ ),
202  getValue( jsonObject, K::assessPropagationTerminationConditionDuringIntegrationSubsteps,
203  defaults.assessPropagationTerminationConditionDuringIntegrationSubsteps_ ) );
204  return;
205  }
206  case rungeKuttaVariableStepSize:
207  {
208  // Check which constructor to use
209  if ( getValue< bool >( jsonObject, K::areTolerancesDefinedAsScalar ) )
210  {
211  // Scalar tolerances
212  RungeKuttaVariableStepSizeSettingsScalarTolerances< TimeType > defaults(
213  0.0, 0.0, RungeKuttaCoefficientSet::rungeKuttaFehlberg45, 0.0, 0.0 );
214 
215  integratorSettings = std::make_shared< RungeKuttaVariableStepSizeSettingsScalarTolerances< TimeType > >(
216  initialTime,
217  getValue< TimeType >( jsonObject, K::initialStepSize ),
218  getValue< RungeKuttaCoefficientSet >( jsonObject, K::rungeKuttaCoefficientSet ),
219  getValue< TimeType >( jsonObject, K::minimumStepSize ),
220  getValue< TimeType >( jsonObject, K::maximumStepSize ),
221  getValue( jsonObject, K::relativeErrorTolerance, defaults.relativeErrorTolerance_ ),
222  getValue( jsonObject, K::absoluteErrorTolerance, defaults.absoluteErrorTolerance_ ),
223  getValue( jsonObject, K::saveFrequency, defaults.saveFrequency_ ),
224  getValue( jsonObject, K::assessPropagationTerminationConditionDuringIntegrationSubsteps,
225  defaults.assessPropagationTerminationConditionDuringIntegrationSubsteps_ ),
226  getValue( jsonObject, K::safetyFactorForNextStepSize,
227  defaults.safetyFactorForNextStepSize_ ),
228  getValue( jsonObject, K::maximumFactorIncreaseForNextStepSize,
229  defaults.maximumFactorIncreaseForNextStepSize_ ),
230  getValue( jsonObject, K::minimumFactorDecreaseForNextStepSize,
231  defaults.minimumFactorDecreaseForNextStepSize_ ) );
232  }
233  else
234  {
235  throw std::runtime_error( "Error while creating Runge-Kutta variable step-size integrator from JSON object. RK "
236  "integrators with vector tolerances are not yet supported via JSON." );
237  }
238  return;
239  }
240  case adamsBashforthMoulton:
241  {
242  AdamsBashforthMoultonSettings< TimeType > defaults(
243  0.0, 0.0, 0.0, 0.0 );
244 
245  integratorSettings = std::make_shared< AdamsBashforthMoultonSettings< TimeType > >(
246  initialTime,
247  getValue< TimeType >( jsonObject, K::initialStepSize ),
248  getValue< TimeType >( jsonObject, K::minimumStepSize ),
249  getValue< TimeType >( jsonObject, K::maximumStepSize ),
250  getValue( jsonObject, K::relativeErrorTolerance, defaults.relativeErrorTolerance_ ),
251  getValue( jsonObject, K::absoluteErrorTolerance, defaults.absoluteErrorTolerance_ ),
252  getValue( jsonObject, K::minimumOrder, defaults.minimumOrder_ ),
253  getValue( jsonObject, K::maximumOrder, defaults.maximumOrder_ ),
254  getValue( jsonObject, K::saveFrequency, defaults.saveFrequency_ ),
255  getValue( jsonObject, K::assessPropagationTerminationConditionDuringIntegrationSubsteps,
256  defaults.assessPropagationTerminationConditionDuringIntegrationSubsteps_ ),
257  getValue( jsonObject, K::bandwidth,
258  defaults.bandwidth_ ) );
259  return;
260  }
261  case bulirschStoer:
262  {
263  BulirschStoerIntegratorSettings< TimeType > defaults(
264  0.0, 0.0, bulirsch_stoer_sequence, 6, std::numeric_limits< double >::epsilon( ),
265  std::numeric_limits< double >::infinity( ) );
266 
267  integratorSettings = std::make_shared< BulirschStoerIntegratorSettings< TimeType > >(
268  initialTime,
269  getValue< TimeType >( jsonObject, K::initialStepSize ),
270  getValue( jsonObject, K::extrapolationSequence, defaults.extrapolationSequence_ ),
271  getValue( jsonObject, K::maximumNumberOfSteps, defaults.maximumNumberOfSteps_ ),
272  getValue< TimeType >( jsonObject, K::minimumStepSize ),
273  getValue< TimeType >( jsonObject, K::maximumStepSize ),
274  getValue( jsonObject, K::relativeErrorTolerance, defaults.relativeErrorTolerance_ ),
275  getValue( jsonObject, K::absoluteErrorTolerance, defaults.absoluteErrorTolerance_ ),
276  getValue( jsonObject, K::saveFrequency, defaults.saveFrequency_ ),
277  getValue( jsonObject, K::assessPropagationTerminationConditionDuringIntegrationSubsteps,
278  defaults.assessPropagationTerminationConditionDuringIntegrationSubsteps_ ),
279  getValue( jsonObject, K::safetyFactorForNextStepSize,
280  defaults.safetyFactorForNextStepSize_ ),
281  getValue( jsonObject, K::maximumFactorIncreaseForNextStepSize,
282  defaults.maximumFactorIncreaseForNextStepSize_ ),
283  getValue( jsonObject, K::minimumFactorDecreaseForNextStepSize,
284  defaults.minimumFactorDecreaseForNextStepSize_ ) );
285  return;
286  }
