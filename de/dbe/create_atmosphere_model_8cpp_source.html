68  // Declare return object.

70 
71  // Check which type of atmosphere model is to be created.
72  switch( atmosphereSettings->getAtmosphereType( ) )

74  case exponential_atmosphere:
75  {
76  // Check whether settings for atmosphere are consistent with its type.
77  boost::shared_ptr< ExponentialAtmosphereSettings > exponentialAtmosphereSettings =
78  boost::dynamic_pointer_cast< ExponentialAtmosphereSettings >( atmosphereSettings );
79  if( exponentialAtmosphereSettings == NULL )
80  {

83  }
84  else
85  {

88  boost::make_shared< ExponentialAtmosphere >(

99  // Check whether settings for atmosphere are consistent with its type

104  throw std::runtime_error(

106  }
107  else
