143  static std::pair< boost::multi_array< double, 1 >, std::vector< std::vector< double > > >
144  readMultiArrayAndIndependentVariables(
145  const std::string fileName )
146  {
147  std::vector< std::vector< double > > independentVariables;
148  Eigen::MatrixXd coefficientBlock;
149 
150  // Read raw data from file
151  readCoefficientsFile( fileName, independentVariables, coefficientBlock );
152 
153  if( !( independentVariables.size( ) == 1 ) )
154  {
155  throw std::runtime_error( "Error when reading 3-D multi-array, wrong number of independent variables found" );
156  }
157 
158  // Set sizes of independent variables
159  std::vector< int > independentVariableSize;
160  independentVariableSize.push_back( independentVariables.at( 0 ).size( ) );
161 
162  // Parse data from file
163  return std::make_pair( parseRawOneDimensionalCoefficientsFromFile(
164  independentVariableSize, coefficientBlock ), independentVariables );
