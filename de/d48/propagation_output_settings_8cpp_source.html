171 std::string getDependentVariableId(
172  const boost::shared_ptr< SingleDependentVariableSaveSettings > dependentVariableSettings )
173 {
174  std::string variableId = getDependentVariableName( dependentVariableSettings->dependentVariableType_ );
175 
176  if( ( dependentVariableSettings->dependentVariableType_ == single_acceleration_dependent_variable ) ||
177  ( dependentVariableSettings->dependentVariableType_ == single_acceleration_norm_dependent_variable ) )
178  {
179  boost::shared_ptr< SingleAccelerationDependentVariableSaveSettings > accelerationDependentVariableSettings =
180  boost::dynamic_pointer_cast< SingleAccelerationDependentVariableSaveSettings >( dependentVariableSettings );
181  if( accelerationDependentVariableSettings == NULL )
182  {
183  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (acceleration type )" );
184  }
185  else
186  {
187  variableId += basic_astrodynamics::getAccelerationModelName(
188  accelerationDependentVariableSettings->accelerationModeType_ );
189  }
190  }
191  else if( ( dependentVariableSettings->dependentVariableType_ == single_torque_dependent_variable ) ||
192  ( dependentVariableSettings->dependentVariableType_ == single_torque_norm_dependent_variable ) )
193  {
194  boost::shared_ptr< SingleTorqueDependentVariableSaveSettings > torqueDependentVariableSettings =
195  boost::dynamic_pointer_cast< SingleTorqueDependentVariableSaveSettings >( dependentVariableSettings );
196  if( torqueDependentVariableSettings == NULL )
197  {
198  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (torque type )" );
199  }
200  else
201  {
202  variableId += basic_astrodynamics::getTorqueModelName(
203  torqueDependentVariableSettings->torqueModeType_ );
204  }
205  }
206  else if( dependentVariableSettings->dependentVariableType_ == intermediate_aerodynamic_rotation_matrix_variable )
207  {
208  boost::shared_ptr< IntermediateAerodynamicRotationVariableSaveSettings > rotationDependentVariableSettings =
209  boost::dynamic_pointer_cast< IntermediateAerodynamicRotationVariableSaveSettings >( dependentVariableSettings );
210  if( rotationDependentVariableSettings == NULL )
211  {
212  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (rotation matrix)" );
213  }
214  else
215  {
216  variableId +=
217  reference_frames::getAerodynamicFrameName( rotationDependentVariableSettings->baseFrame_ ) + "to " +
218  reference_frames::getAerodynamicFrameName( rotationDependentVariableSettings->targetFrame_ );
219  }
220  }
221 
222  else if( dependentVariableSettings->dependentVariableType_ == relative_body_aerodynamic_orientation_angle_variable )
223  {
224  boost::shared_ptr< BodyAerodynamicAngleVariableSaveSettings > angleDependentVariableSettings =
225  boost::dynamic_pointer_cast< BodyAerodynamicAngleVariableSaveSettings >( dependentVariableSettings );
226  if( angleDependentVariableSettings == NULL )
227  {
228  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (angle)" );
229  }
230  else
231  {
232  variableId +=
233  reference_frames::getAerodynamicAngleName( angleDependentVariableSettings->angle_ );
234  }
235  }
236 
237  if( ( dependentVariableSettings->dependentVariableType_ == single_acceleration_dependent_variable ) ||
238  ( dependentVariableSettings->dependentVariableType_ == single_acceleration_norm_dependent_variable ) )
239  {
240  variableId += ", acting on " + dependentVariableSettings->associatedBody_;
241  if( dependentVariableSettings->secondaryBody_ != dependentVariableSettings->associatedBody_ )
242  {
243  variableId += ", exerted by " + dependentVariableSettings->secondaryBody_;
244  }
245  }
246  else
247  {
248  variableId += "of " + dependentVariableSettings->associatedBody_;
249  if( dependentVariableSettings->secondaryBody_ != "" )
250  {
251  variableId += " w.r.t. " + dependentVariableSettings->secondaryBody_;
252  }
253 
254  }
255  return variableId;
256 }
