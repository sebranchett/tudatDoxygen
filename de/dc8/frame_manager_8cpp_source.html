50 void ReferenceFrameManager::setEphemerides(
51  const std::map< std::string, boost::shared_ptr< Ephemeris > >& additionalEphemerides )
52 {
53  // Set list of all frames for which frame level and base frame have not yet been determined.
54  std::map< std::string, boost::shared_ptr< Ephemeris > > unhandledFrames_ = additionalEphemerides;
55 
56  // Set list of available ephemerides and check whether it already exists (not possible incurrent
57  // implementation)
58  for( std::map< std::string, boost::shared_ptr< Ephemeris > >::const_iterator ephemerisIterator =
59  additionalEphemerides.begin( ); ephemerisIterator != additionalEphemerides.end( );
60  ephemerisIterator++ )
61  {
62  if( availableEphemerides_.count( ephemerisIterator->first ) == 0 )
63  {
64  availableEphemerides_[ ephemerisIterator->first ] = ephemerisIterator->second;
65  }
66  else
67  {
68  throw std::runtime_error( "Error when adding ephemerides to frame manager, frame already exists" );
69  }
70  }
71 
72  // Start at frame level 0
73  int currentLevel = 0;
74  std::map< std::string, std::string > singleLevelList;
75 
76  // While there are unhandled frames, continue at next frame level.
77  while( unhandledFrames_.size( ) != 0 )
78  {
79  // Clear frame list for new level.
80  singleLevelList.clear( );
81 
82  // If frame level is 0, base frame equals baseFrameName.
83  if( currentLevel == 0 )
84  {
85  for( std::map< std::string, boost::shared_ptr< Ephemeris > >::iterator
86  frameIterator = unhandledFrames_.begin( );
87  frameIterator != unhandledFrames_.end( ); frameIterator++ )
88  {
89  // If current frame is at level 0, add to list of levels.
90  if( frameIterator->second->getReferenceFrameOrigin( ) == getBaseFrameName( ) )
91  {
92  singleLevelList[ frameIterator->first ] = getBaseFrameName( );
93  }
94  }
95  }
96  // Else check if frame is at current level.
97  else
98  {
99  std::map< std::string, std::string >::iterator previousLevelIterator;
100  for( std::map< std::string, boost::shared_ptr< Ephemeris > >::iterator
101  frameIterator = unhandledFrames_.begin( );
102  frameIterator != unhandledFrames_.end( ); frameIterator++ )
103  {
104  // If base frame of current ephemeris is on previous level, add to list of current
105  // level.
106  previousLevelIterator = baseFrameList_[ currentLevel - 1 ].find(
107  frameIterator->second->getReferenceFrameOrigin( ) );
108  if( previousLevelIterator != baseFrameList_[ currentLevel - 1 ].end( ) )
109  {
110  singleLevelList[ frameIterator->first ]
111  = frameIterator->second->getReferenceFrameOrigin( );
112  }
113  }
114  }
115 
116  // Go through all ephemerides set at current level, add to frame list and remove from
117  // unhandled frame list if present.
118  std::map< std::string, boost::shared_ptr< Ephemeris > >::iterator frameIterator;
119  for( std::map< std::string, std::string >::iterator singleListIterator = singleLevelList.begin( );
120  singleListIterator != singleLevelList.end( ); singleListIterator++ )
121  {
122  // Find current frame in unhandles frame list.
123  frameIterator = unhandledFrames_.find( singleListIterator->first );
124  if( frameIterator != unhandledFrames_.end( ) )
125  {
126  unhandledFrames_.erase( frameIterator );
127  frameIndexList_[ singleListIterator->first ] = currentLevel;
128  }
129  else
130  {
131  throw std::runtime_error( "Error when making frame manager, could not find frame " +
132  singleListIterator->first + " when deleting" );
133  }
134  }
135 
136  // If no new frame matches found and unhandled frame list is not empty: stuck in infinite loop.
137  if( singleLevelList.size( ) == 0 )
138  {
139  throw std::runtime_error( "Error, found no new frame matches at current level, but list is not empty" );
140  }
141 
142  // Add to base frame list.
143  baseFrameList_.push_back( singleLevelList );
144  currentLevel++;
145  }
146 
147  // Check if all frames have same orientation.
148  std::string firstFrameOrientation = availableEphemerides_.begin( )->second->getReferenceFrameOrientation( );
149  for( std::map< std::string, boost::shared_ptr< Ephemeris > >::iterator ephemerisIterator =
150  availableEphemerides_.begin( ); ephemerisIterator != availableEphemerides_.end( ); ephemerisIterator++ )
151  {
152  if( ephemerisIterator->second->getReferenceFrameOrientation( ) != firstFrameOrientation )
153  {
154  throw std::runtime_error(
155  "Error, multiple reference frame orientations of ephemerides currently not supported" );
156  }
157  }
158 }

162 std::vector< boost::shared_ptr< Ephemeris > > ReferenceFrameManager::getDirectEphemerisFromLowerToUpperFrame(
163  const std::string& lowerFrame, const std::string& upperFrame )
164 {
165  // Get indices of frames.
166  int upperIndex = frameIndexList_.at( upperFrame );
167  int lowerIndex = frameIndexList_.at( lowerFrame );
168 
169  std::vector< boost::shared_ptr< Ephemeris > > ephemerisList;
170 
171  // Check validity of input (i.e. upper > lower)
172  if( upperIndex < lowerIndex )
173  {
174  throw std::runtime_error(
175  "Error when making direct ephemeris link in frame manager, upper index is smaller than lower index" );
176  }
177  // If frames are not equal, make list of ephemeris
178  else if( upperIndex != lowerIndex )
179  {
180  // Start list creation at upper frame.
181  int currentIndex = upperIndex;
182  std::string currentFrame = upperFrame;
183 
184  // Continue while lower frame is reached
185  while( currentIndex > lowerIndex )
186  {
187  // Check whether current frame is consistent with current frame index.
188  if( frameIndexList_[ currentFrame ] != currentIndex )
189  {
190  throw std::runtime_error(
191  "Error when making direct constituent ephemeris, frame index inconsistent." );
192  }
193 
194  // Add base frame of current frame.
195  std::string currentBase = baseFrameList_[ currentIndex ][ currentFrame ];
196  ephemerisList.push_back( availableEphemerides_[ currentFrame ] );
197 
198  // Decrement frame level and move to one frame level lower.
199  currentIndex--;
200  currentFrame = currentBase;
201  }
202  }
203  return ephemerisList;
204 }

208 {

211  // Find level of frame.

229 std::pair< std::string, int > ReferenceFrameManager::getNearestCommonFrame( std::vector< std::string > frameList )
230 {
231  // Make lists of current frames and frame levels that are currently being investigated for each
232  // frame in frameList.
233  std::vector< int > currentFrameLevels;
234  currentFrameLevels.resize( frameList.size( ) );
235  std::vector< std::string > currentFrames;
236  currentFrames.resize( frameList.size( ) );
237 
238  // Initialize current maximum frame level to nonsense value.
239  int currentMaximumFrameLevel = -1E8;
240 
241  // Check frame level of all input frames and find maximum level.
242  for( unsigned int i = 0; i < frameList.size( ); i++ )
243  {
244  // Get current frame level.
245  currentFrameLevels[ i ] = frameIndexList_[ frameList[ i ] ];
246  currentFrames[ i ] = frameList[ i ];
247 
248  if( currentFrameLevels[ i ] > currentMaximumFrameLevel )
249  {
250  currentMaximumFrameLevel = currentFrameLevels[ i ];
251  }
252  }
253 
254  bool isConverged = false;
255  std::string firstFrame;
256  bool areAllFramesEqual = true;
257 
258  // Check if all frames are equal, i.e. if already converged.
259  firstFrame = currentFrames[ 0 ];
260  for( unsigned int i = 1; i < currentFrames.size( ); i++ )
261  {
262  if( currentFrames[ i ] != firstFrame )
263  {
264  areAllFramesEqual = false;
265  }
266  }
267  if( areAllFramesEqual )
268  {
269  isConverged = true;
270  }
271 
272  // Decrement maximum frame levels and associated frames until all frames are equal.
273  while( isConverged == false && currentMaximumFrameLevel >= 0 )
274  {
275  // Iterate over all frames.
276  for( unsigned int i = 0; i < frameList.size( ); i++ )
277  {
278  // If frame level of current frame is at maximum, reduce it by one level.
279  if( currentFrameLevels[ i ] == currentMaximumFrameLevel )
280  {
281  // Find base frame of current level.
282  currentFrames[ i ] = baseFrameList_[ currentFrameLevels[ i ] ][ currentFrames[ i ] ];
283 
284  // Reduce level by 1
285  currentFrameLevels[ i ]--;
286  }
287  }
288  currentMaximumFrameLevel--;
289 
290  // Check if all frames are equal.
291  areAllFramesEqual = true;
292  firstFrame = currentFrames[ 0 ];
293  for( unsigned int i = 1; i < currentFrames.size( ); i++ )
294  {
295  if( currentFrames[ i ] != firstFrame )
296  {
297  areAllFramesEqual = false;
298  }
299  }
300 
301  if( areAllFramesEqual )
302  {
303  isConverged = true;
304  }
305  }
306 
307  // If not converged, set global base frame as common frame.
308  std::pair< std::string, int > output;
309  if( isConverged == false )
310  {
311  output = std::make_pair( getBaseFrameName( ), -1 );
312  }
313  else
314  {
315  output = std::make_pair( currentFrames[ 0 ], currentMaximumFrameLevel );
316  }
317 
318  return output;
319 }
