22 Eigen::Vector3d computeGravitationalAccelerationDueToJ3(
23  const Eigen::Vector3d& positionOfBodySubjectToAcceleration,
24  const double gravitationalParameterOfBodyExertingAcceleration,
25  const double equatorialRadiusOfBodyExertingAcceleration,
26  const double j3CoefficientOfGravityField,
27  const Eigen::Vector3d& positionOfBodyExertingAcceleration )
28 {
29  // Set constant values reused for optimal computation of acceleration components.
30  const double distanceBetweenBodies = ( positionOfBodySubjectToAcceleration
31  - positionOfBodyExertingAcceleration ).norm( );
32 
33  const double preMultiplier = -gravitationalParameterOfBodyExertingAcceleration
34  / std::pow( distanceBetweenBodies, 5.0 ) * 2.5 * j3CoefficientOfGravityField
35  * std::pow( equatorialRadiusOfBodyExertingAcceleration, 3.0 );
36 
37  const double scaledZCoordinate = ( positionOfBodySubjectToAcceleration.z( )
38  - positionOfBodyExertingAcceleration.z( ) )
39  / distanceBetweenBodies;
40 
41  const double scaledZCoordinateSquared = scaledZCoordinate * scaledZCoordinate;
42 
43  const double factorForXAndYDirections = ( 3.0 - 7.0 * scaledZCoordinateSquared )
44  * scaledZCoordinate / distanceBetweenBodies;
45 
46  // Compute components of acceleration due to J3-effect.
47  Eigen::Vector3d gravitationalAccelerationDueToJ3 = Eigen::Vector3d::Constant( preMultiplier );
48 
49  gravitationalAccelerationDueToJ3( orbital_element_conversions::xCartesianPositionIndex )
50  *= ( positionOfBodySubjectToAcceleration.x( )
51  - positionOfBodyExertingAcceleration.x( ) ) * factorForXAndYDirections;
52 
53  gravitationalAccelerationDueToJ3( orbital_element_conversions::yCartesianPositionIndex )
54  *= ( positionOfBodySubjectToAcceleration.y( )
55  - positionOfBodyExertingAcceleration.y( ) ) * factorForXAndYDirections;
56 
57  gravitationalAccelerationDueToJ3( orbital_element_conversions::zCartesianPositionIndex )
58  *= ( -0.6 + 6.0 * scaledZCoordinateSquared
59  - 7.0 * scaledZCoordinateSquared * scaledZCoordinateSquared );
60 
61  return gravitationalAccelerationDueToJ3;
62 }
