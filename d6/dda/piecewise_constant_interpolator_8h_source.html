134  DependentVariableType interpolate( const IndependentVariableType targetIndependentVariableValue )
135  {
136  // Check whether boundary handling needs to be applied, if independent variable is beyond its defined range.
137  DependentVariableType interpolatedValue;
138  bool useValue = false;
139  this->checkBoundaryCase( interpolatedValue, useValue, targetIndependentVariableValue );
140  if( useValue )
141  {
142  return interpolatedValue;
143  }
144 
145  // Determine the lower entry in the table corresponding to the target independent variable value.
146  int lowerEntry;
147  if( targetIndependentVariableValue <= independentValues_.at( 0 ) )
148  {
149  lowerEntry = 0;
150  }
151  else if( targetIndependentVariableValue >= independentValues_.at( independentValues_.size( ) - 1 ) )
152  {
153  lowerEntry = independentValues_.size( ) - 1;
154  }
155  else
156  {
157  lowerEntry = lookUpScheme_->findNearestLowerNeighbour( targetIndependentVariableValue );
158  }
159 
160  // Return interpolated value
161  return dependentValues_.at( lowerEntry );
162  }
