106  void updateFilter( const DependentVector& currentMeasurementVector )
107  {
108  // Prediction step
109  DependentVector aPrioriStateEstimate = this->predictState( );
110  DependentMatrix currentStateJacobianMatrix;
111  DependentMatrix currentStateNoiseJacobianMatrix;
112  if ( this->isStateToBeIntegrated_ )
113  {
114  std::pair< DependentMatrix, DependentMatrix > discreteTimeJacobians =
115  discreteTimeStateJacobians_( aPrioriStateEstimate );
116  currentStateJacobianMatrix = discreteTimeJacobians.first;
117  currentStateNoiseJacobianMatrix = discreteTimeJacobians.second;
118  }
119  else
120  {
121  currentStateJacobianMatrix = stateJacobianFunction_( this->currentTime_, aPrioriStateEstimate );
122  currentStateNoiseJacobianMatrix = stateNoiseJacobianFunction_( this->currentTime_, aPrioriStateEstimate );
123  }
124  DependentVector measurementEstimate = this->measurementFunction_( this->currentTime_, aPrioriStateEstimate );
125 
126  // Compute remaining Jacobians
127  DependentMatrix currentMeasurementJacobianMatrix = measurementJacobianFunction_( this->currentTime_, aPrioriStateEstimate );
128  DependentMatrix currentMeasurementNoiseJacobianMatrix = measurementNoiseJacobianFunction_( this->currentTime_, aPrioriStateEstimate );
129 
130  // Prediction step (continued)
131  DependentMatrix aPrioriCovarianceEstimate = currentStateJacobianMatrix * this->aPosterioriCovarianceEstimate_ *
132  currentStateJacobianMatrix.transpose( ) + currentStateNoiseJacobianMatrix * this->systemUncertainty_ *
133  currentStateNoiseJacobianMatrix.transpose( );
134 
135  // Compute Kalman gain
136  DependentMatrix kalmanGain = aPrioriCovarianceEstimate * currentMeasurementJacobianMatrix.transpose( ) * (
137  currentMeasurementJacobianMatrix * aPrioriCovarianceEstimate * currentMeasurementJacobianMatrix.transpose( ) +
138  currentMeasurementNoiseJacobianMatrix * this->measurementUncertainty_ *
139  currentMeasurementNoiseJacobianMatrix.transpose( ) ).inverse( );
140 
141  // Correction step
142  this->currentTime_ += this->filteringStepSize_;
143  this->correctState( aPrioriStateEstimate, currentMeasurementVector, measurementEstimate, kalmanGain );
144  this->correctCovariance( aPrioriCovarianceEstimate, currentMeasurementJacobianMatrix, kalmanGain );
145  }

184  std::pair< DependentMatrix, DependentMatrix > generateDiscreteTimeSystemJacobians(
185  const DependentVector& currentStateVector )
186  {
187  // Pre-compute Jacobians
188  DependentMatrix stateJacobian = stateJacobianFunction_( this->currentTime_, currentStateVector );
189  DependentMatrix noiseJacobian = stateNoiseJacobianFunction_( this->currentTime_, currentStateVector );
190 
191  // Get sizes
192  unsigned int stateDimension = stateJacobian.rows( );
193  unsigned int noiseCols = noiseJacobian.cols( );
194 
195  // Merge Jacobians in one matrix
196  DependentMatrix continuousJacobians = DependentMatrix::Zero( stateDimension + noiseCols, stateDimension + noiseCols );
197  continuousJacobians.block( 0, 0, stateDimension, stateDimension ) = stateJacobian;
198  continuousJacobians.block( 0, stateDimension, stateDimension, noiseCols ) = noiseJacobian;
199 
200  // Generate discrete-time Jacobians
201  DependentMatrix discreteJacobians = ( continuousJacobians * this->filteringStepSize_ ).exp( );
202 
203  // Extract state and noise Jacobians
204  return std::make_pair( discreteJacobians.block( 0, 0, stateDimension, stateDimension ),
205  discreteJacobians.block( 0, stateDimension, stateDimension, noiseCols ) );
206  }
