37 template< typename DirectGravityPartial >
38 basic_astrodynamics::AvailableAcceleration getAccelerationTypeOfThirdBodyGravity(
39  const std::shared_ptr< DirectGravityPartial > directGravityPartial )
40 {
41  using namespace basic_astrodynamics;
42  AvailableAcceleration accelerationType;
43 
44  // Check type of direct partial derivative.
45  if( std::dynamic_pointer_cast< CentralGravitationPartial >( directGravityPartial ) != nullptr )
46  {
47  accelerationType = third_body_central_gravity;
48  }
49  else if( std::dynamic_pointer_cast< SphericalHarmonicsGravityPartial >( directGravityPartial ) != nullptr )
50  {
51  accelerationType = third_body_spherical_harmonic_gravity;
52  }
53  else if( std::dynamic_pointer_cast< MutualSphericalHarmonicsGravityPartial >( directGravityPartial ) != nullptr )
54  {
55  accelerationType = third_body_mutual_spherical_harmonic_gravity;
56  }
57 
58  else
59  {
60  throw std::runtime_error( "Error when getting third body partial type, type not identified" );
61  }
62  return accelerationType;
63 }

363  int setParameterPartialUpdateFunction(
364  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
365  {
366  // Check parameter dependency of direct acceleration.
367  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromDirectGravity =
368  partialOfDirectGravityOnBodyUndergoingAcceleration_->getParameterPartialFunction( parameter );
369  if( partialFunctionFromDirectGravity.second > 0 )
370  {
371  partialOfDirectGravityOnBodyUndergoingAcceleration_->setParameterPartialUpdateFunction( parameter );
372  }
373 
374  // Check parameter dependency of indirect acceleration.
375  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromCentralGravity =
376  partialOfDirectGravityOnCentralBody_->getParameterPartialFunction( parameter );
377  if( partialFunctionFromCentralGravity.second > 0 )
378  {
379  partialOfDirectGravityOnCentralBody_->setParameterPartialUpdateFunction( parameter );
380  }
381 
382  // If either dependency exists, create combined partial.
383  if( partialFunctionFromCentralGravity.second > 0 || partialFunctionFromDirectGravity.second > 0 )
384  {
385  parameterDoublePartialFunctions_[ parameter ] =
386  getCombinedCurrentDoubleParameterFunction(
387  partialOfDirectGravityOnBodyUndergoingAcceleration_,
388  partialOfDirectGravityOnCentralBody_,
389  parameter, partialFunctionFromDirectGravity.second, partialFunctionFromCentralGravity.second, 1 );
390  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
391  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
392  }
393  return std::max( partialFunctionFromDirectGravity.second, partialFunctionFromCentralGravity.second );
394  }

517 inline std::string getCentralBodyNameFromThirdBodyAccelerationPartial(
518  const std::shared_ptr< AccelerationPartial > accelerationPartial )
519 {
520  std::string centralBody;
521  if( !basic_astrodynamics::isAccelerationFromThirdBody( accelerationPartial->getAccelerationType( ) ) )
522  {
523  throw std::runtime_error( "Error, requested third body from acceleration partial, but input is incompatible." );
524  }
525  else
526  {
527  if( accelerationPartial->getAccelerationType( ) == basic_astrodynamics::third_body_point_mass_gravity )
528  {
529  centralBody = std::dynamic_pointer_cast< ThirdBodyGravityPartial< CentralGravitationPartial > >(
530  accelerationPartial )->getCentralBodyName( );
531  }
532  else if( accelerationPartial->getAccelerationType( ) == basic_astrodynamics::third_body_spherical_harmonic_gravity )
533  {
534  centralBody = std::dynamic_pointer_cast< ThirdBodyGravityPartial< SphericalHarmonicsGravityPartial > >(
535  accelerationPartial )->getCentralBodyName( );
536  }
537  else if( accelerationPartial->getAccelerationType( ) == basic_astrodynamics::third_body_mutual_spherical_harmonic_gravity )
538  {
539  centralBody = std::dynamic_pointer_cast< ThirdBodyGravityPartial< MutualSphericalHarmonicsGravityPartial > >(
540  accelerationPartial )->getCentralBodyName( );
541  }
542  else
543  {
544  throw std::runtime_error( "Error, requested third body from acceleration partial, input third-body type not recognized." );
545  }
546  }
547 
548  return centralBody;
549 }
