39  double centralBodyGravitationalParameter = centralBodyGravitationalParameterFunction_( );

43  orbital_element_conversions::convertCartesianToModifiedEquinoctialElements(

47  double auxiliaryParameterW = ( 1.0 + modifiedEquinoctialElements( 1 ) * cos( modifiedEquinoctialElements( 5 ) )
48  + modifiedEquinoctialElements( 2 ) * sin( modifiedEquinoctialElements( 5 ) ) );

50  + modifiedEquinoctialElements( 4 ) * modifiedEquinoctialElements( 4 );

53  double Lap = costates_( 0 ) * 2.0 * modifiedEquinoctialElements( 0 ) / auxiliaryParameterW;
54  double Laf1 = costates_( 1 ) * sin( modifiedEquinoctialElements( 5 ) ) ;
55  double Laf2 = costates_( 1 ) / auxiliaryParameterW *
