40 Eigen::VectorXd Torus::getSurfacePoint( const double majorCircumferentialAngle,
41  const double minorCircumferentialAngle )
42 {
43  // Form cartesian position vector from paranmetrization.
44  cartesianPositionVector_( 0 )
45  = ( majorRadius_ + ( minorRadius_ * std::cos( minorCircumferentialAngle ) ) )
46  * std::cos( majorCircumferentialAngle );
47  cartesianPositionVector_( 1 )
48  = ( majorRadius_ + ( minorRadius_ * std::cos( minorCircumferentialAngle ) ) )
49  * std::sin( majorCircumferentialAngle );
50  cartesianPositionVector_( 2 ) = minorRadius_ * std::sin( minorCircumferentialAngle );
51 
52  // Translate vector.
53  transformPoint( cartesianPositionVector_ );
54 
55  // Return Cartesian position vector.
56  return cartesianPositionVector_;
57 }

60 Eigen::VectorXd Torus::getSurfaceDerivative( const double majorCircumferentialAngle,
61  const double minorCircumferentialAngle,
62  const int powerOfMajorCircumferentialAngleDerivative,
63  const int powerOfMinorCircumferentialAngleDerivative )
64 {
65  // Declare and set size of derivative vector.
66  Eigen::VectorXd derivative_ = Eigen::VectorXd( 3 );
67 
68  // Go through the different possibilities for the values of the power
69  // of the derivative.
70  if ( powerOfMajorCircumferentialAngleDerivative < 0
71  || powerOfMinorCircumferentialAngleDerivative < 0 )
72  {
73  derivative_( 0 ) = 0.0;
74  derivative_( 1 ) = 0.0;
75  derivative_( 2 ) = 0.0;
76  }
77 
78  // When requesting the zeroth derivative with respect to the two
79  // independent variables, the surface point is returned. Note that this
80  // does include the offset.
81  else if ( powerOfMajorCircumferentialAngleDerivative == 0 &&
82  powerOfMinorCircumferentialAngleDerivative == 0 )
83  {
84  derivative_ = getSurfacePoint( majorCircumferentialAngle,
85  minorCircumferentialAngle );
86  }
87 
88  // The contributions of the two parameterizing variables are determined
89  // and then multiplied component-wise.
90  else
91  {
92  // Declare and set sizes of contribution vectors.
93  Eigen::VectorXd derivative1Contribution_ = Eigen::VectorXd( 3 );
94  Eigen::VectorXd derivative2Contribution_ = Eigen::VectorXd( 3 );
95 
96  // Since this derivative is "cyclical", as it is
97  // only dependent on sines and cosines, only the "modulo 4"th
98  // derivatives need to be determined. Derivatives are determined
99  // from the form of the spherical coordinates, see
100  // basic_mathematics::coordinateConversions::convertSphericalToCartesian
101  switch( powerOfMajorCircumferentialAngleDerivative % 4 )
102  {
103  case( 0 ):
104 
105  derivative1Contribution_( 0 ) = std::cos( majorCircumferentialAngle );
106  derivative1Contribution_( 1 ) = std::sin( majorCircumferentialAngle );
107  derivative1Contribution_( 2 ) = 1.0;
108  break;
109 
110  case( 1 ):
111 
112  derivative1Contribution_( 0 ) = -std::sin( majorCircumferentialAngle );
113  derivative1Contribution_( 1 ) = std::cos( majorCircumferentialAngle );
114  derivative1Contribution_( 2 ) = 0.0;
115  break;
116 
117  case( 2 ):
118 
119  derivative1Contribution_( 0 ) = -std::cos( majorCircumferentialAngle );
120  derivative1Contribution_( 1 ) = -std::sin( majorCircumferentialAngle );
121  derivative1Contribution_( 2 ) = 0.0;
122  break;
123 
124  case( 3 ):
125 
126  derivative1Contribution_( 0 ) = std::sin( majorCircumferentialAngle );
127  derivative1Contribution_( 1 ) = -std::cos( majorCircumferentialAngle );
128  derivative1Contribution_( 2 ) = 0.0;
129  break;
130 
131  default:
132 
133  throw std::runtime_error( " Bad value for powerOfMajorCircumferentialAngleDerivative ( mod 4 ) of value is not 0, 1, 2 or 3 in torus" );
134  }
135 
136  // This derivative is "cyclical" in the same manner as the derivative
137  // with respect to the 1st independent variable. One check needs to
138  // made, as the zeroth derivative violates cyclicity
139  switch( powerOfMinorCircumferentialAngleDerivative %4 )
140  {
141  case( 0 ):
142 
143  derivative2Contribution_( 0 ) = minorRadius_ * std::cos( minorCircumferentialAngle );
144  derivative2Contribution_( 1 ) = minorRadius_ * std::cos( minorCircumferentialAngle );
145  derivative2Contribution_( 2 ) = minorRadius_ * std::sin( minorCircumferentialAngle );
146 
147  // For the zeroth derivative, a term needs to be added.
148  if ( powerOfMinorCircumferentialAngleDerivative == 0 )
149  {
150  derivative2Contribution_( 0 ) += majorRadius_;
151  derivative2Contribution_( 1 ) += majorRadius_;
152  }
153 
154  break;
155 
156  case( 1 ):
157 
158  derivative2Contribution_( 0 ) = -minorRadius_ * std::sin( minorCircumferentialAngle );
159  derivative2Contribution_( 1 ) = -minorRadius_ * std::sin( minorCircumferentialAngle );
160  derivative2Contribution_( 2 ) = minorRadius_ * std::cos( minorCircumferentialAngle );
161  break;
162 
163  case( 2 ):
164 
165  derivative2Contribution_( 0 ) = -minorRadius_ * std::cos( minorCircumferentialAngle );
166  derivative2Contribution_( 1 ) = -minorRadius_ * std::cos( minorCircumferentialAngle );
167  derivative2Contribution_( 2 ) = -minorRadius_ * std::sin( minorCircumferentialAngle );
168  break;
169 
170  case( 3 ):
171 
172  derivative2Contribution_( 0 ) = minorRadius_ * std::sin( minorCircumferentialAngle );
173  derivative2Contribution_( 1 ) = minorRadius_ * std::sin( minorCircumferentialAngle );
174  derivative2Contribution_( 2 ) = -minorRadius_ * std::cos( minorCircumferentialAngle );
175  break;
176 
177  default:
178 
179  throw std::runtime_error( " Bad value for powerOfMinorCircumferentialAngleDerivative ( mod 4 ) of value is not 0, 1, 2 or 3 in torus" );
180  }
181 
182  // Construct the full derivative.
183  derivative_( 0 ) = derivative1Contribution_( 0 ) * derivative2Contribution_( 0 );
184  derivative_( 1 ) = derivative1Contribution_( 1 ) * derivative2Contribution_( 1 );
185  derivative_( 2 ) = derivative1Contribution_( 2 ) * derivative2Contribution_( 2 );
186 
187  // Rotate derivatives to take into account rotation of torus.
188  derivative_ = rotationMatrix_ * scalingMatrix_ * derivative_;
189  }
190 
191  // Return derivative vector.
192  return derivative_;
193 }
