98 void AtmosphericFlightConditions::setAerodynamicCoefficientsIndependentVariableFunction(
99  const AerodynamicCoefficientsIndependentVariables independentVariable,
100  const std::function< double( ) > coefficientDependency )
101 {
102  if( ( independentVariable == mach_number_dependent ) ||
103  ( independentVariable == angle_of_attack_dependent ) ||
104  ( independentVariable == angle_of_sideslip_dependent )||
105  ( independentVariable == altitude_dependent ) )
106  {
107  throw std::runtime_error(
108  std::string( "Error when setting aerodynamic coefficient function dependency, value of parameter " ) +
109  std::to_string( independentVariable ) +
110  std::string(", will not be used." ) );
111  }
112  else
113  {
114  customCoefficientDependencies_[ independentVariable ] = coefficientDependency;
115  }

119 void AtmosphericFlightConditions::updateConditions( const double currentTime )
120 {
121  if( !( currentTime == currentTime_ ) )
122  {
123  currentTime_ = currentTime;
124 
125  // Update aerodynamic angles (but not angles w.r.t. body-fixed frame).
126  if( aerodynamicAngleCalculator_!= nullptr )
127  {
128  aerodynamicAngleCalculator_->update( currentTime, false );
129  }
130 
131  // Calculate state of vehicle in global frame and corotating frame.
132  currentBodyCenteredAirspeedBasedBodyFixedState_ = bodyCenteredPseudoBodyFixedStateFunction_( );
133 
134  updateAerodynamicCoefficientInput( );
135 
136  // Update angles from aerodynamic to body-fixed frame (if relevant).
137  if( aerodynamicAngleCalculator_!= nullptr )
138  {
139  aerodynamicAngleCalculator_->update( currentTime, true );
140  updateAerodynamicCoefficientInput( );
141  }
142 
143  // Update aerodynamic coefficients.
144  aerodynamicCoefficientInterface_->updateFullCurrentCoefficients(
145  aerodynamicCoefficientIndependentVariables_, controlSurfaceAerodynamicCoefficientIndependentVariables_ );
146  }

150 double AtmosphericFlightConditions::getAerodynamicCoefficientIndependentVariable(
151  const AerodynamicCoefficientsIndependentVariables independentVariableType,
152  const std::string& secondaryIdentifier )
153 {
154  double currentIndependentVariable;
155  switch( independentVariableType )
156  {
157  //Calculate Mach number if needed.
158  case mach_number_dependent:
159  currentIndependentVariable = getCurrentMachNumber( );
160  break;
161  //Get angle of attack if needed.
162  case angle_of_attack_dependent:
163  if( aerodynamicAngleCalculator_== nullptr )
164  {
165  throw std::runtime_error( "Error, aerodynamic angle calculator is nullptr, but require angle of attack" );
166  }
167  currentIndependentVariable = aerodynamicAngleCalculator_->getAerodynamicAngle(
168  reference_frames::angle_of_attack );
169  break;
170  //Get angle of sideslip if needed.
171  case angle_of_sideslip_dependent:
172  if( aerodynamicAngleCalculator_== nullptr )
173  {
174  throw std::runtime_error( "Error, aerodynamic angle calculator is nullptr, but require angle of sideslip" );
175  }
176  currentIndependentVariable = aerodynamicAngleCalculator_->getAerodynamicAngle(
177  reference_frames::angle_of_sideslip );
178  break;
179  case altitude_dependent:
180  if( aerodynamicAngleCalculator_== nullptr )
181  {
182  throw std::runtime_error( "Error, aerodynamic angle calculator is nullptr, but require angle of sideslip" );
183  }
184  currentIndependentVariable = getCurrentAltitude( );
185  break;
186  case control_surface_deflection_dependent:
187  {
188  try
189  {
190  currentIndependentVariable = controlSurfaceDeflectionFunction_( secondaryIdentifier );
191  }
192  catch( std::runtime_error )
193  {
194  throw std::runtime_error( "Error, control surface " + secondaryIdentifier + "not recognized when updating coefficients" );
195  }
196  break;
197  }
198  default:
199  if( customCoefficientDependencies_.count( independentVariableType ) == 0 )
200  {
201  throw std::runtime_error( "Error, did not recognize aerodynamic coefficient dependency "
202  + std::to_string( independentVariableType ) );
203  }
204  else
205  {
206  currentIndependentVariable = customCoefficientDependencies_.at( independentVariableType )( );
207  }
208  }
209 
210  return currentIndependentVariable;
211 }

214 void AtmosphericFlightConditions::updateAerodynamicCoefficientInput( )
215 {
216  aerodynamicCoefficientIndependentVariables_.clear( );
217  // Calculate independent variables for aerodynamic coefficients.
218  for( unsigned int i = 0; i < aerodynamicCoefficientInterface_->getNumberOfIndependentVariables( ); i++ )
219  {
220  aerodynamicCoefficientIndependentVariables_.push_back(
221  getAerodynamicCoefficientIndependentVariable(
222  aerodynamicCoefficientInterface_->getIndependentVariableName( i ) ) );
223  }
224 
225  controlSurfaceAerodynamicCoefficientIndependentVariables_.clear( );
226  for( unsigned int i = 0; i < aerodynamicCoefficientInterface_->getNumberOfControlSurfaces( ); i++ )
227  {
228  std::string currentControlSurface = aerodynamicCoefficientInterface_->getControlSurfaceName( i );
229  for( unsigned int j = 0; j < aerodynamicCoefficientInterface_->getNumberOfControlSurfaceIndependentVariables( currentControlSurface ); j++ )
230  {
231  controlSurfaceAerodynamicCoefficientIndependentVariables_[ currentControlSurface ].push_back(
232  getAerodynamicCoefficientIndependentVariable(
233  aerodynamicCoefficientInterface_->getControlSurfaceIndependentVariableName(
234  currentControlSurface, j ), currentControlSurface ) );
235  }
236  }
237 }

240 std::shared_ptr< TrimOrientationCalculator > setTrimmedConditions(
241  const std::shared_ptr< AtmosphericFlightConditions > flightConditions )
242 {
243  // Create trim object.
244  std::shared_ptr< TrimOrientationCalculator > trimOrientation =
245  std::make_shared< TrimOrientationCalculator >(
246  flightConditions->getAerodynamicCoefficientInterface( ) );
247 
248  // Create angle-of-attack function from trim object.
249  std::function< std::vector< double >( ) > untrimmedIndependentVariablesFunction =
250  std::bind( &AtmosphericFlightConditions::getAerodynamicCoefficientIndependentVariables,
251  flightConditions );
252  std::function< std::map< std::string, std::vector< double > >( ) > untrimmedControlSurfaceIndependentVariablesFunction =
253  std::bind( &AtmosphericFlightConditions::getControlSurfaceAerodynamicCoefficientIndependentVariables,
254  flightConditions );
255  flightConditions->getAerodynamicAngleCalculator( )->setOrientationAngleFunctions(
256  std::bind( &TrimOrientationCalculator::findTrimAngleOfAttackFromFunction, trimOrientation,
257  untrimmedIndependentVariablesFunction, untrimmedControlSurfaceIndependentVariablesFunction ) );
258 
259  return trimOrientation;
260 }
