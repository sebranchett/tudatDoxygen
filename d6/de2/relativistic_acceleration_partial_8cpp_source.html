20 void computePartialOfSchwarzschildAccelerationCorrectionWrtPosition(
21  const Eigen::Vector6d& relativeState, Eigen::Vector3d& currentAcceleration, Eigen::Matrix3d& partialMatrix,
22  const double gravitationalParameter, const double ppnParameterGamma, const double ppnParameterBeta )
23 {
24  Eigen::Vector3d position = relativeState.segment( 0, 3 );
25  Eigen::Vector3d velocity = relativeState.segment( 3, 3 );
26  double distance = position.norm( );
27 
28  partialMatrix = 2.0 * ( ppnParameterGamma + ppnParameterBeta ) * gravitationalParameter / distance * (
29  Eigen::Matrix3d::Identity( ) - position * position.transpose( ) / ( distance * distance ) );
30  partialMatrix -= ppnParameterGamma * velocity.dot( velocity ) * Eigen::Matrix3d::Identity( );
31  partialMatrix += 2.0 * ( 1.0 + ppnParameterGamma ) * velocity * velocity.transpose( );
32  partialMatrix *= gravitationalParameter * physical_constants::INVERSE_SQUARE_SPEED_OF_LIGHT / ( distance * distance * distance );
33  partialMatrix -= 3.0 * currentAcceleration * position.transpose( ) / ( distance * distance );
34 }

37 void computePartialOfSchwarzschildAccelerationCorrectionWrtVelocity(
38  const Eigen::Vector6d& relativeState, Eigen::Matrix3d& partialMatrix,
39  const double gravitationalParameter, const double ppnParameterGamma )
40 {
41  Eigen::Vector3d position = relativeState.segment( 0, 3 );
42  Eigen::Vector3d velocity = relativeState.segment( 3, 3 );
43  double distance = position.norm( );
44 
45  partialMatrix = gravitationalParameter * physical_constants::INVERSE_SQUARE_SPEED_OF_LIGHT / ( distance * distance * distance ) *
46  ( - 2.0 * ppnParameterGamma * position * velocity.transpose( ) +
47  2.0 * ( 1.0 + ppnParameterGamma ) * (
48  position.dot( velocity ) * Eigen::Matrix3d::Identity( ) + velocity * position.transpose( ) ) );

53 void computePartialOfSchwarzschildAccelerationCorrectionWrtGravitationalParameter(
54  const Eigen::Vector6d& relativeState,
55  const double gravitationalParameter,
56  Eigen::MatrixXd& partialMatrix, const double ppnParameterGamma, const double ppnParameterBeta )
57 {
58  Eigen::Vector3d position = relativeState.segment( 0, 3 );
59  Eigen::Vector3d velocity = relativeState.segment( 3, 3 );
60  double distance = position.norm( );
61 
62  partialMatrix = physical_constants::INVERSE_SQUARE_SPEED_OF_LIGHT / ( distance * distance * distance ) *
63  ( -ppnParameterGamma * ( velocity.dot( velocity ) ) * position +
64  2.0 * ( 1.0 + ppnParameterGamma ) *
65  ( position.dot( velocity ) ) * velocity
66  + 4.0 * gravitationalParameter * ( ppnParameterGamma + ppnParameterBeta ) *
67  position / distance );
68 }

78  double distance = position.norm( );
79 
80  partialMatrix = physical_constants::INVERSE_SQUARE_SPEED_OF_LIGHT * gravitationalParameter / ( distance * distance * distance ) * (

98 std::pair< boost::function< void( Eigen::MatrixXd& ) >, int >
99 RelativisticAccelerationPartial::getParameterPartialFunction(
100  boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
101 {
102  boost::function< void( Eigen::MatrixXd& ) > partialFunction;
103  int numberOfRows = 0;
104 
105  // Create partial function if parameter is central body gravitational parameter
106  if( parameter->getParameterName( ).second.first == acceleratingBody_ )
107  {
108  switch( parameter->getParameterName( ).first )
109  {
110  case estimatable_parameters::gravitational_parameter:
111  partialFunction = boost::bind( &RelativisticAccelerationPartial::wrtGravitationalParameterOfCentralBody, this, _1 );
112  numberOfRows = 1;
113  break;
114  default:
115  break;
116  }
117  }
118  // Create partial function if parameter is PPN parameter beta or gamma
119  else if( parameter->getParameterName( ).second.first == "global_metric" )
120  {
121  switch( parameter->getParameterName( ).first )
122  {
123  case estimatable_parameters::ppn_parameter_gamma:
124  partialFunction = boost::bind( &RelativisticAccelerationPartial::wrtPpnParameterGamma, this, _1 );
125  numberOfRows = 1;
126  break;
127  case estimatable_parameters::ppn_parameter_beta:
128  partialFunction = boost::bind( &RelativisticAccelerationPartial::wrtPpnParameterBeta, this, _1 );
129  numberOfRows = 1;
130  break;
131  default:
132  break;
133  }
134  }
135  return std::make_pair( partialFunction, numberOfRows );
136 }
