65 template< typename TimeType, typename StateScalarType >
66 void determineInitialStates(
67  nlohmann::json& jsonObject,
68  const simulation_setup::NamedBodyMap& bodyMap,
69  const std::shared_ptr< numerical_integrators::IntegratorSettings< TimeType > >& integratorSettings )
70 {
71  using namespace propagators;
72  using namespace basic_astrodynamics;
73  using namespace simulation_setup;
74  using K = Keys::Propagator;
75 
76  // Get propagators
77  nlohmann::json jsonPropagators = jsonObject.at( Keys::propagators );
78 
79  // Update propagators at jsonObject with initial states retrieved from bodies ephemeris
80  bool usedEphemeris = false;
81  if ( jsonPropagators.size( ) == 1 )
82  {
83  nlohmann::json& jsonPropagator = jsonPropagators.front( );
84  if ( ! isDefined( jsonPropagator, K::initialStates ) )
85  {
86  const IntegratedStateType integratedStateType =
87  getValue( jsonPropagator, K::integratedStateType, translational_state );
88 
89  if ( integratedStateType == translational_state )
90  {
91  try
92  {
93  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialStates =
94  getInitialStatesOfBodies< TimeType, StateScalarType >(
95  getValue< std::vector< std::string > >( jsonPropagator, K::bodiesToPropagate ),
96  getValue< std::vector< std::string > >( jsonPropagator, K::centralBodies ),
97  bodyMap,
98  integratorSettings->initialTime_ );
99  jsonPropagator[ K::initialStates ] = initialStates;
100  usedEphemeris = true;
101  }
102  catch ( ... ) { }
103  }
104  }
105  }
106 
107  if ( ! usedEphemeris )
108  {
109  // Update propagators at jsonObject with initial states stored at JSON bodies settings
110  for ( nlohmann::json& jsonPropagator : jsonPropagators )
111  {
112  if ( ! isDefined( jsonPropagator, K::initialStates ) )
113  {
114  // Integrated state type
115  const IntegratedStateType integratedStateType =
116  getValue( jsonPropagator, K::integratedStateType, translational_state );
117 
118  // State size and associated stateKey
119  const unsigned int stateSize = getSingleIntegrationSize( integratedStateType );
120  const std::string stateKey = getAssociatedKey( integratedStateType );
121 
122  // Bodies to propagate
123  const std::vector< std::string > bodiesToPropagate =
124  getValue< std::vector< std::string > >( jsonPropagator, K::bodiesToPropagate );
125 
126  // System initial state
127  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialStates =
128  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero(
129  stateSize * bodiesToPropagate.size( ), 1 );
130 
131  // Get state for each body
132  for ( unsigned int i = 0; i < bodiesToPropagate.size( ); ++i )
133  {
134  const KeyPath stateKeyPath = Keys::bodies / bodiesToPropagate.at( i ) / stateKey;
135  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > bodyState;
136  if ( integratedStateType == translational_state )
137  {
138  const std::string centralBodyName = getValue< std::vector< std::string > >(
139  jsonPropagator, K::centralBodies ).at( i );
140  bodyState = getCartesianState< StateScalarType >(
141  jsonObject, stateKeyPath, bodyMap.at( centralBodyName ),
142  integratorSettings->initialTime_ );
143  }
144  else
145  {
146  bodyState = getValue< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >(
147  jsonObject, stateKeyPath );
148  }
149  initialStates.segment( i * stateSize, stateSize ) = bodyState;
150  }
151 
152  // Update system initial states
153  jsonPropagator[ K::initialStates ] = initialStates;
154  }
155  }
156  }
157 
158  jsonObject[ Keys::propagators ] = jsonPropagators;
159 }

200 template< typename TimeType >
201 TimeType getTerminationEpoch(
202  const std::shared_ptr< propagators::PropagationTerminationSettings >& terminationSettings )
203 {
204  using namespace propagators;
205 
206  std::shared_ptr< PropagationTimeTerminationSettings > timeTerminationSettings =
207  std::dynamic_pointer_cast< PropagationTimeTerminationSettings >( terminationSettings );
208  if ( timeTerminationSettings )
209  {
210  return timeTerminationSettings->terminationTime_;
211  }
212 
213  std::shared_ptr< PropagationHybridTerminationSettings > hybridTerminationSettings =
214  std::dynamic_pointer_cast< PropagationHybridTerminationSettings >( terminationSettings );
215  if ( hybridTerminationSettings )
216  {
217  for ( unsigned int i = 0; i < hybridTerminationSettings->terminationSettings_.size( ); ++i )
218  {
219  const TimeType endEpoch =
220  getTerminationEpoch< TimeType >( hybridTerminationSettings->terminationSettings_.at( i ) );
221  if ( ! isNaN( endEpoch ) )
222  {
223  return endEpoch;
224  }
225  }
226  }
227 
228  return TUDAT_NAN;
229 }

350 template< typename StateScalarType >
351 void from_json( const nlohmann::json& jsonObject,
352  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > >& multiTypePropagatorSettings )
353 {
354  using namespace simulation_setup;
355  using namespace json_interface;
356 
357  // Termination settings. If not provided, stop when epoch > simulation.finalEpoch
358  std::vector< std::shared_ptr< PropagationTerminationSettings > > terminationConditions;
359 
360  // Find user-defined conditions (and determine if time condition is missing)
361  bool timeConditionMissing = true;
362  if ( isDefined( jsonObject, Keys::termination ) )
363  {
364  std::shared_ptr< PropagationHybridTerminationSettings > userConditions =
365  getValue< std::shared_ptr< PropagationHybridTerminationSettings > >( jsonObject, Keys::termination );
366  terminationConditions.push_back( userConditions );
367 
368  for ( std::shared_ptr< PropagationTerminationSettings > condition : userConditions->terminationSettings_ )
369  {
370  std::shared_ptr< PropagationTimeTerminationSettings > timeCondition =
371  std::dynamic_pointer_cast< PropagationTimeTerminationSettings >( condition );
372  if ( timeCondition )
373  {
374  timeConditionMissing = false;
375  break;
376  }
377  }
378  }
379 
380  // If user did not provide conditions, or if finalEpoch is defined but the time condition is missing, create it
381  if ( ! isDefined( jsonObject, Keys::termination ) ||
382  ( isDefined( jsonObject, Keys::finalEpoch ) && timeConditionMissing ) )
383  {
384  terminationConditions.push_back( std::make_shared< PropagationTimeTerminationSettings >(
385  getValue< double >( jsonObject, Keys::finalEpoch ) ) );
386  }
387 
388  // If there's only one condition in total (either user-provided time, user-provided dependent or created time)
389  // use it as termination settings.
390  std::shared_ptr< PropagationTerminationSettings > terminationSettings;
391  if ( terminationConditions.size( ) == 1 )
392  {
393  terminationSettings = terminationConditions.at( 0 );
394  }
395  // If there are two conditions (those provided by the user, and the created time condition), combine them into
396  // hybrid termination settings satisfying any of the two conditions.
397  else
398  {
399  terminationSettings = std::make_shared< PropagationHybridTerminationSettings >( terminationConditions, true );
400  }
401 
402  multiTypePropagatorSettings = std::make_shared< MultiTypePropagatorSettings< StateScalarType > >(
403  getValue< std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >(
404  jsonObject, Keys::propagators ),
405  terminationSettings,
406  std::shared_ptr< DependentVariableSaveSettings >( ),
407  getValue< double >( jsonObject, Keys::options / Keys::Options::printInterval, TUDAT_NAN ) );
408 }

415 template< typename StateScalarType >
416 void to_json( nlohmann::json& jsonObject,
417  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > >& singleArcPropagatorSettings )
418 {
419  if ( ! singleArcPropagatorSettings )
420  {
421  return;
422  }
423  using namespace simulation_setup;
424  using namespace json_interface;
425  using K = Keys::Propagator;
426 
427  // Common keys
428  const IntegratedStateType integratedStateType = singleArcPropagatorSettings->getStateType( );
429  jsonObject[ K::integratedStateType ] = integratedStateType;
430  if ( singleArcPropagatorSettings->getInitialStates( ).rows( ) > 0 )
431  {
432  jsonObject[ K::initialStates ] = singleArcPropagatorSettings->getInitialStates( );
433  }
434 
435  switch ( integratedStateType )
436  {
437  case translational_state:
438  {
439  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
440  translationalStatePropagatorSettings = std::dynamic_pointer_cast<
441  TranslationalStatePropagatorSettings< StateScalarType > >( singleArcPropagatorSettings );
442  assertNonnullptrPointer( translationalStatePropagatorSettings );
443  jsonObject[ K::type ] = translationalStatePropagatorSettings->propagator_;
444  jsonObject[ K::centralBodies ] = translationalStatePropagatorSettings->centralBodies_;
445  jsonObject[ K::bodiesToPropagate ] = translationalStatePropagatorSettings->bodiesToIntegrate_;
446  jsonObject[ K::accelerations ] = translationalStatePropagatorSettings->getAccelerationSettingsMap( );
447  return;
448  }
449  case body_mass_state:
450  {
451  std::shared_ptr< MassPropagatorSettings< StateScalarType > > massPropagatorSettings =
452  std::dynamic_pointer_cast< MassPropagatorSettings< StateScalarType > >( singleArcPropagatorSettings );
453  assertNonnullptrPointer( massPropagatorSettings );
454  jsonObject[ K::bodiesToPropagate ] = massPropagatorSettings->bodiesWithMassToPropagate_;
455  jsonObject[ K::massRateModels ] = massPropagatorSettings->getMassRateSettingsMap( );
456  return;
457  }
458  case rotational_state:
459  {
460  std::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationalStatePropagatorSettings =
461  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >(
462  singleArcPropagatorSettings );
463 
464  assertNonnullptrPointer( rotationalStatePropagatorSettings );
465  jsonObject[ K::type ] = rotationalStatePropagatorSettings->propagator_;
466  jsonObject[ K::bodiesToPropagate ] = rotationalStatePropagatorSettings->bodiesToIntegrate_;
467  jsonObject[ K::torques ] = rotationalStatePropagatorSettings->getTorqueSettingsMap( );
468  return;
469  }
470  case hybrid:
471  {
472  std::cerr << "Multitype (hybrid) propagation is implicitly supported by providing a list of propagators, "
473  << "but multitype propagators cannot be nested inside multitype propagators." << std::endl;
474  throw;
475  }
476  default:
477  handleUnimplementedEnumValue( integratedStateType, integratedStateTypes, unsupportedIntegratedStateTypes );
478  }

482 template< typename StateScalarType >
483 void from_json( const nlohmann::json& jsonObject,
484  std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > >& singleArcPropagatorSettings )
485 {
486  using namespace simulation_setup;
487  using namespace json_interface;
488  using K = Keys::Propagator;
489 
490  // Integrated state type
491  const IntegratedStateType integratedStateType =
492  getValue( jsonObject, K::integratedStateType, translational_state );
493 
494  // Named of bodies to propagate
495  const std::vector< std::string > bodiesToPropagate =
496  getValue< std::vector< std::string > >( jsonObject, K::bodiesToPropagate );
497 
498  // Initial states
499  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialStates =
500  getValue< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >( jsonObject, K::initialStates );
501 
502  // No termination settings ( epoch > TUDAT_NAN will always be false )
503  std::shared_ptr< PropagationTerminationSettings > terminationSettings =
504  std::make_shared< PropagationTimeTerminationSettings >( TUDAT_NAN );
505 
506  switch ( integratedStateType )
507  {
508  case translational_state:
509  {
510  TranslationalStatePropagatorSettings< StateScalarType > defaults(
511  { }, SelectedAccelerationMap( ), { }, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >( ),
512  std::shared_ptr< PropagationTerminationSettings >( ) );
513  singleArcPropagatorSettings = std::make_shared< TranslationalStatePropagatorSettings< StateScalarType > >(
514  getValue< std::vector< std::string > >( jsonObject, K::centralBodies ),
515  getValue< SelectedAccelerationMap >( jsonObject, K::accelerations ),
516  bodiesToPropagate,
517  initialStates,
518  terminationSettings,
519  getValue( jsonObject, K::type, defaults.propagator_ ) );
520  return;
521  }
522  case body_mass_state:
523  {
524  MassPropagatorSettings< StateScalarType > defaults(
525  { }, SelectedMassRateModelMap( ), Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >( ), nullptr );
526  singleArcPropagatorSettings = std::make_shared< MassPropagatorSettings< StateScalarType > >(
527  bodiesToPropagate,
528  getValue< SelectedMassRateModelMap >( jsonObject, K::massRateModels ),
529  initialStates,
530  terminationSettings );
531  return;
532  }
533  case rotational_state:
534  {
535  RotationalStatePropagatorSettings< StateScalarType > defaults(
536  SelectedTorqueMap( ), { }, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >( ), nullptr );
537  singleArcPropagatorSettings = std::make_shared< RotationalStatePropagatorSettings< StateScalarType > >(
538  getValue< SelectedTorqueMap >( jsonObject, K::torques ),
539  bodiesToPropagate,
540  initialStates,
541  terminationSettings,
542  getValue( jsonObject, K::type, defaults.propagator_ ) );
543  return;
544  }
545  case hybrid:
546  {
547  std::cerr << "Multitype (hybrid) propagation is implicitly supported by providing a list of propagators, "
548  << "but multitype propagators cannot be nested inside multitype propagators." << std::endl;
549  throw;
550  }
551  default:
552  handleUnimplementedEnumValue( integratedStateType, integratedStateTypes, unsupportedIntegratedStateTypes );
553  }
554 }
