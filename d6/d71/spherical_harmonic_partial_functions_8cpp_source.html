29 void computePotentialSphericalHessian(
30  const double distance,
31  const double radiusPowerTerm,
32  const double cosineOfOrderLongitude,
33  const double sineOfOrderLongitude,
34  const double cosineOfLatitude,
35  const double sineOfLatitude,
36  const double preMultiplier,
37  const int degree,
38  const int order,
39  const double cosineHarmonicCoefficient,
40  const double sineHarmonicCoefficient,
41  const double legendrePolynomial,
42  const double legendrePolynomialDerivative,
43  const double legendrePolynomialSecondDerivative,
44  Eigen::Matrix3d& sphericalHessian )
45 {
46  sphericalHessian.setZero( );
47 
48  sphericalHessian( 0, 0 ) += static_cast< double >( degree + 1 ) * static_cast< double >( degree + 2 ) /
49  ( distance * distance ) * legendrePolynomial *
50  ( cosineHarmonicCoefficient * cosineOfOrderLongitude + sineHarmonicCoefficient * sineOfOrderLongitude );
51  sphericalHessian( 1, 0 ) += -static_cast< double >( degree + 1 ) / distance * cosineOfLatitude *
52  legendrePolynomialDerivative *
53  ( cosineHarmonicCoefficient * cosineOfOrderLongitude + sineHarmonicCoefficient * sineOfOrderLongitude );
54  sphericalHessian( 2, 0 ) += -static_cast< double >( order ) * static_cast< double >( degree + 1 ) / distance *
55  legendrePolynomial *
56  ( -cosineHarmonicCoefficient * sineOfOrderLongitude + sineHarmonicCoefficient * cosineOfOrderLongitude );
57 
58  sphericalHessian( 0, 1 ) = sphericalHessian( 1, 0 );
59  sphericalHessian( 1, 1 ) = ( cosineOfLatitude * cosineOfLatitude * legendrePolynomialSecondDerivative -
60  sineOfLatitude * legendrePolynomialDerivative ) *
61  ( cosineHarmonicCoefficient * cosineOfOrderLongitude + sineHarmonicCoefficient * sineOfOrderLongitude );
62  sphericalHessian( 2, 1 ) = static_cast< double >( order ) * cosineOfLatitude * legendrePolynomialDerivative *
63  ( -cosineHarmonicCoefficient * sineOfOrderLongitude + sineHarmonicCoefficient * cosineOfOrderLongitude );
64 
65  sphericalHessian( 0, 2 ) = sphericalHessian( 2, 0 );
66  sphericalHessian( 1, 2 ) = sphericalHessian( 2, 1 );
67  sphericalHessian( 2, 2 ) += static_cast< double >( order ) * static_cast< double >( order ) * legendrePolynomial *
68  ( -cosineHarmonicCoefficient * cosineOfOrderLongitude - sineHarmonicCoefficient * sineOfOrderLongitude );
69 
70 
71  sphericalHessian *= preMultiplier * radiusPowerTerm;
72 }

125 Eigen::Matrix3d computeCumulativeSphericalHessian(
126  const Eigen::Vector3d& sphericalPosition,
127  const double referenceRadius,
128  const double gravitionalParameter,
129  const Eigen::MatrixXd cosineHarmonicCoefficients,
130  const Eigen::MatrixXd sineHarmonicCoefficients,
131  const boost::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache )
132 {
133  double preMultiplier = gravitionalParameter / referenceRadius;
134 
135  Eigen::Matrix3d sphericalHessian, sphericalHessianTerm;
136 
137  sphericalHessian.setZero( );
138  for( int i = 0; i < cosineHarmonicCoefficients.rows( ); i++ )
139  {
140  for( int j = 0; ( j <= i && j < cosineHarmonicCoefficients.cols( ) ); j++ )
141  {
142  computePotentialSphericalHessian(
143  sphericalPosition, preMultiplier, i, j, cosineHarmonicCoefficients( i, j ),
144  sineHarmonicCoefficients( i, j ), sphericalHarmonicsCache, sphericalHessianTerm );
145  sphericalHessian += sphericalHessianTerm;
146  }
147  }
148  return sphericalHessian;
149 
150 }

154 Eigen::Matrix3d computePartialDerivativeOfBodyFixedSphericalHarmonicAcceleration(
155  const Eigen::Vector3d& cartesianPosition,
156  const Eigen::Vector3d& sphericalPosition,
157  const double referenceRadius,
158  const double gravitionalParameter,
159  const Eigen::MatrixXd cosineHarmonicCoefficients,
160  const Eigen::MatrixXd sineHarmonicCoefficients,
161  const boost::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache,
162  const Eigen::Vector3d& sphericalPotentialGradient,
163  const Eigen::Matrix3d& sphericalToCartesianGradientMatrix )
164 {
165  // Compute Hessian in spherical coordinates.
166  Eigen::Matrix3d sphericalHessian = computeCumulativeSphericalHessian(
167  sphericalPosition, referenceRadius, gravitionalParameter, cosineHarmonicCoefficients,
168  sineHarmonicCoefficients, sphericalHarmonicsCache );
169 
170  // Convert to Cartesian Hessian
171  Eigen::Matrix3d accelerationPartial =
172  sphericalToCartesianGradientMatrix * sphericalHessian * sphericalToCartesianGradientMatrix.transpose( );
173 
174  // Add effect of direct change in rotation matrix
175  accelerationPartial += coordinate_conversions::getDerivativeOfSphericalToCartesianGradient(
176  sphericalPotentialGradient, cartesianPosition );
177 
178  return accelerationPartial;
179 }

183 Eigen::Matrix3d computePartialDerivativeOfBodyFixedSphericalHarmonicAcceleration(
184  const Eigen::Vector3d& cartesianPosition,
185  const double referenceRadius,
186  const double gravitionalParameter,
187  const Eigen::MatrixXd cosineHarmonicCoefficients,
188  const Eigen::MatrixXd sineHarmonicCoefficients,
189  const boost::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache )
190 {
191  // Compute spherical position.
192  Eigen::Vector3d sphericalPosition =
193  coordinate_conversions::convertCartesianToSpherical( cartesianPosition );
194  sphericalPosition( 1 ) = mathematical_constants::PI / 2.0 - sphericalPosition( 1 );
195 
196  // Compute spherical to Cartesian gradient transformation.
197  Eigen::Matrix3d gradientTransformationMatrix =
198  coordinate_conversions::getSphericalToCartesianGradientMatrix( cartesianPosition );
199 
200  // Compute spherical gradient.
201  Eigen::Vector3d sphericalPotentialGradient = gradientTransformationMatrix.inverse( ) *
202  gravitation::computeGeodesyNormalizedGravitationalAccelerationSum(
203  cartesianPosition, gravitionalParameter, referenceRadius, cosineHarmonicCoefficients,
204  sineHarmonicCoefficients, sphericalHarmonicsCache );
205 
206  return computePartialDerivativeOfBodyFixedSphericalHarmonicAcceleration(
207  cartesianPosition, sphericalPosition, referenceRadius, gravitionalParameter, cosineHarmonicCoefficients,
208  sineHarmonicCoefficients, sphericalHarmonicsCache, sphericalPotentialGradient,

234  basic_mathematics::computePotentialGradient(

245 
246  // Transform partials to Cartesian position and integration frame.
