101 void AerodynamicAngleCalculator::update( const double currentTime, const bool updateBodyOrientation )
102 {
103  // Clear all current rotation matrices.
104  currentRotationMatrices_.clear( );
105 
106  // Get current body-fixed state.
107  if( !( currentTime == currentTime_ ) )
108  {
109 
110  currentBodyFixedGroundSpeedBasedState_ = bodyFixedStateFunction_( );
111  currentRotationFromCorotatingToInertialFrame_ = rotationFromCorotatingToInertialFrame_( );
112 
113  Eigen::Vector3d sphericalCoordinates = coordinate_conversions::convertCartesianToSpherical< double >(
114  currentBodyFixedGroundSpeedBasedState_.segment( 0, 3 ) );
115 
116  // Calculate latitude and longitude.
117  currentAerodynamicAngles_[ latitude_angle ] =
118  mathematical_constants::PI / 2.0 - sphericalCoordinates( 1 );
119  currentAerodynamicAngles_[ longitude_angle ] = sphericalCoordinates( 2 );
120 
121  // Compute wind velocity vector
122  Eigen::Vector3d localWindVelocity = Eigen::Vector3d::Zero( );
123  if( windModel_ != NULL )
124  {
125  localWindVelocity = windModel_->getCurrentWindVelocity(
126  shapeModel_->getAltitude( currentBodyFixedGroundSpeedBasedState_.segment( 0, 3 ) ),
127  currentAerodynamicAngles_[ longitude_angle ],
128  currentAerodynamicAngles_[ latitude_angle ],
129  currentTime );
130  }
131 
132  // Compute airspeed-based velocity vector
133  currentBodyFixedAirspeedBasedState_ = currentBodyFixedGroundSpeedBasedState_;
134  currentBodyFixedAirspeedBasedState_.segment( 3, 3 ) += localWindVelocity;
135 
136  // Calculate vertical <-> aerodynamic <-> body-fixed angles if neede.
137  if( calculateVerticalToAerodynamicFrame_ )
138  {
139  Eigen::Vector3d verticalFrameVelocity =
140  getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
141  currentAerodynamicAngles_.at( longitude_angle ),
142  currentAerodynamicAngles_.at( latitude_angle ) ) *
143  currentBodyFixedAirspeedBasedState_.segment( 3, 3 );
144 
145  currentAerodynamicAngles_[ heading_angle ] = calculateHeadingAngle( verticalFrameVelocity );
146  currentAerodynamicAngles_[ flight_path_angle ] =
147  calculateFlightPathAngle( verticalFrameVelocity );
148  }
149 
150  currentTime_ = currentTime;
151  }
152 
153  if( updateBodyOrientation && !( currentBodyAngleTime_ == currentTime ) )
154  {
155  if( !angleUpdateFunction_.empty( ) )
156  {
157  angleUpdateFunction_( currentTime );
158  }
159 
160  if( !angleOfAttackFunction_.empty( ) )
161  {
162  currentAerodynamicAngles_[ angle_of_attack ] = angleOfAttackFunction_( );
163  }
164 
165  if( !angleOfSideslipFunction_.empty( ) )
166  {
167  currentAerodynamicAngles_[ angle_of_sideslip ] = angleOfSideslipFunction_( );
168  }
169 
170  if( !bankAngleFunction_.empty( ) )
171  {
172  currentAerodynamicAngles_[ bank_angle ] = bankAngleFunction_( );
173  }
174 
175  currentBodyAngleTime_ = currentTime;
176  }
177  else if( !( currentBodyAngleTime_ == currentTime ) )
178  {
179  currentAerodynamicAngles_[ angle_of_attack ] = 0.0;
180  currentAerodynamicAngles_[ angle_of_sideslip ] = 0.0;
181  currentAerodynamicAngles_[ bank_angle ] = 0.0;
182  }
183 }

187 Eigen::Quaterniond AerodynamicAngleCalculator::getRotationQuaternionBetweenFrames(
188  const AerodynamicsReferenceFrames originalFrame,
189  const AerodynamicsReferenceFrames targetFrame )
190 {
191  // Initialize rotation to identity matrix.
192  Eigen::Quaterniond rotationToFrame = Eigen::Quaterniond( Eigen::Matrix3d::Identity( ) );
193 
194 
195  // Check if update settings are consistent with requested frames.
196  if( !calculateVerticalToAerodynamicFrame_ &&
197  ( originalFrame > vertical_frame || targetFrame > vertical_frame ) )
198  {
199  throw std::runtime_error( "Error in AerodynamicAngleCalculator, instance ends at vertical frame" );
200  }
201 
202  // Set current frame pair.
203  std::pair< AerodynamicsReferenceFrames, AerodynamicsReferenceFrames > currentRotationPair =
204  std::make_pair( originalFrame, targetFrame );
205 
206  // Calculate rotation matrix if current rotation is not yet calculated.
207  if( currentRotationMatrices_.count( currentRotationPair ) == 0 )
208  {
209  // Get indices of required frames.
210  int currentFrameIndex = static_cast< int >( originalFrame );
211  int targetFrameIndex = static_cast< int >( targetFrame );
212 
213  // Check if any rotation is needed.
214  if( currentFrameIndex != targetFrameIndex )
215  {
216  // Check 'direction' of transformation through AerodynamicsReferenceFrames list.
217  bool isTargetFrameUp;
218  if( targetFrameIndex > currentFrameIndex )
219  {
220  isTargetFrameUp = 1;
221  }
222  else if( targetFrameIndex < currentFrameIndex )
223  {
224  isTargetFrameUp = 0;
225  }
226  else
227  {
228  throw std::runtime_error(
229  "Error when identifying target frame direction in AerodynamicAngleCalculator." );
230 
231  }
232 
233  // Add rotation sequence until final frame is reached.
234  while( currentFrameIndex != targetFrameIndex )
235  {
236  switch( currentFrameIndex )
237  {
238  case static_cast< int >( inertial_frame ):
239 
240  if( isTargetFrameUp )
241  {
242  rotationToFrame = currentRotationFromCorotatingToInertialFrame_.inverse( ) *
243  rotationToFrame;
244  }
245  else
246  {
247  throw std::runtime_error(
248  "Error, inertial_frame is end frame in AerodynamicAngleCalculator" );
249  }
250  break;
251  case static_cast< int >( corotating_frame ):
252 
253  if( isTargetFrameUp )
254  {
255  rotationToFrame =
256  getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
257  currentAerodynamicAngles_.at( longitude_angle ),
258  currentAerodynamicAngles_.at( latitude_angle ) ) *
259  rotationToFrame;
260  }
261  else
262  {
263  rotationToFrame = currentRotationFromCorotatingToInertialFrame_ *
264  rotationToFrame;
265  }
266  break;
267  case static_cast< int >( vertical_frame ):
268  if( isTargetFrameUp )
269  {
270 
271  rotationToFrame =
272  getLocalVerticalFrameToTrajectoryTransformationQuaternion(
273  currentAerodynamicAngles_.at( flight_path_angle ),
274  currentAerodynamicAngles_.at( heading_angle ) ) * rotationToFrame;
275  }
276  else
277  {
278  rotationToFrame =
279  getLocalVerticalToRotatingPlanetocentricFrameTransformationQuaternion(
280  currentAerodynamicAngles_.at( longitude_angle ),
281  currentAerodynamicAngles_.at( latitude_angle ) ) *
282  rotationToFrame;
283  }
284  break;
285  case static_cast< int >( trajectory_frame ):
286  if( isTargetFrameUp )
287  {
288  rotationToFrame =
289  getTrajectoryToAerodynamicFrameTransformationQuaternion(
290  currentAerodynamicAngles_.at( bank_angle ) ) *
291  rotationToFrame;
292 
293  }
294  else
295  {
296  rotationToFrame =
297  getTrajectoryToLocalVerticalFrameTransformationQuaternion(
298  currentAerodynamicAngles_.at( flight_path_angle ),
299  currentAerodynamicAngles_.at( heading_angle ) ) *
300  rotationToFrame;
301  }
302  break;
303  case static_cast< int >( aerodynamic_frame ):
304  if( isTargetFrameUp )
305  {
306  rotationToFrame =
307  getAirspeedBasedAerodynamicToBodyFrameTransformationQuaternion(
308  currentAerodynamicAngles_.at( angle_of_attack ),
309  currentAerodynamicAngles_.at( angle_of_sideslip ) ) *
310  rotationToFrame;
311  }
312  else
313  {
314  rotationToFrame =
315  getAerodynamicToTrajectoryFrameTransformationQuaternion(
316  currentAerodynamicAngles_.at( bank_angle ) ) *
317  rotationToFrame;
318  }
319  break;
320  case static_cast< int >( body_frame ):
321  if( isTargetFrameUp )
322  {
323 
324  throw std::runtime_error(
325  "Error, body frame is end frame in AerodynamicAngleCalculator." );
326  }
327  else
328  {
329  rotationToFrame =
330  getBodyToAirspeedBasedAerodynamicFrameTransformationQuaternion(
331  currentAerodynamicAngles_.at( angle_of_attack ),
332  currentAerodynamicAngles_.at( angle_of_sideslip ) ) *
333  rotationToFrame;
334  }
335  break;
336  default:
337  throw std::runtime_error(
338  "Error, index " +
339  boost::lexical_cast< std::string>( currentFrameIndex ) +
340  "not found in AerodynamicAngleCalculator" );
341 
342  }
343 
344  // Increment/decrement current frame.
345  if( isTargetFrameUp )
346  {
347  currentFrameIndex++;
348  }
349  else
350  {
351  currentFrameIndex--;
352  }
353  }
354  }
355 
356  // Set current rotation (as well as inverse).
357  currentRotationMatrices_[ currentRotationPair ] = rotationToFrame;
358  currentRotationMatrices_[ std::make_pair( targetFrame, originalFrame ) ] =
359  rotationToFrame.inverse( );
360  }
361  else
362  {
363  rotationToFrame = currentRotationMatrices_.at( currentRotationPair );
364  }
365 
366  return rotationToFrame;
367 }

447 boost::function< Eigen::Vector3d( const Eigen::Vector3d& ) >
448 getAerodynamicForceTransformationFunction(
449  const boost::shared_ptr< AerodynamicAngleCalculator > aerodynamicAngleCalculator,
450  const AerodynamicsReferenceFrames accelerationFrame,
451  const boost::function< Eigen::Quaterniond( ) > bodyFixedToInertialFrameFunction,
452  const AerodynamicsReferenceFrames propagationFrame )
453 {
454  boost::function< Eigen::Vector3d( const Eigen::Vector3d& ) > transformationFunction;
455 
456  // If propagation frame is the inertial frame, use bodyFixedToInertialFrameFunction.
457  if( propagationFrame == inertial_frame )
458  {
459  std::vector< boost::function< Eigen::Vector3d( const Eigen::Vector3d& ) > > rotationsList;
460 
461  // Get accelerationFrame to corotating frame transformation.
462  boost::function< Eigen::Quaterniond( ) > firstRotation =
463  boost::bind( &AerodynamicAngleCalculator::getRotationQuaternionBetweenFrames,
464  aerodynamicAngleCalculator, accelerationFrame, corotating_frame );
465  rotationsList.push_back(
466  boost::bind( &transformVectorFromQuaternionFunction,
467  _1, firstRotation ) );
468 
469  // Add corotating to inertial frame.
470  rotationsList.push_back(
471  boost::bind( &transformVectorFromQuaternionFunction,
472  _1, bodyFixedToInertialFrameFunction ) );
473 
474  // Create transformation function.
475  transformationFunction = boost::bind( &transformVectorFromVectorFunctions,
476  _1, rotationsList );
477  }
478  else
479  {
480  // Get accelerationFrame to propagationFrame frame transformation directly.
481  boost::function< Eigen::Quaterniond( ) > rotationFunction =
482  boost::bind( &AerodynamicAngleCalculator::getRotationQuaternionBetweenFrames,
483  aerodynamicAngleCalculator, accelerationFrame, propagationFrame );
484 
485  // Create transformation function.
486  transformationFunction = boost::bind( &transformVectorFromQuaternionFunction, _1,
487  rotationFunction );
488  }
489 
490  return transformationFunction;
491 }

494 void AerodynamicAnglesClosure::updateAngles( const double currentTime )
495 {
496  // Retrieve rotation matrix that is to be converted to orientation angles.
497  currentRotationFromBodyToTrajectoryFrame_ =
498  ( ( imposedRotationFromInertialToBodyFixedFrame_( currentTime ).toRotationMatrix( ) *
499  aerodynamicAngleCalculator_->getRotationQuaternionBetweenFrames(
500  trajectory_frame, inertial_frame ).toRotationMatrix( ) ) ).transpose( );
501 
502  // Compute associated Euler angles and set as orientation angles.
503  Eigen::Vector3d eulerAngles = reference_frames::get132EulerAnglesFromRotationMatrix(
504  currentRotationFromBodyToTrajectoryFrame_ );
505  currentBankAngle_ = eulerAngles.x( );
506  currentAngleOfSideslip_ = eulerAngles.y( );
507  currentAngleOfAttack_ = -eulerAngles.z( );
508 }
