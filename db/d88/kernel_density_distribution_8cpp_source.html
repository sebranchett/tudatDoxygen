22 double EpanechnikovKernelDistribution::evaluatePdf( const double& independentVariable )
23 {
24  if( ( independentVariable - mean_ ) >= ( -bandWidth_ ) && ( independentVariable - mean_ ) <= ( bandWidth_ ) )
25  {
26  return ( 3.0 / ( 4.0 * bandWidth_ ) ) * ( 1.0 - std::pow( ( independentVariable - mean_ ) / bandWidth_, 2.0 ) );
27  }
28  else
29  {
30  return 0.0;

35 double EpanechnikovKernelDistribution::evaluateCdf( const double& independentVariable )
36 {
37  if( ( independentVariable - mean_ ) >= ( -bandWidth_ ) && ( independentVariable - mean_ ) <= ( bandWidth_ ) )
38  {
39  return ( 3.0 / ( 4.0 * bandWidth_ ) ) *
40  ( ( independentVariable - mean_ ) - ( std::pow( ( independentVariable - mean_ ), 3.0 )
41  / ( 3.0 * std::pow( bandWidth_, 2.0 ) ) ) ) + 0.5;
42  }
43  else if( ( independentVariable - mean_ ) < ( -bandWidth_ ) )
44  {
45  return 0.0;
46  }
47  else
48  {
49  return 1.0;
50  }

55 KernelDensityDistribution::KernelDensityDistribution(
56  const std::vector< Eigen::VectorXd >& samples,
57  const double bandWidthFactor,
58  const KernelType kernelType,
59  const Eigen::VectorXd& standardDeviation,
60  const Eigen::VectorXd& manualBandwidth )
61 {
62  // Load data
63  dataSamples_ = samples;
64  dimensions_ = dataSamples_.at( 0 ).rows( );
65  numberOfSamples_ = static_cast< int >( samples.size( ) );
66 
67  // Check input consistency
68  if( ( standardDeviation.rows( ) > 0 ) && ( standardDeviation.rows( ) != dimensions_ ) )
69  {
70  std::string errorMessage = "Error when creating KernelDensityDistribution, manual standard deviation size is inconsistent, should have size : " +
71  std::to_string( dimensions_ ) + "but has size " +
72  std::to_string( standardDeviation.rows( ) );
73  throw std::runtime_error( errorMessage );
74 
75  }
76 
77  if( ( manualBandwidth.rows( ) > 0 ) && ( manualBandwidth.rows( ) != dimensions_ ) )
78  {
79  std::string errorMessage = "Error when creating KernelDensityDistribution, manual bandwidth size is inconsistent, should have size : " +
80  std::to_string( dimensions_ ) + "but has size " +
81  std::to_string( manualBandwidth.rows( ) );
82  throw std::runtime_error( errorMessage );
83  }
84 
85  for( unsigned int i = 0; i < samples.size( ); i++ )
86  {
87  if( dataSamples_.at( i ).rows( ) != dimensions_ )
88  {
89  std::string errorMessage = "Error when creating KernelDensityDistribution, samples size is inconsistent, should have size : " +
90  std::to_string( dimensions_ ) + "but entry " +
91  std::to_string( i ) + " has size " +
92  std::to_string( dataSamples_.at( i ).rows( ) );
93  throw std::runtime_error( errorMessage );
94  }
95  }
96 
97  // Compute datasample properties
98  computeSampleMean( );
99  computeSampleVariance( );
100 
101  // Adjust samples with standard deviation and recompute datasample properties
102  if( standardDeviation.rows( ) > 0 )
103  {
104  scaleSamplesWithVariance( standardDeviation );
105  computeSampleMean( );
106  computeSampleVariance( );
107  }
108 
109 
110  // Compute bandwidth
111  if( manualBandwidth.rows( ) > 0 )
112  {
113  bandWidth_ = manualBandwidth * bandWidthFactor;
114  }
115  else
116  {
117  computeOptimalBandWidth( );
118  bandWidth_ = optimalBandwidth_ * bandWidthFactor;
119  }
120 
121  // Construct kernel matrix rows: samples, cols: dimensions_
122  kernelType_ = kernelType;
123 
124  generateKernelPointerMatrix( );
125 }

128 void KernelDensityDistribution::generateKernelPointerMatrix( )
129 {
130  // Clear existing kernels.
131  kernelPointersMatrix_.clear( );
132 
133  // Check for numerical problems with bandwidths.
134  for( int i = 0; i < bandWidth_.rows( ); i++ )
135  {
136  if( bandWidth_( i ) < 10.0 * std::numeric_limits< double >::epsilon( ) )
137  {
138  std::string errorMessage = "Error in kernel density distribution, index " +
139  std::to_string( i ) + "has bandwidth " +
140  std::to_string( bandWidth_( i ) );
141  throw std::runtime_error( errorMessage );
142  }
143  }
144 
145  // Fill kernel pointer matrix with distribution pointer objects
146  std::vector< std::shared_ptr< ContinuousProbabilityDistribution< double > > > vector( dataSamples_[ 0 ].rows( ) );
147 
148  // Iterate over all samples; create kernel for each entry in each sample
149  for( unsigned int i = 0; i < dataSamples_.size( ); i++ )
150  {
151  for( unsigned int j = 0; j < vector.size( ); j++ )
152  {
153  vector[ j ] = constructKernel( dataSamples_[ i ]( j ), bandWidth_( j ) );
154  }
155  kernelPointersMatrix_.push_back( vector );
156  }

160 void KernelDensityDistribution::computeSampleMean( )
161 {
162  sampleMean_ = Eigen::VectorXd::Zero( dimensions_ );

171 void KernelDensityDistribution::computeSampleVariance( )
172 {
173  sampleVariance_ = Eigen::VectorXd::Zero( dimensions_ );
174  for( unsigned int i = 0; i < dataSamples_.size( ); i++ )
175  {
176  sampleVariance_ += ( dataSamples_[ i ] - sampleMean_ ).cwiseAbs2( );
177  }
178  sampleVariance_ = sampleVariance_ / ( static_cast< double >( numberOfSamples_ ) - 1.0 );
179  sampleStandardDeviation_ = sampleVariance_.cwiseSqrt( );
180 }

183 void KernelDensityDistribution::scaleSamplesWithVariance(
184  const Eigen::VectorXd& standardDeviation )
185 {
186  for( unsigned int i = 0; i < dataSamples_.size( ); i++ )
187  {
188  for( int j = 0; j < dataSamples_[ i ].rows( ); j++ )
189  {
190  dataSamples_[ i ]( j ) = dataSamples_[ i ]( j ) / sampleStandardDeviation_( j ) * standardDeviation( j );
191  }
192  }

196 void KernelDensityDistribution::computeOptimalBandWidth( )
197 {
198  optimalBandwidth_ = Eigen::VectorXd::Zero( dimensions_, 1);
199 
200  // Calculate sigma (median absolute deviation estimator)
201  Eigen::VectorXd medianOfSamples = getMedian( dataSamples_ );
202 
203  std::vector< Eigen::VectorXd > dataSamples2( dataSamples_.size( ) );
204  for( unsigned int i = 0; i < dataSamples_.size( ); i++ )
205  {
206  dataSamples2[ i ] = dataSamples_[ i ] - medianOfSamples;
207  dataSamples2[ i ] = dataSamples2[ i ].cwiseAbs( );
208  }
209 
210  Eigen::VectorXd medianOfSamples2 = getMedian( dataSamples2 );
211  Eigen::VectorXd sigma = medianOfSamples2 / 0.6745;
212 
213  // Calculate optimal Bandwidth
214  optimalBandwidth_ = std::pow( 4.0 / ( ( static_cast< double >( dimensions_ ) + 2.0 ) *
215  static_cast< double >( numberOfSamples_ ) ),
216  1.0 / ( static_cast< double >( dimensions_ ) + 4.0 ) ) * sigma;
217 }

220 Eigen::VectorXd KernelDensityDistribution::getMedian(
221  const std::vector< Eigen::VectorXd >& samples )
222 {
223  int sampleDimensions = samples[ 0 ].rows( );
224  std::vector< std::vector< double > > dataSamplesVectors( sampleDimensions );
225 
226  // Create vector of j^th entry from each sample
227  std::vector< double > data( 0 );
228  for( int i = 0; i < sampleDimensions; i++ )
229  {
230  for( unsigned int j = 0; j < samples.size( ); j++ )
231  {
232  data.push_back( samples[ j ]( i ) );
233  }
234  dataSamplesVectors[ i ] = data;
235  data.clear( );
236  }
237 
238  // Compute sample median for each entry of samples.
239  Eigen::VectorXd medianOfSamples( sampleDimensions );
240  for( int i = 0; i < sampleDimensions; i++ )
241  {
242  medianOfSamples[ i ] = tudat::statistics::computeSampleMedian( dataSamplesVectors[ i ] );
243  }
244  return medianOfSamples;
245 }

248 double KernelDensityDistribution::evaluatePdf( const Eigen::VectorXd& independentVariables )
249 {
250  double propbabilityDensity = 0.0;
251  double currentKernelPdf = 1.0;
252 
253  // Iterative over all kernels
254  for( int i = 0; i < numberOfSamples_; i++ )
255  {
256  currentKernelPdf = 1.0;
257 
258  // Compute pdf of current kernel
259  for( int currentDimension = 0; currentDimension < dimensions_; currentDimension++ )
260  {
261  currentKernelPdf *= kernelPointersMatrix_[ i ][ currentDimension ]->evaluatePdf(
262  independentVariables( currentDimension ) );
263  }
264  propbabilityDensity += currentKernelPdf;
265  }
266 
267  // Average over all kernels
268  return propbabilityDensity / static_cast< double >( numberOfSamples_ );

348 double KernelDensityDistribution::evaluateCumulativeConditionalMarginalProbability(
349  const std::vector< int >& conditionDimensions,
350  const std::vector< double >& conditions,
351  const int marginalDimension, const double independentVariable )
352 {
353  if( std::find( conditionDimensions.begin( ), conditionDimensions.end( ), marginalDimension ) !=
354  conditionDimensions.end( ) )
355  {
356  throw std::runtime_error( "Error when evaluatiing cumulative conditional marginal kernel density probability, repeated indices found" );
357  }
358 
359  double marginalConditionalCdfOfCurrentKernel = 1.0;
360  double normalizationFactor = 0.0;
361  double marginalValue = 0.0;
362 
363  // Iterate over all kernels
364  for( int i = 0; i < numberOfSamples_; i++ )
365  {
366  marginalConditionalCdfOfCurrentKernel = 1.0;
367 
368  for( unsigned int j = 0; j < conditionDimensions.size( ); j++ )
369  {
370  marginalConditionalCdfOfCurrentKernel *= kernelPointersMatrix_[ i ][ conditionDimensions[ j ] ]->evaluateCdf( conditions[ j ] );
371  }
372 
373  // Current value of marginalConditionalCdfOfCurrentKernel is the marginal cdf at the given conditional
374  normalizationFactor += marginalConditionalCdfOfCurrentKernel;
375 
376  // Compute cdf for current kernel at marginal dimension
377  marginalConditionalCdfOfCurrentKernel *= kernelPointersMatrix_[ i ][ marginalDimension ]->evaluateCdf( independentVariable );
378  marginalValue += marginalConditionalCdfOfCurrentKernel;
379  }
380 
381  return marginalValue / normalizationFactor;
382 }
