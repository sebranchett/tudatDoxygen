222 template< typename StateScalarType, typename TimeType >
223 std::shared_ptr< gravitation::SphericalHarmonicsGravitationalAccelerationModel >
224 getSphericalHarmonicAccelerationForDependentVariables(
225  const std::shared_ptr< SingleDependentVariableSaveSettings > dependentVariableSettings,
226  const std::unordered_map< IntegratedStateType,
227  std::vector< std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > > >& stateDerivativeModels )
228 {
229  std::shared_ptr< gravitation::SphericalHarmonicsGravitationalAccelerationModel > sphericalHarmonicAcceleration;
230 
231  // Retrieve list of suitable acceleration models (size should be one to avoid ambiguities)s
232  std::vector< std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > >
233  listOfSuitableAccelerationModels = getAccelerationBetweenBodies(
234  dependentVariableSettings->associatedBody_,
235  dependentVariableSettings->secondaryBody_, stateDerivativeModels,
236  basic_astrodynamics::spherical_harmonic_gravity );
237 
238 
239  // Check if third-body counterpart of acceleration is found
240  if( listOfSuitableAccelerationModels.size( ) == 0 )
241  {
242  listOfSuitableAccelerationModels = getAccelerationBetweenBodies(
243  dependentVariableSettings->associatedBody_,
244  dependentVariableSettings->secondaryBody_,
245  stateDerivativeModels, basic_astrodynamics::getAssociatedThirdBodyAcceleration(
246  basic_astrodynamics::spherical_harmonic_gravity ) );
247  }
248 
249  if( listOfSuitableAccelerationModels.size( ) != 1 )
250  {
251  std::string errorMessage = "Error when getting spherical harmonic acceleration components between bodies " +
252  dependentVariableSettings->associatedBody_ + " and " +
253  dependentVariableSettings->secondaryBody_ + " of type " +
254  std::to_string(
255  basic_astrodynamics::spherical_harmonic_gravity ) +
256  ", no such acceleration found";
257  throw std::runtime_error( errorMessage );
258  }
259  else
260  {
261  sphericalHarmonicAcceleration =
262  std::dynamic_pointer_cast< gravitation::SphericalHarmonicsGravitationalAccelerationModel >(
263  listOfSuitableAccelerationModels.at( 0 ) );
264  if( sphericalHarmonicAcceleration == nullptr )
265  {
266  std::string errorMessage = "Error when getting spherical harmonic acceleration for dependent variable " +
267  dependentVariableSettings->associatedBody_ + " and " +
268  dependentVariableSettings->secondaryBody_ + " type is ionconsistent";
269  throw std::runtime_error( errorMessage );
270  }
271  }
272 
273  return sphericalHarmonicAcceleration;
274 }
