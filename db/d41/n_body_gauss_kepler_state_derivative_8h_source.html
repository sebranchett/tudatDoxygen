168  Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > convertFromOutputSolution(
169  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic >& cartesianSolution,
170  const TimeType& time )
171  {
172  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > currentState =
173  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( cartesianSolution.rows( ) );
174 
175  // Subtract frame origin and Keplerian states from inertial state.
176  Eigen::Matrix< StateScalarType, 6, 1 > currentCartesianState;
177  Eigen::Matrix< StateScalarType, 6, 1 > currentKeplerianState;
178 
179  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
180  {
181  currentCartesianState = cartesianSolution.block( i * 6, 0, 6, 1 );
182  currentKeplerianState = orbital_element_conversions::convertCartesianToKeplerianElements (
183  currentCartesianState, static_cast< StateScalarType >(
184  centralBodyGravitationalParameters_.at( i )( ) ) );
185  StateScalarType eccentricAnomaly = orbital_element_conversions::convertTrueAnomalyToEccentricAnomaly(
186  currentKeplerianState( 5 ), currentKeplerianState( 1 ) );
187  StateScalarType meanAnomaly = orbital_element_conversions::convertEccentricAnomalyToMeanAnomaly(
188  eccentricAnomaly, currentKeplerianState( 1 ) );
189  currentKeplerianState( 5 ) = meanAnomaly;
190  currentState.segment( i * 6, 6 ) = currentKeplerianState;
191  }
192 
193  return currentState;
194 
195  }
