90 Eigen::Vector6d calculateDoodsonFundamentalArguments(
91  const double tdbTime, const double terrestrialTime, const double universalTime1 )
92 {
93  using mathematical_constants::PI;
94 
95  // Calculate current centuries since J2000.
96  double julianCenturiesSinceJ2000 = basic_astrodynamics::convertSecondsSinceEpochToJulianCenturiesSinceEpoch( tdbTime );
97 
98  // Calculate GMST
99  double greenwichMeanSiderealTime = iauGmst06(
100  basic_astrodynamics::JULIAN_DAY_ON_J2000, universalTime1 / physical_constants::JULIAN_DAY,
101  basic_astrodynamics::JULIAN_DAY_ON_J2000, terrestrialTime / physical_constants::JULIAN_DAY );
102 
103  // Calculate relevant angles.
104  double meanElongationOfMoonFromSun = calculateMeanElongationOfMoonFromSun( julianCenturiesSinceJ2000 );
105  double meanArgumentOfLatitudeOfMoon = calculateMeanArgumentOfLatitudeOfMoon( julianCenturiesSinceJ2000 );
106  double meanAnomalyOfMoon = calculateMeanAnomalyOfMoon( julianCenturiesSinceJ2000 );
107  double meanAnomalyOfSun = calculateMeanAnomalyOfSun( julianCenturiesSinceJ2000 );
108  double longitudeOfMoonsAscendingNode = calculateLongitudeOfMoonsAscendingNode( julianCenturiesSinceJ2000 );
109 
110  // Calculate Doodson arguments.
111  Eigen::Vector6d doodsonArguments;
112  doodsonArguments( 1 ) = meanArgumentOfLatitudeOfMoon + longitudeOfMoonsAscendingNode;
113  doodsonArguments( 0 ) = greenwichMeanSiderealTime + PI - doodsonArguments( 1 );
114  doodsonArguments( 2 ) = doodsonArguments( 1 ) - meanElongationOfMoonFromSun;
115  doodsonArguments( 3 ) = doodsonArguments( 1 ) - meanAnomalyOfMoon;
116  doodsonArguments( 5 ) = doodsonArguments( 2 ) - meanAnomalyOfSun;
117  doodsonArguments( 4 ) = -longitudeOfMoonsAscendingNode;
118 
119  return doodsonArguments;
120 }
