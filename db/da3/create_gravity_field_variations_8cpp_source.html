27 std::shared_ptr< gravitation::GravityFieldVariationsSet > createGravityFieldModelVariationsSet(
28  const std::string& body,
29  const NamedBodyMap& bodyMap,
30  const std::vector< std::shared_ptr< GravityFieldVariationSettings > >&
31  gravityFieldVariationSettings )
32 {
33 
34  using namespace tudat::gravitation;
35 
36  // Declare lists for input to GravityFieldVariationsSet
37  std::vector< std::shared_ptr< GravityFieldVariations > > variationObjects;
38  std::vector< BodyDeformationTypes > variationTypes;
39  std::vector< std::string > variationIdentifiers;
40  std::map< int, std::shared_ptr< interpolators::InterpolatorSettings > > createInterpolators;
41  std::map< int, double > initialTimes;
42  std::map< int, double > finalTimes;
43  std::map< int, double > timeSteps;
44 
45  // Iterate over all variations to create.
46  for( unsigned int i = 0; i < gravityFieldVariationSettings.size( ); i++ )
47  {
48  // Get current type of deformation
49  variationTypes.push_back( gravityFieldVariationSettings.at( i )->getBodyDeformationType( ) );
50 
51  // Set current variation object in list.
52  variationObjects.push_back( createGravityFieldVariationsModel(
53  gravityFieldVariationSettings.at( i ), body, bodyMap ) );
54 
55  if( gravityFieldVariationSettings.at( i )->getBodyDeformationType( ) == basic_solid_body )
56  {
57  std::vector< std::string > deformingBodies =
58  std::dynamic_pointer_cast< BasicSolidBodyGravityFieldVariationSettings >(
59  gravityFieldVariationSettings.at( i ) )->getDeformingBodies( );
60  std::string currentIdentifier = deformingBodies.at( 0 );
61  for( unsigned int j = 1; j < deformingBodies.size( ); j++ )
62  {
63  currentIdentifier += "_" + deformingBodies.at( j );
64  }
65  variationIdentifiers.push_back( currentIdentifier );
66  }
67  else
68  {
69  variationIdentifiers.push_back( "" );
70  }
71 
72  // Check if current variation is interpolated, and set settings if necessary.
73  if( gravityFieldVariationSettings.at( i )->getInterpolatorSettings( ) != nullptr )
74  {
75  createInterpolators[ i ]
76  = gravityFieldVariationSettings.at( i )->getInterpolatorSettings( )
77  ->interpolatorSettings_;
78  initialTimes[ i ]
79  = gravityFieldVariationSettings.at( i )->getInterpolatorSettings( )
80  ->initialTime_;
81  finalTimes[ i ]
82  = gravityFieldVariationSettings.at( i )->getInterpolatorSettings( )->finalTime_;
83  timeSteps[ i ]
84  = gravityFieldVariationSettings.at( i )->getInterpolatorSettings( )->timeStep_;
85  }
86  }
87 
88  // Create object with settings for updating variations from new parameter values.
89  std::shared_ptr< GravityFieldVariationsSet > fieldVariationsSet =
90  std::make_shared< GravityFieldVariationsSet >(
91  variationObjects, variationTypes, variationIdentifiers,
92  createInterpolators, initialTimes, finalTimes, timeSteps );
93 
94  if( std::dynamic_pointer_cast< TimeDependentSphericalHarmonicsGravityField >(
95  bodyMap.at( body )->getGravityFieldModel( ) ) == nullptr )
96  {
97  throw std::runtime_error( "Error when making gravity field variations of body " + body +
98  ", base type is not time dependent" );
99  }
100 
101  std::dynamic_pointer_cast< TimeDependentSphericalHarmonicsGravityField >(
102  bodyMap.at( body )->getGravityFieldModel( ) )->setFieldVariationSettings( fieldVariationsSet, false );
103 
104  return fieldVariationsSet;
105 }

108 std::shared_ptr< gravitation::GravityFieldVariations > createGravityFieldVariationsModel(
109  const std::shared_ptr< GravityFieldVariationSettings > gravityFieldVariationSettings,
110  const std::string body,
111  const NamedBodyMap& bodyMap )
112 {
113  using namespace tudat::gravitation;
114 
115  std::shared_ptr< GravityFieldVariations > gravityFieldVariationModel;
116 
117  // Check type of variation
118  switch( gravityFieldVariationSettings->getBodyDeformationType( ) )
119  {
120  case basic_solid_body:
121  {
122  // Check consistency
123  std::shared_ptr< BasicSolidBodyGravityFieldVariationSettings >
124  basicSolidBodyGravityVariationSettings =
125  std::dynamic_pointer_cast< BasicSolidBodyGravityFieldVariationSettings >(
126  gravityFieldVariationSettings );
127  if( basicSolidBodyGravityVariationSettings == nullptr )
128  {
129  throw std::runtime_error( "Error, expected basic solid body gravity field settings for " + body );
130  }
131  else
132  {
133  // Define list of required input.
134  std::vector< std::string > deformingBodies
135  = basicSolidBodyGravityVariationSettings->getDeformingBodies( );
136  std::function< Eigen::Vector6d( const double ) > deformedBodyStateFunction;
137  std::function< Eigen::Quaterniond( const double ) > deformedBodyOrientationFunction;
138  std::vector< std::function< Eigen::Vector6d( const double ) > >
139  deformingBodyStateFunctions;
140  std::vector< std::function< double( ) > > gravitionalParametersOfDeformingBodies;
141 
142  // Iterate over all bodies causing tidal perturbation.
143  for( unsigned int i = 0; i < deformingBodies.size( ); i++ )
144  {
145  // Check if perturbing body exists.
146  if( bodyMap.count( deformingBodies[ i ] ) == 0 )
147  {
148  throw std::runtime_error( "Error when making basic solid body gravity field variation, " +
149  deformingBodies[ i ] + " deforming body not found." );
150  }
151 
152  // Create body state functions (depending on whether the variation is calculated
153  // directly during propagation, or a priori by an interpolator
154  if( gravityFieldVariationSettings->getInterpolatorSettings( ) != nullptr )
155  {
156  deformingBodyStateFunctions.push_back(
157  std::bind(
158  &Body::getStateInBaseFrameFromEphemeris< double, double >,
159  bodyMap.at( deformingBodies[ i ] ), std::placeholders::_1 ) );
160  }
161  else
162  {
163  deformingBodyStateFunctions.push_back(
164  std::bind( &Body::getState, bodyMap.at( deformingBodies[ i ] ) ) );
165  }
166 
167  // Get gravitational parameter of perturbing bodies.
168  if( bodyMap.at( deformingBodies[ i ] )->getGravityFieldModel( ) == nullptr )
169  {
170  throw std::runtime_error(
171  "Error, could not find gravity field model in body " + deformingBodies[ i ] +
172  " when making basic sh variation for body " + body );
173  }
174  else
175  {
176  gravitionalParametersOfDeformingBodies.push_back(
177  std::bind( &GravityFieldModel::getGravitationalParameter,
178  bodyMap.at( deformingBodies[ i ] )
179  ->getGravityFieldModel( ) ) );
180  }
181  }
182 
183  // Set state and orientation functions of perturbed body.
184  if( gravityFieldVariationSettings->getInterpolatorSettings( ) != nullptr )
185  {
186  deformedBodyStateFunction = std::bind( &Body::getStateInBaseFrameFromEphemeris< double, double >,
187  bodyMap.at( body ), std::placeholders::_1 );
188  deformedBodyOrientationFunction = std::bind(
189  &ephemerides::RotationalEphemeris::getRotationToTargetFrame,
190  bodyMap.at( body )->getRotationalEphemeris( ), std::placeholders::_1 );
191  }
192  else
193  {
194  deformedBodyStateFunction = std::bind( &Body::getState, bodyMap.at( body ) );
195  deformedBodyOrientationFunction = std::bind( &Body::getCurrentRotationToLocalFrame,
196  bodyMap.at( body ) );
197 
198 
199  }
200 
201  std::function< double( ) > gravitionalParameterOfDeformedBody =
202  std::bind( &GravityFieldModel::getGravitationalParameter,
203  bodyMap.at( body )->getGravityFieldModel( ) );
204 
205  // Create basic tidal variation object.
206  gravityFieldVariationModel
207  = std::make_shared< BasicSolidBodyTideGravityFieldVariations >(
208  deformedBodyStateFunction,
209  deformedBodyOrientationFunction,
210  deformingBodyStateFunctions,
211  basicSolidBodyGravityVariationSettings->getBodyReferenceRadius( ),
212  gravitionalParameterOfDeformedBody,
213  gravitionalParametersOfDeformingBodies,
214  basicSolidBodyGravityVariationSettings->getLoveNumbers( ),
215  deformingBodies );
216  }
217  break;
218  }
219  case tabulated_variation:
220  {
221  // Check input consistency
222  std::shared_ptr< TabulatedGravityFieldVariationSettings > tabulatedGravityFieldVariationSettings
223  = std::dynamic_pointer_cast< TabulatedGravityFieldVariationSettings >(
224  gravityFieldVariationSettings );
225  if( tabulatedGravityFieldVariationSettings == nullptr )
226  {
227  throw std::runtime_error( "Error, expected tabulated gravity field variation settings for " + body );
228  }
229  else
230  {
231  // Create variation.
232  gravityFieldVariationModel = std::make_shared< TabulatedGravityFieldVariations >
233  ( tabulatedGravityFieldVariationSettings->getCosineCoefficientCorrections( ),
234  tabulatedGravityFieldVariationSettings->getSineCoefficientCorrections( ),
235  tabulatedGravityFieldVariationSettings->getMinimumDegree( ),
236  tabulatedGravityFieldVariationSettings->getMinimumOrder( ),
237  tabulatedGravityFieldVariationSettings->getInterpolatorSettings( )->interpolatorSettings_ );
238  }
239  break;
240  }
241  default:
242  {
243  throw std::runtime_error( "Error, case " + std::to_string(
244  gravityFieldVariationSettings->getBodyDeformationType( ) ) +
245  " not implemented for gravity field variations." );
246  }
247 
248  }
249 
250  if( gravityFieldVariationModel == nullptr )
251  {
252  throw std::runtime_error( "Gravity variation model IS nullptr after creation." );
253  }
254 
255  return gravityFieldVariationModel;
256 
257 }
