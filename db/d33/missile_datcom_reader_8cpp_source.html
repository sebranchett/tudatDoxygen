36 void MissileDatcomReader::readFor004( const std::string& fileNameAndPath )
37 {
38  // Set path and open test text file, skip lines with "CONTENTS".
39  openFile( fileNameAndPath );
40  readAndStoreData( "CONTENTS" );
41 
42  // Retreive data from the textFileReader
43  missileDatcomDataContainer_ = containerOfDataFromFile_;
44 
45  // Loop through all the strings stored in the container.
46  for ( iteratorContainerOfData_ = missileDatcomDataContainer_.begin( ); iteratorContainerOfData_
47  != missileDatcomDataContainer_.end( ); iteratorContainerOfData_++ )
48  {
49  // Empty the dataVector such that it can be reused.
50  dataVector_.clear( );
51 
52  // Split the long string with muliple entries into a vector of small strings.
53  split( iteratorContainerOfData_->second,' ' ,dataVector_ );
54 
55  // Convert Strings to doubles and place in the result vector.
56  for ( unsigned int iteratorInt_ = 0; iteratorInt_ < dataVector_.size( ); iteratorInt_++ )
57  {
58  missileDatcomData_.push_back( stringToDouble( dataVector_[ iteratorInt_ ] ) );
59  }
60  }
61 
62  dataFile_.close( );
63 }

80 void MissileDatcomReader::split( const std::string& dataString, char separator,
81  std::vector< std::string >& dataVector )
82 {
83  std::string::size_type i = 0;
84  std::string::size_type j = dataString.find( separator );
85 
86  // Loop over the whole string and place characters into new string until the separator is
87  // found. Then start with a new string at the next entry in the vector. Until the end of the
88  // original string is reached.
89  while ( j != std::string::npos )
90  {
91  // to prevent placements of empty entries if multiple seperators after eachother are placed.
92  if ( !dataString.substr( i, j-i ).empty() )
93  {
94  dataVector.push_back( dataString.substr( i, j-i ) );
95  }
96 
97  i = ++j;
98  j = dataString.find( separator, j );
99 
100  if ( j == std::string::npos )
101  {
102  dataVector.push_back( dataString.substr(i, dataString.length( ) ) );
103  }
104  }
105 }

108 void MissileDatcomReader::readAndStoreData( const std::string& skipKeyword )
109 {
110  // Local variable for reading a single line
111  std::string stringOfData;
112 
113  int lineCounter = 1;
114 
115  // Whilst the end of the data file has not been reached, continue reading
116  // from lines from data file.
117  while( !dataFile_.eof( ) )
118  {
119  // Get next line of data from data file and store in a string.
120  getline( dataFile_, stringOfData );
121 
122  // Check if string doesn't start with set starting character, if string
123  // is not empty, and if the skip keyword is not in the string.
124  if ( ( ( !skipKeyword.empty( ) && stringOfData.find( skipKeyword ) == std::string::npos )
125  || ( skipKeyword.empty( ) ) ) && !stringOfData.empty( ) )
126  {
127  // Store string in container.
128  containerOfDataFromFile_[ lineCounter ] = stringOfData;
129  }
130 
131  // Increment line counter.
132  lineCounter++;
133  }
134 }
