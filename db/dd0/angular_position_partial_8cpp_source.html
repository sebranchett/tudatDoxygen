20 Eigen::Matrix< double, 1, 3 > calculatePartialOfRightAscensionWrtLinkEndPosition(
21  const Eigen::Vector3d& relativeRangeVector,
22  const bool isLinkEndReceiver )
23 {
24  // Define multiplier of patial vector
25  double partialMultiplier = ( ( isLinkEndReceiver ) ? 1.0 : -1.0 );
26 
27  // Set partial vector
28  Eigen::Matrix< double, 1, 3 > partial = Eigen::Matrix< double, 1, 3 >::Zero( );
29  partial( 0 ) = -relativeRangeVector( 1 );
30  partial( 1 ) = relativeRangeVector( 0 );
31  partial /= ( partialMultiplier * ( relativeRangeVector( 0 ) * relativeRangeVector( 0 ) +
32  relativeRangeVector( 1 ) * relativeRangeVector( 1 ) ) );
33  return partial;
34 }

37 Eigen::Matrix< double, 1, 3 > calculatePartialOfDeclinationWrtLinkEndPosition(
38  Eigen::Vector3d relativeRangeVector,
39  const bool isLinkEndReceiver )
40 {
41  // Define multiplier of patial vector
42  double partialMultiplier = ( ( isLinkEndReceiver ) ? 1.0 : -1.0 );
43 
44  // Set partial vector
45  double range = relativeRangeVector.norm( );
46  Eigen::Matrix< double, 1, 3 > partial = partialMultiplier * relativeRangeVector.transpose( ) / range;
47 
48  partial *= relativeRangeVector( 2 ) / ( range * range );
49  partial += -partialMultiplier * ( Eigen::Vector3d::UnitZ( ) ).transpose( ) / range;
50 
51  return partial;
52 }

69 void AngularPositionScaling::update( const std::vector< Eigen::Vector6d >& linkEndStates,
70  const std::vector< double >& times,
71  const observation_models::LinkEndType fixedLinkEnd,
72  const Eigen::VectorXd currentObservation )
73 {
74  Eigen::Vector3d relativeRangeVector = ( linkEndStates[ 1 ] - linkEndStates[ 0 ] ).segment( 0, 3 );
75  Eigen::Vector3d normalizedRelativeRangeVector = relativeRangeVector.normalized( );
76 
77  // Compute common scaling factor
78  scalingFactor_ = calculatePartialOfAngularPositionWrtLinkEndPosition( relativeRangeVector, true );
79 
80  // Compute scaling for receiver reference
81  if( fixedLinkEnd == observation_models::receiver )
82  {
83  referenceLightTimeCorrectionScaling_ = scalingFactor_ * linkEndStates[ 0 ].segment( 3, 3 ) /
84  ( physical_constants::SPEED_OF_LIGHT - linkEndStates[ 0 ].segment( 3, 3 ).dot( normalizedRelativeRangeVector ) );
85  referenceScalingFactor_ =
86  scalingFactor_ *
87  ( Eigen::Matrix3d::Identity( ) + linkEndStates[ 0 ].segment( 3, 3 ) * normalizedRelativeRangeVector.transpose( ) /
88  ( physical_constants::SPEED_OF_LIGHT - linkEndStates[ 0 ].segment( 3, 3 ).dot( normalizedRelativeRangeVector ) ) );
89  }
90  // Compute scaling for transmitter reference
91  else if( fixedLinkEnd == observation_models::transmitter )
92  {
93  referenceLightTimeCorrectionScaling_ = scalingFactor_ * linkEndStates[ 1 ].segment( 3, 3 ) /
94  ( physical_constants::SPEED_OF_LIGHT - linkEndStates[ 1 ].segment( 3, 3 ).dot( normalizedRelativeRangeVector ) );
95  referenceScalingFactor_ =
96  scalingFactor_ *
97  ( Eigen::Matrix3d::Identity( ) + linkEndStates[ 1 ].segment( 3, 3 ) * normalizedRelativeRangeVector.transpose( ) /
98  ( physical_constants::SPEED_OF_LIGHT - linkEndStates[ 1 ].segment( 3, 3 ).dot( normalizedRelativeRangeVector ) ) );
99  }

101  currentLinkEndType_ = fixedLinkEnd;

106 AngularPositionPartial::AngularPositionPartialReturnType AngularPositionPartial::calculatePartial(
107  const std::vector< Eigen::Vector6d >& states,
108  const std::vector< double >& times,
109  const observation_models::LinkEndType linkEndOfFixedTime,
110  const Eigen::Vector2d& currentObservation )
111 {
112  if( linkEndOfFixedTime != angularPositionScaler_->getCurrentLinkEndType( ) )
113  {
114  throw std::runtime_error( "Error angular position partial and scaling are inconsistent" );
115  }
116 
117  AngularPositionPartialReturnType returnPartial;
118 
119  // Iterate over all link ends
120  for( positionPartialIterator_ = positionPartialList_.begin( ); positionPartialIterator_ != positionPartialList_.end( );
121  positionPartialIterator_++ )
122  {
123  if( positionPartialIterator_->first == observation_models::transmitter )
124  {
125  currentState_ = states[ 0 ];
126  currentTime_ = times[ 0 ];
127  }
128  else if( positionPartialIterator_->first == observation_models::receiver )
129  {
130  currentState_ = states[ 1 ];
131  currentTime_ = times[ 1 ];
132  }
133 
134  // Scale position partials
135  returnPartial.push_back(
136  std::make_pair(
137  angularPositionScaler_->getScalingFactor( positionPartialIterator_->first ) *
138  ( positionPartialIterator_->second->calculatePartialOfPosition(
139  currentState_ , currentTime_ ) ), currentTime_ ) );
140  }
141 
142  // Add scaled light-time correcion partials.
143  for( unsigned int i = 0; i < lighTimeCorrectionPartialsFunctions_.size( ); i++ )
144  {
145  currentLinkTimeCorrectionPartial_ = lighTimeCorrectionPartialsFunctions_.at( i )( states, times );
146  returnPartial.push_back(
147  std::make_pair( angularPositionScaler_->getLightTimePartialScalingFactor( ) *
148  physical_constants::SPEED_OF_LIGHT * currentLinkTimeCorrectionPartial_.first,
149  currentLinkTimeCorrectionPartial_.second ) );
150  }
151 
152 
153  return returnPartial;
154 }
