74 template< typename StateScalarType >
75 Eigen::Matrix< StateScalarType, 6, 1 > getCartesianState(
76  const nlohmann::json& jsonObject,
77  const KeyPath& keyPath = KeyPath( ),
78  const std::shared_ptr< simulation_setup::Body >& centralBody = nullptr,
79  double epoch = TUDAT_NAN )
80 {
81  using namespace basic_astrodynamics;
82  using namespace orbital_element_conversions;
83  using K = Keys::Body::State;
84 
85  const nlohmann::json jsonState = getValue< nlohmann::json >( jsonObject, keyPath );
86  Eigen::Matrix< StateScalarType, 6, 1 > bodyState = Eigen::Matrix< StateScalarType, 6, 1 >::Zero( );
87 
88  // Instead of a vector, an object can be used to provide initial translational state
89  if ( jsonState.is_object( ) )
90  {
91  const double centralBodyAverageRadius = ! isDefined( jsonState, K::centralBodyAverageRadius ) && centralBody ?
92  centralBody->getShapeModel( )->getAverageRadius( ) :
93  getValue< double >( jsonState, K::centralBodyAverageRadius, TUDAT_NAN );
94  bool usedCentralBodyAverageRadius = false;
95 
96  const StateType stateType = getValue< StateType >( jsonState, K::type );
97  switch ( stateType ) {
98  case cartesianElements:
99  {
100  updateFromJSONIfDefined( bodyState( xCartesianPositionIndex ), jsonState, K::x );
101  updateFromJSONIfDefined( bodyState( yCartesianPositionIndex ), jsonState, K::y );
102  updateFromJSONIfDefined( bodyState( zCartesianPositionIndex ), jsonState, K::z );
103  updateFromJSONIfDefined( bodyState( xCartesianVelocityIndex ), jsonState, K::vx );
104  updateFromJSONIfDefined( bodyState( yCartesianVelocityIndex ), jsonState, K::vy );
105  updateFromJSONIfDefined( bodyState( zCartesianVelocityIndex ), jsonState, K::vz );
106  break;
107  }
108  case keplerianElements:
109  {
110  const double centralBodyGravitationalParameter =
111  ! isDefined( jsonState, K::centralBodyGravitationalParameter ) && centralBody ?
112  centralBody->getGravityFieldModel( )->getGravitationalParameter( ) :
113  getValue< double >( jsonState, K::centralBodyGravitationalParameter );
114 
115  // Detemrine semiMajorAxis, eccentricity, argumentOfPeriapsis
116 
117  StateScalarType semiMajorAxis = TUDAT_NAN;
118  StateScalarType eccentricity = TUDAT_NAN;
119  StateScalarType argumentOfPeriapsis = TUDAT_NAN;
120 
121  if ( isDefined( jsonState, K::radius ) || isDefined( jsonState, K::altitude ) ) // circular
122  {
123  if ( isDefined( jsonState, K::altitude ) )
124  {
125  semiMajorAxis = centralBodyAverageRadius + getValue< StateScalarType >( jsonState, K::altitude );
126  usedCentralBodyAverageRadius = true;
127  }
128  else
129  {
130  semiMajorAxis = getValue< StateScalarType >( jsonState, K::radius );
131  }
132  eccentricity = 0.0;
133  argumentOfPeriapsis = 0.0;
134  }
135  else // generic
136  {
137  argumentOfPeriapsis = getValue< StateScalarType >( jsonState, K::argumentOfPeriapsis, 0.0 );
138 
139  if ( isDefined( jsonState, K::apoapsisDistance ) || isDefined( jsonState, K::apoapsisAltitude ) ||
140  isDefined( jsonState, K::periapsisDistance ) || isDefined( jsonState, K::periapsisAltitude ) )
141  {
142  StateScalarType apoapsisDistance = TUDAT_NAN;
143  if ( isDefined( jsonState, K::apoapsisAltitude ) )
144  {
145  apoapsisDistance = centralBodyAverageRadius +
146  getValue< StateScalarType >( jsonState, K::apoapsisAltitude );
147  usedCentralBodyAverageRadius = true;
148  }
149  else if ( isDefined( jsonState, K::apoapsisDistance ) )
150  {
151  apoapsisDistance = getValue< StateScalarType >( jsonState, K::apoapsisDistance );
152  }
153 
154  StateScalarType periapsisDistance = TUDAT_NAN;
155  if ( isDefined( jsonState, K::periapsisAltitude ) )
156  {
157  periapsisDistance = centralBodyAverageRadius +
158  getValue< StateScalarType >( jsonState, K::periapsisAltitude );
159  usedCentralBodyAverageRadius = true;
160  }
161  else if ( isDefined( jsonState, K::periapsisDistance ) )
162  {
163  periapsisDistance = getValue< StateScalarType >( jsonState, K::periapsisDistance );
164  }
165 
166  if ( ! isNaN( apoapsisDistance) && ! isNaN( periapsisDistance ) )
167  {
168  // r_a, r_p -> a, e
169  semiMajorAxis = ( apoapsisDistance + periapsisDistance ) / 2.0;
170  eccentricity = ( apoapsisDistance - periapsisDistance ) / ( 2.0 * semiMajorAxis );
171  }
172  else if ( ! isNaN( apoapsisDistance) )
173  {
174  if ( isDefined( jsonState, K::semiMajorAxis ) )
175  {
176  semiMajorAxis = getValue< StateScalarType >( jsonState, K::semiMajorAxis );
177  // r_a, a -> e
178  eccentricity = apoapsisDistance / semiMajorAxis - 1.0;
179  }
180  else
181  {
182  eccentricity = getValue< StateScalarType >( jsonState, K::eccentricity );
183  // r_a, e -> a
184  semiMajorAxis = apoapsisDistance / ( 1.0 + eccentricity );
185  }
186  }
187  else
188  {
189  if ( isDefined( jsonState, K::semiMajorAxis ) )
190  {
191  semiMajorAxis = getValue< StateScalarType >( jsonState, K::semiMajorAxis );
192  // r_p, a -> e
193  eccentricity = 1.0 - periapsisDistance / semiMajorAxis;
194  }
195  else
196  {
197  eccentricity = getValue< StateScalarType >( jsonState, K::eccentricity );
198  // r_p, e -> a
199  semiMajorAxis = periapsisDistance / ( 1.0 - eccentricity );
200  }
201  }
202  }
203  else
204  {
205  eccentricity = getValue< StateScalarType >( jsonState, K::eccentricity, 0.0 );
206 
207  if ( isDefined( jsonState, K::semiLatusRectum ) )
208  {
209  semiMajorAxis = getValue< StateScalarType >( jsonState, K::semiLatusRectum ) /
210  ( 1.0 - std::pow( eccentricity, 2.0 ) );
211  }
212  else if ( isDefined( jsonState, K::meanMotion ) || isDefined( jsonState, K::period ) )
213  {
214  StateScalarType meanMotion;
215  if ( isDefined( jsonState, K::meanMotion ) )
216  {
217  meanMotion = getValue< StateScalarType >( jsonState, K::meanMotion );
218  }
219  else
220  {
221  meanMotion = 2.0 * mathematical_constants::PI /
222  getValue< StateScalarType >( jsonState, K::period );
223  }
224  semiMajorAxis = std::pow( ( centralBodyGravitationalParameter /
225  std::pow( meanMotion, 2.0 ) ), 1.0/3.0 );
226  }
227  else
228  {
229  semiMajorAxis = getValue< StateScalarType >( jsonState, K::semiMajorAxis );
230  }
231  }
232  }
233 
234  // Determine inclination, longitudeOfAscendingNode
235  StateScalarType inclination = getValue< StateScalarType >( jsonState, K::inclination, 0.0 );
236  StateScalarType longitudeOfAscendingNode =
237  getValue< StateScalarType >( jsonState, K::longitudeOfAscendingNode, 0.0 );
238 
239  // Determine trueAnomaly
240  StateScalarType trueAnomaly;
241  if ( isDefined( jsonState, K::meanAnomaly ) || isDefined( jsonState, K::eccentricAnomaly ) )
242  {
243  StateScalarType eccentricAnomaly;
244  if ( isDefined( jsonState, K::meanAnomaly ) )
245  {
246  eccentricAnomaly = convertMeanAnomalyToEccentricAnomaly(
247  eccentricity,
248  getValue< StateScalarType >( jsonState, K::meanAnomaly ) );
249  }
250  else
251  {
252  eccentricAnomaly = getValue< StateScalarType >( jsonState, K::eccentricAnomaly );
253  }
254  trueAnomaly = convertEccentricAnomalyToTrueAnomaly( eccentricAnomaly, eccentricity );
255  }
256  else
257  {
258  trueAnomaly = getValue< StateScalarType >( jsonState, K::trueAnomaly, 0.0 );
259  }
260 
261  // Full state
262  Eigen::Matrix< StateScalarType, 6, 1 > keplerianElements;
263  keplerianElements( semiMajorAxisIndex ) = semiMajorAxis;
264  keplerianElements( eccentricityIndex ) = eccentricity;
265  keplerianElements( inclinationIndex ) = inclination;
266  keplerianElements( argumentOfPeriapsisIndex ) = argumentOfPeriapsis;
267  keplerianElements( longitudeOfAscendingNodeIndex ) = longitudeOfAscendingNode;
268  keplerianElements( trueAnomalyIndex ) = trueAnomaly;
269 
270  // Convert to Cartesian elements
271  bodyState = convertKeplerianToCartesianElements( keplerianElements, centralBodyGravitationalParameter );
272 
273  break;
274  }
275  case sphericalElements:
276  {
277  assertNonnullptrPointer( centralBody );
278 
279  double radius;
280  if ( isDefined( jsonState, K::altitude ) )
281  {
282  radius = centralBodyAverageRadius + getValue< StateScalarType >( jsonState, K::altitude );
283  usedCentralBodyAverageRadius = true;
284  }
285  else
286  {
287  radius = getValue< StateScalarType >( jsonState, K::radius );
288  }
289 
290  Eigen::Matrix< StateScalarType, 6, 1 > sphericalElements;
291  sphericalElements( radiusIndex ) = radius;
292  sphericalElements( latitudeIndex ) = getValue< StateScalarType >( jsonState, K::latitude, 0.0 );
293  sphericalElements( longitudeIndex ) = getValue< StateScalarType >( jsonState, K::longitude, 0.0 );
294  sphericalElements( speedIndex ) = getValue< StateScalarType >( jsonState, K::speed );
295  sphericalElements( flightPathIndex ) = getValue< StateScalarType >( jsonState, K::flightPathAngle, 0.0 );
296  sphericalElements( headingAngleIndex ) = getValue< StateScalarType >( jsonState, K::headingAngle, 0.0 );
297 
298  // Convert to Cartesian elements
299  if ( isDefined( jsonState, K::epoch ) || isNaN( epoch ) )
300  {
301  epoch = getValue< StateScalarType >( jsonState, K::epoch );
302  }
303  bodyState = tudat::ephemerides::transformStateToGlobalFrame(
304  convertSphericalOrbitalToCartesianState( sphericalElements ),
305  epoch, centralBody->getRotationalEphemeris( ) );
306 
307  break;
308  }
309  default:
310  handleUnimplementedEnumValue( stateType, stateTypes, unsupportedStateTypes );
311  }
312 
313  if ( usedCentralBodyAverageRadius )
314  {
315  if ( isNaN( centralBodyAverageRadius ) )
316  {
317  // Print error saying that key centralBodyAverageRadius is missing
318  getValue< double >( jsonState, K::centralBodyAverageRadius );
319  }
320  else if ( centralBody )
321  {
322  if ( ! std::dynamic_pointer_cast< SphericalBodyShapeModel >( centralBody->getShapeModel( ) ) )
323  {
324  std::cerr << "Using average radius of a non-spherical body "
325  "to determine the initial Cartesian state of body." << std::endl;
326  }
327  }
328  }
329  }
330  else // could not get the state as Cartesian, Keplerian components... then try to convert directly
331  {
332  bodyState = getValue< Eigen::Matrix< StateScalarType, 6, 1 > >( jsonObject, keyPath );
333  }
334 
335  return bodyState;
336 }
