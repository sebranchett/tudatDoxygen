47 Eigen::Vector4d convertModifiedRodriguesParametersToQuaternionElements( const Eigen::Vector4d& modifiedRodriguesParameterElements )
48 {
49  // Declare eventual output vector
50  Eigen::Vector4d convertedQuaternionElements;
51 
52  // Extract modified Rodrigues parameters

54  modifiedRodriguesParameterElements.segment( sigma1ModifiedRodriguesParametersIndex, 3 );
55 
56  // Precompute often used variables
57  double modifiedRodriguesParametersMagnitudeSquared = modifiedRodriguesParametersVector.squaredNorm( );
58 
59  // Convert modified Rodrigues parameters to quaternions
60  double conversionSign = ( int( modifiedRodriguesParameterElements( shadowFlagModifiedRodriguesParametersIndex ) ) == 1 ) ?
61  - 1.0 : 1.0; // converion is slightly different for SMRP and MRP
62  convertedQuaternionElements( etaQuaternionIndex ) = conversionSign *
63  ( 1.0 - modifiedRodriguesParametersMagnitudeSquared ) /
64  ( 1.0 + modifiedRodriguesParametersMagnitudeSquared );
65  convertedQuaternionElements.segment( epsilon1QuaternionIndex, 3 ) = conversionSign *
66  2.0 / ( 1.0 + modifiedRodriguesParametersMagnitudeSquared ) *
67  modifiedRodriguesParametersVector;

74 Eigen::Vector4d convertQuaternionsToExponentialMapElements( const Eigen::Vector4d& quaternionElements )
75 {
76  using mathematical_constants::PI;
77 
78  // Define the tolerance of a singularity
79  const double singularityTolerance = 20.0 * std::numeric_limits< double >::epsilon( );
80 
81  // Declare eventual output vector
82  Eigen::Vector4d convertedExponentialMapElements;
83 
84  // Convert quaternions to exponential map (or SEM)
85  double exponentialMapMagnitude = 2.0 * std::acos( quaternionElements( etaQuaternionIndex ) );
86  bool shadowFlag = std::fabs( exponentialMapMagnitude ) > PI;
87  Eigen::Vector3d exponentialMapVector = quaternionElements.segment( epsilon1QuaternionIndex, 3 );
88  if ( std::fabs( exponentialMapMagnitude ) < singularityTolerance )
89  {
90  exponentialMapVector *= 48.0 / ( 24.0 - exponentialMapMagnitude * exponentialMapMagnitude );
91  }
92  else
93  {
94  exponentialMapVector *= shadowFlag ?
95  - ( 2.0 * PI - exponentialMapMagnitude ) / std::sin( 0.5 * exponentialMapMagnitude ) : // shadow exponential map
96  exponentialMapMagnitude / std::sin( 0.5 * exponentialMapMagnitude ); // exponential map
97  }
98  convertedExponentialMapElements.segment( e1ExponentialMapIndex, 3 ) = exponentialMapVector;
99  convertedExponentialMapElements( shadowFlagExponentialMapIndex ) = shadowFlag ? 1.0 : 0.0; // set flag
100 
101  // Give output
102  return convertedExponentialMapElements;
103 }

106 Eigen::Vector4d convertExponentialMapToQuaternionElements( const Eigen::Vector4d& exponentialMapElements )
107 {
108  // Define the tolerance of a singularity
109  const double singularityTolerance = 20.0 * std::numeric_limits< double >::epsilon( );
110 
111  // Declare eventual output vector
112  Eigen::Vector4d convertedQuaternionElements;
113 
114  // Extract exponential map
115  Eigen::Vector3d exponentialMapVector = exponentialMapElements.segment( e1ExponentialMapIndex, 3 );
116  double exponentialMapMagnitude = exponentialMapVector.norm( );
117 
118  // Convert exponential map to quaternions
119  convertedQuaternionElements( etaQuaternionIndex ) = std::cos( 0.5 * exponentialMapMagnitude );
120  if ( std::fabs( exponentialMapMagnitude ) < singularityTolerance )
121  {
122  convertedQuaternionElements.segment( epsilon1QuaternionIndex, 3 ) =
123  exponentialMapVector * ( 0.5 - exponentialMapMagnitude * exponentialMapMagnitude / 48.0 );
124  }
125  else
126  {
127  convertedQuaternionElements.segment( epsilon1QuaternionIndex, 3 ) =
128  exponentialMapVector / exponentialMapMagnitude * std::sin( 0.5 * exponentialMapMagnitude );
129  }
130 
131  // Change sign based on shadow flag
132  double conversionSign = ( int( exponentialMapElements( shadowFlagExponentialMapIndex ) ) == 1 ) ?
133  - 1.0 : 1.0; // converion is slightly different for SEM and EM
134 
135  // Give output
136  return conversionSign * convertedQuaternionElements;
