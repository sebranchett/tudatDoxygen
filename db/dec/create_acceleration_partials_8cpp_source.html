22 std::vector< std::shared_ptr< orbit_determination::TidalLoveNumberPartialInterface > > createTidalLoveNumberInterfaces(
23  const NamedBodyMap& bodyMap,
24  const std::string& acceleratingBodyName )
25 {
26  // Create return map.
27  std::vector< std::shared_ptr< orbit_determination::TidalLoveNumberPartialInterface > > loveNumberInterfaces;
28 
29  // Check if any gravity field variations are present
30  if( bodyMap.at( acceleratingBodyName )->getGravityFieldVariationSet( ) != nullptr )
31  {
32  // Get list of tidal gravity field variations.
33  std::vector< std::shared_ptr< gravitation::BasicSolidBodyTideGravityFieldVariations > > variationObjectList =
34  utilities::dynamicCastSVectorToTVector< gravitation::GravityFieldVariations,
35  gravitation::BasicSolidBodyTideGravityFieldVariations >(
36  bodyMap.at( acceleratingBodyName )->getGravityFieldVariationSet( )->
37  getDirectTidalGravityFieldVariations( ) );
38 
39  // Create partial for each gravity field variation objet
40  if( variationObjectList.size( ) > 0 )
41  {
42  // Get state/rotation functions for deformed body
43  std::function< Eigen::Vector3d( ) > deformedBodyPositionFunction =
44  std::bind( &Body::getPosition, bodyMap.at( acceleratingBodyName ) );
45  std::function< Eigen::Quaterniond( ) > rotationToDeformedBodyFrameFrameFunction =
46  std::bind( &Body::getCurrentRotationToLocalFrame, bodyMap.at( acceleratingBodyName ) );
47 
48  for( unsigned int i = 0; i < variationObjectList.size( ); i++ )
49  {
50  if( variationObjectList.at( i ) != nullptr )
51  {
52  // Get state/rotation functions for deforming bodyies
53  std::vector< std::function< Eigen::Vector3d( ) > > deformingBodyStateFunctions;
54  std::vector< std::string > deformingBodies = variationObjectList.at( i )->getDeformingBodies( );
55  for( unsigned int i = 0; i < deformingBodies.size( ); i++ )
56  {
57  deformingBodyStateFunctions.push_back(
58  std::bind( &Body::getPosition, bodyMap.at( deformingBodies.at( i ) ) ) );
59  }
60  // Get state/rotation functions for deformed body
61 
62  // Create partial object
63  loveNumberInterfaces.push_back(
64  std::make_shared< orbit_determination::TidalLoveNumberPartialInterface >(
65  variationObjectList.at( i ),
66  deformedBodyPositionFunction,
67  deformingBodyStateFunctions,
68  rotationToDeformedBodyFrameFrameFunction,
69  acceleratingBodyName ) );
70  }
71  }
72  }
73  }
74  return loveNumberInterfaces;
75 }
