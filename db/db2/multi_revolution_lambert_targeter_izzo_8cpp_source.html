60 void MultiRevolutionLambertTargeterIzzo::sanityCheckNumberOfRevolutions( )
61 {
62  // If not yet defined, calculate number of revolutions possible.
63  if ( maximumNumberOfRevolutions == NO_MAXIMUM_REVOLUTIONS )
64  {
65  // Temporarily store specified number, as numberOfRevolutions is needed to calculate max
66  // (this is a tricky way to work, but on the other hand this makes this approach decidedly
67  // different from PyKEP routines and it also happens only once per object).
68  int copyOfOriginalNumberOfRevolutions = numberOfRevolutions;
69 
70  // Calculate first guess of maximum, by dividing the time of flight of the minimum energy
71  // ellipse by the normalized time of flight.
72  numberOfRevolutions = static_cast< int >(
73  normalizedTimeOfFlight / (
74  mathematical_constants::PI / 2.0
75  * std::sqrt( 2.0 * normalizedSemiPerimeter
76  * normalizedSemiPerimeter
77  * normalizedSemiPerimeter ) ) );
78 
79  // If the current guess for the maximum is non-zero, then additional analysis is required to
80  // determine the correct maximum.
81  if( numberOfRevolutions != 0)
82  {
83  // The following try-block is meant to check whether the solution converges or not. If
84  // the current guess for the maximum number of revolutions is correct, then the problem
85  // will converge. If it does not, an exception will be thrown stating that it did not
86  // converge. Catching this exception allows to decrease the guess only when the
87  // exception occurs, and not under other circumstances.
88  try
89  {
90  // Compute root (no further information is required)
91  computeRootTimeOfFlight();
92  }
93  catch( basic_mathematics::ConvergenceException )
94  {
95  // If the rootfinder did not converge, then the current guess is wrong and needs to
96  // be decreased
97  numberOfRevolutions--;
98  }
99  }
100  // No further analysis is needed of the current guess is equal to zero.
101 
102  // Maximum is now found.
103  maximumNumberOfRevolutions = numberOfRevolutions;
104 
105  // Reinstating original number of revolutions specified.
106  numberOfRevolutions = copyOfOriginalNumberOfRevolutions;
107  }
108 
109  // Default: compare maximum with specified number of revolutions.
110  // If specified is larger than maximum, no solution is possible.
111  if ( numberOfRevolutions > maximumNumberOfRevolutions )
112  {
113  // Throw exception.
114  throw std::runtime_error( "Number of revolutions specified in Lambert problem is larger than possible. Specified number of revolutions is " +
115  std::to_string( numberOfRevolutions ) + " while the maximum is " +
116  std::to_string( maximumNumberOfRevolutions ) );
117  }
118  // Else, nothing wrong.
119 }

152 double MultiRevolutionLambertTargeterIzzo::computeTimeOfFlight( const double xParameter )
153 {
154  // Determine semi-major axis.
155  const double semiMajorAxis = normalizedMinimumEnergySemiMajorAxis
156  / ( 1.0 - xParameter * xParameter );
157 
158  // If x < 1, the solution is an ellipse.
159  if ( xParameter < 1.0 )
160  {
161  // Alpha parameter in Lagrange's equation (no explanation available).
162  const double alphaParameter = 2.0 * std::acos( xParameter );
163 
164  // Beta parameter in Lagrange's equation (no explanation available).
165  double betaParameter;
166 
167  // If long transfer arc.
168  if ( isLongway )
169  {
170  betaParameter = -2.0 * std::asin(
171  std::sqrt( ( normalizedSemiPerimeter - normalizedChord )
172  / ( 2.0 * semiMajorAxis ) ) );
173  }
174  // Otherwise short transfer arc.
175  else
176  {
177  betaParameter = 2.0 * std::asin(
178  std::sqrt( ( normalizedSemiPerimeter - normalizedChord )
179  / ( 2.0 * semiMajorAxis ) ) );
180  }
181 
182  // Time-of-flight according to Lagrange including multiple revolutions.
183  const double timeOfFlight = semiMajorAxis * std::sqrt( semiMajorAxis ) *
184  ( ( alphaParameter - std::sin( alphaParameter ) )
185  - ( betaParameter - std::sin( betaParameter ) )
186  + 2.0 * mathematical_constants::PI
187  * numberOfRevolutions );
188 
189  return timeOfFlight;
190  }
191  // Otherwise it is a hyperbola.
192  else
193  {
194  // Alpha parameter in Lagrange's equation (no explanation available).
195  const double alphaParameter = 2.0 * boost::math::acosh( xParameter );
196 
197  // Beta parameter in Lagrange's equation (no explanation available).
198  double betaParameter;
199 
200  // If long transfer arc.
201  if ( isLongway )
202  {
203  betaParameter = -2.0 * boost::math::asinh( std::sqrt( ( normalizedSemiPerimeter
204  - normalizedChord )
205  / ( -2.0 * semiMajorAxis ) ) );
206  }
207  // Otherwise short transfer arc
208  else
209  {
210  betaParameter = 2.0 * boost::math::asinh( std::sqrt( ( normalizedSemiPerimeter
211  - normalizedChord )
212  / ( -2.0 * semiMajorAxis ) ) );
213  }
214 
215  // Time-of-flight according to Lagrange.
216  const double timeOfFlightLagrange = -semiMajorAxis * std::sqrt( -semiMajorAxis ) *
217  ( ( std::sinh( alphaParameter ) - alphaParameter )
218  - ( std::sinh( betaParameter ) - betaParameter ) );
219 
220  return timeOfFlightLagrange;
221  }

225 double MultiRevolutionLambertTargeterIzzo::computeRootTimeOfFlight( )
226 {
227  using mathematical_constants::PI;
228 
229  // Define initial guesses for abcissae (x) and ordinates (y).
230  double x1, x2;
231 
232  if ( isRightBranch )
233  { // right branch solution.
234  x1 = std::tan( .7234 * PI / 2.0 );
235  x2 = std::tan( .5234 * PI / 2.0 );
236  }
237  else
238  { // left branch solution.
239  x1 = std::tan( -.5234 * PI / 2.0 );
240  x2 = std::tan( -.2234 * PI / 2.0 );
241  }
242 
243  double y1 = computeTimeOfFlight( std::atan( x1 ) * 2.0 / PI ) - normalizedTimeOfFlight;
244 
245  double y2 = computeTimeOfFlight( std::atan( x2 ) * 2.0 / PI ) - normalizedTimeOfFlight;
246 
247  // Declare and initialize root-finding parameters.
248  double rootFindingError = 1.0, xNew = 0.0, yNew = 0.0;
249  int iterator = 0;
250 
251  // Root-finding loop.
252  while ( ( rootFindingError > convergenceTolerance ) && ( y1 != y2 )
253  && ( iterator < maximumNumberOfIterations ) )
254  {
255  // Update iterator.
256  iterator++;
257 
258  // Compute new x-value.
259  xNew = ( x1 * y2 - y1 * x2 ) / ( y2 - y1 );
260 
261  // Compute corresponding y-value.
262  yNew = computeTimeOfFlight( std::atan( xNew ) * 2.0 / PI ) - normalizedTimeOfFlight;
263 
264  // Update abcissae and ordinates.
265  x1 = x2;
266  y1 = y2;
267  x2 = xNew;
268  y2 = yNew;
269 
270  // Compute root-finding error.
271  rootFindingError = std::fabs( x1 - xNew );
272  }
273 
274  // Verify that root-finder has converged.
275  if ( iterator == maximumNumberOfIterations )
276  {
277  throw basic_mathematics::ConvergenceException(
278  "Multi-Revolution Lambert targeter failed to converge to a solution. Reached the maximum number of iterations: %d"
279  + std::to_string( maximumNumberOfIterations ) );;
280  }
281 
282  // Revert to x parameter.
283  double xParameter = std::atan( xNew ) * 2.0 / PI;
284  return xParameter;
285 }
