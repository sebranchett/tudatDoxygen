276  ObservationScalarType calculateLightTimeWithLinkEndsStates(
277  StateType& receiverStateOutput,
278  StateType& transmitterStateOutput,
279  const TimeType time,
280  const bool isTimeAtReception = 1,
281  const ObservationScalarType tolerance =
282  ( getDefaultLightTimeTolerance< ObservationScalarType >( ) ) )
283  {
284  using physical_constants::SPEED_OF_LIGHT;
285  using std::fabs;
286 
287  // Initialize reception and transmission times and states to initial guess (zero light time)
288  TimeType receptionTime = time;
289  TimeType transmissionTime = time;
290  StateType receiverState = stateFunctionOfReceivingBody_( receptionTime );
291  StateType transmitterState =
292  stateFunctionOfTransmittingBody_( transmissionTime );
293 
294  // Set initial light-time correction.
295  setTotalLightTimeCorrection(
296  transmitterState, receiverState, transmissionTime, receptionTime );
297 
298  // Calculate light-time solution assuming infinte speed of signal as initial estimate.
299  ObservationScalarType previousLightTimeCalculation =
300  calculateNewLightTimeEstime( receiverState, transmitterState );
301 
302  // Set variables for iteration
303  ObservationScalarType newLightTimeCalculation = 0.0;
304  bool isToleranceReached = false;
305 
306  // Recalculate light-time solution until tolerance is reached.
307  int counter = 0;
308 
309  // Set variable determining whether to update the light time each iteration.
310  bool updateLightTimeCorrections = false;
311  if( iterateCorrections_ )
312  {
313  updateLightTimeCorrections = true;
314  }
315 
316  // Iterate until tolerance reached.
317  while( !isToleranceReached )
318  {
319  // Update light-time corrections, if necessary.
320  if( updateLightTimeCorrections )
321  {
322  setTotalLightTimeCorrection(
323  transmitterState, receiverState, transmissionTime, receptionTime );
324  }
325 
326  // Update light-time estimate for this iteration.
327  if( isTimeAtReception )
328  {
329  receptionTime = time;
330  transmissionTime = time - previousLightTimeCalculation;
331  transmitterState = ( stateFunctionOfTransmittingBody_( transmissionTime ) );
332  }
333  else
334  {
335  receptionTime = time + previousLightTimeCalculation;
336  transmissionTime = time;
337  receiverState = ( stateFunctionOfReceivingBody_( receptionTime ) );
338  }
339  newLightTimeCalculation = calculateNewLightTimeEstime( receiverState, transmitterState );
340 
341  // Check for convergence.
342  if( fabs( newLightTimeCalculation - previousLightTimeCalculation ) < tolerance )
343  {
344  // If convergence reached, but light-time corrections not iterated,
345  // perform 1 more iteration to check for change in correction.
346  if( !updateLightTimeCorrections )
347  {
348  updateLightTimeCorrections = true;
349  }
350  else
351  {
352  isToleranceReached = true;
353  }
354  }
355  else
356  {
357  // Get out of infinite loop (for instance due to low accuracy state functions,
358  // to stringent tolerance or limit case for trop. corrections).
359  if( counter == 50 )
360  {
361  isToleranceReached = true;
362  std::string errorMessage =
363  "Warning, light time unconverged at level " +
364  std::to_string(
365  fabs( newLightTimeCalculation - previousLightTimeCalculation ) ) +
366  "; current light-time corrections are: " +
367  std::to_string( currentCorrection_ ) + " and input time was " +
368  std::to_string( static_cast< double >( time ) );
369  std::cerr << errorMessage << std::endl;
370  }
371 
372  // Update light time for new iteration.
373  previousLightTimeCalculation = newLightTimeCalculation;
374  }
375 
376  counter++;
377  }
378 
379  // Set output variables and return the light time.
380  receiverStateOutput = receiverState;
381  transmitterStateOutput = transmitterState;
382 
383  return newLightTimeCalculation;
384  }

396  const StateType& transmitterState,
