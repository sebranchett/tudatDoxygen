83 template< typename DataType = double >
84 std::shared_ptr< RootFinderCore< DataType > > createRootFinder(
85  const std::shared_ptr< RootFinderSettings > rootFinderSettings,
86  const DataType lowerBound = TUDAT_NAN, const DataType upperBound = TUDAT_NAN,
87  const DataType previousGuess = TUDAT_NAN )
88 {
89  std::shared_ptr< RootFinderCore< DataType > > rootFinder;
90  switch( rootFinderSettings->rootFinderType_ )
91  {
92  case bisection_root_finder:
93  if( !( lowerBound == lowerBound ) || !( upperBound == upperBound ) )
94  {
95  throw std::runtime_error( "Error when making bisection root finder, lower/upped bound not provided" );
96  }
97  rootFinder = std::make_shared< BisectionCore< DataType > >(
98  rootFinderSettings->terminationTolerance_, rootFinderSettings->maximumNumberOfIterations_,
99  lowerBound, upperBound );
100  break;
101  case halley_root_finder:
102  rootFinder = std::make_shared< HalleyRootFinderCore< DataType > >(
103  rootFinderSettings->terminationTolerance_, rootFinderSettings->maximumNumberOfIterations_ );
104  break;
105  case newton_raphson_root_finder:
106  rootFinder = std::make_shared< NewtonRaphsonCore< DataType > >(
107  rootFinderSettings->terminationTolerance_, rootFinderSettings->maximumNumberOfIterations_ );
108  break;
109  case secant_root_finder:
110  if( !( previousGuess == previousGuess ) )
111  {
112  throw std::runtime_error( "Error when making secant root finder, initial guess not provided" );
113  }
114  rootFinder = std::make_shared< SecantRootFinderCore< DataType > >(
115  rootFinderSettings->terminationTolerance_, rootFinderSettings->maximumNumberOfIterations_,
116  previousGuess );
117  break;
118  default:
119  throw std::runtime_error( "Error when creating root finder, did not recognize root finder type" );
120  }
121  return rootFinder;
122 }
