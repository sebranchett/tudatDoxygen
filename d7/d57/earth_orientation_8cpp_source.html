23 std::pair< Eigen::Vector2d, double > getPositionOfCipInGcrs(
24  const double terrestrialTime, const double julianDaysEpochShift,
25  const basic_astrodynamics::IAUConventions precessionNutationTheory )
26 {
27  // Declare Sofa function return arguments (by reference)
28  double xAngle, yAngle;
29  double originLocator;
30 
31  // Check for IAU convention and retrieve requested values.
32  switch( precessionNutationTheory )
33  {
34  case basic_astrodynamics::iau_2000_a:
35  iauXys00a( julianDaysEpochShift, terrestrialTime / physical_constants::JULIAN_DAY,
36  &xAngle, &yAngle, &originLocator );
37  break;
38 
39  case basic_astrodynamics::iau_2000_b:
40  iauXys00b( julianDaysEpochShift, terrestrialTime / physical_constants::JULIAN_DAY,
41  &xAngle, &yAngle, &originLocator );
42  break;
43 
44  case basic_astrodynamics::iau_2006:
45  iauXys06a( julianDaysEpochShift, terrestrialTime / physical_constants::JULIAN_DAY,
46  &xAngle, &yAngle, &originLocator );
47  break;
48  default:
49  throw std::runtime_error( "Warning, precession nutation theory selection not recongnized" );
50 
51  }
52 
53  // Set and return requested values.
54  Eigen::Vector2d cioPosition;
55  cioPosition << xAngle, yAngle;
56  return std::pair< Eigen::Vector2d, double >( cioPosition, originLocator );
57 }

60 double calculateGreenwichMeanSiderealTime(
61  const double terrestrialTime, const double universalTime1,
62  const double referenceJulianDay, const basic_astrodynamics::IAUConventions iauConvention )
63 {
64  // Declare GMST variable
65  double gmst = TUDAT_NAN;
66 
67  // Check for IAU convention and retrieve requested GMST
68  switch( iauConvention )
69  {
70  case basic_astrodynamics::iau_2000_a:
71  gmst = iauGmst00( referenceJulianDay, universalTime1 / physical_constants::JULIAN_DAY,
72  referenceJulianDay, terrestrialTime / physical_constants::JULIAN_DAY );
73  break;
74 
75  case basic_astrodynamics::iau_2000_b:
76  gmst = iauGmst00( referenceJulianDay, universalTime1 / physical_constants::JULIAN_DAY,
77  referenceJulianDay, terrestrialTime / physical_constants::JULIAN_DAY );
78  break;
79 
80  case basic_astrodynamics::iau_2006:

84  default:
