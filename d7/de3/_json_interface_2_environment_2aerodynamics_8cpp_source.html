23 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< AerodynamicCoefficientSettings >& aerodynamicSettings )
24 {
25  if ( ! aerodynamicSettings )
26  {
27  return;
28  }
29  using namespace json_interface;
30  using K = Keys::Body::Aerodynamics;
31 
32  const AerodynamicCoefficientTypes aerodynamicCoefficientType =
33  aerodynamicSettings->getAerodynamicCoefficientType( );
34  jsonObject[ K::coefficientsType ] = aerodynamicCoefficientType;
35  jsonObject[ K::referenceArea ] = aerodynamicSettings->getReferenceArea( );
36 
37  switch ( aerodynamicCoefficientType ) {
38  case constant_aerodynamic_coefficients:
39  {
40  std::shared_ptr< ConstantAerodynamicCoefficientSettings > constantAerodynamicSettings =
41  std::dynamic_pointer_cast< ConstantAerodynamicCoefficientSettings >( aerodynamicSettings );
42  assertNonnullptrPointer( constantAerodynamicSettings );
43  jsonObject[ K::forceCoefficients ] = constantAerodynamicSettings->getConstantForceCoefficient( );
44 
45  // Moments
46  if ( ! isNaN( constantAerodynamicSettings->getReferenceLength( ) ) )
47  {
48  jsonObject[ K::referenceLength ] = constantAerodynamicSettings->getReferenceLength( );
49  jsonObject[ K::lateralReferenceLength ] = constantAerodynamicSettings->getLateralReferenceLength( );
50  jsonObject[ K::momentReferencePoint ] = constantAerodynamicSettings->getMomentReferencePoint( );
51  jsonObject[ K::momentCoefficients ] = constantAerodynamicSettings->getConstantMomentCoefficient( );
52  }
53 
54  jsonObject[ K::areCoefficientsInAerodynamicFrame ] =
55  constantAerodynamicSettings->getAreCoefficientsInAerodynamicFrame( );
56  jsonObject[ K::areCoefficientsInNegativeAxisDirection ] =
57  constantAerodynamicSettings->getAreCoefficientsInNegativeAxisDirection( );
58  return;
59  }
60  case tabulated_coefficients:
61  {
62  std::shared_ptr< TabulatedAerodynamicCoefficientSettingsBase > fromFileAerodynamicCoefficientSettings =
63  std::dynamic_pointer_cast< TabulatedAerodynamicCoefficientSettingsBase >( aerodynamicSettings );
64  assertNonnullptrPointer( fromFileAerodynamicCoefficientSettings );
65  const bool hasMoments = ! isNaN( fromFileAerodynamicCoefficientSettings->getReferenceLength( ) );
66 
67  std::shared_ptr< TabulatedAerodynamicCoefficientSettings< 1 > > tabulated1AerodynamicSettings =
68  std::dynamic_pointer_cast< TabulatedAerodynamicCoefficientSettings< 1 > >( aerodynamicSettings );
69  if ( tabulated1AerodynamicSettings ) // uni-dimensional
70  {
71  jsonObject[ K::independentVariableValues ] =
72  getMapKeys< std::map >( tabulated1AerodynamicSettings->getForceCoefficients( ) );
73  jsonObject[ K::forceCoefficients ] =
74  getMapValues< std::map >( tabulated1AerodynamicSettings->getForceCoefficients( ) );
75 
76  if ( hasMoments )
77  {
78  jsonObject[ K::momentCoefficients ] =
79  getMapValues< std::map >( tabulated1AerodynamicSettings->getMomentCoefficients( ) );
80  }
81 
82  jsonObject[ K::interpolator ] = tabulated1AerodynamicSettings->getInterpolatorSettings( );
83  }
84  else // N-dimensional
85  {
86  const std::map< int, std::string > forceCoefficientsFilesMap =
87  fromFileAerodynamicCoefficientSettings->getForceCoefficientsFiles( );
88  jsonObject[ K::forceCoefficients ] = { "", "", "" };
89  for ( auto entry : forceCoefficientsFilesMap )
90  {
91  jsonObject[ K::forceCoefficients ][ entry.first ] = boost::filesystem::path( entry.second );
92  }
93 
94  if ( hasMoments )
95  {
96  const std::map< int, std::string > momentCoefficientsFilesMap =
97  fromFileAerodynamicCoefficientSettings->getMomentCoefficientsFiles( );
98  jsonObject[ K::momentCoefficients ] = { "", "", "" };
99  for ( auto entry : momentCoefficientsFilesMap )
100  {
101  jsonObject[ K::momentCoefficients ][ entry.first ] = boost::filesystem::path( entry.second );
102  }
103  }
104  }
105 
106  if ( hasMoments )
107  {
108  jsonObject[ K::referenceLength ] = fromFileAerodynamicCoefficientSettings->getReferenceLength( );
109  jsonObject[ K::lateralReferenceLength ] =
110  fromFileAerodynamicCoefficientSettings->getLateralReferenceLength( );
111  jsonObject[ K::momentReferencePoint ] = fromFileAerodynamicCoefficientSettings->getMomentReferencePoint( );
112  }
113 
114  jsonObject[ K::independentVariableNames ] =
115  fromFileAerodynamicCoefficientSettings->getIndependentVariableNames( );
116  jsonObject[ K::areCoefficientsInAerodynamicFrame ] =
117  fromFileAerodynamicCoefficientSettings->getAreCoefficientsInAerodynamicFrame( );
118  jsonObject[ K::areCoefficientsInNegativeAxisDirection ] =
119  fromFileAerodynamicCoefficientSettings->getAreCoefficientsInNegativeAxisDirection( );
120 
121  return;
122  }
123  default:
124  handleUnimplementedEnumValue( aerodynamicCoefficientType, aerodynamicCoefficientTypes,
125  unsupportedAerodynamicCoefficientTypes );
126  }

130 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< AerodynamicCoefficientSettings >& aerodynamicSettings )
131 {
132  using namespace aerodynamics;
133  using namespace interpolators;
134  using namespace json_interface;
135  using K = Keys::Body::Aerodynamics;
136 
137  // Aerodynamic coefficient type (constant by default)
138  const AerodynamicCoefficientTypes coefficientsType =
139  getValue( jsonObject, K::coefficientsType, constant_aerodynamic_coefficients );
140 
141  // Reference area (either from the current object or from the current object's parent, i.e. the body)
142  const double referenceArea =
143  getValue< double >( jsonObject, { K::referenceArea, SpecialKeys::up / Keys::Body::referenceArea } );
144 
145  switch ( coefficientsType )
146  {
147  case constant_aerodynamic_coefficients:
148  {
149  // Read forceCoefficients. If not defined, use [ dragCoefficient, 0, 0 ].
150  Eigen::Vector3d forceCoefficients = Eigen::Vector3d::Zero( );
151  if ( isDefined( jsonObject, K::forceCoefficients ) )
152  {
153  forceCoefficients = getValue< Eigen::Vector3d >( jsonObject, K::forceCoefficients );
154  }
155  else
156  {
157  forceCoefficients( 0 ) = getValue< double >( jsonObject, K::dragCoefficient );
158  }
159 
160  if ( isDefined( jsonObject, K::momentCoefficients ) ) // moments
161  {
162  ConstantAerodynamicCoefficientSettings defaults( TUDAT_NAN, TUDAT_NAN, TUDAT_NAN,
163  Eigen::Vector3d( ), Eigen::Vector3d( ) );
164  aerodynamicSettings = std::make_shared< ConstantAerodynamicCoefficientSettings >(
165  getValue< double >( jsonObject, K::referenceLength ),
166  referenceArea,
167  getValue< double >( jsonObject, K::lateralReferenceLength ),
168  getValue< Eigen::Vector3d >( jsonObject, K::momentReferencePoint ),
169  forceCoefficients,
170  getValue< Eigen::Vector3d >( jsonObject, K::momentCoefficients ),
171  getValue( jsonObject, K::areCoefficientsInAerodynamicFrame,
172  defaults.getAreCoefficientsInAerodynamicFrame( ) ),
173  getValue( jsonObject, K::areCoefficientsInNegativeAxisDirection,
174  defaults.getAreCoefficientsInNegativeAxisDirection( ) ) );
175  }
176  else // no moments
177  {
178  ConstantAerodynamicCoefficientSettings defaults( TUDAT_NAN, Eigen::Vector3d( ) );
179  aerodynamicSettings = std::make_shared< ConstantAerodynamicCoefficientSettings >(
180  referenceArea,
181  forceCoefficients,
182  getValue( jsonObject, K::areCoefficientsInAerodynamicFrame,
183  defaults.getAreCoefficientsInAerodynamicFrame( ) ),
184  getValue( jsonObject, K::areCoefficientsInNegativeAxisDirection,
185  defaults.getAreCoefficientsInNegativeAxisDirection( ) ) );
186  }
187  return;
188  }
189  case tabulated_coefficients:
190  {
191  const nlohmann::json jsonForceCoefficients = getValue< nlohmann::json >( jsonObject, K::forceCoefficients );
192  const nlohmann::json jsonMomentCoefficients = getValue( jsonObject, K::momentCoefficients, nlohmann::json( ) );
193  const bool useMoments = ! jsonMomentCoefficients.is_null( );
194 
195  bool readFromFiles = true;
196  std::map< int, std::string > forceCoefficientsFiles;
197  std::vector< Eigen::Vector3d > forceCoefficients;
198  std::map< int, std::string > momentCoefficientsFiles;
199  std::vector< Eigen::Vector3d > momentCoefficients;
200  try
201  {
202  const std::vector< boost::filesystem::path > forceCoefficientsFilesVector =
203  getAs< std::vector< boost::filesystem::path > >( jsonForceCoefficients );
204  for ( unsigned int i = 0; i < forceCoefficientsFilesVector.size( ); ++i )
205  {
206  if ( ! jsonObject.at( K::forceCoefficients ).at( i ).get< std::string >( ).empty( ) )
207  {
208  forceCoefficientsFiles[ i ] = forceCoefficientsFilesVector.at( i ).string( );
209  }
210  }
211 
212  if ( useMoments )
213  {
214  const std::vector< boost::filesystem::path > momentCoefficientsFilesVector =
215  getAs< std::vector< boost::filesystem::path > >( jsonMomentCoefficients );
216  for ( unsigned int i = 0; i < momentCoefficientsFilesVector.size( ); ++i )
217  {
218  if ( ! jsonObject.at( K::momentCoefficients ).at( i ).get< std::string >( ).empty( ) )
219  {
220  momentCoefficientsFiles[ i ] = momentCoefficientsFilesVector.at( i ).string( );
221  }
222  }
223  }
224  }
225  catch ( ... )
226  {
227  readFromFiles = false;
228  forceCoefficients = getAs< std::vector< Eigen::Vector3d > >( jsonForceCoefficients );
229  if ( useMoments )
230  {
231  momentCoefficients = getAs< std::vector< Eigen::Vector3d > >( jsonMomentCoefficients );
232  }
233  }
234 
235  const std::vector< AerodynamicCoefficientsIndependentVariables > independentVariableNames =
236  getValue< std::vector< AerodynamicCoefficientsIndependentVariables > >(
237  jsonObject, K::independentVariableNames );
238  const bool areCoefficientsInAerodynamicFrame =
239  getValue( jsonObject, K::areCoefficientsInAerodynamicFrame, true );
240  const bool areCoefficientsInNegativeAxisDirection =
241  getValue( jsonObject, K::areCoefficientsInNegativeAxisDirection, true );
242 
243  if ( readFromFiles )
244  {
245  if ( useMoments )
246  {
247  aerodynamicSettings = readTabulatedAerodynamicCoefficientsFromFiles(
248  forceCoefficientsFiles,
249  momentCoefficientsFiles,
250  getValue< double >( jsonObject, K::referenceLength ),
251  referenceArea,
252  getValue< double >( jsonObject, K::lateralReferenceLength ),
253  getValue< Eigen::Vector3d >( jsonObject, K::momentReferencePoint ),
254  independentVariableNames,
255  areCoefficientsInAerodynamicFrame,
256  areCoefficientsInNegativeAxisDirection );
257  }
258  else
259  {
260  aerodynamicSettings = readTabulatedAerodynamicCoefficientsFromFiles(
261  forceCoefficientsFiles,
262  referenceArea,
263  independentVariableNames,
264  areCoefficientsInAerodynamicFrame,
265  areCoefficientsInNegativeAxisDirection );
266  }
267  }
268  else
269  {
270  if ( useMoments )
271  {
272  aerodynamicSettings = std::make_shared< TabulatedAerodynamicCoefficientSettings< 1 > >(
273  getValue< std::vector< double > >( jsonObject, K::independentVariableValues ),
274  forceCoefficients,
275  momentCoefficients,
276  getValue< double >( jsonObject, K::referenceLength ),
277  referenceArea,
278  getValue< double >( jsonObject, K::lateralReferenceLength ),
279  getValue< Eigen::Vector3d >( jsonObject, K::momentReferencePoint ),
280  independentVariableNames.front( ),
281  areCoefficientsInAerodynamicFrame,
282  areCoefficientsInNegativeAxisDirection,
283  getValue< std::shared_ptr< InterpolatorSettings > >( jsonObject, K::interpolator ) );
284  }
285  else
286  {
287  aerodynamicSettings = std::make_shared< TabulatedAerodynamicCoefficientSettings< 1 > >(
288  getValue< std::vector< double > >( jsonObject, K::independentVariableValues ),
289  forceCoefficients,
290  referenceArea,
291  independentVariableNames.front( ),
292  areCoefficientsInAerodynamicFrame,
293  areCoefficientsInNegativeAxisDirection,
294  getValue< std::shared_ptr< InterpolatorSettings > >( jsonObject, K::interpolator ) );
295  }
296  }
297  return;
298  }
299  default:
300  handleUnimplementedEnumValue( coefficientsType, aerodynamicCoefficientTypes,
301  unsupportedAerodynamicCoefficientTypes );
302  }
303 }
