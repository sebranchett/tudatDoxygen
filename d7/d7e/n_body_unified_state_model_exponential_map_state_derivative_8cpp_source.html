27 Eigen::Vector7d computeStateDerivativeForUnifiedStateModelExponentialMap(
28  const Eigen::Vector7d& currentUnifiedStateModelElements,
29  const Eigen::Vector3d& accelerationsInRswFrame,
30  const double sineLambda,
31  const double cosineLambda,
32  const double gammaParameter,
33  const Eigen::Vector3d& rotationalVelocityVector,
34  const Eigen::Vector3d& pAuxiliaryVector )
35 {
36  // Compute matrix for dynamic equation
37  Eigen::Matrix3d hodographMatrix = Eigen::Matrix3d::Zero( );
38  hodographMatrix( 0, 1 ) = - pAuxiliaryVector( 0 );
39  hodographMatrix( 1, 0 ) = cosineLambda;
40  hodographMatrix( 1, 1 ) = - ( 1.0 + pAuxiliaryVector( 0 ) ) * sineLambda;
41  hodographMatrix( 1, 2 ) = - gammaParameter * pAuxiliaryVector( 1 );
42  hodographMatrix( 2, 0 ) = sineLambda;
43  hodographMatrix( 2, 1 ) = ( 1.0 + pAuxiliaryVector( 0 ) ) * cosineLambda;
44  hodographMatrix( 2, 2 ) = gammaParameter * pAuxiliaryVector( 2 );
45 
46  // Evaluate USMEM equations.
47  Eigen::Vector7d stateDerivative;
48  stateDerivative.segment( 0, 3 ) = hodographMatrix * accelerationsInRswFrame;
49  stateDerivative.segment( 3, 4 ) = calculateExponentialMapDerivative( currentUnifiedStateModelElements.segment( 3, 4 ),
50  rotationalVelocityVector );
51 
52  // Give output
53  return stateDerivative;
54 }

57 Eigen::Vector7d computeStateDerivativeForUnifiedStateModelExponentialMap(
58  const Eigen::Vector7d& currentUnifiedStateModelElements,
59  const Eigen::Vector3d& accelerationsInRswFrame,
60  const double centralBodyGravitationalParameter )
61 {
62  using namespace orbital_element_conversions;
63 
64  // Convert exponential map to quaternions
65  Eigen::Vector4d quaternionElements = convertExponentialMapToQuaternionElements(
66  currentUnifiedStateModelElements.segment( e1USMEMIndex, 4 ) );
67  double etaQuaternionParameter = quaternionElements( etaQuaternionIndex );
68  Eigen::Vector3d epsilonQuaternionVector = quaternionElements.segment( epsilon1QuaternionIndex, 3 );
69 
70  // Compute supporting parameters
71  double denominator = std::pow( epsilonQuaternionVector( 2 ), 2 ) + std::pow( etaQuaternionParameter, 2 );
72  double sineLambda = ( 2 * epsilonQuaternionVector( 2 ) * etaQuaternionParameter ) / denominator;
73  double cosineLambda = ( std::pow( etaQuaternionParameter, 2 ) - std::pow( epsilonQuaternionVector( 2 ), 2 ) ) / denominator;
74  double gammaParameter = ( epsilonQuaternionVector( 0 ) * epsilonQuaternionVector( 2 ) -
75  epsilonQuaternionVector( 1 ) * etaQuaternionParameter ) / denominator;
76 
77  double velocityHodographParameter = currentUnifiedStateModelElements( CHodographUSMEMIndex ) -
78  currentUnifiedStateModelElements( Rf1HodographUSMEMIndex ) * sineLambda +
79  currentUnifiedStateModelElements( Rf2HodographUSMEMIndex ) * cosineLambda;
80  Eigen::Vector3d rotationalVelocityVector = Eigen::Vector3d::Zero( );
81  rotationalVelocityVector( 0 ) = accelerationsInRswFrame( 2 ) / velocityHodographParameter;
82  rotationalVelocityVector( 2 ) = std::pow( velocityHodographParameter, 2 ) *
83  currentUnifiedStateModelElements( CHodographUSMEMIndex ) / centralBodyGravitationalParameter;
84 
85  Eigen::Vector3d pAuxiliaryVector = Eigen::Vector3d::Zero( );
86  pAuxiliaryVector( 0 ) = currentUnifiedStateModelElements( CHodographUSMEMIndex );
87  pAuxiliaryVector( 1 ) = currentUnifiedStateModelElements( Rf2HodographUSMEMIndex );
88  pAuxiliaryVector( 2 ) = currentUnifiedStateModelElements( Rf1HodographUSMEMIndex );
89  pAuxiliaryVector /= velocityHodographParameter;
90 
91  // Evaluate USMEM equations
92  return computeStateDerivativeForUnifiedStateModelExponentialMap(
93  currentUnifiedStateModelElements, accelerationsInRswFrame, sineLambda,
94  cosineLambda, gammaParameter, rotationalVelocityVector, pAuxiliaryVector );
95 }
