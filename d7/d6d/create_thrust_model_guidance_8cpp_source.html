22 std::shared_ptr< propulsion::BodyFixedForceDirectionGuidance > createThrustGuidanceModel(
23  const std::shared_ptr< ThrustDirectionGuidanceSettings > thrustDirectionGuidanceSettings,
24  const NamedBodyMap& bodyMap,
25  const std::string& nameOfBodyWithGuidance,
26  const std::function< Eigen::Vector3d( ) > bodyFixedThrustOrientation,
27  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >& magnitudeUpdateSettings )
28 {
29  std::shared_ptr< propulsion::BodyFixedForceDirectionGuidance > thrustGuidance;
30 
31  // Determine thrust direction type
32  switch( thrustDirectionGuidanceSettings->thrustDirectionType_ )
33  {
34  case colinear_with_state_segment_thrust_direction:
35  {
36  // Check input consistency
37  std::shared_ptr< ThrustDirectionFromStateGuidanceSettings > thrustDirectionFromStateGuidanceSettings =
38  std::dynamic_pointer_cast< ThrustDirectionFromStateGuidanceSettings >( thrustDirectionGuidanceSettings );
39  if( thrustDirectionFromStateGuidanceSettings == nullptr )
40  {
41  throw std::runtime_error( "Error when thrust colinear with state, input is inconsistent" );
42  }
43  else
44  {
45  // Retrieve state function of body for which thrust is to be computed.
46  std::function< Eigen::Vector6d( ) > bodyStateFunction =
47  std::bind( &Body::getState, bodyMap.at( nameOfBodyWithGuidance ) );
48  std::function< Eigen::Vector6d( ) > centralBodyStateFunction;
49 
50  // Retrieve state function of central body (or set to zero if inertial)
51  if( thrustDirectionFromStateGuidanceSettings->relativeBody_ != "SSB" &&
52  thrustDirectionFromStateGuidanceSettings->relativeBody_ != "" )
53  {
54  centralBodyStateFunction = std::bind( &Body::getState, bodyMap.at(
55  thrustDirectionFromStateGuidanceSettings->relativeBody_ ) );
56  magnitudeUpdateSettings[ propagators::body_translational_state_update ].push_back(
57  thrustDirectionFromStateGuidanceSettings->relativeBody_ );
58  }
59  else
60  {
61  centralBodyStateFunction = [ ]( ){ return Eigen::Vector6d::Zero( ); };
62  }
63 
64  // Define relative state function
65  std::function< void( Eigen::Vector6d& ) > stateFunction =
66  std::bind(
67  &ephemerides::getRelativeState, std::placeholders::_1, bodyStateFunction, centralBodyStateFunction );
68  std::function< Eigen::Vector3d( const double ) > thrustDirectionFunction;
69 
70  // Create force direction function.
71  if( thrustDirectionFromStateGuidanceSettings->isColinearWithVelocity_ )
72  {
73  thrustDirectionFunction =
74  std::bind( &propulsion::getForceDirectionColinearWithVelocity, stateFunction, std::placeholders::_1,
75  thrustDirectionFromStateGuidanceSettings->directionIsOppositeToVector_ );
76  }
77  else
78  {
79  thrustDirectionFunction =
80  std::bind( &propulsion::getForceDirectionColinearWithPosition, stateFunction, std::placeholders::_1,
81  thrustDirectionFromStateGuidanceSettings->directionIsOppositeToVector_ );
82  }
83 
84  // Create direction guidance
85  thrustGuidance = std::make_shared< propulsion::DirectionBasedForceGuidance >(
86  thrustDirectionFunction, thrustDirectionFromStateGuidanceSettings->relativeBody_,
87  bodyFixedThrustOrientation );
88  }
89  break;
90  }
91  case thrust_direction_from_existing_body_orientation:
92  {
93  std::shared_ptr< Body > bodyWithGuidance = bodyMap.at( nameOfBodyWithGuidance );
94 
95  std::function< Eigen::Quaterniond( const double ) > rotationFunction;
96 
97  // Retrieve existing body rotation model and set associated update settings.
98  if( bodyWithGuidance->getFlightConditions( ) != nullptr )
99  {
100  rotationFunction = std::bind(
101  &simulation_setup::Body::getCurrentRotationToGlobalFrame,
102  bodyWithGuidance );
103 
104  magnitudeUpdateSettings[ propagators::vehicle_flight_conditions_update ].push_back( nameOfBodyWithGuidance );
105  magnitudeUpdateSettings[ propagators::body_rotational_state_update ].push_back( nameOfBodyWithGuidance );
106  magnitudeUpdateSettings[ propagators::body_rotational_state_update ].push_back(
107  thrustDirectionGuidanceSettings->relativeBody_ );
108  magnitudeUpdateSettings[ propagators::body_translational_state_update ].push_back(
109  thrustDirectionGuidanceSettings->relativeBody_);
110  }
111  else if( bodyWithGuidance->getRotationalEphemeris( ) != nullptr )
112  {
113  rotationFunction = std::bind(
114  &simulation_setup::Body::getCurrentRotationToGlobalFrame,
115  bodyWithGuidance );
116  magnitudeUpdateSettings[ propagators::body_rotational_state_update ].push_back( nameOfBodyWithGuidance );
117  }
118  else
119  {
120  throw std::runtime_error( "Error, requested thrust orientation from existing model, but no such model found" );
121  }
122 
123  thrustGuidance = std::make_shared< propulsion::OrientationBasedForceGuidance >(
124  rotationFunction, bodyFixedThrustOrientation );
125 
126  break;
127  }
128  case custom_thrust_direction:
129  {
130  // Check input consistency
131  std::shared_ptr< CustomThrustDirectionSettings > customThrustGuidanceSettings =
132  std::dynamic_pointer_cast< CustomThrustDirectionSettings >( thrustDirectionGuidanceSettings );
133  if( customThrustGuidanceSettings == nullptr )
134  {
135  throw std::runtime_error( "Error when getting thrust guidance with custom_thrust_direction, input is inconsistent" );
136  }
137  else
138  {
139  // Create direction guidance
140  std::function< Eigen::Vector3d( const double ) > thrustDirectionFunction =
141  customThrustGuidanceSettings->thrustDirectionFunction_;
142  thrustGuidance = std::make_shared< propulsion::DirectionBasedForceGuidance >(
143  thrustDirectionFunction, "", bodyFixedThrustOrientation );
144  }
145 
146  break;
147  }
148  case custom_thrust_orientation:
149  {
150  // Check input consistency
151  std::shared_ptr< CustomThrustOrientationSettings > customThrustOrientationSettings =
152  std::dynamic_pointer_cast< CustomThrustOrientationSettings >( thrustDirectionGuidanceSettings );
153  if( customThrustOrientationSettings == nullptr )
154  {
155  throw std::runtime_error( "Error when getting thrust guidance with custom_thrust_orientation, input is inconsistent" );
156  }
157  else
158  {
159  // Create direction guidance
160  thrustGuidance = std::make_shared< propulsion::OrientationBasedForceGuidance >(
161  customThrustOrientationSettings->thrustOrientationFunction_,
162  bodyFixedThrustOrientation );
163  magnitudeUpdateSettings[ propagators::body_rotational_state_update ].push_back( nameOfBodyWithGuidance );
164  }
165  break;
166  }
167  case mee_costate_based_thrust_direction:
168  {
169  // Check input consistency
170  std::shared_ptr< MeeCostateBasedThrustDirectionSettings > meeCostateBasedThrustSettings =
171  std::dynamic_pointer_cast< MeeCostateBasedThrustDirectionSettings >( thrustDirectionGuidanceSettings );
172 
173  if( meeCostateBasedThrustSettings == nullptr )
174  {
175  throw std::runtime_error( "Error when getting thrust guidance with mee_costate_based_thrust_direction, input is inconsistent" );
176  }
177  else
178  {
179  // Check whether all required environment properties exist
180  if( bodyMap.count( meeCostateBasedThrustSettings->relativeBody_ ) == 0 )
181  {
182  throw std::runtime_error( "Error when getting thrust guidance with mee_costate_based_thrust_direction, central body " +
183  meeCostateBasedThrustSettings->relativeBody_ + " not found." );
184  }
185  else if( bodyMap.count( meeCostateBasedThrustSettings->vehicleName_ ) == 0 )
186  {
187  throw std::runtime_error( "Error when getting thrust guidance with mee_costate_based_thrust_direction, thrusting body " +
188  meeCostateBasedThrustSettings->vehicleName_ + " not found." );
189  }
190  else if( bodyMap.at( meeCostateBasedThrustSettings->relativeBody_ )->getGravityFieldModel( ) == nullptr )
191  {
192  throw std::runtime_error( "Error when getting thrust guidance with mee_costate_based_thrust_direction, central body " +
193  meeCostateBasedThrustSettings->relativeBody_ + " has no gravity field." );
194  }
195  else
196  {
197  // Retrieve required functions and create guidance object
198  std::function< Eigen::Vector6d( ) > thrustingBodyStateFunction =
199  std::bind( &simulation_setup::Body::getState,
200  bodyMap.at( meeCostateBasedThrustSettings->vehicleName_ ) );
201  std::function< Eigen::Vector6d( ) > centralBodyStateFunction =
202  std::bind( &simulation_setup::Body::getState,
203  bodyMap.at( meeCostateBasedThrustSettings->relativeBody_ ) );
204  std::function< double( ) > centralBodyGravitationalParameterFunction =
205  std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
206  bodyMap.at( meeCostateBasedThrustSettings->relativeBody_ )->getGravityFieldModel( ) );
207 
208  thrustGuidance = std::make_shared< propulsion::MeeCostateBasedThrustGuidance >(
209  thrustingBodyStateFunction, centralBodyStateFunction,
210  centralBodyGravitationalParameterFunction,
211  meeCostateBasedThrustSettings->costateFunction_,
212  bodyFixedThrustOrientation );
213  }
214  }
215  break;
216  }
217  default:
218  throw std::runtime_error( "Error, could not find thrust guidance type when creating thrust guidance." );
219  }
220  return thrustGuidance;
221 }

224 Eigen::Vector3d getCombinedThrustDirection(
225  const std::vector< std::function< Eigen::Vector3d( )> >& thrustDirections,
226  const std::vector< std::function< double( )> >& thrustMagnitudes )
227 {
228  Eigen::Vector3d thrustDirection = Eigen::Vector3d::Zero( );
229  double totalThrust = 0.0;
230 
231  for( unsigned int i = 0; i < thrustDirections.size( ); i++ )
232  {
233  thrustDirection += thrustMagnitudes.at( i )( ) * thrustDirections.at( i )( );
234  totalThrust += thrustMagnitudes.at( i )( );
235  }
236  return thrustDirection / totalThrust;
237 }

240 std::function< Eigen::Vector3d( ) > getBodyFixedThrustDirection(
241  const std::shared_ptr< ThrustMagnitudeSettings > thrustMagnitudeSettings,
242  const NamedBodyMap& bodyMap,
243  const std::string bodyName )
244 {
245  std::function< Eigen::Vector3d( ) > thrustDirectionFunction;
246 
247  // Identify magnitude settings type
248  switch( thrustMagnitudeSettings->thrustMagnitudeGuidanceType_ )
249  {
250  case constant_thrust_magnitude:
251  {
252  // Check input consistency
253  std::shared_ptr< ConstantThrustMagnitudeSettings > constantThrustMagnitudeSettings =
254  std::dynamic_pointer_cast< ConstantThrustMagnitudeSettings >( thrustMagnitudeSettings );
255  if( constantThrustMagnitudeSettings == nullptr )
256  {
257  throw std::runtime_error( "Error when creating body-fixed thrust direction of type constant_thrust_magnitude, input is inconsistent" );
258  }
259  else
260  {
261  thrustDirectionFunction = [ = ]( ){ return constantThrustMagnitudeSettings->bodyFixedThrustDirection_; };
262  }
263  break;
264  }
265  case from_engine_properties_thrust_magnitude:
266  {
267  // Check input consistency
268  std::shared_ptr< FromBodyThrustMagnitudeSettings > fromEngineThrustMagnitudeSettings =
269  std::dynamic_pointer_cast< FromBodyThrustMagnitudeSettings >( thrustMagnitudeSettings );
270  if( fromEngineThrustMagnitudeSettings == nullptr )
271  {
272  throw std::runtime_error( "Error when creating body-fixed thrust direction of type from_engine_properties_thrust_magnitude, input is inconsistent" );
273  }
274  if( bodyMap.at( bodyName )->getVehicleSystems( ) == nullptr )
275  {
276  throw std::runtime_error( "Error when creating body-fixed thrust direction of type from_engine_properties_thrust_magnitude, no vehicle systems found" );
277 
278  }
279 
280  // Retrieve single engine
281  if( fromEngineThrustMagnitudeSettings->useAllEngines_ == false )
282  {
283  // Check if engine model exists
284  if( ( bodyMap.at( bodyName )->getVehicleSystems( )->getEngineModels( ).count(
285  thrustMagnitudeSettings->thrustOriginId_ ) == 0 ) )
286  {
287  throw std::runtime_error( "Error when creating body-fixed thrust direction of type from_engine_properties_thrust_magnitude, no engine of right ID found" );
288  }
289  else
290  {
291  thrustDirectionFunction =
292  std::bind( &system_models::EngineModel::getBodyFixedThrustDirection,
293  bodyMap.at( bodyName )->getVehicleSystems( )->getEngineModels( ).at(
294  thrustMagnitudeSettings->thrustOriginId_ ) );
295  }
296 
297  }
298  // Retrieve mean thrust direction from all engines
299  else
300  {
301  // Print warning if there are no engines (zero thrust)
302  if( ( bodyMap.at( bodyName )->getVehicleSystems( )->getEngineModels( ).size( ) == 0 ) )
303  {
304  std::cerr << "Warning when creating body-fixed thrust direction of type from_engine_properties_thrust_magnitude; no engines found: returning 0 thrust" << std::endl;
305  }
306 
307  // Retrieve force directions/magnitudes
308  std::vector< std::function< Eigen::Vector3d( )> > thrustDirections;
309  std::vector< std::function< double( )> > thrustMagnitudes;
310 
311  std::map< std::string, std::shared_ptr< system_models::EngineModel > > engineModels =
312  bodyMap.at( bodyName )->getVehicleSystems( )->getEngineModels( );
313 
314  for( std::map< std::string, std::shared_ptr< system_models::EngineModel > >::const_iterator engineIterator =
315  engineModels.begin( ); engineIterator != engineModels.end( ); engineIterator++ )
316  {
317  thrustDirections.push_back(
318  std::bind( &system_models::EngineModel::getBodyFixedThrustDirection, engineIterator->second ) );
319  thrustMagnitudes.push_back(
320  std::bind( &system_models::EngineModel::getCurrentThrust, engineIterator->second ) );
321  }
322 
323  // Create effective thrust direction function.
324  thrustDirectionFunction = std::bind(
325  &getCombinedThrustDirection, thrustDirections, thrustMagnitudes );
326 
327  }
328  break;
329 
330  }
331  case thrust_magnitude_from_time_function:
332  {
333  // Check input consistency
334  std::shared_ptr< FromFunctionThrustMagnitudeSettings > fromFunctionThrustMagnitudeSettings =
335  std::dynamic_pointer_cast< FromFunctionThrustMagnitudeSettings >( thrustMagnitudeSettings );
336  if( fromFunctionThrustMagnitudeSettings == nullptr )
337  {
338  throw std::runtime_error( "Error when creating body-fixed thrust direction of type thrust_magnitude_from_time_function, input is inconsistent" );
339  }
340  else
341  {
342  thrustDirectionFunction = fromFunctionThrustMagnitudeSettings->bodyFixedThrustDirection_;
343  }
344  break;
345 
346  }
347  case thrust_magnitude_from_dependent_variables:
348  {
349  // Check input consistency
350  std::shared_ptr< ParameterizedThrustMagnitudeSettings > fromFunctionThrustMagnitudeSettings =
351  std::dynamic_pointer_cast< ParameterizedThrustMagnitudeSettings >( thrustMagnitudeSettings );
352  if( fromFunctionThrustMagnitudeSettings == nullptr )
353  {
354  throw std::runtime_error( "Error when creating body-fixed thrust direction of type thrust_magnitude_from_dependent_variables, input is inconsistent" );
355  }
356  else
357  {
358  thrustDirectionFunction = [ = ]( ){ return fromFunctionThrustMagnitudeSettings->bodyFixedThrustDirection_; };
359  }
360  break;
361  }
362  default:
363  throw std::runtime_error( "Error when creating body-fixed thrust direction, type not identified" );
364  }
365  return thrustDirectionFunction;
366 }

369 std::shared_ptr< propulsion::ThrustMagnitudeWrapper > createThrustMagnitudeWrapper(
370  const std::shared_ptr< ThrustMagnitudeSettings > thrustMagnitudeSettings,
371  const NamedBodyMap& bodyMap,
372  const std::string& nameOfBodyWithGuidance,
373  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >& magnitudeUpdateSettings )
374 {
375  std::shared_ptr< propulsion::ThrustMagnitudeWrapper > thrustMagnitudeWrapper;
376 
377  // Identify magnitude settings type
378  switch( thrustMagnitudeSettings->thrustMagnitudeGuidanceType_ )
379  {
380  case constant_thrust_magnitude:
381  {
382  // Check input consistency
383  std::shared_ptr< ConstantThrustMagnitudeSettings > constantThrustMagnitudeSettings =
384  std::dynamic_pointer_cast< ConstantThrustMagnitudeSettings >( thrustMagnitudeSettings );
385  if( constantThrustMagnitudeSettings == nullptr )
386  {
387  throw std::runtime_error( "Error when creating constant thrust magnitude wrapper, input is inconsistent" );
388  }
389 
390  thrustMagnitudeWrapper = std::make_shared< propulsion::CustomThrustMagnitudeWrapper >(
391  [ = ]( const double ){ return constantThrustMagnitudeSettings->thrustMagnitude_; },
392  [ = ]( const double ){ return constantThrustMagnitudeSettings->specificImpulse_; } );
393  break;
394 
395  }
396  case from_engine_properties_thrust_magnitude:
397  {
398  // Check input consistency
399  std::shared_ptr< FromBodyThrustMagnitudeSettings > fromEngineThrustMagnitudeSettings =
400  std::dynamic_pointer_cast< FromBodyThrustMagnitudeSettings >( thrustMagnitudeSettings );
401  if( fromEngineThrustMagnitudeSettings == nullptr )
402  {
403  throw std::runtime_error( "Error when creating from-engine thrust magnitude wrapper, input is inconsistent" );
404  }
405  if( bodyMap.at( nameOfBodyWithGuidance )->getVehicleSystems( ) == nullptr )
406  {
407  throw std::runtime_error( "Error when creating from-engine thrust magnitude wrapper, no vehicle systems found" );
408 
409  }
410 
411  // Retrieve single engine thrust
412  if( fromEngineThrustMagnitudeSettings->useAllEngines_ == false )
413  {
414  if( ( bodyMap.at( nameOfBodyWithGuidance )->getVehicleSystems( )->getEngineModels( ).count(
415  thrustMagnitudeSettings->thrustOriginId_ ) == 0 ) )
416  {
417  throw std::runtime_error( "Error when creating from-engine thrust magnitude wrapper, no engine of right ID found" );
418  }
419  else
420  {
421  thrustMagnitudeWrapper = std::make_shared< propulsion::ThrustMagnitudeFromEngineWrapper >(
422  bodyMap.at( nameOfBodyWithGuidance )->getVehicleSystems( )->getEngineModels( ).at(
423  thrustMagnitudeSettings->thrustOriginId_ ) );
424  }
425  }
426  // Retrieve total engine thrust
427  else
428  {
429  if( ( bodyMap.at( nameOfBodyWithGuidance )->getVehicleSystems( )->getEngineModels( ).size( ) == 0 ) )
430  {
431  std::cerr << "Warning when creating from-engine thrust magnitude wrapper for all engines; no engines found: returning 0 thrust" << std::endl;
432  }
433  thrustMagnitudeWrapper = std::make_shared< propulsion::ThrustMagnitudeFromEngineWrapper >(
434  utilities::createVectorFromMapValues< std::shared_ptr< system_models::EngineModel >, std::string >(
435  bodyMap.at( nameOfBodyWithGuidance )->getVehicleSystems( )->getEngineModels( ) ));
436  }
437  break;
438 
439  }
440  case thrust_magnitude_from_time_function:
441  {
442  // Check input consistency
443  std::shared_ptr< FromFunctionThrustMagnitudeSettings > fromFunctionThrustMagnitudeSettings =
444  std::dynamic_pointer_cast< FromFunctionThrustMagnitudeSettings >( thrustMagnitudeSettings );
445  if( fromFunctionThrustMagnitudeSettings == nullptr )
446  {
447  throw std::runtime_error( "Error when creating from-function thrust magnitude wrapper, input is inconsistent" );
448  }
449  thrustMagnitudeWrapper = std::make_shared< propulsion::CustomThrustMagnitudeWrapper >(
450  fromFunctionThrustMagnitudeSettings->thrustMagnitudeFunction_,
451  fromFunctionThrustMagnitudeSettings->specificImpulseFunction_,
452  fromFunctionThrustMagnitudeSettings->isEngineOnFunction_,
453  fromFunctionThrustMagnitudeSettings->customThrustResetFunction_ );
454  break;
455 
456  }
457  case thrust_magnitude_from_dependent_variables:
458  {
459  // Check input consistency
460  std::shared_ptr< ParameterizedThrustMagnitudeSettings > parameterizedThrustMagnitudeSettings =
461  std::dynamic_pointer_cast< ParameterizedThrustMagnitudeSettings >( thrustMagnitudeSettings );
462  if( parameterizedThrustMagnitudeSettings == nullptr )
463  {
464  throw std::runtime_error( "Error when creating from-function thrust magnitude wrapper, input is inconsistent" );
465  }
466 
467  // Create indpendent variable functions
468  std::vector< std::function< double( ) > > thrustInputVariableFunctions =
469  getPropulsionInputVariables(
470  bodyMap.at( nameOfBodyWithGuidance ), parameterizedThrustMagnitudeSettings->thrustIndependentVariables_,
471  parameterizedThrustMagnitudeSettings->thrustGuidanceInputVariables_ );
472  std::vector< std::function< double( ) > > specificInputVariableFunctions =
473  getPropulsionInputVariables(
474  bodyMap.at( nameOfBodyWithGuidance ), parameterizedThrustMagnitudeSettings->specificImpulseDependentVariables_,
475  parameterizedThrustMagnitudeSettings->specificImpulseGuidanceInputVariables_ );
476 
477  // Create thrust magnitude wrapper
478  thrustMagnitudeWrapper = std::make_shared< propulsion::ParameterizedThrustMagnitudeWrapper >(
479  parameterizedThrustMagnitudeSettings->thrustMagnitudeFunction_,
480  parameterizedThrustMagnitudeSettings->specificImpulseFunction_,
481  thrustInputVariableFunctions,
482  specificInputVariableFunctions,
483  parameterizedThrustMagnitudeSettings->thrustIndependentVariables_,
484  parameterizedThrustMagnitudeSettings->specificImpulseDependentVariables_,
485  parameterizedThrustMagnitudeSettings->inputUpdateFunction_ );
486 
487  break;
488 
489  }
490  default:
491  throw std::runtime_error( "Error when creating thrust magnitude wrapper, type not identified" );
492  }
493 
494  return thrustMagnitudeWrapper;
495 }
