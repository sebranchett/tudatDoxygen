32 Eigen::Vector6d ApproximatePlanetPositionsCircularCoplanar::
33 getCartesianState( const double secondsSinceEpoch )
34 {
35  // Set Julian date.
36  julianDate_ = basic_astrodynamics::convertSecondsSinceEpochToJulianDay(
37  secondsSinceEpoch, referenceJulianDate_ );
38 
39  // Compute number of centuries past J2000.
40  numberOfCenturiesPastJ2000_ = ( julianDate_ - 2451545.0 ) / 36525.0;
41 
42  // Compute mean longitude of planet at given Julian date.
43  meanLongitudeAtGivenJulianDate_ = approximatePlanetPositionsDataContainer_.meanLongitude_
44  + ( approximatePlanetPositionsDataContainer_.rateOfChangeOfMeanLongitude_
45  * numberOfCenturiesPastJ2000_ );
46 
47  // Convert mean longitude at given Julian date from degrees to radians.
48  meanLongitudeAtGivenJulianDate_ =
49  unit_conversions::convertDegreesToRadians(
50  meanLongitudeAtGivenJulianDate_);
51 
52  // Get semi-major axis at J2000 and assume constant radius of circular orbit.
53  constantOrbitalRadius_ =
54  unit_conversions::convertAstronomicalUnitsToMeters(
55  approximatePlanetPositionsDataContainer_.semiMajorAxis_ );
56 
57  // Convert to Cartesian position.
58  Eigen::VectorXd planetCartesianStateAtGivenJulianDate( 6 );
59  planetCartesianStateAtGivenJulianDate.segment( 0, 3 )
60  = coordinate_conversions::convertSphericalToCartesian(
61  Eigen::Vector3d( constantOrbitalRadius_,
62  0.5 * mathematical_constants::PI,
63  meanLongitudeAtGivenJulianDate_ ) );
64 
65  // Compute orbital velocity.
66  double circularOrbitalVelocity = std::sqrt( sunGravitationalParameter /
67  constantOrbitalRadius_ );
68 
69  // Convert to Cartesian velocity.
70  planetCartesianStateAtGivenJulianDate( 3 ) = -sin( meanLongitudeAtGivenJulianDate_ ) *
71  circularOrbitalVelocity;
72  planetCartesianStateAtGivenJulianDate( 4 ) = cos( meanLongitudeAtGivenJulianDate_ ) *
73  circularOrbitalVelocity;
74  planetCartesianStateAtGivenJulianDate( 5 ) = 0.0;
75 
76  // Return Cartesian state of planet at given Julian date.
77  return planetCartesianStateAtGivenJulianDate;
78 }
