38 NRLMSISE00Input nrlmsiseInputFunction( const double altitude, const double longitude,
39  const double latitude, const double time,
40  const tudat::input_output::solar_activity::SolarActivityDataMap& solarActivityMap,
41  const bool adjustSolarTime,
42  const double localSolarTime ) {

44 
45  // Declare input data class member
46  NRLMSISE00Input nrlmsiseInputData;
47 
48  // Julian dates
49  double julianDate = tudat::basic_astrodynamics::convertSecondsSinceEpochToJulianDay(
50  time, basic_astrodynamics::JULIAN_DAY_ON_J2000 );
51  double julianDay = std::floor( julianDate - 0.5 ) + 0.5;
52 
53  // Check if solar activity is found for current day.
54  if( solarActivityMap.count( julianDay ) == 0 )
55  {
56  std::string errorMessage = "Solar activity data could not be found for this julian date: "
57  + boost::lexical_cast< std::string >( julianDay ) + " in nrlmsiseInputFunction";
58  throw std::runtime_error( errorMessage );
59  }
60  SolarActivityDataPtr solarActivity = solarActivityMap.at( julianDay );
61 
62 
63  // Compute julian date at the first of januari
64  double julianDate1Jan = tudat::basic_astrodynamics::convertCalendarDateToJulianDay(
65  solarActivity->year, 1, 1, 0, 0, 0.0 );
66 
67  nrlmsiseInputData.year = solarActivity->year; // int
68  nrlmsiseInputData.dayOfTheYear = julianDay - julianDate1Jan + 1;
69  nrlmsiseInputData.secondOfTheDay = time -
70  tudat::basic_astrodynamics::convertJulianDayToSecondsSinceEpoch( julianDay,
71  tudat::basic_astrodynamics::JULIAN_DAY_ON_J2000 );
72 
73  if( solarActivity->fluxQualifier == 1 )
74  { // requires adjustment
75  nrlmsiseInputData.f107 = solarActivity->solarRadioFlux107Adjusted;
76  nrlmsiseInputData.f107a = solarActivity->centered81DaySolarRadioFlux107Adjusted;
77  }
78  else
79  { // no adjustment required
80  nrlmsiseInputData.f107 = solarActivity->solarRadioFlux107Observed;
81  nrlmsiseInputData.f107a = solarActivity->centered81DaySolarRadioFlux107Observed;
82  }
83  nrlmsiseInputData.apDaily = solarActivity->planetaryEquivalentAmplitudeAverage;
84  nrlmsiseInputData.apVector = eigenToStlVector( solarActivity->planetaryEquivalentAmplitudeVector );
85 
86  // Compute local solar time
87  // Hrs since begin of the day at longitude 0 (GMT) + Hrs passed at current longitude
88  if( adjustSolarTime )
89  {
90  nrlmsiseInputData.localSolarTime = localSolarTime;
91  }
92  else
93  {
94  nrlmsiseInputData.localSolarTime = nrlmsiseInputData.secondOfTheDay / 3600.0
95  + longitude / ( tudat::mathematical_constants::PI / 12.0 );
96  }
97 
98  return nrlmsiseInputData;
99 }
