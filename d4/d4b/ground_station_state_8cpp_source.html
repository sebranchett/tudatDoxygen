103 Eigen::Quaterniond getRotationQuaternionFromBodyFixedToTopocentricFrame(
104  const std::shared_ptr< basic_astrodynamics::BodyShapeModel > bodyShapeModel,
105  const double geocentricLatitude,
106  const double geocentricLongitude,
107  const Eigen::Vector3d localPoint )
108 {
109  // Declare unit vectors of topocentric frame, to be calculated.
110  std::vector< Eigen::Vector3d > topocentricUnitVectors;
111 
112  bool isSurfaceModelRecognized = 1;
113 
114  // Identify type of body shape model
115  if( std::dynamic_pointer_cast< basic_astrodynamics::SphericalBodyShapeModel >( bodyShapeModel ) != nullptr )
116  {
117  // For a sphere the topocentric and geocentric frames are equal.
118  topocentricUnitVectors = getGeocentricLocalUnitVectors(
119  geocentricLatitude, geocentricLongitude );
120  }
121  else if( std::dynamic_pointer_cast< basic_astrodynamics::OblateSpheroidBodyShapeModel >( bodyShapeModel ) != nullptr )
122  {
123  std::shared_ptr< basic_astrodynamics::OblateSpheroidBodyShapeModel > oblateSphericalShapeModel =
124  std::dynamic_pointer_cast< basic_astrodynamics::OblateSpheroidBodyShapeModel >( bodyShapeModel );
125 
126  // Calculate geodetic latitude.
127  double flattening = oblateSphericalShapeModel->getFlattening( );
128  double equatorialRadius = oblateSphericalShapeModel->getEquatorialRadius( );
129  double geodeticLatitude = coordinate_conversions::calculateGeodeticLatitude(
130  localPoint, equatorialRadius, flattening, 1.0E-4 );
131 
132  // Calculte unit vectors of topocentric frame.
133  topocentricUnitVectors = getGeocentricLocalUnitVectors( geodeticLatitude, geocentricLongitude );
134  }
135  else
136  {
137  isSurfaceModelRecognized = 0;
138  throw std::runtime_error( "Error when making transformation to topocentric frame, shape model not recognized" );
139  }
140 
141  // Create rotation matrix
142 
143  Eigen::Matrix3d bodyFixedToTopocentricFrame;
144 
145  if( isSurfaceModelRecognized == 1 )
146  {
147  bodyFixedToTopocentricFrame.block( 0, 0, 1, 3 ) = topocentricUnitVectors[ 0 ].transpose( );
148  bodyFixedToTopocentricFrame.block( 1, 0, 1, 3 ) = topocentricUnitVectors[ 1 ].transpose( );
149  bodyFixedToTopocentricFrame.block( 2, 0, 1, 3 ) = topocentricUnitVectors[ 2 ].transpose( );
150  }
151  else
152  {
153  bodyFixedToTopocentricFrame = Eigen::Matrix3d::Identity( );
154  }
155 
156 
157 
158  // Convert to quaternion and return.
159  return Eigen::Quaterniond( bodyFixedToTopocentricFrame );
160 }
