66 double LambertTargeterIzzo::getTransverseVelocityAtDeparture( )

109 double LambertTargeterIzzo::getSemiMajorAxis( )
110 {
111  // Compute specific orbital energy: eps = v^2/ - mu/r.
112  const double specificOrbitalEnergy = cartesianVelocityAtDeparture.squaredNorm( ) / 2.0
113  - gravitationalParameter / cartesianPositionAtDeparture.norm( );
114 
115  // Compute semi-major axis: a = -mu / 2*eps.
116  return -gravitationalParameter / ( 2.0 * specificOrbitalEnergy );
117 }
