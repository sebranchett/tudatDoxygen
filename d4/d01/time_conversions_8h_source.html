172 template< typename TimeScalarType = double >
173 TimeScalarType convertCalendarDateToJulianDaysSinceEpoch( const int calendarYear,
174  const int calendarMonth,
175  const int calendarDay,
176  const int calendarHour,
177  const int calendarMinutes,
178  const TimeScalarType calendarSeconds,
179  const TimeScalarType referenceJulianDay )
180 {
181  // Calculate julian day of calendar date.
182  TimeScalarType julianDay =
183  static_cast< TimeScalarType >( boost::gregorian::date( calendarYear, calendarMonth, calendarDay ).
184  julian_day( ) ) - referenceJulianDay;
185 
186  //Compute day fraction
187  const TimeScalarType dayFraction =
188  static_cast< TimeScalarType >( calendarHour ) /
189  mathematical_constants::getFloatingInteger< TimeScalarType >( 24 ) +
190  static_cast< TimeScalarType >( calendarMinutes ) /
191  mathematical_constants::getFloatingInteger< TimeScalarType >( 24 * 60 ) +
192  calendarSeconds / mathematical_constants::getFloatingInteger< TimeScalarType >( 24 * 3600 );
193 
194 
195  // Compute Julian day by adding day fraction and subtracting 0.5 to reference to midnight instead of noon..
196  return julianDay + dayFraction - mathematical_constants::getFloatingFraction< TimeScalarType >( 1, 2 );
197 }
