25 void LawgsPartGeometry::setMesh( boost::shared_ptr< SingleSurfaceGeometry > originalSurface,
26  int numberOfLinesIn, int numberOfPointsIn )
27 {
28  // Set (temporary name) of part.
29  name_ = "copied surface";
30 
31  // Set size of mesh.
32  numberOfLines_ = numberOfLinesIn;
33  numberOfPoints_ = numberOfPointsIn;
34 
35  // Allocate mesh points.
36  meshPoints_.resize( boost::extents[ numberOfLines_ ][ numberOfPoints_ ] );
37 
38  // Set grid sizes from requested number of sample points.
39  double independentVariableGridSize1 =
40  ( originalSurface->getMaximumIndependentVariable( 1 ) -
41  originalSurface->getMinimumIndependentVariable( 1 ) ) /
42  ( static_cast< double >( numberOfLines_ - 1 ) );
43 
44  double independentVariableGridSize2 =
45  ( originalSurface->getMaximumIndependentVariable( 2 ) -
46  originalSurface->getMinimumIndependentVariable( 2 ) ) /
47  ( static_cast< double >( numberOfPoints_ - 1 ) );
48 
49  // Declare sample point variables.
50  double variable1, variable2;
51 
52  // Declare and set minimum values of independent variables.
53  double minimumIndependentVariable1 = originalSurface->getMinimumIndependentVariable( 1 );
54  double minimumIndependentVariable2 = originalSurface->getMinimumIndependentVariable( 2 );
55 
56  // Loop through the number of lines and points specified and sample
57  // geometry at fixed intervals.
58  for ( int i = 0; i < numberOfLines_; i++ )
59  {
60  for ( int j = 0; j < numberOfPoints_; j++ )
61  {
62  // Set sampling point of original geometry.
63  variable1 = minimumIndependentVariable1 + i * independentVariableGridSize1;
64  variable2 = minimumIndependentVariable2 + j * independentVariableGridSize2;
65 
66  // Set new mesh point.
67  meshPoints_[ i ][ j ] = originalSurface->getSurfacePoint( variable1, variable2 );
68  }
69  }
70 
71  // Perform panel calculations for mesh.
72  performPanelCalculations( );
73 }

93 Eigen::VectorXd LawgsPartGeometry::getSurfacePoint( const double independentVariable1,
94  const double independentVariable2 )
95 {
96  // Declare local variables denoting 'start' of panel.
97  int pointIndex, lineIndex;
98 
99  // Declare local variable denoting surface point.
100  Eigen::VectorXd point = Eigen::VectorXd( 3 );
101 
102  // Set local variables.
103  pointIndex = static_cast< int > ( floor( independentVariable2 ) );
104  lineIndex = static_cast< int > ( floor( independentVariable1 ) );
105 
106  // Start with panel centroid.
107  point = panelCentroids_[ lineIndex ][ pointIndex ];
108 
109  // Move back to panel corner.
110  point -= 0.5 *( meshPoints_[ lineIndex + 1 ][ pointIndex + 1 ] -
111  meshPoints_[ lineIndex ][ pointIndex ]);
112 
113  // Add contribution of 1st independent variable on panel.
114  point += ( independentVariable1 - lineIndex ) *
115  ( meshPoints_[ lineIndex + 1 ][ pointIndex ]
116  - meshPoints_[ lineIndex ][ pointIndex ] );
117 
118  // Add contribution of 2nd independent variable on panel.
119  point += ( independentVariable2 - pointIndex ) *
120  ( meshPoints_[ lineIndex ][ pointIndex + 1 ] -
121  meshPoints_[ lineIndex ][ pointIndex ] );
122 
123  return point;
124 }
