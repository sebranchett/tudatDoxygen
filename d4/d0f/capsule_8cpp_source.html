39 Capsule::Capsule( const double noseRadius,
40  const double middleRadius,
41  const double rearLength,
42  const double rearAngle,
43  const double sideRadius )
44 {
45  using std::sin;
46  using std::cos;
47  using mathematical_constants::PI;
48 
49  // Call set functions for number of single and composite surface geometries
50  // with predetermined values.
51  setNumberOfCompositeSurfaceGeometries( 0 );
52  setNumberOfSingleSurfaceGeometries( 4 );
53 
54  // Set member shape variables
55  noseRadius_ = noseRadius;
56  middleRadius_ = middleRadius;
57  rearLength_ = rearLength;
58  rearAngle_ = rearAngle;
59  sideRadius_ = sideRadius;
60 
61  // Determine and set extent of spherical nose part.
62  double noseSphereAngle_ = asin( ( middleRadius_ - sideRadius_ )
63  / ( noseRadius_ - sideRadius_ ) );
64 
65  // Create nose sphere.
66  boost::shared_ptr< SphereSegment > noseSphere_ = boost::make_shared< SphereSegment >(
67  noseRadius_, 0, 2 * PI, 0, noseSphereAngle_ );
68 
69  // Declare translation vector.
70  Eigen::VectorXd translationVector_ = Eigen::VectorXd( 3 );
71  translationVector_( 2 ) = 0.0;
72  translationVector_( 1 ) = 0.0;
73  translationVector_( 0 ) = - noseRadius_ * cos( noseSphereAngle_ );
74 
75  // Set nose translation vector.
76  noseSphere_->setOffset( translationVector_ );
77 
78  // Set noseSphere_ in singleSurfaceList_.
79  setSingleSurfaceGeometry( noseSphere_, 0 );
80 
81  // Create rear cone, fully revolved.
82  boost::shared_ptr< ConicalFrustum > cone_ = boost::make_shared< ConicalFrustum >(
83  rearAngle_, middleRadius_ - sideRadius_ * ( 1.0 - cos( rearAngle_ ) ),
84  rearLength_ );
85 
86  // Set translation vector of cone.
87  translationVector_( 0 ) = -sideRadius_ * ( sin( PI / 2.0 - noseSphereAngle_ )
88  + sin ( -rearAngle_ ) );
89  cone_->setOffset( translationVector_ );
90 
91  // Set cone in singleSurfaceList_.
92  setSingleSurfaceGeometry( cone_, 1 );
93 
94  // Calculate end radius of cone.
95  double endRadius_ = cone_->getStartRadius( ) + rearLength_ * tan( rearAngle_ );
96 
97  // Calculate rear sphere radius.
98  double rearNoseRadius_ = endRadius_ / cos( -rearAngle_ );
99 
100  // Create rear sphere ( "end cap" ), fully revolved.
101  boost::shared_ptr< SphereSegment > rearSphere_ = boost::make_shared< SphereSegment >(
102  rearNoseRadius_, 0.0, 2.0 * PI, PI / 2.0 - rearAngle_, PI );
103 
104  // Set translation vector of rear sphere.
105  translationVector_( 0 ) = ( rearNoseRadius_ * sin( -rearAngle_ ) ) - rearLength_
106  - ( sideRadius_ * ( sin( PI / 2.0 - noseSphereAngle_ ) + sin ( -rearAngle_ ) ) );
107  rearSphere_->setOffset( translationVector_ );
108  setSingleSurfaceGeometry( rearSphere_, 2 );
109 
110  // Create torus section of capsule.
111  double torusMajorRadius_ = ( noseRadius_ - sideRadius_ ) * sin( noseSphereAngle_ );
112  boost::shared_ptr< Torus > torus_ = boost::make_shared< Torus >(
113  torusMajorRadius_, sideRadius_, 0.0, 2.0 * PI, PI / 2.0 - noseSphereAngle_, rearAngle_ );
114 
115  // Set translation vector of rear sphere.
116  translationVector_( 0 ) = -cos( noseSphereAngle_ ) * sideRadius_;
117  torus_->setOffset( translationVector_ );
118  setSingleSurfaceGeometry( torus_, 3 );
119 
120  // Set rotation matrix fo each part to be compatible with flow direction in
121  // aerodynamic analysis.
122  Eigen::MatrixXd rotationMatrix = Eigen::MatrixXd( 3, 3 );
123  double angle_ = PI / 2.0;
124  rotationMatrix( 0, 0 ) = cos( angle_ );
125  rotationMatrix( 0, 1 ) = 0.0;
126  rotationMatrix( 0, 2 ) = sin( angle_ );
127  rotationMatrix( 1, 0 ) = 0.0;
128  rotationMatrix( 1, 1 ) = 1.0;
129  rotationMatrix( 1, 2 ) = 0.0;
130  rotationMatrix( 2, 0 ) = -sin( angle_ );
131  rotationMatrix( 2, 1 ) = 0.0;
132  rotationMatrix( 2, 2 ) = cos( angle_ );
133 
134  // Set rotation matrix for single surface geometries.
135  for ( unsigned i = 0; i < numberOfSingleSurfaceGeometries_ ; i++ )
136  {
137  singleSurfaceGeometryList_[ i ]->setRotationMatrix( rotationMatrix );
138  }
139 }
