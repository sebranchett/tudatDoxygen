43  EmpiricalAccelerationCoefficientsParameter(
44  const std::shared_ptr< basic_astrodynamics::EmpiricalAcceleration > empiricalAcceleration,
45  const std::string& associatedBody,
46  const std::map< basic_astrodynamics::EmpiricalAccelerationComponents,
47  std::vector< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes > > componentsToEstimate ):
48  EstimatableParameter< Eigen::VectorXd >( empirical_acceleration_coefficients, associatedBody ),
49  empiricalAcceleration_( empiricalAcceleration )
50  {
51  parameterSize_ = 0;
52 
53  // Set components of empirical accelerations that are to be estimated
54  for( std::map< basic_astrodynamics::EmpiricalAccelerationComponents,
55  std::vector< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes > >::
56  const_iterator inputComponentIterator = componentsToEstimate.begin( );
57  inputComponentIterator != componentsToEstimate.end( );
58  inputComponentIterator++ )
59  {
60  int currentComponentIndex = static_cast< int >( inputComponentIterator->first );
61 
62  // Iterate over all components
63  for( unsigned int i = 0; i < inputComponentIterator->second.size( ); i++ )
64  {
65  // Check feasibility of input
66  if( accelerationIndices_.count( inputComponentIterator->second.at( i ) ) > 0 )
67  {
68  std::vector< int > currentIndices = accelerationIndices_.at( inputComponentIterator->second.at( i ) );
69  if( std::find( currentIndices.begin( ), currentIndices.end( ), currentComponentIndex ) !=
70  currentIndices.end( ) )
71  {
72  throw std::runtime_error(
73  "Error when creating empirical acceleration parameter object, found duplicate of component." );
74  }
75  }
76  accelerationIndices_[ inputComponentIterator->second.at( i ) ].push_back( currentComponentIndex );
77  parameterSize_++;
78  }
79  }

91  Eigen::VectorXd getParameterValue( )
92  {
93  Eigen::VectorXd parameter = Eigen::VectorXd::Zero( parameterSize_ );
94  int currentIndex = 0;
95 
96  // Iterate over all components and retrieve required values
97  Eigen::Matrix3d accelerationComponents = empiricalAcceleration_->getAccelerationComponents( );
98  for( std::map< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes, std::vector< int > >::const_iterator
99  componentIterator = accelerationIndices_.begin( );
100  componentIterator != accelerationIndices_.end( ); componentIterator++ )
101  {
102  Eigen::Vector3d accelerations =
103  accelerationComponents.block( 0, static_cast< int >( componentIterator->first ), 3, 1 );
104 
105  for( unsigned int i = 0; i < componentIterator->second.size( ); i++ )
106  {
107  parameter( currentIndex ) = accelerations( componentIterator->second.at( i ) );
108  currentIndex++;
109  }
110  }
111 
112  // Check consistency
113  if( currentIndex != parameterSize_ )
114  {
115  throw std::runtime_error( "Error when getting empirical parameter; inconsistent sizes found." );
116  }
117 
118  return parameter;

126  void setParameterValue( Eigen::VectorXd parameterValue )
127  {
128  int currentIndex = 0;
129 
130  // Iterate over all components and set required values
131  Eigen::Matrix3d accelerationComponents = empiricalAcceleration_->getAccelerationComponents( );
132  for( std::map< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes, std::vector< int > >::const_iterator
133  componentIterator = accelerationIndices_.begin( );
134  componentIterator != accelerationIndices_.end( ); componentIterator++ )
135  {
136  Eigen::Vector3d accelerations =
137  accelerationComponents.block( 0, static_cast< int >( componentIterator->first ), 3, 1 );
138  for( unsigned int i = 0; i < componentIterator->second.size( ); i++ )
139  {
140  accelerations( componentIterator->second.at( i ) ) = parameterValue( currentIndex );
141  currentIndex++;
142  }
143  accelerationComponents.block( 0, static_cast< int >( componentIterator->first ), 3, 1 ) = accelerations;
144  }
145 
146  // Reset components in acceleration model
147  empiricalAcceleration_->resetAccelerationComponents( accelerationComponents );
148 
149  if( currentIndex != parameterSize_ )
150  {
151  throw std::runtime_error(
152  "Error when getting empirical parameter size; inconsistent sizes found." );
153  }
154  }
