120  int findNearestLowerNeighbour( const IndependentVariableType valueToLookup )
121  {
122  // Initialize return value.
123  int newNearestLowerIndex = 0;
124 
125  // If this is first call of function, use binary search.
126  if ( !isFirstLookupDone )
127  {
128  newNearestLowerIndex = basic_mathematics::computeNearestLeftNeighborUsingBinarySearch
129  < IndependentVariableType >( independentVariableValues_, valueToLookup );
130  isFirstLookupDone = 1;
131  }
132 
133  else
134  {
135  // If requested value is in same interval, return same value as previous time.
136  if ( basic_mathematics::isIndependentVariableInInterval< IndependentVariableType >
137  ( previousNearestLowerIndex_, valueToLookup, independentVariableValues_ ) )
138  {
139  newNearestLowerIndex = previousNearestLowerIndex_;
140  }
141 
142  // Otherwise, perform hunting algorithm.
143  else
144  {
145  newNearestLowerIndex =
146  basic_mathematics::findNearestLeftNeighbourUsingHuntingAlgorithm<
147  IndependentVariableType >
148  ( valueToLookup, previousNearestLowerIndex_, independentVariableValues_ );
149  }
150  }
151 
152  // Set calculated value for use in next call.
153  previousNearestLowerIndex_ = newNearestLowerIndex;
154 
155  return newNearestLowerIndex;
156  }
