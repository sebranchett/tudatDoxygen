34 Eigen::MatrixXd readMatrixFromFile( const std::string& relativePath, const std::string& separators,
35  const std::string& skipLinesCharacter )
36 {
37  // Open input and output.
38  std::fstream file( relativePath.c_str( ), std::ios::in );
39  if ( file.fail( ) )
40  {
41  throw std::runtime_error( "Data file could not be opened: " + relativePath );
42  }
43 
44  std::stringstream filteredStream( std::ios::in | std::ios::out );
45  {
46  // Filter the file stream. This needs to be in its own scope, because filtering_stream::
47  // flush( ) does not work if the underlying end point is a stringstream, so the flush has
48  // to be forced by letting the filtering_stream go out of scope.
49  boost::iostreams::filtering_ostream filterProcessor;
50  for ( unsigned int i = 0; i < skipLinesCharacter.size( ); i++ )
51  {
52  // Remove all comments from the stream.
53  filterProcessor.push( input_output::stream_filters::RemoveComment(
54  skipLinesCharacter[ i ], true ) );
55  }
56 
57  // Add the output to the filter chain.
58  filterProcessor.push( filteredStream );
59 
60  // Copy the input to the filter.
61  boost::iostreams::copy( file, filterProcessor );
62  }
63 
64  // Seek stream back to start.
65  filteredStream.seekg( 0, std::ios::beg );
66 
67  // Read the filtered stream into lines.
68  std::vector< std::string > lines_;
69  while ( !filteredStream.eof( ) )
70  {
71  std::string line_;
72  getline( filteredStream, line_ );
73  if ( !line_.empty( ) )
74  {
75  boost::trim_all( line_ );
76  lines_.push_back( line_ );
77  }
78  }
79 
80  // If there are no lines, return an empty matrix.
81  if ( lines_.empty( ) )
82  {
83  return Eigen::MatrixXd( );
84  }
85 
86  const std::string realSeparators = std::string( separators ) + " ";
87 
88  // Determine the number of columns from.
89  std::vector< std::string > lineSplit_;
90  boost::algorithm::split( lineSplit_, lines_[ 0 ], boost::is_any_of( realSeparators ),
91  boost::algorithm::token_compress_on );
92  const unsigned int numberOfColumns = lineSplit_.size( );
93 
94  // Initialize the matrix with sizes obtained from the number of lines and the entries in the
95  // first line.
96  Eigen::MatrixXd dataMatrix_( lines_.size( ), numberOfColumns );
97  for ( int rowIndex = 0; rowIndex < dataMatrix_.rows( ); rowIndex++ )
98  {
99  lineSplit_.clear( );
100 
101  // Read current line and split into separate entries.
102  boost::algorithm::split( lineSplit_, lines_[ rowIndex ],
103  boost::is_any_of( realSeparators ),
104  boost::algorithm::token_compress_on );
105 
106  // Check if number of column entries in line matches the number of columns in the matrix.
107  // If not, throw a runtime error.
108  if ( lineSplit_.size( ) != numberOfColumns )
109  {
110  throw std::runtime_error(
111  "Number of columns in row " + std::to_string( rowIndex ) + " is " +
112  std::to_string( lineSplit_.size( ) ) + " should be " +
113  std::to_string( numberOfColumns ) );
114  }
115 
116  // Put single line entries into matrix as doubles.
117  for ( int columnIndex = 0; columnIndex < dataMatrix_.cols( ); columnIndex++ )
118  {
119  boost::trim( lineSplit_.at( columnIndex ) );
120  dataMatrix_( rowIndex, columnIndex ) =
121  std::stod( lineSplit_.at( columnIndex ) );
122  }
123  }
124 
125  return dataMatrix_;
126 }
