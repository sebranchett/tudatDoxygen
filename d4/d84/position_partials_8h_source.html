317  const std::shared_ptr< PositionObservationScaling > positionObservationScaler,
318  const std::map< observation_models::LinkEndType, std::shared_ptr< CartesianStatePartial > >& positionPartialList,
319  const estimatable_parameters::EstimatebleParameterIdentifier parameterIdentifier ):

338  virtual PositionObservationPartialReturnType calculatePartial(
339  const std::vector< Eigen::Vector6d >& states,
340  const std::vector< double >& times,
341  const observation_models::LinkEndType linkEndOfFixedTime,
342  const Eigen::Vector3d& currentObservation = Eigen::Vector3d::Constant( TUDAT_NAN ) )
343  {
344  PositionObservationPartialReturnType returnPartial;
345 
346 
347  // Iterate over all link ends.
348  for( positionPartialIterator_ = positionPartialList_.begin( );
349  positionPartialIterator_ != positionPartialList_.end( );
350  positionPartialIterator_++ )
351  {
352  // Retrieve link end time and state
353  if( positionPartialIterator_->first == observation_models::observed_body )
354  {
355  currentState_ = states[ 0 ];
356  currentTime_ = times[ 0 ];
357  }
358  else
359  {
360  throw std::runtime_error(
361  "Error when calculating position observation partial, invalid link end type requested" );
362  }
363 
364  // Get position partial and scale with associated term.
365  returnPartial.push_back(
366  std::make_pair(
367  positionObservationScaler_->getScalingFactor(
368  positionPartialIterator_->first ) *
369  ( positionPartialIterator_->second->calculatePartialOfPosition(
370  currentState_, currentTime_ ) ), currentTime_ ) );
371  }
372 
373  return returnPartial;
374  }
375 
376 protected:
377 
379  std::shared_ptr< PositionObservationScaling > positionObservationScaler_;
380 
382  std::map< observation_models::LinkEndType, std::shared_ptr< CartesianStatePartial > > positionPartialList_;
383 
385  std::map< observation_models::LinkEndType, std::shared_ptr< CartesianStatePartial > >::iterator positionPartialIterator_;
386 
387 
389  Eigen::Vector6d currentState_;
