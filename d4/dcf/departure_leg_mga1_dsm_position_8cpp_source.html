136 void DepartureLegMga1DsmPosition::calculateDsmLocation( )
137 {
138  // Calculate the required unit vectors
139  const Eigen::Vector3d unitVector1 = departureBodyPosition_ / departureBodyPosition_.norm( ) ;
140  const Eigen::Vector3d unitVector3 = unitVector1.cross( departureBodyVelocity_ ) /
141  ( unitVector1.cross( departureBodyVelocity_ ) ).norm( );
142  const Eigen::Vector3d unitVector2 = unitVector3.cross( unitVector1 );
143 
144  // Calculate the absolute DSM radius.
145  const double absoluteRadiusDsm = dimensionlessRadiusDsm_ * departureBodyPosition_.norm( );
146 
147  // Calculate the radius in the central body reference frame.
148  dsmLocation_ = std::cos( inPlaneAngle_ ) * std::cos( outOfPlaneAngle_ ) * absoluteRadiusDsm *
149  unitVector1 +
150  std::sin( inPlaneAngle_ ) * std::cos( outOfPlaneAngle_ ) * absoluteRadiusDsm *
151  unitVector2 +
152  std::sin( outOfPlaneAngle_ ) * absoluteRadiusDsm * unitVector3;
153 }
