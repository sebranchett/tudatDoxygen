28 Eigen::Vector6d StateDerivativeCircularRestrictedThreeBodyProblem::computeStateDerivative(
29  const double time, const Eigen::Vector6d& cartesianState )
30 {
31  using namespace orbital_element_conversions;
32 
33  TUDAT_UNUSED_PARAMETER( time );
34 
35  // Compute distance to primary body.
36  const double xCoordinateToPrimaryBodySquared =
37  ( cartesianState( xCartesianPositionIndex ) + massParameter )
38  * ( cartesianState( xCartesianPositionIndex ) + massParameter );
39 
40  const double yCoordinateSquared = cartesianState( yCartesianPositionIndex )
41  * cartesianState( yCartesianPositionIndex );
42 
43  const double zCoordinateSquared = cartesianState( zCartesianPositionIndex )
44  * cartesianState( zCartesianPositionIndex );
45 
46  const double normDistanceToPrimaryBodyCubed = pow(
47  xCoordinateToPrimaryBodySquared + yCoordinateSquared + zCoordinateSquared, 1.5 );
48 
49  // Compute distance to secondary body.
50  const double xCoordinateSecondaryBodySquared =
51  ( cartesianState( xCartesianPositionIndex ) - ( 1.0 - massParameter ) )
52  * ( cartesianState( xCartesianPositionIndex ) - ( 1.0 - massParameter ) );
53 
54  double normDistanceToSecondaryBodyCubed = pow(
55  xCoordinateSecondaryBodySquared + yCoordinateSquared + zCoordinateSquared, 1.5 );
56 
57  // Compute derivative of state.
58  Eigen::VectorXd stateDerivative( 6 );
59 
60  stateDerivative.segment( xCartesianPositionIndex, 3 ) = cartesianState.segment( xCartesianVelocityIndex, 3 );
61 
62  stateDerivative( 3 ) = cartesianState( xCartesianPositionIndex )
63  - ( ( 1.0 - massParameter ) / normDistanceToPrimaryBodyCubed )
64  * ( cartesianState( xCartesianPositionIndex ) + massParameter )
65  - ( massParameter / normDistanceToSecondaryBodyCubed )
66  * ( cartesianState( xCartesianPositionIndex ) - ( 1.0 - massParameter ) )
67  + 2.0 * cartesianState( yCartesianVelocityIndex );
68  stateDerivative( 4 ) = cartesianState( yCartesianPositionIndex )
69  * ( 1.0 - ( ( 1.0 - massParameter ) / normDistanceToPrimaryBodyCubed )
70  - ( massParameter / normDistanceToSecondaryBodyCubed ) )
71  - 2.0 * cartesianState( xCartesianVelocityIndex );
72  stateDerivative( 5 ) = -cartesianState( zCartesianPositionIndex )
73  * ( ( ( 1.0 - massParameter ) / normDistanceToPrimaryBodyCubed )
74  + ( massParameter / normDistanceToSecondaryBodyCubed ) );
75 
76  // Return computed state derivative.
77  return stateDerivative;
78 }
