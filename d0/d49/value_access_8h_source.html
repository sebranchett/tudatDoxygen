187 template< typename ValueType >
188 ValueType getValue( const nlohmann::json& jsonObject, const KeyPath& keyPath )
189 {
190  nlohmann::json rootObject;
191  nlohmann::json currentObject = jsonObject;
192  KeyPath currentKeyPath = keyPath;
193  KeyPath canonicalKeyPath = keyPath;
194 
195  if ( ! contains( currentKeyPath, SpecialKeys::keyPath ) )
196  {
197  canonicalKeyPath = currentKeyPath.canonical( getKeyPath( currentObject ) );
198  if ( ! contains( currentKeyPath, SpecialKeys::rootObject ) )
199  {
200  if ( isDefined( currentObject, SpecialKeys::rootObject ) )
201  {
202  rootObject = getValue< nlohmann::json >( currentObject, SpecialKeys::rootObject );
203  }
204 
205  if ( currentKeyPath.size( ) > 0 )
206  {
207  if ( currentKeyPath.front( ) == SpecialKeys::root || contains( currentKeyPath, SpecialKeys::up ) )
208  {
209  // Path is absolute or contains up-key
210 
211  // Use absolute key path
212  currentKeyPath = canonicalKeyPath;
213 
214  // Use jsonObject's root object instead
215  if ( rootObject.is_null( ) )
216  {
217  throw UndefinedKeyError( canonicalKeyPath );
218  }
219  currentObject = rootObject;
220  }
221  }
222  }
223  }
224 
225  if ( currentKeyPath.isAbsolute( ) )
226  {
227  // Remove "~"
228  currentKeyPath.erase( currentKeyPath.begin( ) );
229  }
230 
231  accessedKeyPaths.insert( canonicalKeyPath );
232 
233  try
234  {
235  currentObject = valueAt( currentObject, currentKeyPath );
236  }
237  catch ( ... )
238  {
239  // At least one of the keys was missing
240  throw UndefinedKeyError( canonicalKeyPath );
241  }
242 
243  if ( ! containsAnyOf( currentKeyPath, SpecialKeys::all ) )
244  {
245  // If jsonObject is an array containing objects, convert to object
246  convertToObjectIfContainsObjects( currentObject );
247 
248  if ( currentObject.is_object( ) )
249  {
250  // Define keys rootObject and keyPath of jsonObject to be returned
251  currentObject[ SpecialKeys::rootObject ] = rootObject.is_null( ) ? jsonObject : rootObject;
252  currentObject[ SpecialKeys::keyPath ] = canonicalKeyPath;
253  }
254  }
255 
256  try
257  {
258  // Try to convert to the requested type
259  const ValueType convertedObject = currentObject;
260  return convertedObject;
261  }
262  catch ( const nlohmann::detail::type_error& )
263  {
264  // Could not convert to the requested type
265  throw IllegalValueError( canonicalKeyPath, currentObject, typeid( ValueType ) );
266  }
267  catch ( const UnknownEnumError& )
268  {
269  // Could not convert string to enum
270  throw IllegalValueError( canonicalKeyPath, currentObject, typeid( ValueType ) );
271  }
272  catch ( const UndefinedKeyError& error )
273  {
274  // Some of the keys needed during creation of object of type ValueType were missing
275  throw error;
276  }
277  catch ( const IllegalValueError& error )
278  {
279  // Some of the values for the keys needed during creation of object of type ValueType were illegal
280  throw error;
281  }
282 }

344 template< typename ValueType >
345 ValueType getValue( const nlohmann::json& jsonObject, const std::vector< KeyPath >& keyPaths )
346 {
347  for ( unsigned int i = 0; i < keyPaths.size( ); ++i )
348  {
349  try
350  {
351  return getValue< nlohmann::json >( jsonObject, keyPaths.at( i ) );
352  }
353  catch ( const UndefinedKeyError& error )
354  {
355  error.rethrowIfNotTriggeredByMissingValueAt( keyPaths.at( i ) );
356  if ( i == keyPaths.size( ) - 1 )
357  {
358  throw error;
359  }
360  }
361  }
362  throw; // silence warning may reach end of non-void function
363 }
