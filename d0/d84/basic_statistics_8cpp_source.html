32 double computeStandardDeviationOfVectorComponents( const Eigen::VectorXd& vectorOfData )
33 {
34  // Compute average of components.
35  double averageOfComponents = computeAverageOfVectorComponents( vectorOfData );
36 
37  // Declare variance of components.
38  double varianceOfComponents = 0.0;
39 
40  // Compute variance of components.
41  for ( int i = 0 ; i < vectorOfData.rows( ) ; i++ )
42  {
43  varianceOfComponents += std::pow( ( vectorOfData( i ) - averageOfComponents ), 2.0 );
44  }
45 
46  varianceOfComponents /= static_cast< double >( vectorOfData.rows( ) - 1 );
47 
48  // Return square root of variance ( = standard deviation ).
49  return std::sqrt( varianceOfComponents );
50 }

70 double computeSampleVariance( const std::vector< double >& sampleData )
71 {
72  // Declare local variables.

74  double sampleMean = computeSampleMean( sampleData );
75 
76  // Declare and initialize sum of residuals squared.
77  double sumOfResidualsSquared = 0.0;
78 
79  // Compute sum of residuals of sample data squared.
80  for ( unsigned int i = 0; i < sampleData.size( ); i++ )

90 double computeSampleMedian( std::vector< double > sampleData )
91 {
92  // Sort data
93  std::sort( sampleData.begin() , sampleData.end());
94 
95  // Check if odd number of samples or even
96  double numberOfSamples = static_cast< double >( sampleData.size( ) );
97  int odd = static_cast< int >( basic_mathematics::computeModulo( numberOfSamples, 2.0 ) );
98 
99  // Calculate sample median
100  double sampleMedian;
101  if( odd == 0 ) // even
102  {
103  // 0 .. 99 (100) ->
104  unsigned int index = static_cast< unsigned int >( numberOfSamples / 2.0 - 0.5 );
105  sampleMedian = ( sampleData[ index ] + sampleData[ index + 1 ] ) / 2.0;
106  }
107  else // odd
108  {
109  // 0 .. 100 (101) -> 50
110  unsigned int index = static_cast< unsigned int >( numberOfSamples / 2.0 - 0.5 );
111  sampleMedian = sampleData[ index ];
112  }
113 
114  return sampleMedian;
115 }

118 Eigen::VectorXd computeSampleVariance( const std::vector< Eigen::VectorXd >& sampleData )
119 {
120  // Declare local variables.
121  // Declare and compute sample mean.
122  Eigen::VectorXd sampleMean = computeSampleMean( sampleData );
123 
124  // Declare and initialize sum of residuals squared.
125  Eigen::VectorXd sumOfResidualsSquared = Eigen::VectorXd::Zero( sampleData.at( 0 ).rows( ) );
126 
127  // Compute sum of residuals of sample data squared.
128  for ( unsigned int i = 0; i < sampleData.size( ); i++ )
129  {
130  sumOfResidualsSquared += ( sampleData.at( i ) - sampleMean ).cwiseProduct( sampleData.at( i ) - sampleMean );

132 
133  // Return sample variance.
134  return 1.0 / ( static_cast< double >( sampleData.size( ) ) - 1.0 ) * sumOfResidualsSquared;

138 Eigen::VectorXd computeMovingAverage( const Eigen::VectorXd& sampleData, const unsigned int numberOfAveragingPoints )
139 {
140  // Declare moving average vector
141  int numberOfSamplePoints = sampleData.rows( );
142  Eigen::VectorXd movingAverage;
143  movingAverage.resize( numberOfSamplePoints );
144 
145  // Check that number of samples to be used for averaging is odd
146  if ( ( numberOfAveragingPoints % 2 ) == 0 )
147  {
148  throw std::runtime_error( "Error while computing moving average. The number of points used to average has to be odd." );
149  }
150  int numberOfPointsOnEachSide = ( numberOfAveragingPoints - 1 ) / 2;
151 
152  // Loop over each element and compute moving average
153  unsigned int lowerIndex, upperIndex;
154  for ( int i = 0; i < numberOfSamplePoints; i++ )
155  {
156  lowerIndex = ( ( i - numberOfPointsOnEachSide ) < 0 ) ? 0 : ( i - numberOfPointsOnEachSide );
157  upperIndex = ( ( i + numberOfPointsOnEachSide ) > ( numberOfSamplePoints - 1 ) ) ?
158  ( numberOfSamplePoints - 1 ) : ( i + numberOfPointsOnEachSide );
159  movingAverage[ i ] = computeAverageOfVectorComponents( sampleData.segment( lowerIndex, ( upperIndex - lowerIndex + 1 ) ) );
160  }
161  return movingAverage;
162 }
