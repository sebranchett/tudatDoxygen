45 void TimeDependentSphericalHarmonicsGravityField::update( const double time )
46 {
47  // Initialize current coefficients to nominal values.
48  sineCoefficients_ = nominalSineCoefficients_;
49  cosineCoefficients_ = nominalCosineCoefficients_;
50 
51  // Iterate over all corrections.
52  for( unsigned int i = 0; i < correctionFunctions_.size( ); i++ )
53  {
54  // Add correction of this iteration to current coefficients.
55  correctionFunctions_[ i ]( time, sineCoefficients_, cosineCoefficients_ );
