246  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
247  {
248  // Loop over each body
249  Eigen::Matrix< StateScalarType, 3, 1 > exponentialMapVector;
250  StateScalarType exponentialMapMagnitude;
251  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
252  {
253  // Convert to/from shadow exponential map (SEM) (transformation is the same either way)
254  exponentialMapVector = unprocessedState.block( i * 7 + 3, 0, 3, 1 );
255  exponentialMapMagnitude = exponentialMapVector.norm( );
256  if ( exponentialMapMagnitude >= mathematical_constants::PI )
257  {
258  // Invert flag
259  unprocessedState.segment( i * 7 + 6, 1 ) = ( unprocessedState.block( i * 7 + 6, 0, 1, 1 ) -
260  Eigen::Matrix< StateScalarType, 1, 1 >::Ones( ) ).cwiseAbs( );
261 
262  // Convert to EM/SEM
263  exponentialMapVector *= ( 1.0 - ( 2.0 * mathematical_constants::PI / exponentialMapMagnitude ) );
264 
265  // Replace EM with SEM, or vice-versa
266  unprocessedState.segment( i * 7 + 3, 3 ) = exponentialMapVector;
267  }
268  }
269  }
