163 {
164  if ( response == printWarning || response == throwError )
165  {

228 nlohmann::json getAsArray( const nlohmann::json& jsonObject )
229 {
230  nlohmann::json jsonArray = jsonObject;
231 
232  bool isObjectWithIntConvertibleKeys = jsonObject.is_object( );
233  std::vector< unsigned int > indices;
234  std::vector< nlohmann::json > values;
235  if ( isObjectWithIntConvertibleKeys )
236  {
237  for ( nlohmann::json::const_iterator it = jsonObject.begin( ); it != jsonObject.end( ); ++it )
238  {
239  const std::string key = it.key( );
240  if ( ! contains( SpecialKeys::all, key ) )
241  {
242  const int intKey = indexFromKey( key );
243  if ( intKey >= 0 )
244  {
245  indices.push_back( intKey );
246  }
247  else
248  {
249  isObjectWithIntConvertibleKeys = false;
250  break;
251  }
252  values.push_back( getValue< nlohmann::json >( jsonObject, key ) );
253  }
254  }
255  }
256 
257  if ( isObjectWithIntConvertibleKeys )
258  {
259  if ( values.empty( ) )
260  {
261  jsonArray = nlohmann::json( );
262  }
263  else
264  {
265  std::vector< nlohmann::json > vector( *std::max_element( indices.begin( ), indices.end( ) ) + 1 );
266  for ( unsigned int i = 0; i < indices.size( ); ++i )
267  {
268  vector[ indices.at( i ) ] = values.at( i );
269  }
270  jsonArray = vector;
271  }
272  }
273 
274  return jsonArray;
275 }
