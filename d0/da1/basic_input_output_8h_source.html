184 template< typename OutputStream, typename ScalarType,
185  int NumberOfRows, int NumberOfColumns, int Options, int MaximumRows, int MaximumCols >
186 void writeValueToStream( OutputStream& stream, const Eigen::Matrix< ScalarType,
187  NumberOfRows, NumberOfColumns, Options,
188  MaximumRows, MaximumCols >& value,
189  const int precision, const std::string& delimiter,
190  const bool endLineAfterRow = 0 )
191 {
192  for ( int i = 0; i < value.rows( ); i++ )
193  {
194  for ( int j = 0; j < value.cols( ); j++ )
195  {
196  stream << delimiter << " "
197  << std::setprecision( precision ) << std::left
198  << std::setw( precision + 1 )
199  << value( i, j );
200  }
201  if( endLineAfterRow )
202  {
203  stream << std::endl;
204  }
205  }
206  stream << std::endl;
207 }

241 template< typename InputIterator >
242 void writeDataMapToTextFile(
243  InputIterator iteratorDataMap, InputIterator last,
244  const std::string& outputFilename,
245  const boost::filesystem::path& outputDirectory, const std::string& fileHeader,
246  const int precisionOfKeyType, const int precisionOfValueType,
247  const std::string& delimiter )
248 {
249  // Check if output directory exists; create it if it doesn't.
250  if ( !boost::filesystem::exists( outputDirectory ) )
251  {
252  boost::filesystem::create_directories( outputDirectory );
253  }
254 
255  // Open output file.
256  std::string outputDirectoryAndFilename = outputDirectory.string( ) + "/" + outputFilename;
257  std::ofstream outputFile_( outputDirectoryAndFilename.c_str( ) );
258 
259  // Write file header to file.
260  outputFile_ << fileHeader;
261 
262  // Loop over map of propagation history.
263  for ( ; iteratorDataMap != last; iteratorDataMap++ )
264  {
265  // Print map data to output file.
266  outputFile_ << std::setprecision( precisionOfKeyType )
267  << std::left << std::setw( precisionOfKeyType + 1 )
268  << iteratorDataMap->first;
269  writeValueToStream( outputFile_, iteratorDataMap->second, precisionOfValueType,
270  delimiter );
271  }
272 
273  // Close output file.
274  outputFile_.close( );
275 }
