47 template< typename ScalarType = double >
48 ScalarType computeSemiLatusRectum(
49  const ScalarType eccentricity,
50  const ScalarType semiMajorAxis,
51  const ScalarType tolerance )
52 {
53  // Declare semi-latus rectum.
54  ScalarType semiLatusRectum;
55 
56  // Compute semi-latus rectum in the case it is not a parabola.
57  if ( std::fabs( eccentricity - mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ) >
58  tolerance )
59  {
60  semiLatusRectum = semiMajorAxis * (
61  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
62  eccentricity * eccentricity );
63  }
64 
65  // Else set the semi-latus rectum given for a parabola as the first element in the vector
66  // of Keplerian elements..
67  else
68  {
69  semiLatusRectum = semiMajorAxis;
70  }
71  return semiLatusRectum;
72 }

116 template< typename ScalarType = double >
117 Eigen::Matrix< ScalarType, 6, 1 > convertKeplerianToCartesianElements(
118  const Eigen::Matrix< ScalarType, 6, 1 >& keplerianElements,
119  const ScalarType centralBodyGravitationalParameter )
120 {
121  using std::cos;
122  using std::fabs;
123  using std::pow;
124  using std::sin;
125  using std::sqrt;
126 
127  // Set tolerance.
128  const ScalarType tolerance_ = std::numeric_limits< ScalarType >::epsilon( );
129 
130  // Set local keplerian elements.
131  ScalarType semiMajorAxis_ = keplerianElements( semiMajorAxisIndex );
132  ScalarType eccentricity_ = keplerianElements( eccentricityIndex );
133  ScalarType inclination_ = keplerianElements( inclinationIndex );
134  ScalarType argumentOfPeriapsis_ = keplerianElements( argumentOfPeriapsisIndex );
135  ScalarType longitudeOfAscendingNode_ = keplerianElements( longitudeOfAscendingNodeIndex );
136  ScalarType trueAnomaly_ = keplerianElements( trueAnomalyIndex );
137 
138  // Pre-compute sines and cosines of involved angles for efficient computation.
139  ScalarType cosineOfInclination_ = cos( inclination_ );
140  ScalarType sineOfInclination_ = sin( inclination_ );
141  ScalarType cosineOfArgumentOfPeriapsis_ = cos( argumentOfPeriapsis_ );
142  ScalarType sineOfArgumentOfPeriapsis_ = sin( argumentOfPeriapsis_ );
143  ScalarType cosineOfLongitudeOfAscendingNode_ = cos( longitudeOfAscendingNode_ );
144  ScalarType sineOfLongitudeOfAscendingNode_ = sin( longitudeOfAscendingNode_ );
145  ScalarType cosineOfTrueAnomaly_ = cos( trueAnomaly_ );
146  ScalarType sineOfTrueAnomaly_ = sin( trueAnomaly_ );
147 
148  // Declare semi-latus rectum.
149  ScalarType semiLatusRectum_ = computeSemiLatusRectum< ScalarType >( eccentricity_, semiMajorAxis_, tolerance_ );
150 
151  // Definition of position in the perifocal coordinate system.
152  Eigen::Matrix< ScalarType, 2, 1 > positionPerifocal_;
153  positionPerifocal_.x( ) = semiLatusRectum_ * cosineOfTrueAnomaly_
154  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
155  eccentricity_ * cosineOfTrueAnomaly_ );
156  positionPerifocal_.y( ) = semiLatusRectum_ * sineOfTrueAnomaly_
157  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
158  eccentricity_ * cosineOfTrueAnomaly_ );
159 
160  // Definition of velocity in the perifocal coordinate system.
161  Eigen::Matrix< ScalarType, 2, 1 > velocityPerifocal_(
162  -sqrt( centralBodyGravitationalParameter / semiLatusRectum_ ) * sineOfTrueAnomaly_,
163  sqrt( centralBodyGravitationalParameter / semiLatusRectum_ )
164  * ( eccentricity_ + cosineOfTrueAnomaly_ ) );
165 
166  // Definition of the transformation matrix.
167  Eigen::Matrix< ScalarType, 3, 2 > transformationMatrix_;
168 
169  // Compute the transformation matrix.
170  transformationMatrix_( 0, 0 ) = cosineOfLongitudeOfAscendingNode_
171  * cosineOfArgumentOfPeriapsis_ -sineOfLongitudeOfAscendingNode_
172  * sineOfArgumentOfPeriapsis_ * cosineOfInclination_;
173  transformationMatrix_( 0, 1 ) = -cosineOfLongitudeOfAscendingNode_
174  * sineOfArgumentOfPeriapsis_ -sineOfLongitudeOfAscendingNode_
175  * cosineOfArgumentOfPeriapsis_ * cosineOfInclination_;
176  transformationMatrix_( 1, 0 ) = sineOfLongitudeOfAscendingNode_
177  * cosineOfArgumentOfPeriapsis_ + cosineOfLongitudeOfAscendingNode_
178  * sineOfArgumentOfPeriapsis_ * cosineOfInclination_;
179  transformationMatrix_( 1, 1 ) = -sineOfLongitudeOfAscendingNode_
180  * sineOfArgumentOfPeriapsis_ + cosineOfLongitudeOfAscendingNode_
181  * cosineOfArgumentOfPeriapsis_ * cosineOfInclination_;
182  transformationMatrix_( 2, 0 ) = sineOfArgumentOfPeriapsis_ * sineOfInclination_;
183  transformationMatrix_( 2, 1 ) = cosineOfArgumentOfPeriapsis_ * sineOfInclination_;
184 
185  // Declare converted Cartesian elements.
186  Eigen::Matrix< ScalarType, 6, 1 > convertedCartesianElements_;
187 
188  // Compute value of position in Cartesian coordinates.
189  Eigen::Matrix< ScalarType, 3, 1 > position_ = transformationMatrix_ * positionPerifocal_;
190  convertedCartesianElements_.segment( 0, 3 ) = position_;
191 
192  // Compute value of velocity in Cartesian coordinates.
193  Eigen::Matrix< ScalarType, 3, 1 > velocity_ = transformationMatrix_ * velocityPerifocal_;
194  convertedCartesianElements_.segment( 3, 3 ) = velocity_;
195 
196  // Return Cartesian elements.
197  return convertedCartesianElements_;
198 }

236 template< typename ScalarType = double >
237 Eigen::Matrix< ScalarType, 6, 1 > convertCartesianToKeplerianElements(
238  const Eigen::Matrix< ScalarType, 6, 1 >& cartesianElements,
239  const ScalarType centralBodyGravitationalParameter )
240 {
241  // Set tolerance.
242  const ScalarType tolerance = 20.0 * std::numeric_limits< ScalarType >::epsilon( );
243 
244  // Declare converted Keplerian elements.
245  Eigen::Matrix< ScalarType, 6, 1 > computedKeplerianElements_;
246 
247  // Set position and velocity vectors.
248  const Eigen::Matrix< ScalarType, 3, 1 > position_( cartesianElements.segment( 0, 3 ) );
249  const Eigen::Matrix< ScalarType, 3, 1 > velocity_( cartesianElements.segment( 3, 3 ) );
250 
251  // Compute orbital angular momentum vector.
252  const Eigen::Matrix< ScalarType, 3, 1 > angularMomentum_( position_.cross( velocity_ ) );
253 
254  // Compute semi-latus rectum.
255  const ScalarType semiLatusRectum_ = angularMomentum_.squaredNorm( )
256  / centralBodyGravitationalParameter;
257 
258  // Compute unit vector to ascending node.
259  Eigen::Matrix< ScalarType, 3, 1 > unitAscendingNodeVector_(
260  ( Eigen::Matrix< ScalarType, 3, 1 >::UnitZ( ).cross(
261  angularMomentum_.normalized( ) ) ).normalized( ) );
262 
263  // Compute eccentricity vector.
264  Eigen::Matrix< ScalarType, 3, 1 > eccentricityVector_(
265  velocity_.cross( angularMomentum_ ) / centralBodyGravitationalParameter
266  - position_.normalized( ) );
267 
268  // Store eccentricity.
269  computedKeplerianElements_( eccentricityIndex ) = eccentricityVector_.norm( );
270 
271  // Compute and store semi-major axis.
272  // Check if orbit is parabolic. If it is, store the semi-latus rectum instead of the
273  // semi-major axis.
274  if ( std::fabs( computedKeplerianElements_( eccentricityIndex ) -
275  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ) < tolerance )
276  {
277  computedKeplerianElements_( semiLatusRectumIndex ) = semiLatusRectum_;
278  }
279 
280  // Else the orbit is either elliptical or hyperbolic, so store the semi-major axis.
281  else
282  {
283  computedKeplerianElements_( semiMajorAxisIndex ) = semiLatusRectum_
284  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
285  computedKeplerianElements_( eccentricityIndex )
286  * computedKeplerianElements_( eccentricityIndex ) );
287  }
288 
289  // Compute and store inclination.
290  computedKeplerianElements_( inclinationIndex ) = std::acos( angularMomentum_.z( )
291  / angularMomentum_.norm( ) );
292 
293  // Compute and store longitude of ascending node.
294  // Define the quadrant condition for the argument of perigee.
295  ScalarType argumentOfPeriapsisQuandrantCondition = eccentricityVector_.z( );
296 
297  // Check if the orbit is equatorial. If it is, set the vector to the line of nodes to the
298  // x-axis.
299  if ( std::fabs( computedKeplerianElements_( inclinationIndex ) ) < tolerance )
300  {
301  unitAscendingNodeVector_ = Eigen::Matrix< ScalarType, 3, 1 >::UnitX( );
302 
303  // If the orbit is equatorial, eccentricityVector_.z( ) is zero, therefore the quadrant
304  // condition is taken to be the y-component, eccentricityVector_.y( ).
305  argumentOfPeriapsisQuandrantCondition = eccentricityVector_.y( );
306  }
307 
308  // Compute and store the resulting longitude of ascending node.
309  computedKeplerianElements_( longitudeOfAscendingNodeIndex )
310  = std::acos( unitAscendingNodeVector_.x( ) );
311 
312  // Check if the quandrant is correct.
313  if ( unitAscendingNodeVector_.y( ) <
314  mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
315  {
316  computedKeplerianElements_( longitudeOfAscendingNodeIndex ) =
317  mathematical_constants::getFloatingInteger< ScalarType >( 2 ) *
318  mathematical_constants::getPi< ScalarType >( ) -
319  computedKeplerianElements_( longitudeOfAscendingNodeIndex );
320  }
321 
322  // Compute and store argument of periapsis.
323  // Define the quadrant condition for the true anomaly.
324  ScalarType trueAnomalyQuandrantCondition = position_.dot( velocity_ );
325 
326  // Check if the orbit is circular. If it is, set the eccentricity vector to unit vector
327  // pointing to the ascending node, i.e. set the argument of periapsis to zero.
328  if ( std::fabs( computedKeplerianElements_( eccentricityIndex ) ) < tolerance )
329  {
330  eccentricityVector_ = unitAscendingNodeVector_;
331 
332  computedKeplerianElements_( argumentOfPeriapsisIndex ) =
333  mathematical_constants::getFloatingInteger< ScalarType >( 0 );
334 
335  // Check if orbit is also equatorial and set true anomaly quandrant check condition
336  // accordingly.
337  if ( unitAscendingNodeVector_ == Eigen::Matrix< ScalarType, 3, 1 >::UnitX( ) )
338  {
339  // If the orbit is circular, position_.dot( velocity_ ) = 0, therefore this value
340  // cannot be used as a quadrant condition. Moreover, if the orbit is equatorial,
341  // position_.z( ) is also zero and therefore the quadrant condition is taken to be the
342  // y-component, position_.y( ).
343  trueAnomalyQuandrantCondition = position_.y( );
344  }
345 
346  else
347  {
348  // If the orbit is circular, position_.dot( velocity_ ) = 0, therefore the quadrant
349  // condition is taken to be the z-component of the position, position_.z( ).
350  trueAnomalyQuandrantCondition = position_.z( );
351  }
352  }
353 
354  // Else, compute the argument of periapsis as the angle between the eccentricity vector and
355  // the unit vector to the ascending node.
356  else
357  {
358  ScalarType eccentricityAscendingNodeDotProduct = eccentricityVector_.normalized( ).dot( unitAscendingNodeVector_ );
359 
360  // Check whether dot product is in bounds (might be out of bounds due to numerical noise).
361  if( eccentricityAscendingNodeDotProduct < mathematical_constants::getFloatingInteger< ScalarType >( -1 ) )
362  {
363  computedKeplerianElements_( argumentOfPeriapsisIndex ) = mathematical_constants::getPi< ScalarType >( );
364  }
365  else if( eccentricityAscendingNodeDotProduct > mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
366  {
367  computedKeplerianElements_( argumentOfPeriapsisIndex ) =
368  mathematical_constants::getFloatingInteger< ScalarType >( 0 );
369  }
370  else
371  {
372  computedKeplerianElements_( argumentOfPeriapsisIndex ) = std::acos( eccentricityAscendingNodeDotProduct );
373  }
374 
375  // Check if the quadrant is correct.
376  if ( argumentOfPeriapsisQuandrantCondition <
377  mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
378  {
379  computedKeplerianElements_( argumentOfPeriapsisIndex ) =
380  mathematical_constants::getFloatingInteger< ScalarType >( 2 ) *
381  mathematical_constants::getPi< ScalarType >( ) -
382  computedKeplerianElements_( argumentOfPeriapsisIndex );
383  }
384  }
385 
386  // Compute dot-product of position and eccentricity vectors.
387  ScalarType dotProductPositionAndEccentricityVectors
388  = position_.normalized( ).dot( eccentricityVector_.normalized( ) );
389 
390  // Check if the dot-product is one of the limiting cases: 0.0, -1.0 or 1.0
391  // (within prescribed tolerance).
392  if ( std::fabs( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
393  dotProductPositionAndEccentricityVectors ) < tolerance )
394  {
395  dotProductPositionAndEccentricityVectors =
396  mathematical_constants::getFloatingInteger< ScalarType >( 1 );
397  }
398 
399  if ( std::fabs( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
400  dotProductPositionAndEccentricityVectors ) < tolerance )
401  {
402  dotProductPositionAndEccentricityVectors =
403  -mathematical_constants::getFloatingInteger< ScalarType >( 1 );
404  }
405 
406  if ( std::fabs( dotProductPositionAndEccentricityVectors ) < tolerance )
407  {
408  dotProductPositionAndEccentricityVectors =
409  mathematical_constants::getFloatingInteger< ScalarType >( 0 );
410  }
411 
412  // Compute and store true anomaly.
413  computedKeplerianElements_( trueAnomalyIndex )
414  = std::acos( dotProductPositionAndEccentricityVectors );
415 
416  // Check if the quandrant is correct.
417  if ( trueAnomalyQuandrantCondition <
418  mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
419  {
420  computedKeplerianElements_( trueAnomalyIndex ) =
421  mathematical_constants::getFloatingInteger< ScalarType >( 2 ) *
422  mathematical_constants::getPi< ScalarType >( ) -
423  computedKeplerianElements_( trueAnomalyIndex );
424  }
425 
426  // Return converted Keplerian elements.
427  return computedKeplerianElements_;
428 }

440 template< typename ScalarType = double >
441 ScalarType convertTrueAnomalyToEllipticalEccentricAnomaly(
442  const ScalarType trueAnomaly, const ScalarType eccentricity )
443 
444 {
445  using std::cos;
446  using std::sqrt;
447  using std::atan2;
448 
449  if ( eccentricity >= mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ||
450  eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
451  {
452  throw std::runtime_error( "Eccentricity is invalid when converting true to elliptical eccentric anomaly." );
453  }
454  else
455  {
456  // Declare and compute sine and cosine of eccentric anomaly.
457  ScalarType sineOfEccentricAnomaly_ =
458  sqrt( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
459  eccentricity * eccentricity ) * std::sin( trueAnomaly ) /
460  ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
461  eccentricity * cos( trueAnomaly ) );
462  ScalarType cosineOfEccentricAnomaly_ = ( eccentricity + cos( trueAnomaly ) )
463  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
464  eccentricity * cos( trueAnomaly ) );
465 
466  // Return elliptic eccentric anomaly.
467  return atan2( sineOfEccentricAnomaly_, cosineOfEccentricAnomaly_ );
468  }
469 }

480 ScalarType convertTrueAnomalyToHyperbolicEccentricAnomaly( const ScalarType trueAnomaly,
481  const ScalarType eccentricity )
482 {
483  if ( eccentricity <= mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
484  {
485  throw std::runtime_error( "Eccentricity is invalid when converting true to hyperbolic eccentric anomaly." );
486  }

489  {
490  using std::cos;
491 
492  // Compute hyperbolic sine and hyperbolic cosine of hyperbolic eccentric anomaly.
493  ScalarType hyperbolicSineOfHyperbolicEccentricAnomaly_
494  = std::sqrt( eccentricity * eccentricity -
495  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )

498  cos( trueAnomaly ) );

501  = ( cos( trueAnomaly ) + eccentricity ) /

524 template< typename ScalarType = double >
525 ScalarType convertTrueAnomalyToEccentricAnomaly( const ScalarType trueAnomaly,
526  const ScalarType eccentricity )
527 {
528  // Declare computed eccentric anomaly.
529  ScalarType eccentricAnomaly_ = 0.0;
530 
531  // Check if eccentricity is invalid and throw an error if true.
532  if ( eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
533  {
534  throw std::runtime_error( "Eccentricity is invalid when converting true to eccentric anomaly." );
535  }
536 
537  // Check if orbit is parabolic and throw an error if true.
538  else if ( std::fabs( eccentricity -
539  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ) <
540  std::numeric_limits< ScalarType >::epsilon( ) )
541  {
542  throw std::runtime_error( "Parabolic orbits have not yet been implemented when converting true to eccentric anomaly." );
543  }
544 
545  // Check if orbit is elliptical and compute eccentric anomaly.
546  else if ( eccentricity >= mathematical_constants::getFloatingInteger< ScalarType >( 0 ) &&
547  eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
548  {
549  eccentricAnomaly_ = convertTrueAnomalyToEllipticalEccentricAnomaly< ScalarType >(
550  trueAnomaly, eccentricity );
551  }
552 
553  else if ( eccentricity > mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
554  {
555  eccentricAnomaly_ = convertTrueAnomalyToHyperbolicEccentricAnomaly< ScalarType >(
556  trueAnomaly, eccentricity );
557  }

559  // Return computed eccentric anomaly.
560  return eccentricAnomaly_;

571 template< typename ScalarType = double >
572 ScalarType convertEllipticalEccentricAnomalyToTrueAnomaly(
573  const ScalarType ellipticEccentricAnomaly,
574  const ScalarType eccentricity )
575 {
576  if ( eccentricity >= mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ||
577  eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
578  {
579  throw std::runtime_error( "Eccentricity is invalid when converting elliptical eccentric to true anomaly." );
580  }
581 
582  else
583  {
584  using std::cos;
585  using std::sqrt;
586 
587  // Compute sine and cosine of true anomaly.
588  ScalarType sineOfTrueAnomaly_ =
589  sqrt( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
590  eccentricity * eccentricity ) *
591  std::sin( ellipticEccentricAnomaly )
592  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
593  eccentricity * cos( ellipticEccentricAnomaly ) );
594 
595  ScalarType cosineOfTrueAnomaly_ = ( cos( ellipticEccentricAnomaly ) - eccentricity )
596  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
597  eccentricity * cos( ellipticEccentricAnomaly ) );
598 
599  // Return true anomaly.
600  return std::atan2( sineOfTrueAnomaly_, cosineOfTrueAnomaly_ );
601  }
