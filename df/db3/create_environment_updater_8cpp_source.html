23 void checkValidityOfRequiredEnvironmentUpdates(
24  const std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >&
25  requestedUpdates, const simulation_setup::NamedBodyMap& bodyMap )
26 {
27  using namespace propagators;
28 
29  // Iterate over all environment update types.
30  for( std::map< propagators::EnvironmentModelsToUpdate,
31  std::vector< std::string > >::const_iterator
32  updateIterator = requestedUpdates.begin( );
33  updateIterator != requestedUpdates.end( ); updateIterator++ )
34  {
35  // Iterate over all updated bodies.
36  for( unsigned int i = 0; i < updateIterator->second.size( ); i++ )
37  {
38  // Ignore global required updates.
39  if( updateIterator->second.at( i ) != "" )
40  {
41  // Check if body exists.
42  if( bodyMap.count( updateIterator->second.at( i ) ) == 0 )
43  {
44  throw std::runtime_error(
45  "Error when making environment model update settings, could not find body "
46  + updateIterator->second.at( i ) );
47  }
48 
49  // Check if requested environment model exists.
50  switch( updateIterator->first )
51  {
52  case body_translational_state_update:
53  {
54  if( bodyMap.at( updateIterator->second.at( i ) )->getEphemeris( ) == nullptr )
55  {
56  throw std::runtime_error(
57  "Error when making environment model update settings, could not find ephemeris of body "
58  + updateIterator->second.at( i ) );
59  }
60  break;
61  }
62  case body_rotational_state_update:
63  {
64  if( ( bodyMap.at( updateIterator->second.at( i ) )->
65  getRotationalEphemeris( ) == nullptr ) &&
66  ( bodyMap.at( updateIterator->second.at( i ) )->getDependentOrientationCalculator( ) == nullptr ) )
67  {
68  throw std::runtime_error(
69  "Error when making environment model update settings, could not find rotational ephemeris or dependent orientation calculator of body "
70  + updateIterator->second.at( i ) );
71  }
72  break;
73  }
74  case spherical_harmonic_gravity_field_update:
75  {
76  std::shared_ptr< gravitation::SphericalHarmonicsGravityField >
77  gravityFieldModel =
78  std::dynamic_pointer_cast< gravitation::SphericalHarmonicsGravityField >(
79  bodyMap.at( updateIterator->second.at( i ) )->getGravityFieldModel( ) );
80  if( gravityFieldModel == nullptr )
81  {
82  throw std::runtime_error(
83  "Error when making environment model update settings, could not find spherical harmonic gravity field of body "
84  + updateIterator->second.at( i ) );
85  }
86  break;
87  }
88  case vehicle_flight_conditions_update:
89  {
90  std::shared_ptr< aerodynamics::FlightConditions > flightConditions = bodyMap.at(
91  updateIterator->second.at( i ) )->getFlightConditions( );
92  if( flightConditions == nullptr )
93  {
94  throw std::runtime_error(
95  "Error when making environment model update settings, could not find flight conditions of body "
96  + updateIterator->second.at( i ) );
97  }
98  break;
99  }
100  case radiation_pressure_interface_update:
101  {
102  std::map< std::string, std::shared_ptr< electro_magnetism::RadiationPressureInterface > >
103  radiationPressureInterfaces = bodyMap.at(
104  updateIterator->second.at( i ) )->getRadiationPressureInterfaces( );
105  if( radiationPressureInterfaces.size( ) == 0 )
106  {
107  throw std::runtime_error(
108  "Error when making environment model update settings, could not find radiation pressure interface of body "
109  + updateIterator->second.at( i ) );
110  }
111  break;
112  }
113  case body_mass_update:
114  if( bodyMap.at( updateIterator->second.at( i ) )->getBodyMassFunction( ) == nullptr )
115  {
116  throw std::runtime_error(
117  "Error when making environment model update settings, no body mass function of body "
118  + updateIterator->second.at( i ) );
119  }
120 
121  break;
122  }
123  }
124  }
125  }

129 void removePropagatedStatesFomEnvironmentUpdates(
130  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >& environmentModelsToUpdate,
131  const std::map< IntegratedStateType, std::vector< std::pair< std::string, std::string > > >& integratedStateList )
132 {
133  // Iterate over all propagated states
134  for( auto it = integratedStateList.begin( ); it != integratedStateList.end( ); it++ )
135  {
136  // Iterate over all propagated bodies for current state
137  for( unsigned int i = 0; i < it->second.size( ); i++ )
138  {
139  switch( it->first )
140  {
141  // Check for propagated translational states in update list, and remove if necessary
142  case translational_state:
143  if( environmentModelsToUpdate.count( body_translational_state_update ) > 0 )
144  {
145  std::vector< std::string > bodiesToUpdate = environmentModelsToUpdate.at( body_translational_state_update );
146  std::vector< std::string >::iterator findIterator =
147  std::find( bodiesToUpdate.begin( ), bodiesToUpdate.end( ), it->second.at( i ).first );
148 
149  if( findIterator != bodiesToUpdate.end( ) )
150  {
151  bodiesToUpdate.erase( findIterator );
152  environmentModelsToUpdate[ body_translational_state_update ] = bodiesToUpdate;
153 
154  }
155  }
156  break;
157  // Check for propagated rotational states in update list, and remove if necessary
158  case rotational_state:
159  if( environmentModelsToUpdate.count( body_rotational_state_update ) > 0 )
160  {
161  std::vector< std::string > bodiesToUpdate = environmentModelsToUpdate.at( body_rotational_state_update );
162  std::vector< std::string >::iterator findIterator =
163  std::find( bodiesToUpdate.begin( ), bodiesToUpdate.end( ), it->second.at( i ).first );
164 
165  if( findIterator != bodiesToUpdate.end( ) )
166  {
167  bodiesToUpdate.erase( findIterator );
168  environmentModelsToUpdate[ body_rotational_state_update ] = bodiesToUpdate;
169 
170  }
171  }
172  break;
173  // Check for propagated mass states in update list, and remove if necessary
174  case body_mass_state:
175  if( environmentModelsToUpdate.count( body_mass_update ) > 0 )
176  {
177  std::vector< std::string > bodiesToUpdate = environmentModelsToUpdate.at( body_mass_update );
178  std::vector< std::string >::iterator findIterator =
179  std::find( bodiesToUpdate.begin( ), bodiesToUpdate.end( ), it->second.at( i ).first );
180 
181  if( findIterator != bodiesToUpdate.end( ) )
182  {
183  bodiesToUpdate.erase( findIterator );
184  environmentModelsToUpdate[ body_mass_update ] = bodiesToUpdate;
185 
186  }
187  }
188  break;
189  case custom_state:
190  break;
191  default:
192  throw std::runtime_error( "Error when removing propagated states from environment updates, state type " +
193  std::to_string( it->first ) + " not recognized." );
194  }
195  }
196  }

200 std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >
201 createRotationalEquationsOfMotionEnvironmentUpdaterSettings(
202  const basic_astrodynamics::TorqueModelMap& torqueModels, const simulation_setup::NamedBodyMap& bodyMap )
203 {
204  using namespace basic_astrodynamics;
205 
206  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > environmentModelsToUpdate;
207  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > singleTorqueUpdateNeeds;
208 
209  // Iterate over all bodies on which torques are being exerting
210  for( TorqueModelMap::const_iterator acceleratedBodyIterator = torqueModels.begin( );
211  acceleratedBodyIterator != torqueModels.end( ); acceleratedBodyIterator++ )
212  {
213  // Iterate over all bodies exerting on current body
214  for( SingleBodyTorqueModelMap::const_iterator torqueModelIterator = acceleratedBodyIterator->second.begin( );
215  torqueModelIterator != acceleratedBodyIterator->second.end( ); torqueModelIterator++ )
216  {
217  singleTorqueUpdateNeeds.clear( );
218  for( unsigned int i = 0; i < torqueModelIterator->second.size( ); i++ )
219  {
220  AvailableTorque currentTorqueModelType =
221  getTorqueModelType( torqueModelIterator->second.at( i ) );
222 
223  switch( currentTorqueModelType )
224  {
225  case second_order_gravitational_torque:
226  break;
227  case aerodynamic_torque:
228  singleTorqueUpdateNeeds[ body_rotational_state_update ].push_back(
229  torqueModelIterator->first );
230  singleTorqueUpdateNeeds[ vehicle_flight_conditions_update ].push_back(
231  acceleratedBodyIterator->first );
232  break;
233  default:
234  std::cerr << "Error, update information not found for torque model " << currentTorqueModelType << std::endl;
235  break;
236  }
237  }
238 
239  checkValidityOfRequiredEnvironmentUpdates( singleTorqueUpdateNeeds, bodyMap );
240  addEnvironmentUpdates( environmentModelsToUpdate, singleTorqueUpdateNeeds );
241  }
242  }
243 
244  return environmentModelsToUpdate;
245 }

249 std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >
250 createTranslationalEquationsOfMotionEnvironmentUpdaterSettings(
251  const basic_astrodynamics::AccelerationMap& translationalAccelerationModels,
252  const simulation_setup::NamedBodyMap& bodyMap )
253 {
254  using namespace basic_astrodynamics;
255  using namespace propagators;
256 
257  std::map< propagators::EnvironmentModelsToUpdate,
258  std::vector< std::string > > environmentModelsToUpdate;
259  std::map< propagators::EnvironmentModelsToUpdate,
260  std::vector< std::string > > singleAccelerationUpdateNeeds;
261 
262  // Iterate over all bodies being accelerated
263  for( AccelerationMap::const_iterator acceleratedBodyIterator
264  = translationalAccelerationModels.begin( );
265  acceleratedBodyIterator != translationalAccelerationModels.end( );
266  acceleratedBodyIterator++ )
267  {
268  // Iterate over all bodies exerting acceleration.
269  for( SingleBodyAccelerationMap::const_iterator accelerationModelIterator
270  = acceleratedBodyIterator->second.begin( );
271  accelerationModelIterator != acceleratedBodyIterator->second.end( );
272  accelerationModelIterator++ )
273  {
274  singleAccelerationUpdateNeeds.clear( );
275  for( unsigned int i = 0; i < accelerationModelIterator->second.size( ); i++ )
276  {
277  AvailableAcceleration currentAccelerationModelType =
278  getAccelerationModelType( accelerationModelIterator->second.at( i ) );
279 
280  // Add translational state of both bodies to update list for current acceleration model.
281  if( translationalAccelerationModels.count( accelerationModelIterator->first ) == 0 )
282  {
283  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
284  accelerationModelIterator->first );
285  }
286 
287  // Check acceleration model type and change environment update list accordingly.
288  switch( currentAccelerationModelType )
289  {
290  case central_gravity:
291  break;
292  case third_body_central_gravity:
293  {
294  std::shared_ptr< gravitation::ThirdBodyCentralGravityAcceleration >
295  thirdBodyAcceleration = std::dynamic_pointer_cast<
296  gravitation::ThirdBodyCentralGravityAcceleration >(
297  accelerationModelIterator->second.at( i ) );
298  if( thirdBodyAcceleration != nullptr && translationalAccelerationModels.count(
299  thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
300  {
301  if( translationalAccelerationModels.count( thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
302  {
303  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
304  thirdBodyAcceleration->getCentralBodyName( ) );
305  }
306  }
307  else if( thirdBodyAcceleration == nullptr )
308  {
309  throw std::runtime_error(
310  std::string( "Error, incompatible input (ThirdBodyCentralGravityAcceleration) to" )
311  + std::string( "createTranslationalEquationsOfMotionEnvironmentUpdaterSettings" ) );
312  }
313  break;
314  }
315  case aerodynamic:
316  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
317  accelerationModelIterator->first );
318  singleAccelerationUpdateNeeds[ vehicle_flight_conditions_update ].push_back(
319  acceleratedBodyIterator->first );
320  singleAccelerationUpdateNeeds[ body_mass_update ].push_back(
321  acceleratedBodyIterator->first );
322  break;
323  case cannon_ball_radiation_pressure:
324  singleAccelerationUpdateNeeds[ radiation_pressure_interface_update ].push_back(
325  acceleratedBodyIterator->first );
326  singleAccelerationUpdateNeeds[ body_mass_update ].push_back(
327  acceleratedBodyIterator->first );
328  break;
329  case spherical_harmonic_gravity:
330  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
331  accelerationModelIterator->first );
332  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
333  push_back( accelerationModelIterator->first );
334  break;
335  case mutual_spherical_harmonic_gravity:
336  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
337  accelerationModelIterator->first );
338  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
339  accelerationModelIterator->first );
340  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
341  acceleratedBodyIterator->first );
342  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
343  acceleratedBodyIterator->first );
344  break;
345  case third_body_spherical_harmonic_gravity:
346  {
347  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
348  accelerationModelIterator->first );
349  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
350  push_back( accelerationModelIterator->first );
351 
352  std::shared_ptr<
353  gravitation::ThirdBodySphericalHarmonicsGravitationalAccelerationModel >
354  thirdBodyAcceleration = std::dynamic_pointer_cast<
355  gravitation::ThirdBodySphericalHarmonicsGravitationalAccelerationModel >(
356  accelerationModelIterator->second.at( i ) );;
357  if( thirdBodyAcceleration != nullptr && translationalAccelerationModels.count(
358  thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
359  {
360  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
361  thirdBodyAcceleration->getCentralBodyName( ) );
362  }
363  else if( thirdBodyAcceleration == nullptr )
364  {
365  throw std::runtime_error(
366  std::string( "Error, incompatible input (ThirdBodySphericalHarmonicsGravitational" )
367  + std::string( "AccelerationModel) to createTranslationalEquationsOfMotion ")
368  + std::string( "EnvironmentUpdaterSettings" ) );
369  }
370  break;
371  }
372  case third_body_mutual_spherical_harmonic_gravity:
373  {
374  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
375  accelerationModelIterator->first );
376  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
377  accelerationModelIterator->first );
378  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
379  acceleratedBodyIterator->first );
380  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
381  acceleratedBodyIterator->first );
382 
383  std::shared_ptr< gravitation::ThirdBodyMutualSphericalHarmonicsGravitationalAccelerationModel >
384  thirdBodyAcceleration = std::dynamic_pointer_cast<
385  gravitation::ThirdBodyMutualSphericalHarmonicsGravitationalAccelerationModel >(
386  accelerationModelIterator->second.at( i ) );
387  if( thirdBodyAcceleration != nullptr && translationalAccelerationModels.count(
388  thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
389  {
390  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
391  thirdBodyAcceleration->getCentralBodyName( ) );
392  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
393  thirdBodyAcceleration->getCentralBodyName( ) );
394  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
395  thirdBodyAcceleration->getCentralBodyName( ) );
396  }
397  else if( thirdBodyAcceleration == nullptr )
398  {
399  throw std::runtime_error(
400  std::string( "Error, incompatible input (ThirdBodyMutualSphericalHarmonicsGravitational" ) +
401  std::string( "AccelerationModel) to createTranslationalEquationsOfMotion ") +
402  std::string( "EnvironmentUpdaterSettings" ) );
403  }
404  break;
405  }
406  case thrust_acceleration:
407  {
408  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > thrustModelUpdates =
409  std::dynamic_pointer_cast< propulsion::ThrustAcceleration >(
410  accelerationModelIterator->second.at( i ) )->getRequiredModelUpdates( );
411  addEnvironmentUpdates( singleAccelerationUpdateNeeds, thrustModelUpdates );
412 
413  singleAccelerationUpdateNeeds[ body_mass_update ].push_back(
414  acceleratedBodyIterator->first );
415 
416  break;
417  }
418  case relativistic_correction_acceleration:
419  {
420  std::shared_ptr< relativity::RelativisticAccelerationCorrection >
421  accelerationCorrection = std::dynamic_pointer_cast< relativity::RelativisticAccelerationCorrection >(
422  accelerationModelIterator->second.at( i ) );
423  if( accelerationCorrection->getCalculateDeSitterCorrection( ) )
424  {
425  std::string primaryBody = accelerationCorrection->getPrimaryBodyName( );
426  if( translationalAccelerationModels.count(primaryBody ) == 0 )
427  {
428  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
429  primaryBody );
430  }
431  }
432  break;
433  }
434  case direct_tidal_dissipation_in_central_body_acceleration:
435  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
436  accelerationModelIterator->first );
437  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
438  push_back( accelerationModelIterator->first );
439  break;
440  case direct_tidal_dissipation_in_orbiting_body_acceleration:
441  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
442  accelerationModelIterator->first );
443  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
444  push_back( accelerationModelIterator->first );
445  break;
446  case empirical_acceleration:
447  break;
448  default:
449  throw std::runtime_error( std::string( "Error when setting acceleration model update needs, model type not recognized: " ) +
450  std::to_string( currentAccelerationModelType ) );
451  break;
452  }
453 
454  }
455 
456  // Check whether requested updates are possible.
457  checkValidityOfRequiredEnvironmentUpdates( singleAccelerationUpdateNeeds, bodyMap );
458 
459  // Add requested updates of current acceleration model to
460  // full list of environment updates.
461  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
462  }
463  }
464 
465  return environmentModelsToUpdate;
466 }

821 std::vector< std::string > > createFullEnvironmentUpdaterSettings(
822  const simulation_setup::NamedBodyMap& bodyMap )
823 {
824  using namespace basic_astrodynamics;
825  using namespace electro_magnetism;
826  using namespace gravitation;
827  using namespace propagators;
828 
829  std::map< propagators::EnvironmentModelsToUpdate,
830  std::vector< std::string > > environmentModelsToUpdate;
831  std::map< propagators::EnvironmentModelsToUpdate,
832  std::vector< std::string > > singleAccelerationUpdateNeeds;
833 
834  // Iterate over all bodies.
835  for( simulation_setup::NamedBodyMap::const_iterator
836  bodyIterator = bodyMap.begin( ); bodyIterator != bodyMap.end( ); bodyIterator++ )
837  {
838  singleAccelerationUpdateNeeds.clear( );
839 
840  // Check if current body is a vehicle.
841 
842  // Check if current body has flight conditions set.
843  if( bodyIterator->second ->getFlightConditions( ) != nullptr )
844  {
845  // If vehicle has flight conditions, add flight conditions update function to update list.
846  singleAccelerationUpdateNeeds[ vehicle_flight_conditions_update ].
847  push_back( bodyIterator->first );
848  }
849 
850  // Get body radiation pressure interface(s) (one per source)
851  std::map< std::string,
852  std::shared_ptr< RadiationPressureInterface > > radiationPressureInterfaces
853  = bodyIterator->second->getRadiationPressureInterfaces( );
854 
855  // Add each interface update function to update list.
856  for( std::map< std::string, std::shared_ptr< RadiationPressureInterface > >::iterator
857  iterator = radiationPressureInterfaces.begin( );
858  iterator != radiationPressureInterfaces.end( ); iterator++ )
859  {
860  singleAccelerationUpdateNeeds[ radiation_pressure_interface_update ].
861  push_back( bodyIterator->first );
862  }
863 
864  // If body has rotation model, update rotational state in each time step.;
865  if( ( bodyIterator->second->getRotationalEphemeris( ) != nullptr ) ||
866  ( bodyIterator->second->getDependentOrientationCalculator( ) != nullptr ) )
867  {
868  singleAccelerationUpdateNeeds[ body_rotational_state_update ].
869  push_back( bodyIterator->first );
870  }
871 
872 
873  std::shared_ptr< TimeDependentSphericalHarmonicsGravityField > gravityField =
874  std::dynamic_pointer_cast< TimeDependentSphericalHarmonicsGravityField >
875  ( bodyIterator->second->getGravityFieldModel( ) );
876  if( gravityField != nullptr )
877  {
878  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
879  push_back( bodyIterator->first );
880  }
881 
882  singleAccelerationUpdateNeeds[ body_mass_update ].push_back( bodyIterator->first );
883 
884  // Check whether requested updates are possible.
885  checkValidityOfRequiredEnvironmentUpdates( singleAccelerationUpdateNeeds, bodyMap );
886 
887  // Add requested updates of current acceleration model to full list of environment updates.
888  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
889  }
890  return environmentModelsToUpdate;
