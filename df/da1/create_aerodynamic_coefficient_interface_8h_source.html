863 template< unsigned int NumberOfDimensions >
864 std::shared_ptr< AerodynamicCoefficientSettings >
865 readGivenSizeTabulatedAerodynamicCoefficientsFromFiles(
866  const std::map< int, std::string > forceCoefficientFiles,
867  const std::map< int, std::string > momentCoefficientFiles,
868  const double referenceLength,
869  const double referenceArea,
870  const double lateralReferenceLength,
871  const Eigen::Vector3d& momentReferencePoint,
872  const std::vector< aerodynamics::AerodynamicCoefficientsIndependentVariables > independentVariableNames,
873  const bool areCoefficientsInAerodynamicFrame = true,
874  const bool areCoefficientsInNegativeAxisDirection = true,
875  const std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings = nullptr )
876 {
877  std::pair< boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) >,
878  std::vector< std::vector< double > > >
879  aerodynamicForceCoefficients =
880  input_output::readAerodynamicCoefficients< NumberOfDimensions >( forceCoefficientFiles );
881  std::pair< boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfDimensions ) >,
882  std::vector< std::vector< double > > >
883  aerodynamicMomentCoefficients =
884  input_output::readAerodynamicCoefficients< NumberOfDimensions >( momentCoefficientFiles );
885 
886  if( !input_output::compareIndependentVariables(
887  aerodynamicForceCoefficients.second, aerodynamicMomentCoefficients.second ) )
888  {
889  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, "
890  "force and moment independent variables are inconsistent" );
891  }
892 
893  if( independentVariableNames.size( ) != NumberOfDimensions )
894  {
895  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, input sizes are inconsistent" );
896  }
897 
898  // Create coefficient settings.
899  std::shared_ptr< TabulatedAerodynamicCoefficientSettings< NumberOfDimensions > > tabulatedCoefficients =
900  std::make_shared< TabulatedAerodynamicCoefficientSettings< NumberOfDimensions > >(
901  aerodynamicForceCoefficients.second, aerodynamicForceCoefficients.first, aerodynamicMomentCoefficients.first,
902  referenceLength, referenceArea, lateralReferenceLength, momentReferencePoint, independentVariableNames,
903  areCoefficientsInAerodynamicFrame, areCoefficientsInNegativeAxisDirection, interpolatorSettings );
904  tabulatedCoefficients->setForceCoefficientsFiles( forceCoefficientFiles );
905  tabulatedCoefficients->setMomentCoefficientsFiles( momentCoefficientFiles );
906  return tabulatedCoefficients;
907 }
