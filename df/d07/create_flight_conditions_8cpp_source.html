31 std::shared_ptr< aerodynamics::AtmosphericFlightConditions > createAtmosphericFlightConditions(
32  const std::shared_ptr< Body > bodyWithFlightConditions,
33  const std::shared_ptr< Body > centralBody,
34  const std::string& nameOfBodyUndergoingAcceleration,
35  const std::string& nameOfBodyExertingAcceleration,
36  const std::function< double( ) > angleOfAttackFunction,
37  const std::function< double( ) > angleOfSideslipFunction,
38  const std::function< double( ) > bankAngleFunction,
39  const std::function< void( const double ) > angleUpdateFunction )
40 {
41  // Check whether all required environment models are set.
42  if( centralBody->getAtmosphereModel( ) == nullptr )
43  {
44  throw std::runtime_error(
45  "Error when making flight conditions, body " + nameOfBodyExertingAcceleration +
46  " has no atmosphere model." );
47  }
48 
49  if( centralBody->getShapeModel( ) == nullptr )
50  {
51  throw std::runtime_error(
52  "Error when making flight conditions, body " + nameOfBodyExertingAcceleration +
53  " has no shape model." );
54  }
55 
56  if( centralBody->getRotationalEphemeris( ) == nullptr )
57  {
58  throw std::runtime_error(
59  "Error when making flight conditions, body " + nameOfBodyExertingAcceleration +
60  " has no rotation model." );
61  }
62 
63  if( bodyWithFlightConditions->getAerodynamicCoefficientInterface( ) == nullptr )
64  {
65  throw std::runtime_error(
66  "Error when making flight conditions, body " + nameOfBodyUndergoingAcceleration +
67  " has no aerodynamic coefficients." );
68  }
69 
70 
71  // Create function to rotate state from intertial to body-fixed frame.
72  std::function< Eigen::Quaterniond( ) > rotationToFrameFunction =
73  std::bind( &Body::getCurrentRotationToLocalFrame, centralBody );
74  std::function< Eigen::Matrix3d( ) > rotationMatrixToFrameDerivativeFunction =
75  std::bind( &Body::getCurrentRotationMatrixDerivativeToLocalFrame, centralBody );
76 
77  std::function< Eigen::Matrix< double, 6, 1 >( ) > bodyStateFunction = std::bind( &Body::getState, bodyWithFlightConditions );
78  std::function< Eigen::Matrix< double, 6, 1 >( ) > centralBodyStateFunction = std::bind( &Body::getState, centralBody );
79 
80  std::function< Eigen::Matrix< double, 6, 1 >( ) > relativeBodyFixedStateFunction =
81  std::bind( &ephemerides::transformRelativeStateToFrame< double >,
82  bodyStateFunction, centralBodyStateFunction,
83  rotationToFrameFunction,
84  rotationMatrixToFrameDerivativeFunction );
85 
86  // Create aerodynamic angles calculator and set in flight conditions.
87  std::shared_ptr< reference_frames::AerodynamicAngleCalculator > aerodynamicAngleCalculator =
88  std::make_shared< reference_frames::AerodynamicAngleCalculator >(
89  relativeBodyFixedStateFunction,
90  std::bind( &simulation_setup::Body::getCurrentRotationToGlobalFrame, centralBody ),
91  nameOfBodyExertingAcceleration, 1,
92  angleOfAttackFunction, angleOfSideslipFunction, bankAngleFunction, angleUpdateFunction );
93 
94  // Add wind model if present
95  if( centralBody->getAtmosphereModel( )->getWindModel( ) != nullptr )
96  {
97  if( centralBody->getShapeModel( ) == nullptr )
98  {
99  std::cerr << "Warnning, body " << nameOfBodyExertingAcceleration << " has wind model, but no shape model, cannot compute wind as function of altitude " << std::endl;
100  }
101  else
102  {
103  aerodynamicAngleCalculator->setWindModel(
104  centralBody->getAtmosphereModel( )->getWindModel( ), centralBody->getShapeModel( ) );
105  }
106  }
107 
108 
109  // Create flight conditions.
110  std::function< double( const std::string& )> controlSurfaceDeflectionFunction;
111  if( bodyWithFlightConditions->getVehicleSystems( ) != nullptr )
112  {
113  controlSurfaceDeflectionFunction = std::bind(
114  &system_models::VehicleSystems::getCurrentControlSurfaceDeflection,
115  bodyWithFlightConditions->getVehicleSystems( ), std::placeholders::_1 );
116  }
117  std::shared_ptr< aerodynamics::AtmosphericFlightConditions > flightConditions =
118  std::make_shared< aerodynamics::AtmosphericFlightConditions >(
119  centralBody->getAtmosphereModel( ), centralBody->getShapeModel( ),
120  bodyWithFlightConditions->getAerodynamicCoefficientInterface( ), aerodynamicAngleCalculator,
121  controlSurfaceDeflectionFunction );
122 
123  return flightConditions;
124 
125 
126 }

179 std::shared_ptr< aerodynamics::TrimOrientationCalculator > setTrimmedConditions(
180  const std::shared_ptr< aerodynamics::AtmosphericFlightConditions > flightConditions )
181 {
182  // Create trim object.
183  std::shared_ptr< aerodynamics::TrimOrientationCalculator > trimOrientation =
184  std::make_shared< aerodynamics::TrimOrientationCalculator >(
185  flightConditions->getAerodynamicCoefficientInterface( ) );
186 
187  // Create angle-of-attack function from trim object.
188  std::function< std::vector< double >( ) > untrimmedIndependentVariablesFunction =
189  std::bind( &aerodynamics::AtmosphericFlightConditions::getAerodynamicCoefficientIndependentVariables,
190  flightConditions );
191  std::function< std::map< std::string, std::vector< double > >( ) > untrimmedControlSurfaceIndependentVariableFunction =
192  std::bind( &aerodynamics::AtmosphericFlightConditions::getControlSurfaceAerodynamicCoefficientIndependentVariables,
193  flightConditions );
194 
195  flightConditions->getAerodynamicAngleCalculator( )->setOrientationAngleFunctions(
196  std::bind( &aerodynamics::TrimOrientationCalculator::findTrimAngleOfAttackFromFunction, trimOrientation,
197  untrimmedIndependentVariablesFunction, untrimmedControlSurfaceIndependentVariableFunction ) );
198 
199  return trimOrientation;
200 }
