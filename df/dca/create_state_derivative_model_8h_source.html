47 template< typename StateScalarType, typename TimeType >
48 boost::shared_ptr< CentralBodyData< StateScalarType, TimeType > > createCentralBodyData(
49  const std::vector< std::string >& centralBodies,
50  const std::vector< std::string >& bodiesToIntegrate,
51  const simulation_setup::NamedBodyMap& bodyMap )
52 {
53 
54  // Check whether the bodies that are to e integrated exist in bodyMap
55  for( unsigned int i = 0; i < bodiesToIntegrate.size( ); i++ )
56  {
57  if( bodyMap.count( bodiesToIntegrate.at( i ) ) == 0 )
58  {
59  throw std::runtime_error(
60  "Warning when creating CentralBodyData, body " + bodiesToIntegrate.at( i )
61  + " not present in provided body map." );
62  }
63  }
64 
65  std::vector< std::string > centralBodiesToUse;
66  // Generate central body data for requested settings; if no
67  // central bodies provided, get inertial SSB as central body
68  if( centralBodies.size( ) == 0 )
69  {
70  for( unsigned int i = 0; i < bodiesToIntegrate.size( ); i++ )
71  {
72  centralBodiesToUse.push_back( "SSB" );
73  }
74  }
75  else
76  {
77  centralBodiesToUse = centralBodies;
78  }
79 
80  std::map< std::string, boost::function< Eigen::Matrix< StateScalarType, 6, 1 >
81  ( const TimeType ) > > bodyStateFunctions;
82 
83  // Retrieve frame origin state functions
84  for( unsigned int i = 0; i < centralBodiesToUse.size( ); i++ )
85  {
86  if( centralBodiesToUse.at( i ) != "SSB" )
87  {
88  bodyStateFunctions[ centralBodiesToUse.at( i ) ] =
89  boost::bind( &simulation_setup::Body::getStateInBaseFrameFromEphemeris
90  < StateScalarType, TimeType >,
91  bodyMap.at( centralBodiesToUse.at( i ) ), _1 );
92  }
93  else
94  {
95  bodyStateFunctions[ centralBodiesToUse.at( i ) ] = boost::lambda::constant(
96  Eigen::Matrix< StateScalarType, 6, 1 >::Zero( ) );
97  }
98  }
99  return boost::make_shared< CentralBodyData< StateScalarType, TimeType > >(
100  centralBodiesToUse, bodiesToIntegrate, bodyStateFunctions );
101 }

112 template< typename StateScalarType = double, typename TimeType = double >
113 boost::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > >
114 createTranslationalStateDerivativeModel(
115  const boost::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
116  translationPropagatorSettings,
117  const simulation_setup::NamedBodyMap& bodyMap,
118  const TimeType propagationStartTime )
119 {
120 
121  // Create object for frame origin transformations.
122  boost::shared_ptr< CentralBodyData< StateScalarType, TimeType > > centralBodyData =
123  createCentralBodyData< StateScalarType, TimeType >(
124  translationPropagatorSettings->centralBodies_,
125  translationPropagatorSettings->bodiesToIntegrate_,
126  bodyMap );
127 
128  boost::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > stateDerivativeModel;
129 
130  // Check propagator type and create corresponding state derivative object.
131  switch( translationPropagatorSettings->propagator_ )
132  {
133  case cowell:
134  {
135  stateDerivativeModel = boost::make_shared<
136  NBodyCowellStateDerivative< StateScalarType, TimeType > >
137  ( translationPropagatorSettings->accelerationsMap_, centralBodyData,
138  translationPropagatorSettings->bodiesToIntegrate_ );
139  break;
140  }
141  case encke:
142  {
143  // Calculate initial Kepler elements for Encke propagator
144  std::vector< Eigen::Matrix< StateScalarType, 6, 1 > > initialKeplerElements;
145  initialKeplerElements.resize( translationPropagatorSettings->bodiesToIntegrate_.size( ) );
146  std::vector< std::string > centralBodies = translationPropagatorSettings->centralBodies_;
147 
148  for( unsigned int i = 0; i < translationPropagatorSettings->bodiesToIntegrate_.size( ); i++ )
149  {
150  if( bodyMap.count( centralBodies[ i ] ) == 0 )
151  {
152  std::string errorMessage = "Error when creating Encke propagator, did not find central body " +
153  boost::lexical_cast< std::string >( centralBodies[ i ] );
154  throw std::runtime_error( errorMessage );
155  }
156  initialKeplerElements[ i ] = orbital_element_conversions::convertCartesianToKeplerianElements< StateScalarType >(
157  translationPropagatorSettings->getInitialStates( ).segment( i * 6, 6 ), static_cast< StateScalarType >(
158  bodyMap.at( centralBodies[ i ] )->getGravityFieldModel( )->getGravitationalParameter( ) ) );
159  }
160 
161  // Create Encke state derivative object.
162  stateDerivativeModel = boost::make_shared< NBodyEnckeStateDerivative< StateScalarType, TimeType > >
163  ( translationPropagatorSettings->accelerationsMap_, centralBodyData, translationPropagatorSettings->bodiesToIntegrate_,
164  initialKeplerElements, propagationStartTime );
165 
166  break;
167  }
168  case gauss_keplerian:
169  {
170  // Create Encke state derivative object.
171  stateDerivativeModel = boost::make_shared< NBodyGaussKeplerStateDerivative< StateScalarType, TimeType > >
172  ( translationPropagatorSettings->accelerationsMap_, centralBodyData,
173  translationPropagatorSettings->bodiesToIntegrate_ );
174 
175  break;
176  }
177  case gauss_modified_equinoctial:
178  {
179  std::vector< Eigen::Matrix< StateScalarType, 6, 1 > > initialKeplerElements;
180  std::vector< std::string > centralBodies = translationPropagatorSettings->centralBodies_;
181 
182  for( unsigned int i = 0; i < translationPropagatorSettings->bodiesToIntegrate_.size( ); i++ )
183  {
184  if( bodyMap.count( centralBodies[ i ] ) == 0 )
185  {
186  std::string errorMessage = "Error when creating Encke propagator, did not find central body " +
187  boost::lexical_cast< std::string >( centralBodies[ i ] );
188  throw std::runtime_error( errorMessage );
189  }
190  initialKeplerElements.push_back( orbital_element_conversions::convertCartesianToKeplerianElements< StateScalarType >(
191  translationPropagatorSettings->getInitialStates( ).segment( i * 6, 6 ), static_cast< StateScalarType >(
192  bodyMap.at( centralBodies[ i ] )->getGravityFieldModel( )->getGravitationalParameter( ) ) ) );
193  }
194 
195  // Create Encke state derivative object.:
196  stateDerivativeModel = boost::make_shared< NBodyGaussModifiedEquinictialStateDerivative< StateScalarType, TimeType > >
197  ( translationPropagatorSettings->accelerationsMap_, centralBodyData,
198  translationPropagatorSettings->bodiesToIntegrate_, initialKeplerElements );
199 
200  break;
201  }
202  default:
203  throw std::runtime_error(
204  "Error, did not recognize translational state propagation type: " +
205  boost::lexical_cast< std::string >( translationPropagatorSettings->propagator_ ) );
206  }
207  return stateDerivativeModel;
208 }

262 template< typename StateScalarType = double, typename TimeType = double >
263 boost::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > >
264 createStateDerivativeModel(
265  const boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
266  const simulation_setup::NamedBodyMap& bodyMap,
267  const TimeType propagationStartTime )
268 {
269  boost::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > stateDerivativeModel;
270 
271  // Check dynamics type and call associated function to create
272  // specific type of state derivative model.
273  switch( propagatorSettings->getStateType( ) )
274  {
275  case transational_state:
276  {
277  // Check input consistency.
278  boost::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
279  translationPropagatorSettings =
280  boost::dynamic_pointer_cast<
281  TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
282  if( translationPropagatorSettings == NULL )
283  {
284  throw std::runtime_error(
285  "Error, expected translational state propagation settings when making state derivative model" );
286  }
287  else
288  {
289  stateDerivativeModel = createTranslationalStateDerivativeModel< StateScalarType, TimeType >(
290  translationPropagatorSettings, bodyMap, propagationStartTime );
291  }
292  break;
293  }
294  case rotational_state:
295  {
296  boost::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationPropagatorSettings =
297  boost::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
298  if( rotationPropagatorSettings == NULL )
299  {
300  throw std::runtime_error(
301  "Error, expected rotation state propagation settings when making state derivative model" );
302  }
303  else
304  {
305  stateDerivativeModel = createRotationalStateDerivativeModel< StateScalarType, TimeType >(
306  rotationPropagatorSettings, bodyMap, propagationStartTime );
307  }
308  break;
309  }
310  case body_mass_state:
311  {
312  // Check input consistency.
313  boost::shared_ptr< MassPropagatorSettings< StateScalarType > > massPropagatorSettings =
314  boost::dynamic_pointer_cast< MassPropagatorSettings< StateScalarType > >( propagatorSettings );
315  if( massPropagatorSettings == NULL )
316  {
317  throw std::runtime_error(
318  "Error, expected mass propagation settings when making state derivative model" );
319  }
320  else
321  {
322  stateDerivativeModel = createBodyMassStateDerivativeModel< StateScalarType, TimeType >(
323  massPropagatorSettings, bodyMap );
324  }
325  break;
326  }
327  case custom_state:
328  {
329  // Check input consistency.
330  boost::shared_ptr< CustomStatePropagatorSettings< StateScalarType, TimeType > > customPropagatorSettings =
331  boost::dynamic_pointer_cast< CustomStatePropagatorSettings< StateScalarType, TimeType > >(
332  propagatorSettings );
333  if( customPropagatorSettings == NULL )
334  {
335  throw std::runtime_error(
336  "Error, expected custom propagation settings when making state derivative model" );
337  }
338  else
339  {
340  stateDerivativeModel = boost::make_shared< CustomStateDerivative< StateScalarType, TimeType > >(
341  customPropagatorSettings->stateDerivativeFunction_, customPropagatorSettings->stateSize_ );
342  }
343  break;
344  }
345  default:
346  throw std::runtime_error(
347  "Error, could not process state type "
348  + boost::lexical_cast< std::string >( propagatorSettings->getStateType( ) )
349  + " when making state derivative model" );
350  }
351  return stateDerivativeModel;
352 }

354 template< typename StateScalarType = double >
355 void setMultiTypePropagationClosure(
356  const boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
357  const simulation_setup::NamedBodyMap& bodyMap )
358 {
359  boost::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
360  boost::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
361  if( multiTypePropagatorSettings != NULL )
362  {
363  if( multiTypePropagatorSettings->propagatorSettingsMap_.count( transational_state ) > 0 &&
364  multiTypePropagatorSettings->propagatorSettingsMap_.count( rotational_state ) > 0 )
365  {
366  std::vector< boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > translationalStateSettings =
367  multiTypePropagatorSettings->propagatorSettingsMap_.at( transational_state );
368  std::vector< boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > rotationalStateSettings =
369  multiTypePropagatorSettings->propagatorSettingsMap_.at( rotational_state );
370 
371  std::vector< std::string > bodiesWithAerodynamicAcceleration;
372  for( unsigned int i = 0; i < translationalStateSettings.size( ); i++ )
373  {
374  boost::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > > currentTranslationalState =
375  boost::dynamic_pointer_cast< TranslationalStatePropagatorSettings< StateScalarType > >(
376  translationalStateSettings.at( i ) );
377  basic_astrodynamics::AccelerationMap currentAccelerationsMap = currentTranslationalState->accelerationsMap_;
378  for( basic_astrodynamics::AccelerationMap::const_iterator accelerationIterator = currentAccelerationsMap.begin( );
379  accelerationIterator != currentAccelerationsMap.end( ); accelerationIterator++ )
380  {
381  for( basic_astrodynamics::SingleBodyAccelerationMap::const_iterator singleBodyIterator =
382  accelerationIterator->second.begin( ); singleBodyIterator != accelerationIterator->second.end( );
383  singleBodyIterator++ )
384  {
385  for( unsigned int j = 0; j < singleBodyIterator->second.size( ); j++ )
386  {
387  if( std::find( bodiesWithAerodynamicAcceleration.begin( ), bodiesWithAerodynamicAcceleration.end( ),
388  accelerationIterator->first ) == bodiesWithAerodynamicAcceleration.end( ) )
389  {
390  if( basic_astrodynamics::getAccelerationModelType( singleBodyIterator->second.at( j ) ) ==
391  basic_astrodynamics::aerodynamic )
392  {
393  bodiesWithAerodynamicAcceleration.push_back( accelerationIterator->first );
394  }
395  }
396  }
397  }
398  }
399  }
400 
401  std::vector< std::string > bodiesWithPropagatedRotation;
402  for( unsigned int i = 0; i < rotationalStateSettings.size( ); i++ )
403  {
404  boost::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > currentTranslationalState =
405  boost::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >(
406  rotationalStateSettings.at( i ) );
407  std::vector< std::string > currentBodiesWithPropagatedRotation = currentTranslationalState->bodiesToIntegrate_;
408  for( unsigned int j = 0; j < currentBodiesWithPropagatedRotation.size( ); j++ )
409  {
410  if( std::find( bodiesWithPropagatedRotation.begin( ), bodiesWithPropagatedRotation.end( ),
411  currentBodiesWithPropagatedRotation.at( j ) ) == bodiesWithPropagatedRotation.end( ) )
412  {
413  bodiesWithPropagatedRotation.push_back( currentBodiesWithPropagatedRotation.at( j ) );
414  }
415  }
416  }
417 
418  std::vector< std::string > bodiesWithAerodynamicRotationalClosure;
419  for( unsigned int i = 0; i < bodiesWithPropagatedRotation.size( ); i++ )
420  {
421  if( std::find( bodiesWithAerodynamicAcceleration.begin( ), bodiesWithAerodynamicAcceleration.end( ),
422  bodiesWithPropagatedRotation.at( i ) ) != bodiesWithAerodynamicAcceleration.end( ) )
423  {
424  bodiesWithAerodynamicRotationalClosure.push_back( bodiesWithPropagatedRotation.at( i ) );
425  }
426  }
427 
428  for( unsigned int i = 0; i < bodiesWithPropagatedRotation.size( ); i++ )
429  {
430  std::cout<<"Setting closure "<<std::endl;
431  boost::shared_ptr< aerodynamics::FlightConditions > currentFlightConditions =
432  bodyMap.at( bodiesWithPropagatedRotation.at( i ) )->getFlightConditions( );
433  reference_frames::setAerodynamicDependentOrientationCalculatorClosure(
434  boost::bind( &simulation_setup::Body::getCurrentRotationToLocalFrame,
435  bodyMap.at( bodiesWithPropagatedRotation.at( i ) ) ),
436  currentFlightConditions->getAerodynamicAngleCalculator( ) );
437  }
438  }
439  }
440 }

453 template< typename StateScalarType = double, typename TimeType = double >
454 std::vector< boost::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > >
455 createStateDerivativeModels(
456  const boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
457  const simulation_setup::NamedBodyMap& bodyMap,
458  const TimeType propagationStartTime )
459 {
460  std::vector< boost::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > >
461  stateDerivativeModels;
462 
463  // Check type of state derivative model and call associated create function.
464  switch( propagatorSettings->getStateType( ) )
465  {
466  // If hybrid, call create function separately for each entry.
467  case hybrid:
468  {
469  boost::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
470  boost::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
471 
472  // Iterate over all propagation settings
473  for( typename std::map< IntegratedStateType,
474  std::vector< boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >::iterator
475  propagatorIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
476  propagatorIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); propagatorIterator++ )
477  {
478  for( unsigned int i = 0; i < propagatorIterator->second.size( ); i++ )
479  {
480  // Call create function for current propagation settings.
481  if( propagatorIterator->first != hybrid )
482  {
483  stateDerivativeModels.push_back( createStateDerivativeModel< StateScalarType, TimeType >(
484  propagatorIterator->second.at( i ), bodyMap, propagationStartTime ) );
485  }
486  else
487  {
488  throw std::runtime_error(
489  "Error when making state derivative model, cannot process nested hybrid propagators" );
490  }
491  }
492  }
493 
494  setMultiTypePropagationClosure( propagatorSettings, bodyMap );
495 
496  break;
497  }
498  // If not hybrid, call create function for single object directly.
499  default:
500  stateDerivativeModels.push_back( createStateDerivativeModel< StateScalarType, TimeType >(
501  propagatorSettings, bodyMap, propagationStartTime ) );
502  }
503 
504  return stateDerivativeModels;
505 }
