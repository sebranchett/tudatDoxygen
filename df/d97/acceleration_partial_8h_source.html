66  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
67  getDerivativeFunctionWrtStateOfIntegratedBody(
68  const std::pair< std::string, std::string >& stateReferencePoint,
69  const propagators::IntegratedStateType integratedStateType )
70  {
71  // Initialize to empty function; 0 parameter size.
72  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
73  partialFunction = std::make_pair( std::function< void( Eigen::Block< Eigen::MatrixXd > ) >( ), 0 );
74 
75  // Check if state dependency exists
76  switch( integratedStateType )
77  {
78  case propagators::translational_state:
79  {
80  // Check if reference id is consistent.
81  if( stateReferencePoint.second != "" )
82  {
83  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for dynamics" );
84  }
85  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
86  else if( stateReferencePoint.first == acceleratedBody_ )
87  {
88  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAcceleratedBody, this, std::placeholders::_1 ), 3 );
89  }
90  else if( stateReferencePoint.first == acceleratingBody_ )
91  {
92  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAcceleratingBody, this, std::placeholders::_1 ), 3 );
93  }
94  else if( isAccelerationPartialWrtAdditionalBodyNonnullptr( stateReferencePoint.first ) )
95  {
96  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAdditionalBody,
97  this, std::placeholders::_1, stateReferencePoint.first ), 3 );
98  }
99  break;
100  }
101  case propagators::body_mass_state:
102  {
103  // Check if reference id is consistent.
104  if( stateReferencePoint.second != "" )
105  {
106  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for body mass" );
107  }
108  else if( isStateDerivativeDependentOnIntegratedNonTranslationalState( stateReferencePoint, integratedStateType ) )
109  {
110  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtNonTranslationalStateOfAdditionalBody,
111  this, std::placeholders::_1, stateReferencePoint, integratedStateType ), 1 );
112  }
113  }
114  case propagators::custom_state:
115  {
116  break;
117  }
118  default:
119  std::string errorMessage =
120  "Error when getting state derivative partial acceleration model, dynamics type " +
121  std::to_string( integratedStateType ) + "not recognized" ;
122  throw std::runtime_error( errorMessage );
123  break;
124  }
125 
126 
127  return partialFunction;
128  }

152  bool isStateDerivativeDependentOnIntegratedState(
153  const std::pair< std::string, std::string >& stateReferencePoint,
154  const propagators::IntegratedStateType integratedStateType )
155  {
156  bool isDependent = 0;
157 
158  // Check if state is translational.
159  switch( integratedStateType )
160  {
161  case propagators::translational_state:
162  {
163  // Check if reference id is consistent.
164  if( stateReferencePoint.second != "" )
165  {
166  throw std::runtime_error( "Error when checking state derivative partial dependency of acceleration model, cannot have reference point on body for dynamics" );
167  }
168 
169  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
170  else if( stateReferencePoint.first == acceleratedBody_ || stateReferencePoint.first == acceleratingBody_ ||
171  isAccelerationPartialWrtAdditionalBodyNonnullptr( stateReferencePoint.first ) )
172  {
173  isDependent = 1;
174  }
175  break;
176  }
177  default:
178  isDependent = isStateDerivativeDependentOnIntegratedNonTranslationalState( stateReferencePoint, integratedStateType );
179  break;
180  }
181  return isDependent;
182  }
