67  std::pair< boost::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
68  getDerivativeFunctionWrtStateOfIntegratedBody(
69  const std::pair< std::string, std::string >& stateReferencePoint,
70  const propagators::IntegratedStateType integratedStateType )
71  {
72  // Initialize to empty function; 0 parameter size.
73  std::pair< boost::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
74  partialFunction = std::make_pair( boost::function< void( Eigen::Block< Eigen::MatrixXd > ) >( ), 0 );
75 
76  // Check if state dependency exists
77  switch( integratedStateType )
78  {
79  case propagators::transational_state:
80  {
81  // Check if reference id is consistent.
82  if( stateReferencePoint.second != "" )
83  {
84  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for dynamics" );
85  }
86  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
87  else if( stateReferencePoint.first == acceleratedBody_ )
88  {
89  partialFunction = std::make_pair( boost::bind( &AccelerationPartial::wrtStateOfAcceleratedBody, this, _1 ), 3 );
90  }
91  else if( stateReferencePoint.first == acceleratingBody_ )
92  {
93  partialFunction = std::make_pair( boost::bind( &AccelerationPartial::wrtStateOfAcceleratingBody, this, _1 ), 3 );
94  }
95  else if( isAccelerationPartialWrtAdditionalBodyNonNull( stateReferencePoint.first ) )
96  {
97  partialFunction = std::make_pair( boost::bind( &AccelerationPartial::wrtStateOfAdditionalBody,
98  this, _1, stateReferencePoint.first ), 3 );
99  }
100  break;
101  }
102  case propagators::body_mass_state:
103  {
104  // Check if reference id is consistent.
105  if( stateReferencePoint.second != "" )
106  {
107  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for body mass" );
108  }
109  else if( isStateDerivativeDependentOnIntegratedNonTranslationalState( stateReferencePoint, integratedStateType ) )
110  {
111  partialFunction = std::make_pair( boost::bind( &AccelerationPartial::wrtNonTranslationalStateOfAdditionalBody,
112  this, _1, stateReferencePoint, integratedStateType ), 1 );
113  }
114  }
115  case propagators::custom_state:
116  {
117  break;
118  }
119  default:
120  std::string errorMessage =
121  "Error when getting state derivative partial acceleration model, dynamics type " +
122  std::to_string( integratedStateType ) + "not recognized" ;
123  throw std::runtime_error( errorMessage );
124  break;
125  }
126 
127 
128  return partialFunction;
129  }

156  {
157  bool isDependent = 0;

161  {
162  case propagators::transational_state:
163  {
164  // Check if reference id is consistent.
165  if( stateReferencePoint.second != "" )
166  {
167  throw std::runtime_error( "Error when checking state derivative partial dependency of acceleration model, cannot have reference point on body for dynamics" );
168  }
169 
170  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
171  else if( stateReferencePoint.first == acceleratedBody_ || stateReferencePoint.first == acceleratingBody_ ||
172  isAccelerationPartialWrtAdditionalBodyNonNull( stateReferencePoint.first ) )
173  {
174  isDependent = 1;
175  }
176  break;
177  }
178  default:
179  isDependent = isStateDerivativeDependentOnIntegratedNonTranslationalState( stateReferencePoint, integratedStateType );
180  break;
181  }
182  return isDependent;
