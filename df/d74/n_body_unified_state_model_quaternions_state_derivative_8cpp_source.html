22 Eigen::Vector7d computeStateDerivativeForUnifiedStateModelQuaternions(
23  const Eigen::Vector7d& currentUnifiedStateModelElements,
24  const Eigen::Vector3d& accelerationsInRswFrame,
25  const double sineLambda,
26  const double cosineLambda,
27  const double gammaParameter,
28  const Eigen::Vector3d& rotationalVelocityVector,
29  const Eigen::Vector3d& pAuxiliaryVector )
30 {
31  // Compute supporting parameters
32  Eigen::Matrix3d hodographMatrix = Eigen::Matrix3d::Zero( );
33  hodographMatrix( 0, 1 ) = - pAuxiliaryVector( 0 );
34  hodographMatrix( 1, 0 ) = cosineLambda;
35  hodographMatrix( 1, 1 ) = - ( 1.0 + pAuxiliaryVector( 0 ) ) * sineLambda;
36  hodographMatrix( 1, 2 ) = - gammaParameter * pAuxiliaryVector( 1 );
37  hodographMatrix( 2, 0 ) = sineLambda;
38  hodographMatrix( 2, 1 ) = ( 1.0 + pAuxiliaryVector( 0 ) ) * cosineLambda;
39  hodographMatrix( 2, 2 ) = gammaParameter * pAuxiliaryVector( 2 );
40 
41  // Evaluate USM7 equations.
42  Eigen::Vector7d stateDerivative;
43  stateDerivative.segment( 0, 3 ) = hodographMatrix * accelerationsInRswFrame;
44  stateDerivative.segment( 3, 4 ) = calculateQuaternionDerivative( currentUnifiedStateModelElements.segment( 3, 4 ),
45  rotationalVelocityVector );
46 
47  // Give output
48  return stateDerivative;
49 }

52 Eigen::Vector7d computeStateDerivativeForUnifiedStateModelQuaternions(
53  const Eigen::Vector7d& currentUnifiedStateModelElements,
54  const Eigen::Vector3d& accelerationsInRswFrame,
55  const double centralBodyGravitationalParameter )
56 {
57  using namespace orbital_element_conversions;
58 
59  // Retrieve USM elements
60  double etaQuaternionParameter = currentUnifiedStateModelElements( etaUSM7Index );
61  double epsilon1QuaternionParameter = currentUnifiedStateModelElements( epsilon1USM7Index );
62  double epsilon2QuaternionParameter = currentUnifiedStateModelElements( epsilon2USM7Index );
63  double epsilon3QuaternionParameter = currentUnifiedStateModelElements( epsilon3USM7Index );
64 
65  // Compute supporting parameters
66  double denominator = std::pow( epsilon3QuaternionParameter, 2 ) + std::pow( etaQuaternionParameter, 2 );
67  double sineLambda = ( 2 * epsilon3QuaternionParameter * etaQuaternionParameter ) / denominator;
68  double cosineLambda = ( std::pow( etaQuaternionParameter, 2 ) - std::pow( epsilon3QuaternionParameter, 2 ) ) / denominator;
69  double gammaParameter = ( epsilon1QuaternionParameter * epsilon3QuaternionParameter -
70  epsilon2QuaternionParameter * etaQuaternionParameter ) / denominator;
71 
72  double velocityHodographParameter = currentUnifiedStateModelElements( CHodographUSM7Index ) -
73  currentUnifiedStateModelElements( Rf1HodographUSM7Index ) * sineLambda +
74  currentUnifiedStateModelElements( Rf2HodographUSM7Index ) * cosineLambda;
75  Eigen::Vector3d rotationalVelocityVector = Eigen::Vector3d::Zero( );
76  rotationalVelocityVector( 0 ) = accelerationsInRswFrame( 2 ) / velocityHodographParameter;
77  rotationalVelocityVector( 2 ) = std::pow( velocityHodographParameter, 2 ) *
78  currentUnifiedStateModelElements( CHodographUSM7Index ) / centralBodyGravitationalParameter;
79 
80  Eigen::Vector3d pAuxiliaryVector = Eigen::Vector3d::Zero( );
81  pAuxiliaryVector( 0 ) = currentUnifiedStateModelElements( CHodographUSM7Index );
82  pAuxiliaryVector( 1 ) = currentUnifiedStateModelElements( Rf2HodographUSM7Index );
83  pAuxiliaryVector( 2 ) = currentUnifiedStateModelElements( Rf1HodographUSM7Index );
84  pAuxiliaryVector /= velocityHodographParameter;
85 
86  // Evaluate USM7 equations
87  return computeStateDerivativeForUnifiedStateModelQuaternions(
88  currentUnifiedStateModelElements, accelerationsInRswFrame, sineLambda,
89  cosineLambda, gammaParameter, rotationalVelocityVector, pAuxiliaryVector );
90 }
