116 void from_json( const nlohmann::json& jsonObject,
117  std::shared_ptr< PropagationTerminationSettings >& terminationSettings )
118 {
119  using namespace json_interface;
120  using K = Keys::Termination;
121 
122  // Hybrid
123  if ( isDefined( jsonObject, K::allOf ) || isDefined( jsonObject, K::anyOf ) )
124  {
125  terminationSettings = getAs< std::shared_ptr< PropagationHybridTerminationSettings > >( jsonObject );
126  return;
127  }
128 
129  const std::shared_ptr< VariableSettings > variable =
130  getValue< std::shared_ptr< VariableSettings > >( jsonObject, K::variable );
131  switch ( variable->variableType_ )
132  {
133  case independentVariable:
134  {
135  terminationSettings = std::make_shared< PropagationTimeTerminationSettings >(
136  getValue< double >( jsonObject, { K::upperLimit, SpecialKeys::root / Keys::finalEpoch } ) );
137  return;
138  }
139  case cpuTimeVariable:
140  {
141  terminationSettings = std::make_shared< PropagationCPUTimeTerminationSettings >(
142  getValue< double >( jsonObject, K::upperLimit ) );
143  return;
144  }
145  case dependentVariable:
146  {
147  // If both lower limit and upper limit, create hybrid satistying any of the two conditions
148  if ( isDefined( jsonObject, K::lowerLimit ) && isDefined( jsonObject, K::upperLimit ) )
149  {
150  // Lower limit
151  nlohmann::json lowerLimitObject = jsonObject;
152  lowerLimitObject.erase( K::upperLimit );
153  std::shared_ptr< PropagationTerminationSettings > lowerLimitCondition =
154  getAs< std::shared_ptr< PropagationTerminationSettings > >( lowerLimitObject );
155 
156  // Upper limit
157  nlohmann::json upperLimitObject = jsonObject;
158  upperLimitObject.erase( K::lowerLimit );
159  std::shared_ptr< PropagationTerminationSettings > upperLimitCondition =
160  getAs< std::shared_ptr< PropagationTerminationSettings > >( upperLimitObject );
161 
162  // Combine
163  terminationSettings = std::make_shared< PropagationHybridTerminationSettings >(
164  std::vector< std::shared_ptr< PropagationTerminationSettings > >(
165  { lowerLimitObject, upperLimitObject } ), true );
166  return;
167  }
168 
169  const std::shared_ptr< SingleDependentVariableSaveSettings > dependentVar =
170  std::dynamic_pointer_cast< SingleDependentVariableSaveSettings >( variable );
171  assertNonnullptrPointer( dependentVar );
172 
173  // Limit value
174  double limitValue;
175  bool useLowerLimit;
176  if ( isDefined( jsonObject, K::lowerLimit ) )
177  {
178  limitValue = getValue< double >( jsonObject, K::lowerLimit );
179  useLowerLimit = true;
180  }
181  else
182  {
183  limitValue = getValue< double >( jsonObject, K::upperLimit );
184  useLowerLimit = false;
185  }
186 
187  terminationSettings = std::make_shared< PropagationDependentVariableTerminationSettings >(
188  dependentVar, limitValue, useLowerLimit );
189  return;
190  }
191  default:
192  throw std::runtime_error( "Termination settings cannot be defined as a function of variable: " +
193  getVariableId( variable ) );
194  }
